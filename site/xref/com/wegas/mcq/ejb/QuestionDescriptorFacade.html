82      /**
83       *
84       * @param question
85       * @param player
86       * @param time
87       * @throws ScriptException
88       */
89      public void setCurrentTime(QuestionDescriptor question, Player player, Long time) throws ScriptException, WegasException {
90          QuestionInstance questionInstance = (QuestionInstance) question.getInstance(player);
91          for (Reply reply : questionInstance.getReplies()) {
92              //logger.warn(reply.getStartTime()+"*"+reply.getChoiceDescriptor().getDuration()+"*"+time);
93              if (reply.getStartTime() + reply.getResult().getChoiceDescriptor().getDuration() + 1 == time) {
94                  this.validateReply(player, reply);
95              }
96          }
97      }

99      /**
100      *
101      * @param choiceId
102      * @param playerId
103      * @param startTime
104      * @return
105      */
106     public Reply selectChoice(Long choiceId, Player player, Long startTime) throws WegasException {
107         ChoiceDescriptor choice = em.find(ChoiceDescriptor.class, choiceId);
108 
109         QuestionDescriptor questionDescriptor = (QuestionDescriptor) variableDescriptorFacade.findParentListDescriptor(choice);
110 
111         QuestionInstance questionInstance = (QuestionInstance) questionDescriptor.getInstance(player);
112         Reply reply = new Reply();
113 
114         reply.setStartTime(startTime);
115         reply.setResult(this.getCurrentResult(player, choice));
116         questionInstance.addReply(reply);
117 
118         em.flush();
119         em.refresh(reply);
120 
121         return reply;
122     }

135     private Result getCurrentResult(Player p, ChoiceDescriptor choice) throws WegasException {
136         Result r = choice.getInstance(p).getCurrentResult();
137         if (r == null) {
138             try {
139                 r = choice.getResults().get(0);
140             }
141             catch (ArrayIndexOutOfBoundsException ex) {
142                 throw new WegasException("No result found for choice \"" + choice.getEditorLabel() + "\"", ex);
143             }
144         }
145         return r;
146     }

159     /**
160      *
161      * @param player
162      * @param reply
163      * @throws ScriptException
164      */
165     public void validateReply(Player player, Reply reply) throws ScriptException, WegasException {
166         HashMap<String, AbstractEntity> arguments = new HashMap<String, AbstractEntity>();
167         ChoiceDescriptor choiceDescriptor = reply.getResult().getChoiceDescriptor();
168         //choiceInstance = reply.getResult().getChoiceDescriptor().getInstance(player);
169 
170         reply.setResult(this.getCurrentResult(player, choiceDescriptor));       // Refresh the current result
171 
172         arguments.put("selectedReply", reply);
173         arguments.put("selectedChoice", choiceDescriptor.getInstance(player));
174         arguments.put("selectedQuestion", reply.getQuestionInstance());
175         scriptManager.eval(player, reply.getResult().getImpact(), arguments);
176         scriptManager.eval(player, reply.getResult().getChoiceDescriptor().getImpact(), arguments);
177     }
