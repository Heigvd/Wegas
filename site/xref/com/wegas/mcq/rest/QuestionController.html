63              @PathParam("playerId") Long playerId,

65  
66          checkPermissions(playerFacade.find(playerId).getGame().getId(), playerId);
67          
68          Reply reply = questionDescriptorFacade.selectChoice(choiceId, playerId, new Long(0));
69          //Reply reply =
