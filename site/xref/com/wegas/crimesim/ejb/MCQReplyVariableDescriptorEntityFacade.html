60      public MCQReplyVariableInstanceEntity selectReply(Long replyVariableDescriptorId, Long playerId, Long startTime) {
61          MCQReplyVariableDescriptorEntity reply = this.find(replyVariableDescriptorId);
62          PlayerEntity player = playerEntityFacade.find(playerId);
63  
64          VariableDescriptorEntity vd = reply.getMCQVariableDescriptor();
65          MCQVariableInstanceEntity vi = (MCQVariableInstanceEntity) vd.getVariableInstance(player);
66          MCQReplyVariableInstanceEntity replyInstance = new MCQReplyVariableInstanceEntity();
67  
68          replyInstance.setAnswer(reply.getAnswer());
69          replyInstance.setDescription(reply.getDescription());
70          replyInstance.setName(reply.getName());
71          replyInstance.setStartTime(startTime);
72          replyInstance.setDuration(reply.getDuration());
73          vi.addReply(replyInstance);
74  
75          return replyInstance;
76      }
77  
78      public List<VariableInstanceEntity> validateReply(Long replyVariableInstanceId, Long playerId) {
79          MCQReplyVariableInstanceEntity reply = this.mCQReplyVariableInstanceEntityFacade.find(replyVariableInstanceId);
80  
81          ScriptEntity s = new ScriptEntity();                                    // Run the impact
82          s.setContent(reply.getImpact());
83  
84          return sm.runScript(reply.getMCQVariableInstance().getScope().getVariableDescriptor().getGameModel().getId(), playerId, s);
85      }
