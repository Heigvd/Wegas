25      /**
26       *
27       * @param entities
28       * @param selectOne
29       * @return
30       */
31      public static SelectItem[] getSelectItems(List<?> entities, boolean selectOne) {
32          int size = selectOne ? entities.size() + 1 : entities.size();
33          SelectItem[] items = new SelectItem[size];
34          int i = 0;
35          if (selectOne) {
36              items[0] = new SelectItem("", "---");
37              i++;
38          }
39          for (Object x : entities) {
40              items[i++] = new SelectItem(x, x.toString());
41          }
42          return items;
43      }

45      /**
46       *
47       * @param ex
48       * @param defaultMsg
49       */
50      public static void addErrorMessage(Exception ex, String defaultMsg) {
51          String msg = ex.getLocalizedMessage();
52          if (msg != null && msg.length() > 0) {
53              addErrorMessage(msg);
54          } else {
55              addErrorMessage(defaultMsg);
56          }
57      }
58  
59      /**
60       *
61       * @param messages
62       */
63      public static void addErrorMessages(List<String> messages) {
64          for (String message : messages) {
65              addErrorMessage(message);
66          }
67      }
68  
69      /**
70       *
71       * @param msg
72       */
73      public static void addErrorMessage(String msg) {
74          FacesMessage facesMsg = new FacesMessage(FacesMessage.SEVERITY_ERROR, msg, msg);
75          FacesContext.getCurrentInstance().addMessage(null, facesMsg);
76      }
77  
78      /**
79       *
80       * @param msg
81       */
82      public static void addSuccessMessage(String msg) {
83          FacesMessage facesMsg = new FacesMessage(FacesMessage.SEVERITY_INFO, msg, msg);
84          FacesContext.getCurrentInstance().addMessage("successInfo", facesMsg);
85      }

92      public static String getRequestParameter(String key) {

96      /**
97       *
98       * @param requestParameterName
99       * @param converter
100      * @param component
101      * @return
102      */
103     public static Object getObjectFromRequestParameter(String requestParameterName, Converter converter, UIComponent component) {
104         String theId = JsfUtil.getRequestParameter(requestParameterName);
105         return converter.getAsObject(FacesContext.getCurrentInstance(), component, theId);
