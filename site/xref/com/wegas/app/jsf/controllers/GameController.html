26  @ManagedBean(name = "gameController")
27  @RequestScoped
28  public class GameController extends AbstractGameController {
29  
30      /**
31       *
32       * @throws IOException if the target we dispatch to do not exist
33       */
34      @PostConstruct
35      public void init() throws IOException {
36          final ExternalContext externalContext = FacesContext.getCurrentInstance().getExternalContext();
37  
38          if (this.playerId != null) {                                            // If a playerId is provided, we use it
39             
40              currentPlayer = playerFacade.find(this.getPlayerId());
41              if (currentPlayer == null){
42                  externalContext.dispatch("/wegas-app/view/error/accessdenied.xhtml");
43              } else if (!userFacade.matchCurrentUser(currentPlayer.getId())) {
44                  externalContext.dispatch("/wegas-app/view/error/accessdenied.xhtml");
45              }
46              
47              SecurityUtils.getSubject().checkPermission("Game:View:g" + currentPlayer.getGame().getId());
48              
49          } else if (this.gameModelId != null) {                                  // If we only have a gameModel id, we select the 1st player of the 1st team of the 1st game
50              
51              final GameModel gameModel = gameModelFacade.find(this.gameModelId);
52              currentPlayer = gameModel.getGames().get(0).getTeams().get(0).getPlayers().get(0);
53          
54          } else if (this.gameId != null) {                                       // If we only have a gameModel id,
55              
56  //            SecurityUtils.getSubject().checkPermission("Game:View:g" + this.gameId);
57              
58              try {
59                  currentPlayer = playerFacade.findByGameIdAndUserId(this.gameId, // we try to check if current shiro user is registered to the target game
60                          userFacade.getCurrentUser().getId());
61              
62              } catch (PersistenceException e) {                                               // If we still have nothing                //Cas 1 player lobby
63                  externalContext.dispatch("/wegas-app/view/error/accessdenied.xhtml");
64              }
65          }
66          if (currentPlayer == null) {                                            // If no player could be found, we redirect to an error page
67              externalContext.dispatch("/wegas-app/view/error/gameerror.xhtml");
68          }
69      }
