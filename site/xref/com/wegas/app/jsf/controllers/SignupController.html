28  @ManagedBean(name = "signupController")
29  @SessionScoped
30  public class SignupController implements Serializable {
31  
32      /**
33       *
34       */
35      private String userName;
36      /**
37       *
38       */
39      private String userPass;
40      /**
41       *
42       */
43      private String msg = "";
44  
45      /**
46       *
47       */
48      public SignupController() {
49      }
50  
51      /**
52       *
53       */
54      public void signup() {
55          JNDIAndSaltAwareJdbcRealm cRealm = null;
56          for (Realm r : ( (RealmSecurityManager) SecurityUtils.getSecurityManager() ).getRealms()) {
57              if (r instanceof JNDIAndSaltAwareJdbcRealm) {
58                  cRealm = (JNDIAndSaltAwareJdbcRealm) r;
59              }
60          }
61          try {
62              cRealm.createUser(userName, userPass);
63              this.msg = ResourceBundle.getBundle("com.wegas.app.Bundle").getString("SignupPage_UserCreatedMsg");
64          }
65          catch (SQLException ex) {
66              this.msg = ResourceBundle.getBundle("com.wegas.app.Bundle").getString("SignupPage_UserCreationErrorMsg");
67              Logger.getLogger(SignupController.class.getName()).log(Level.SEVERE, null, ex);
68          }
69      }
70  
71      /**
72       * @return the msg
73       */
74      public String getMsg() {
75          return msg;
76      }
77  
78      /**
79       * @param msg the msg to set
80       */
81      public void setMsg(final String msg) {
82          this.msg = msg;
83      }
84

86       * @return the userName
87       */
88      public String getUserName() {

90      }
