50      /**
51       *
52       * @param to
53       * @param from
54       * @param subject
55       * @param body
56       */
57      public void send(String to, String from,
58              String subject, String body) {
59  
60          Properties props = System.getProperties();
61  
62          ResourceBundle res = ResourceBundle.getBundle("wegas");
63          props.put("mail.smtp.host", res.getString("mail.smtp.host"));           // Attaching to default Session, or we could start a new one
64  
65          Session session = Session.getDefaultInstance(props, null);
66  
67          javax.mail.Message msg = new MimeMessage(session);                                 // Create a new message
68  
69          try {
70              msg.setFrom(new InternetAddress(from));                             // Set the FROM and TO fields
71              msg.setRecipients(javax.mail.Message.RecipientType.TO,
72                      InternetAddress.parse(to, false));
73              // -- We could include CC recipients too --
74              // if (cc != null)
75              // msg.setRecipients(Message.RecipientType.CC
76              // ,InternetAddress.parse(cc, false));
77  
78              msg.setSubject(subject);                                            // Set the subject and body text
79              msg.setText(body);
80              msg.setSentDate(new Date());
81              Transport.send(msg);                                                // Send the message
82              System.out.println("Message sent OK.");
83          }
84          catch (MessagingException ex) {
85              Logger.getLogger(EMailFacade.class.getName()).log(Level.SEVERE, null, ex);
86          }
87      }
