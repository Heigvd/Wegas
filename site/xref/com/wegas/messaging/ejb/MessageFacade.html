28  /**
29   *
30   * @author Francois-Xavier Aeberhard <fx@red-agent.com>
31   */
32  @Stateless
33  @LocalBean
34  public class MessageFacade extends AbstractFacadeImpl<Message> {
35  
36      final static private Logger logger = LoggerFactory.getLogger(MessageFacade.class);
37      /**
38       *
39       */
40      @PersistenceContext(unitName = "wegasPU")
41      private EntityManager em;
42      /**
43       *
44       */
45      @EJB
46      private VariableDescriptorFacade variableDescriptorFacade;
47  
48      /**
49       *
50       */
51      public MessageFacade() {
52          super(Message.class);
53      }
54  
55      /**
56       *
57       * @param messageEvent
58       */
59      public void listener(@Observes MessageEvent messageEvent) {
60          logger.info("Message received for player {}.", messageEvent.getPlayer());
61          this.send(messageEvent.getPlayer(), messageEvent.getMessage());
62      }
63  
64      /**
65       *
66       * @param p
67       * @param msg
68       */
69      public void send(Player p, Message msg) {
70          VariableDescriptor vd = variableDescriptorFacade.findByName(p.getGameModel(), "inbox");
71          InboxInstance inbox = (InboxInstance) vd.getInstance(p);
72          inbox.sendMessage(msg);
73      }
74  
75      /**
76       *
77       * @param p
78       * @param subject
79       * @param body
80       * @param from
81       */
82      public void send(Player p, String subject, String body, String from) {
83          Message msg = new Message();
84          msg.setSubject(subject);
85          msg.setBody(body);
86          msg.setFrom(from);
87          this.send(p, msg);
88      }
89      
90      /**
91       *
92       * @param p
93       * @param subject
94       * @param body
95       * @param from
96       * @param attachements
97       */
98      public void send(Player p, String subject, String body, String from, List<String> attachements) {
99          Message msg = new Message();
100         msg.setSubject(subject);
101         msg.setBody(body);
102         msg.setFrom(from);
103         msg.setAttachements(attachements);
104         this.send(p, msg);
105     }
106 
107     @Override
108     protected EntityManager getEntityManager() {
109         return this.em;
110     }
