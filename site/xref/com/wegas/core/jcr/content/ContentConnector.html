153     public void deleteWorkspace() throws RepositoryException {
154         //throw new UnsupportedOperationException("Jackrabbit: There is currently no programmatic way to delete workspaces. You can delete a workspace by manually removing the workspace directory when the repository instance is not running.");
155         String name = session.getWorkspace().getName();
156         SessionHolder.closeSession(workspace);
157         Session s = SessionHolder.getSession(null);
158         try {
159             s.getWorkspace().deleteWorkspace(name);
160         } catch (UnsupportedRepositoryOperationException ex) {
161             logger.warn("UnsupportedRepositoryOperationException : fallback to clear workspace. Further : improve to remove workspace");
162             session = SessionHolder.getSession(workspace);
163             this.clearWorkspace();
164             SessionHolder.closeSession(workspace);
165         }
166     }

168     public void cloneWorkspace(Long oldGameModelId) throws RepositoryException {
169         ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(oldGameModelId);
170         NodeIterator it = connector.listChildren("/");
171         String path;
172         while (it.hasNext()) {
173             path = it.nextNode().getPath();
174             session.getWorkspace().clone("GM_" + oldGameModelId, path, path, true);
175         }
176         session.save();
177     }

179     public void clearWorkspace() throws RepositoryException {
180         NodeIterator it = this.listChildren("/");
181         while (it.hasNext()) {
182             it.nextNode().remove();
183         }
184         this.save();
185     }

197     public void exportXML(OutputStream out) throws RepositoryException, IOException, SAXException {
198         NodeIterator it = this.listChildren("/");
199         OutputFormat format = new OutputFormat("XML", "UTF-8", true);
200         XMLSerializer serializer = new XMLSerializer(out, format);
201         ContentHandler handler = serializer.asContentHandler();
202         handler.startDocument();
203         handler.startElement("", "", "root", null);
204         while (it.hasNext()) {
205             session.exportDocumentView(it.nextNode().getPath(), handler, false, false);
206         }
207         handler.endElement("", "", "root");
208         handler.endDocument();
209     }

211     /**
212      * Check for custom namespaces and register them if they don't exist
213      *
214      * @throws RepositoryException
215      */
216     private void initializeNamespaces() throws RepositoryException {
217         for (String prefix : WFSConfig.namespaces.keySet()) {
218             try {
219                 session.getWorkspace().getNamespaceRegistry().getURI(prefix);
220             } catch (NamespaceException e) {
221                 session.getWorkspace().getNamespaceRegistry().registerNamespace(prefix, WFSConfig.namespaces.get(prefix));
222             }
223         }
