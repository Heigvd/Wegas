27  @XmlRootElement
28  public class DirectoryDescriptor extends AbstractContentDescriptor {
29  
30      /**
31       * Directory mime-type
32       */
33      public static final String MIME_TYPE = "application/wfs-directory";
34  
35      public DirectoryDescriptor(String absolutePath, ContentConnector contentConnector) {
36          super(absolutePath, contentConnector);
37          this.mimeType = MIME_TYPE;
38      }
39  
40      public DirectoryDescriptor(String name, String path, ContentConnector contentConnector) {
41          super(name, path, contentConnector);
42          this.mimeType = MIME_TYPE;
43      }
44  
45      @XmlTransient
46      public boolean isRootDirectory() {
47          return this.fileSystemAbsolutePath.equals("/");
48      }
49  
50      @JsonProperty("bytes")
51      @Override
52      public Long getBytes() {
53          List<AbstractContentDescriptor> nodes = new ArrayList<>();
54          try {
55              nodes = this.list();
56          } catch (RepositoryException ex) {
57          }
58          Long sum = new Long(0);
59          for (AbstractContentDescriptor n : nodes) {
60              sum += n.getBytes();
61          }
62          return sum;
63      }
64  
65      @XmlTransient
66      public List<AbstractContentDescriptor> list() throws RepositoryException {
67          NodeIterator nodeIterator = this.connector.listChildren(this.fileSystemAbsolutePath);
68          List<AbstractContentDescriptor> files = new ArrayList<>();
69          while (nodeIterator.hasNext()) {
70              files.add(DescriptorFactory.getDescriptor(nodeIterator.nextNode(), this.connector));
71          }
72          return files;
73      }
