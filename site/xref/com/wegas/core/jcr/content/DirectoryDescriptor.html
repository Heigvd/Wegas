21   *

24  @XmlRootElement
25  public class DirectoryDescriptor extends AbstractContentDescriptor {
26  
27      /**
28       * Directory mime-type
29       */
30      public static final String MIME_TYPE = "application/wfs-directory";
31  
32      public DirectoryDescriptor(String absolutePath, ContentConnector contentConnector) {
33          super(absolutePath, contentConnector);
34          this.mimeType = MIME_TYPE;
35      }
36  
37      public DirectoryDescriptor(String name, String path, ContentConnector contentConnector) {
38          super(name, path, contentConnector);
39          this.mimeType = MIME_TYPE;
40      }
41  
42      @XmlTransient
43      public boolean isRootDirectory() {
44          return this.fileSystemAbsolutePath.equals("/");
45      }
46  
47      @XmlTransient
48      public List<AbstractContentDescriptor> list() throws RepositoryException {
49          NodeIterator nodeIterator = this.connector.listChildren(this.fileSystemAbsolutePath);
50          List<AbstractContentDescriptor> files = new ArrayList<>();
51          while (nodeIterator.hasNext()) {
52              files.add(DescriptorFactory.getDescriptor(nodeIterator.nextNode(), this.connector));
53          }
54          return files;
55      }
