11  package com.wegas.core.jcr.content;

13  import javax.jcr.Node;
14  import javax.jcr.RepositoryException;
15  import org.slf4j.LoggerFactory;

17  /**
18   *
19   * @author Cyril Junod <cyril.junod at gmail.com>
20   */
21  public class DescriptorFactory {
22  
23      static final private org.slf4j.Logger logger = LoggerFactory.getLogger(DescriptorFactory.class);
24  
25      public static AbstractContentDescriptor getDescriptor(Node node, ContentConnector contentConnector) {
26          AbstractContentDescriptor abstractContentDescriptor = null;
27          String nodePath = null;
28          String mimeType = null;
29  
30          try {
31              nodePath = node.getPath();
32              if (nodePath.equals("/")) {
33                  return new DirectoryDescriptor(nodePath, contentConnector);     //Root Node
34              }
35              mimeType = node.getProperty(WFSConfig.WFS_MIME_TYPE).getString();
36              abstractContentDescriptor =
37                      DirectoryDescriptor.MIME_TYPE.equals(mimeType)
38                      ? new DirectoryDescriptor(nodePath, contentConnector)
39                      : new FileDescriptor(nodePath, mimeType, node.getProperty(WFSConfig.WFS_LAST_MODIFIED).getDate(), node.getProperty(WFSConfig.WFS_DATA).getBinary().getSize(), contentConnector);
40              if (abstractContentDescriptor.exist()) {
41                  abstractContentDescriptor.getContentFromRepository();
42              }
43          } catch (RepositoryException ex) {
44              logger.error("WFS: node error (path:{}, type:{})", nodePath, mimeType);
45          }
46          return abstractContentDescriptor;
47      }
48  
49      public static AbstractContentDescriptor getDescriptor(String absolutePath, ContentConnector contentConnector) throws RepositoryException {
50          AbstractContentDescriptor abstractContentDescriptor = new AbstractContentDescriptor(absolutePath, contentConnector) {};
51          Node node;
52  
53          node = contentConnector.getNode(abstractContentDescriptor.fileSystemAbsolutePath);
54          if (node == null) {
55              return new DirectoryDescriptor(absolutePath, contentConnector);     //return a directory (inexistant)
56          }
57          return getDescriptor(node, contentConnector);
58      }
