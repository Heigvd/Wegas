123     public void propagateDefaultVariableInstance(boolean force) {
124         VariableDescriptorEntity vd = this.getVariableDescriptor();
125         GameModelEntity gm = vd.getGameModel();
126         for (GameEntity g : gm.getGames()) {
127             for (TeamEntity t : g.getTeams()) {
128                 for (PlayerEntity p : t.getPlayers()) {
129                     VariableInstanceEntity vi = this.variableInstances.get(p.getId());
130                     if (vi == null) {
131                         this.setVariableInstance(p.getId(), vd.getDefaultVariableInstance().clone());
132                     } else if (force) {
133                         vi.merge(vd.getDefaultVariableInstance());
134                     }
135                 }
136             }
137         }
