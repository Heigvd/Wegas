89      @PostLoad
90      public void onLoad() {
91          try {
92              this.triggerEvent = this.getStates().get(1L).getTransitions().get(0).getTriggerCondition();
93          } catch (NullPointerException e) {
94              this.triggerEvent = null;
95          }
96          try {
97              if (this.oneShot) {
98                  this.postTriggerEvent = this.getStates().get(2L).getOnEnterEvent();
99              } else {
100                 this.postTriggerEvent = this.getStates().get(1L).getOnEnterEvent();
101             }
102         } catch (NullPointerException e) {
103             this.postTriggerEvent = null;
104         }
105     }
106 
107     @PrePersist
108     public void buildStateMachine() {
109         State initialState = new State();
110         Transition transition = new Transition();
111         transition.setTriggerCondition(this.triggerEvent);
112         List<Transition> transitions = new ArrayList<>(1);
113         transitions.add(transition);
114         initialState.setTransitions(transitions);
115         HashMap<Long, State> states = new HashMap<>();
116         states.put(1L, initialState);
117         if (this.oneShot != null && this.oneShot) {
118             transition.setNextStateId(2L);
119             State finalState = new State();
120             finalState.setOnEnterEvent(this.postTriggerEvent);
121             states.put(2L, finalState);
122         } else {
123             initialState.setOnEnterEvent(this.postTriggerEvent);
124             transition.setNextStateId(1L);
125         }
126         ((TriggerInstance) this.getDefaultInstance()).setCurrentStateId(1L);
127         this.setStates(states);
128     }
