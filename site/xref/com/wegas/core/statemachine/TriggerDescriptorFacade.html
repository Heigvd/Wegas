28  @Stateless
29  public class TriggerDescriptorFacade extends AbstractFacade<TriggerDescriptorEntity> {
30  
31      @PersistenceContext(unitName = "wegasPU")
32      private EntityManager em;
33      @EJB
34      private GameModelFacade gameModelEntityFacade;
35  
36      public TriggerDescriptorFacade() {
37          super(TriggerDescriptorEntity.class);
38      }
39  
40      public void create(Long gameModelId, VariableDescriptorEntity triggerDescriptor) {
41          this.gameModelEntityFacade.find(gameModelId).addVariableDescriptor(triggerDescriptor);
42          em.persist(triggerDescriptor);
43          //create initial state
44          State tmpInitialState = ((StateMachineDescriptorEntity) triggerDescriptor).getStates().get(((StateMachineDescriptorEntity) triggerDescriptor).getInitialStateId());
45          StateMachineInstanceEntity defaultInstance = (StateMachineInstanceEntity) triggerDescriptor.getDefaultVariableInstance();
46          defaultInstance.setCurrentState(tmpInitialState);
47          ((StateMachineDescriptorEntity) triggerDescriptor).setInitialStateId(tmpInitialState.getId());
48          //reset instance
49          triggerDescriptor.getScope().propagateDefaultInstance(true);
50          em.flush();
51          em.refresh(triggerDescriptor);
52      }
53  
54      @Override
55      protected EntityManager getEntityManager() {
56          return this.em;
57      }
58  }
