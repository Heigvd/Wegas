37  @RequestScoped
38  public class StateMachineRunner implements Serializable {
39  
40      static final private org.slf4j.Logger logger = LoggerFactory.getLogger(StateMachineRunner.class);
41      @EJB
42      private VariableDescriptorFacade variableDescriptorFacade;
43      @EJB
44      private ScriptManager scriptManager;
45      /**
46       * StateMachineRunner is running
47       */
48      private Boolean run = false;
49      private Integer steps = 0;
50      private HashSet<StateMachineInstanceEntity> stateMachines = new HashSet<>();
51      private HashSet<Transition> passedTransitions = new HashSet<>();
52      @Inject
53      private GameManager gameManager;
54  
55      public StateMachineRunner() {
56      }
57  
58      public void entityUpdateListener(@Observes GameManager.PlayerAction playerAction) {
59          if (run) {
60              logger.info("Running, received changed {}", gameManager.getUpdatedInstances());
61              return;
62          }
63          //gameManager.clearUpdatedInstances();
64          //TODO: Should Eval without firing Event, lock SMInstance (concurrency)
65          run = true;
66          if (stateMachines.isEmpty()) {                                          // load stateMachines only once
67              Long gmId = gameManager.getGameModel().getId();
68              List<VariableDescriptorEntity> stateMachineDescriptors = variableDescriptorFacade.findByClassAndGameModelId(StateMachineDescriptorEntity.class, gmId);
69              for (VariableDescriptorEntity stateMachineDescriptor : stateMachineDescriptors) {
70                  stateMachines.add((StateMachineInstanceEntity) stateMachineDescriptor.getScope().getVariableInstance(gameManager.getCurrentPlayer()));
71              }
72              logger.info("StateMachineInstance(s) found: {}", stateMachines);
73          }
74          //Put that in the SM Facade
75          ArrayList<ScriptEntity> impacts = new ArrayList<>();
76          for (StateMachineInstanceEntity stateMachine : stateMachines) {
77              List<Transition> transitions = stateMachine.getCurrentState().getTransitions();
78              for (Transition transition : transitions) {
79                  Boolean validTransition = false;
80                  try {
81                      validTransition = (Boolean) scriptManager.eval(gameManager.getCurrentPlayer(), transition.getTriggerCondition());
82                  } catch (ScriptException ex) {
83                      logger.error("Script Failed : {} returned: {}", transition.getTriggerCondition(), ex);
84                  }
85                  if (validTransition) {
86                      if (passedTransitions.contains(transition)) {
87                          logger.warn("Loop detected, already marked {} IN {}", transition, passedTransitions);
88                      } else {
89                          stateMachine.setCurrentStateId(transition.getNextStateId());
90                          impacts.add(stateMachine.getCurrentState().getOnEnterEvent());
91                          passedTransitions.add(transition);
92                          break;
93                      }
94                  }
95  
96              }
97          }
98          steps += 1;
99          run = false;
100         try {
101             scriptManager.eval(gameManager.getCurrentPlayer(), impacts, null);
102         } catch (ScriptException ex) {
103             logger.error("Script Failed : {} returned: {}", impacts, ex);
104         }
105     }
106 
107     @PreDestroy
108     public void logFinalState() {
109         logger.info("#steps[" + steps + "] - Player {} triggered transition(s):{}", gameManager.getCurrentPlayer().getId(), passedTransitions);
110     }
111 }
