76      /**
77       *
78       * @param player
79       * @param scripts A list of ScriptEntities to evaluate, all programming
80       * language should be the same
81       * @param arguments
82       * @return
83       */
84      public Object eval(PlayerEntity player, List<ScriptEntity> scripts, Map<String, AbstractEntity> arguments) throws ScriptException {
85          if (scripts.isEmpty()) {
86              return null;
87          }
88          arguments = (Map<String, AbstractEntity>) (arguments != null ? arguments : new HashMap<>());
89          ScriptEngineManager mgr = new ScriptEngineManager();
90          ScriptEngine engine = mgr.getEngineByName(scripts.get(0).getLanguage());
91          // Invocable invocableEngine = (Invocable) engine;
92          GameModelEntity gm = player.getTeam().getGame().getGameModel();
93          List<VariableInstanceEntity> vis = new ArrayList<VariableInstanceEntity>();
94  
95          gameManager.setCurrentPlayer(player);                                // Set up request execution context
96          gameManager.setGameModel(gm);
97          engine.put("self", player);                                             // Inject constants
98  
99          for (Entry<String, AbstractEntity> arg : arguments.entrySet()) {        // Inject the arguments
100             engine.put(arg.getKey(), arg.getValue());
101         }
102 
103         for (VariableDescriptorEntity vd : gm.getVariableDescriptors()) {       // We inject the variable instances in the script
104             VariableInstanceEntity vi = vd.getVariableInstance(player);
105             engine.put(vd.getName(), vi);
106         }
107         String script = "";
108         for (ScriptEntity scriptEntity : scripts) {
109             script += scriptEntity.getContent() + ";";
110         }
111 
112         Object result = engine.eval(script);
113         em.flush();
114 
115         gameManager.commit();
116 
117         return result;
118     }
