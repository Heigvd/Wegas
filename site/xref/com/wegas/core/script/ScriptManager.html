53      public List<VariableInstanceEntity> runScript(Long gameModelId, Long playerId, ScriptEntity s) {
54          ScriptEngineManager mgr = new ScriptEngineManager();
55          ScriptEngine engine = mgr.getEngineByName("JavaScript");
56          // Invocable invocableEngine = (Invocable) engine;
57          // GameModelEntity gm = gameModelEntityFacade.find(gameModelId);
58          PlayerEntity p = playerEntityFacade.find(playerId);
59          GameModelEntity gm = p.getTeam().getGame().getGameModel();
60  
61          List<VariableInstanceEntity> vis = new ArrayList<VariableInstanceEntity>();
62  
63          try {
64              for (VariableDescriptorEntity vd : gm.getVariableDescriptors()) {   // We inject the variable instances in the script
65                  VariableInstanceEntity vi = vd.getVariableInstance(p);
66                  engine.put(vd.getName(), vi);
67                  vis.add(vi);
68              }
69              engine.eval(s.getContent());                                        // Then we evaluate the script
70          }
71          catch (ScriptException ex) {
72              Logger.getLogger(ScriptManager.class.getName()).log(Level.SEVERE, null, ex);
73          }
74          return vis;
75      }
