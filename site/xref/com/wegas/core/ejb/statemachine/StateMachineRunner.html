75      public void playerUpdated(Player player) {
76  
77          Integer steps = 0;
78          logger.debug("Updated instances {}, transition done : {}", requestManager.getUpdatedInstances());
79          if (run) {
80              logger.debug("Running, received changed {}", requestManager.getUpdatedInstances());
81              return;
82          }
83  
84          run = true;
85          //gameManager.clearUpdatedInstances();
86          //TODO: lock SMInstance (concurrency)
87  
88          /* Find players for each instance */
89          //List<Player> players = new ArrayList<>();
90          Map<Player, HashSet<Transition>> playerTransitions = new HashMap<>();   // store passed transitions.
91          if (player != null) {
92              playerTransitions.put(player, new HashSet<Transition>());
93          } else {
94              for (VariableInstance instance : requestManager.getUpdatedInstances()) {
95                  try {
96                      /* Setup player's passed transitions */
97                      playerTransitions.put(variableInstanceFacade.findAPlayer(instance), new HashSet<Transition>());
98                  } catch (NoPlayerException e) {
99                      logger.warn("{} : {}", e.getClass().getSimpleName(), e.getMessage());
100                 }
101             }
102         }
103         if (playerTransitions.size() < 1) {
104             logger.warn("No players found");
105             return;
106         }
107         /* build a list with each statemachine instance relevant to players (MAP) */
108         Map<StateMachineInstance, Player> statemachinesPlayerMap = new HashMap<>();
109         GameModel gamemodel = ((Player) playerTransitions.keySet().toArray()[0]).getGameModel();
110         List<VariableDescriptor> stateMachineDescriptors = variableDescriptorFacade.findByClass(gamemodel, StateMachineDescriptor.class);
111         for (VariableDescriptor stateMachineDescriptor : stateMachineDescriptors) {
112             for (Player p : playerTransitions.keySet()) {
113                 statemachinesPlayerMap.put((StateMachineInstance) stateMachineDescriptor.getInstance(p), p); //won't duplicate (HashMap) if players share same instance
114             }
115         }
116 
117         logger.debug("StateMachineInstance(s) found: {}", statemachinesPlayerMap);
118         while (this.run(statemachinesPlayerMap, playerTransitions)) {
119             steps += 1;
120         }
121         run = false;
122         if (steps > 0) {
123             for (Player p : playerTransitions.keySet()) {
124                 logger.info("#steps[" + steps + "] - Player {} triggered transition(s):{}", p.getName(), playerTransitions.get(p));
125             }
126         }
127     }

129     private Boolean run(Map<StateMachineInstance, Player> statemachinesPlayerMap, Map<Player, HashSet<Transition>> playerTransitions) {
130 
131         Boolean transitionTriggered = false;
132         Map<Player, List<Script>> playerImpacts = new HashMap<>();
133         Player currentPlayer;
134         List<Script> impacts;
135         HashSet<Transition> passedTransitions;
136 
137         for (StateMachineInstance stateMachine : statemachinesPlayerMap.keySet()) {
138             /* Skip disabled Statemachines */
139             if (!stateMachine.getEnabled()) {
140                 continue;
141             }
142             currentPlayer = statemachinesPlayerMap.get(stateMachine);
143             passedTransitions = playerTransitions.get(currentPlayer);
144 
145 
146             /* Setup player's impact list */
147             if (playerImpacts.get(currentPlayer) == null) {
148                 impacts = new ArrayList<>();
149                 playerImpacts.put(currentPlayer, impacts);
150             } else {
151                 impacts = playerImpacts.get(currentPlayer);
152             }
153             List<Transition> transitions = stateMachine.getCurrentState().getTransitions();
154             /* Loop on each state's transtions until a valid transition is found */
155             for (Transition transition : transitions) {
156                 Boolean validTransition = false;
157 
158                 if (!(transition instanceof DialogueTransition)
159                         && transition.getTriggerCondition() != null) {      //Do not eval Dialogue transition, no condition means invalid transition
160                     requestManager.setPlayer(currentPlayer);
161                     try {
162                         validTransition = (Boolean) scriptManager.eval(transition.getTriggerCondition());
163                     } catch (ScriptException ex) {
164                         validTransition = false;
165                     }
166                 }
167 
168                 if (validTransition == null) {
169                     throw new WegasException("Please review condition [" + stateMachine.getDescriptor().getName() + "]:\n"
170                             + transition.getTriggerCondition().getContent());
171                 } else if (validTransition) {
172                     /* A valid transition has been found */
173                     if (passedTransitions.contains(transition)) {
174                         /* Loop prevention : that player already passed through this transiton */
175                         logger.debug("Loop detected, already marked {} IN {}", transition, passedTransitions);
176                     } else {
177                         passedTransitions.add(transition);                      //Store transition to avoid doing it again
178                         stateMachine.setCurrentStateId(transition.getNextStateId());//Change statemachine's state
179                         impacts.add(transition.getPreStateImpact());            //Prepare for eval
180                         impacts.add(stateMachine.getCurrentState().getOnEnterEvent()); //Prepare for eval
181                         stateMachine.transitionHistoryAdd(transition.getId());  // Adding transition.id to history
182                         transitionTriggered = true;
183                         break;                                                  // A transition has bean found stop searching
184                     }
185                 }
186             }
187         }
188         this.compute(playerImpacts);
189         return transitionTriggered;
190     }

192     /**
193      * Compute for a list of players their impact list
194      *
195      * @param playerImpacts Map a player with a list of corresponding Script
196      * impacts
197      */
198     private void compute(Map<Player, List<Script>> playerImpacts) {
199         for (Iterator<Player> it = playerImpacts.keySet().iterator(); it.hasNext();) {
200             Player p = it.next();
201             requestManager.setPlayer(p);
202             if (playerImpacts.get(p).size() > 0) {
203                 logger.debug("Compute for {}\n{}", p.getName(), playerImpacts.get(p));
204                 try {
205                     scriptManager.eval(playerImpacts.get(p));
206                 } catch (ScriptException | WegasException ex) {
207                     logger.warn("Script failed ", ex);
208                 }
209             }
210         }
211     }
