40  @RequestScoped
41  public class StateMachineRunner implements Serializable {
42  
43      static final private org.slf4j.Logger logger = LoggerFactory.getLogger(StateMachineRunner.class);
44      @EJB
45      private VariableDescriptorFacade variableDescriptorFacade;
46      @EJB
47      private ScriptFacade scriptManager;
48      /**
49       * StateMachineRunner is running
50       */
51      private Boolean run = false;
52      private Integer steps = 0;
53      private HashSet<StateMachineInstance> stateMachines = new HashSet<>();
54      private HashSet<Transition> passedTransitions = new HashSet<>();
55      @Inject
56      private RequestManager requestManager;
57  
58      public StateMachineRunner() {
59      }
60  
61      public void entityUpdateListener(@Observes RequestManager.PlayerAction playerAction) throws WegasException {
62          if (run) {
63              logger.info("Running, received changed {}", requestManager.getUpdatedInstances());
64              return;
65          }
66          //gameManager.clearUpdatedInstances();
67          //TODO: Should Eval without firing Event, lock SMInstance (concurrency)
68          run = true;
69          if (stateMachines.isEmpty()) {                                          // load stateMachines only once
70              GameModel gamemodel = requestManager.getCurrentGameModel();
71              List<VariableDescriptor> stateMachineDescriptors = variableDescriptorFacade.findByClass(gamemodel, StateMachineDescriptor.class);
72              for (VariableDescriptor stateMachineDescriptor : stateMachineDescriptors) {
73                  stateMachines.add((StateMachineInstance) stateMachineDescriptor.getInstance());
74              }
75              logger.info("StateMachineInstance(s) found: {}", stateMachines);
76          }
77          //Put that in the SM Facade
78          ArrayList<Script> impacts = new ArrayList<>();
79          for (StateMachineInstance stateMachine : stateMachines) {
80              List<Transition> transitions = stateMachine.getCurrentState().getTransitions();
81              for (Transition transition : transitions) {
82                  Boolean validTransition = false;
83                  try {
84                      if (!(transition instanceof DialogueTransition) && transition.getTriggerCondition() != null) { //Do not eval Dialogue transition
85                          validTransition = (Boolean) scriptManager.eval(transition.getTriggerCondition());
86                      }
87                  } catch (ScriptException ex) {
88                      logger.error("Script Failed : {} returned: {}", transition.getTriggerCondition(), ex);
89                  }
90                  if (validTransition == null) {
91                      throw new WegasException("Please review condition [" + stateMachine.getDescriptor().getName() + "]:\n" + transition.getTriggerCondition().getContent());
92                  } else if (validTransition) {
93                      if (passedTransitions.contains(transition)) {
94                          logger.warn("Loop detected, already marked {} IN {}", transition, passedTransitions);
95                      } else {
96                          stateMachine.setCurrentStateId(transition.getNextStateId());
97                          if (stateMachine.getCurrentState().getOnEnterEvent() != null) {
98                              impacts.add(stateMachine.getCurrentState().getOnEnterEvent());
99                          }
100                         impacts.add(transition.getPreStateImpact());
101                         passedTransitions.add(transition);
102                         stateMachine.transitionHistoryAdd(transition.getId());  // Adding transition.id to history
103                         break;
104                     }
105                 }
106 
107             }
108         }
109         steps += 1;
110         run = false;
111         try {
112             scriptManager.eval(impacts);
113         } catch (ScriptException ex) {
114             logger.error("Script Failed : {} returned: {}", impacts, ex);
115         }
116     }
117 
118     @PreDestroy
119     public void logFinalState() {
120         logger.info("#steps[" + steps + "] - Player {} triggered transition(s):{}", requestManager.getPlayer().getId(), passedTransitions);
121     }
