49      /**
50       *
51       * @param name
52       * @return
53       */
54      public VariableDescriptorEntity findByName(String name) {
55          CriteriaBuilder cb = em.getCriteriaBuilder();
56          CriteriaQuery cq = cb.createQuery();
57          Root<UserEntity> variableDescriptor = cq.from(VariableDescriptorEntity.class);
58          //cq.where(cb.equal(variableDescriptor.get(VariableDescriptorEntity_.name), name));
59          cq.where(cb.equal(variableDescriptor.get("name"), name));
60          Query q = em.createQuery(cq);
61          return (VariableDescriptorEntity) q.getSingleResult();
62      }

71          findVariableDescriptorsByRootGameModelId.setParameter("gameModelId", gameModelId);
72          return findVariableDescriptorsByRootGameModelId.getResultList();
73      }
