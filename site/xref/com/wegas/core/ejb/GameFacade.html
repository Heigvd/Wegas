74      /**
75       *
76       * @param token
77       * @return
78       */
79      public Game findByToken(String token) {
80          CriteriaBuilder cb = em.getCriteriaBuilder();
81          CriteriaQuery cq = cb.createQuery();
82          Root<Game> game = cq.from(Game.class);
83          cq.where(cb.equal(game.get(Game_.token), token));
84          Query q = em.createQuery(cq);
85          Game g;
86          try {
87              g = (Game) q.getSingleResult();
88          } catch (NoResultException ex) {
89              g = null;
90          } catch (NonUniqueResultException ex) {
91              g = (Game) q.getResultList().get(0);
92          }
93          return g;                                     // If there is more than one game with this token, use the 1st one
94          //return (Game) q.getSingleResult();
95      }

128     /**
129      * Metod return all public games
130      *
131      * @return Collection<Game>
132      */
133     public Collection<Game> getPublicGames(Long userId) {
134 
135         Role pRolle = roleFacade.findByName("Public");
136         Collection<Game> registerdGame = userFacade.registeredGames(userId);
137         Collection<Game> games = new ArrayList<>();
138         for (String permission : pRolle.getPermissions()) {
139             String splitedPermission[] = permission.split(":g");
140             String f = splitedPermission[1].substring(0, 1);
141             if (!f.equals("m") && splitedPermission[0].equals("Game:View")) {
142                 Game g = this.find(Long.parseLong(splitedPermission[1]));
143                 this.em.detach(g);
144                 boolean registerd = false;
145                 for (Game aRegisterdG : registerdGame) {
146                     if (g.equals(aRegisterdG)) {
147                         registerd = true;
148                         break;
149                     }
150                 }
151                 if (!registerd) {
152                     g.setName(g.getGameModel().getName() + " : " + g.getName());
153                     games.add(g);
154                 }
155             }
156         }
157         return games;
158     }
