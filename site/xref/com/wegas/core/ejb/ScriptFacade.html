82      /**
83       *
84       * Fires an engineInvocationEvent, which should be intercepted to customize
85       * engine scope.
86       *
87       * @param player
88       * @param scripts A list of ScriptEntities to evaluate, all programming
89       * language should be the same
90       * @param arguments
91       * @return
92       * @throws ScriptException
93       */
94      public Object eval(List<Script> scripts, Map<String, AbstractEntity> arguments) throws ScriptException, WegasException {
95          while (scripts.remove(null)) {
96          }                                           //remove null scripts
97          if (scripts.isEmpty()) {
98              return null;
99          }
100 
101         ScriptEngineManager mgr = new ScriptEngineManager();                    // Instantiate the corresponding script engine
102         ScriptEngine engine;
103         try {
104             engine = mgr.getEngineByName(scripts.get(0).getLanguage());
105         } catch (NullPointerException ex) {
106             logger.error("Wrong language ?", ex.getMessage(), ex.getStackTrace());
107             return null;
108         }
109         // Invocable invocableEngine = (Invocable) engine;
110 
111         engine.put("self", requestManager.getPlayer());                         // Inject current player
112         Object result = null;
113         try {
114             engineInvocationEvent.fire(
115                     new EngineInvocationEvent(requestManager.getPlayer(), engine)); // Fires the engine invocation event, to allow extensions
116         } catch (ObserverException ex) {
117             throw (WegasException) ex.getCause();
118         } finally {                                                             //Try finishing evaluation
119             for (Entry<String, AbstractEntity> arg : arguments.entrySet()) {        // Inject the arguments
120                 engine.put(arg.getKey(), arg.getValue());
121             }
122 
123             // @fixme test the most performant version
124 
125             String script = "";
126             for (Script s : scripts) {                                              // Evaluate each script
127                 try {
128                     script += s.getContent() + ";";
129                 } catch (NullPointerException ex) {
130                     //script does not exist
131                 }
132                 //result = engine.eval(s.getContent());
133             }
134             try {
135                 result = engine.eval(script);
136             } catch (ScriptException ex) {
137                 logger.warn("{} in\n{}", ex.getMessage(), script);
138                 throw new ScriptException(ex.getMessage(), script, ex.getLineNumber());
139             }
140 
141             em.flush();                                                             // Commit the transaction
142             requestManager.commit();
143         }
144         return result;
145     }

147     /**
148      * Default customization of our engine: inject the script library, the root
149      * variable instances and some libraries.
150      *
151      * @param messageEvent
152      */
153     public void onEngineInstantiation(@Observes ScriptFacade.EngineInvocationEvent evt) throws ScriptException, WegasException {
154         evt.getEngine().put("VariableDescriptorFacade", variableDescriptorFacade); // Inject the variabledescriptor facade
155         evt.getEngine().eval("importPackage(com.wegas.core.script)");           // Inject factory object
156         List<String> errorVariable = new ArrayList<>();
157         for (Entry<String, String> arg : evt.getPlayer().getGameModel().getScriptLibrary().entrySet()) {        // Inject the arguments
158             try {
159                 evt.getEngine().eval(arg.getValue());
160             } catch (ScriptException ex) {
161                 logger.warn("{} in\n{}", ex.getMessage(), arg.getValue());
162                 throw new ScriptException(ex.getMessage(), arg.getValue(), ex.getLineNumber());
163             }
164         }
165 
166         for (VariableDescriptor vd : evt.getPlayer().getGameModel().getVariableDescriptors()) { // We inject the variable instances in the script
167             VariableInstance vi = vd.getInstance(evt.getPlayer());
168             try {
169                 evt.getEngine().put(vd.getName(), vi);
170             } catch (IllegalArgumentException ex) {
171                 errorVariable.add(vd.getEditorLabel());
172 
173             }
174         }
175         if (errorVariable.size() > 0) {
176             StringBuilder allVars = new StringBuilder();
177             Iterator<String> si = errorVariable.iterator();
178             while (si.hasNext()) {
179                 allVars.append(si.next());
180                 if (si.hasNext()) {
181                     allVars.append(",");
182                 }
183             }
184             throw new WegasException("Missing name for Variable [" + allVars.toString() + "]");
185         }
186     }
