29  @Stateless
30  @LocalBean
31  public class TeamFacade extends AbstractFacadeImpl<Team> {

33      /**

35       */
36      @EJB
37      private UserFacade userFacade;
38      /**
39       *
40       */
41      @EJB
42      private GameFacade gameFacade;
43      /**
44       *
45       */
46      @PersistenceContext(unitName = "wegasPU")
47      private EntityManager em;

49      /**
50       *
51       * @param gameId
52       * @param t
53       */
54      public void create(Long gameId, Team t) {
55          Game g = gameFacade.find(gameId);
56          g.addTeam(t);
57          em.flush();
58          em.refresh(t);
59          g.getGameModel().propagateDefaultInstance(false);
60      }
61  
62      public Team findByToken(String token) throws PersistenceException {
63          Query findByToken = em.createNamedQuery("findTeamByToken");
64          findByToken.setParameter("token", token);
65          return (Team) findByToken.getSingleResult();
66      }
67  
68      /**
69       *
70       * @param teamId
71       * @param userId
72       * @return
73       */
74      public Player joinTeam(Team team, User user) {
75          // logger.log(Level.INFO, "Adding user " + userId + " to team: " + teamId + ".");
76          Player p = new Player();
77          p.setUser(user);
78          this.joinTeam(team, p);
79          return p;
80      }
81  
82      public Player joinTeam(Long teamId, Long userId) {
83          // logger.log(Level.INFO, "Adding user " + userId + " to team: " + teamId + ".");
84          return this.joinTeam(this.find(teamId), userFacade.find(userId));
85      }
86  
87      public void joinTeam(Team team, Player player) {
88          team.addPlayer(player);
89          em.flush();
90          em.refresh(player);
91          team.getGame().getGameModel().propagateDefaultInstance(false);
92      }
93  
94      /**
95       *
96       * @param teamId
97       * @param p
98       * @return
99       */
100     public Player createPlayer(Long teamId, Player p) {
101         Team t = this.find(teamId);
102         this.joinTeam(t, p);
103         return p;
104     }
