53      private final String FILENAME_REGEXP = "(\\w|\\.| |-|_)+";

55      /**
56       *
57       * @param gameModelId
58       * @param name
59       * @param note
60       * @param path
61       * @param file
62       * @param details
63       * @return
64       * @throws RepositoryException
65       */
66      @POST
67      @Consumes(MediaType.MULTIPART_FORM_DATA)
68      @Produces(MediaType.APPLICATION_JSON)
69      @Path("upload{directory : .*?}")
70      public AbstractContentDescriptor upload(@PathParam("gameModelId") String gameModelId,
71              @FormDataParam("name") String name,
72              @FormDataParam("note") String note,
73              @FormDataParam("note") String description,
74              @PathParam("directory") String path,
75              @FormDataParam("file") InputStream file,
76              @FormDataParam("file") FormDataBodyPart details) throws RepositoryException, WegasException {
77          logger.debug("File name: {}", details.getContentDisposition().getFileName());
78          if (name == null) {
79              name = details.getContentDisposition().getFileName();
80          }
81          Pattern pattern = Pattern.compile(FILENAME_REGEXP);
82          Matcher matcher = pattern.matcher(name);
83          if (name.equals("") || !matcher.matches()) {
84              throw new WegasException(name + " is not a valid filename.");
85          }
86          ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
87  
88          AbstractContentDescriptor detachedFile = null;
89          AbstractContentDescriptor dir = DescriptorFactory.getDescriptor(path, connector);
90          if (dir.exist()) {                                                  //directory has to exist
91              if (details == null || details.getContentDisposition().getFileName() == null || details.getContentDisposition().getFileName().equals("")) {       //Assuming an empty filename means a directory
92                  detachedFile = new DirectoryDescriptor(name, path, connector);
93              } else {
94                  logger.debug("File name: {}", details.getContentDisposition().getFileName());
95                  detachedFile = new FileDescriptor(name, path, connector);
96              }
97              if (!detachedFile.exist()) {                                        //Node should not exist
98                  note = note == null ? "" : note;
99                  detachedFile.setNote(note);
100                 detachedFile.setDescription(description);
101                 if (detachedFile instanceof FileDescriptor) {
102                     //TODO : check allowed mime-types
103                     try {
104                         ((FileDescriptor) detachedFile).setBase64Data(file, details.getMediaType().toString());
105                         logger.info(details.getFormDataContentDisposition().getFileName() + "(" + details.getMediaType() + ") uploaded as \"" + name + "\"");
106                     } catch (IOException ex) {
107                         logger.error("Error reading uploaded file :", ex);
108                         connector.save();
109                     }
110                 } else {
111                     detachedFile.sync();
112                     logger.info("Directory {} created at {}", detachedFile.getName(), detachedFile.getPath());
113                 }
114             } else {
115                 throw new WegasException(detachedFile.getPath() + "/" + name + " already exists");
116             }
117         } else {
118             logger.debug("Parent Directory does not exist");
119         }
120         connector.save();
121         return detachedFile;
122     }

130     @GET
131     @Path("read{absolutePath : .*?}")
132     public Response read(@PathParam("gameModelId") String gameModelId, @PathParam("absolutePath") String name) {
133         logger.debug("Asking file (/{})", name);
134         AbstractContentDescriptor fileDescriptor;
135         ContentConnector connector = null;
136         Response.ResponseBuilder response = Response.status(404);
137         try {
138             connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
139             fileDescriptor = DescriptorFactory.getDescriptor(name, connector);
140         } catch (PathNotFoundException e) {
141             logger.debug("Asked path does not exist: {}", e.getMessage());
142             connector.save();
143             return response.build();
144         } catch (RepositoryException e) {
145             logger.error("Need to check those errors", e);
146             return response.build();
147         }
148         if (fileDescriptor instanceof FileDescriptor) {
149             response = Response.ok(((FileDescriptor) fileDescriptor).getBase64Data());
150             response.header("Content-Type", fileDescriptor.getMimeType());
151             response.header("Description", fileDescriptor.getDescription());
152             CacheControl cc = new CacheControl();
153             cc.setMaxAge(3600);
154             cc.setPrivate(true);
155             response.cacheControl(cc);
156             response.lastModified(((FileDescriptor) fileDescriptor).getDataLastModified().getTime());
157 
158             try {
159                 ((FileDescriptor) fileDescriptor).getBase64Data().close();
160             } catch (IOException ex) {
161                 Logger.getLogger(FileController.class.getName()).log(Level.SEVERE, null, ex);
162             } finally {
163                 connector.save();
164             }
165         }
166         return response.build();
167     }

180         try {
181             ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
182             AbstractContentDescriptor dir = DescriptorFactory.getDescriptor(directory, connector);
183             if (!dir.exist() || dir instanceof FileDescriptor) {
184                 connector.save();
185                 return null;
186             } else if (dir instanceof DirectoryDescriptor) {
187                 List<AbstractContentDescriptor> ret = ((DirectoryDescriptor) dir).list();
188                 connector.save();
189                 return ret;
190             }
191         } catch (LoginException ex) {
192             Logger.getLogger(FileController.class.getName()).log(Level.SEVERE, null, ex);
193         } catch (RepositoryException ex) {

228     @DELETE
229     @Path("{force: (force/)?}delete{absolutePath : .*?}")
230     @Produces(MediaType.APPLICATION_JSON)
231     public Object delete(@PathParam("gameModelId") String gameModelId,
232             @PathParam("absolutePath") String absolutePath,
233             @PathParam("force") String force) {
234 
235         boolean recursive = force.equals("") ? false : true;
236         logger.debug("Asking delete for node ({}), force {}", absolutePath, recursive);
237         try {
238             ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
239             AbstractContentDescriptor descriptor = DescriptorFactory.getDescriptor(absolutePath, connector);
240             if (descriptor.exist()) {
241                 descriptor.sync();
242                 if (descriptor instanceof DirectoryDescriptor && ((DirectoryDescriptor) descriptor).isRootDirectory()) {
243                     return Response.notModified("Unable to erase Root Directory").build();
244                 }
245                 try {
246                     descriptor.delete(recursive);
247                 } catch (ItemExistsException e) {
248                     return Response.notModified(e.getMessage()).build();
249                 }
250                 connector.save();
251                 return descriptor;
252             } else {
253                 connector.save();
254                 return Response.notModified("Path" + absolutePath + " does not exist").build();
255             }
256         } catch (RepositoryException ex) {
257             logger.error("Really what append here ??", ex);
258         }
259         return null;
260     }
