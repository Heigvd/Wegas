68      /**
69       *
70       * @param gameModelId
71       * @param playerId
72       * @param stateMachineDescriptorId
73       * @param transitionId
74       * @return StateMachineInstance
75       * @throws ScriptException
76       */
77      @GET
78      @Path("{stateMachineDescriptorId : [1-9][0-9]*}/Player/{playerId : [1-9][0-9]*}/Do/{transitionId : [1-9][0-9]*}")
79      @Produces(MediaType.APPLICATION_JSON)
80      public StateMachineInstance doTransition(
81              @PathParam("gameModelId") Long gameModelId,
82              @PathParam("playerId") Long playerId, @PathParam("stateMachineDescriptorId") Long stateMachineDescriptorId,
83              @PathParam("transitionId") Long transitionId)
84              throws ScriptException, WegasException {
85          StateMachineDescriptor stateMachineDescriptorEntity = (StateMachineDescriptor) stateMachineDescriptorFacade.find(stateMachineDescriptorId);
86          StateMachineInstance stateMachineInstanceEntity = (StateMachineInstance) stateMachineDescriptorEntity.getInstance(playerFacade.find(playerId));
87          State currentState = stateMachineInstanceEntity.getCurrentState();
88          List<Transition> transitions = currentState.getTransitions();
89  
90          for (Transition transition : transitions) {
91              if (transition instanceof DialogueTransition && transition.getId().equals(transitionId)) {
92                  //TODO : eval attached script (AND)
93                  stateMachineInstanceEntity.setCurrentStateId(transition.getNextStateId());
94                  stateMachineInstanceEntity.transitionHistoryAdd(transitionId);
95                  if (stateMachineInstanceEntity.getCurrentState().getOnEnterEvent() != null) {
96                      scriptManager.eval(playerId, stateMachineInstanceEntity.getCurrentState().getOnEnterEvent());
97                  }
98                  break;
99              }
100         }
101         return (StateMachineInstance) variableInstanceFacade.update(stateMachineInstanceEntity.getId(), stateMachineInstanceEntity);
102     }
