47      /**

68       * @throws RepositoryException
69       */
70      @GET
71      @Path("/{pageId : [1-9][0-9]*}")
72      @Produces(MediaType.APPLICATION_JSON)
73      public JsonNode getPage(@PathParam("gameModelId") String gameModelId, @PathParam("pageId") String pageId) throws RepositoryException {
74          GameModel gm = gmFacade.find(new Long(gameModelId));
75          Pages pages = new Pages(gm.getName());
76          Page page = pages.getPage(new Integer(pageId));

86       *
87       * @param gameModelId The GameModel's ID
88       * @param pageId The specific page to replace

90       * @return The stored page
91       * @throws RepositoryException

93      @POST
94      @Path("/{pageId : [1-9][0-9]*}")
95      @Consumes(MediaType.APPLICATION_JSON)

97      public JsonNode setPage(@PathParam("gameModelId") String gameModelId, @PathParam("pageId") String pageId, JsonNode content) throws RepositoryException, IOException {
98          GameModel gm = gmFacade.find(new Long(gameModelId));
99          Pages pages = new Pages(gm.getName());
100         Page page = new Page(new Integer(pageId), content);
101         pages.store(page);

105     /**
106      * Merges exisiting gameModelId's pages with the provided content, replacing
107      * existing pages with those given, storing new pages and keeping previous
108      * pages.
109      *
110      * @param gameModelId The GameMoldel's ID
111      * @param content A JSON map <Integer, JSONObject>
112      * @return The merge result
113      * @throws RepositoryException
114      * @throws JSONException
115      */
116     @POST
117     @Consumes(MediaType.APPLICATION_JSON)
118     @Produces(MediaType.APPLICATION_JSON)
119     public Map<Integer, JsonNode> setPages(@PathParam("gameModelId") String gameModelId, Map<Integer, JsonNode> pageMap) throws RepositoryException, JSONException {
120         GameModel gm = gmFacade.find(new Long(gameModelId));
121         Pages pages = new Pages(gm.getName());
122         //pages.delete();                                                       //remove first existing Pages currently merges, uncomment to replace
123         Iterator kIterator = pageMap.keySet().iterator();
124         while (kIterator.hasNext()) {
125             Integer key = (Integer) kIterator.next();
126             Page page = new Page(key, pageMap.get(key));
127             pages.store(page);
128         }
129         return getPages(gameModelId);
130     }
