42       */
43      @EJB

45      /**
46       *
47       */
48      @EJB
49      private ScriptManager scriptManager;
50  
51      /**
52       *
53       * @return
54       */
55      @Override
56      @GET
57      @Produces(MediaType.APPLICATION_JSON)
58      public Collection<AbstractEntity> index() {
59          Long gameModelId = this.getGameModelId();
60          GameModelEntity gameModel = gameModelFacade.find(gameModelId);
61          return (Collection) gameModel.getVariableDescriptors();
62      }
63  
64      @Override
65      public AbstractEntity create(AbstractEntity entity) {
66          this.variableDescriptorFacade.create(new Long(this.getPathParam("gameModelId")),
67                  (VariableDescriptorEntity) entity);
68          return entity;
69      }
70  
71      /**
72       * Resets all the variables of a given game model
73       *
74       * @param gameModelId game model id
75       * @return OK
76       */
77      @GET
78      @Path("reset")
79      @Produces(MediaType.APPLICATION_JSON)
80      public Collection<VariableDescriptorEntity> reset(
81              @PathParam("gameModelId") Long gameModelId) {
82          return gameModelFacade.reset(gameModelId).getVariableDescriptors();
83      }
84  
85      private Long getGameModelId() {
86          return new Long(this.getPathParam("gameModelId"));
87      }
88  
89      /**
