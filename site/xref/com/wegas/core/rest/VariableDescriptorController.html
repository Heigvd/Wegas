17  import javax.ejb.EJB;
18  import javax.ejb.Stateless;
19  import javax.ws.rs.GET;
20  import javax.ws.rs.Path;
21  import javax.ws.rs.PathParam;

28  /**
29   *
30   * @author Francois-Xavier Aeberhard <fx@red-agent.com>
31   */
32  @Stateless
33  @Path("GameModel/{gameModelId : [1-9][0-9]*}/VariableDescriptor")
34  public class VariableDescriptorController extends AbstractRestController<VariableDescriptorFacade, VariableDescriptor> {
35  
36      private static final Logger logger = LoggerFactory.getLogger(VariableDescriptorController.class);
37      /**
38       *
39       */
40      @EJB
41      private VariableDescriptorFacade variableDescriptorFacade;
42      /**
43       *
44       */
45      @EJB
46      private GameModelFacade gameModelFacade;
47  
48      /**
49       *
50       * @return
51       */
52      @Override
53      @GET
54      @Produces(MediaType.APPLICATION_JSON)
55      public Collection<VariableDescriptor> index() {
56          Long gameModelId = this.getGameModelId();
57          GameModel gameModel = gameModelFacade.find(gameModelId);
58          return gameModel.getChildVariableDescriptors();
59      }
60  
61      @Override
62      public VariableDescriptor create(VariableDescriptor entity) {
63          this.variableDescriptorFacade.create(new Long(this.getPathParam("gameModelId")),
64                  entity);
65          return entity;
66      }
67  
68      /**
69       * Resets all the variables of a given game model
70       *
71       * @param gameModelId game model id
72       * @return OK
73       */
74      @GET
75      @Path("reset")
76      @Produces(MediaType.APPLICATION_JSON)
77      public Response reset(@PathParam("gameModelId") Long gameModelId) {
78          gameModelFacade.reset(gameModelId);
79          return Response.ok().build();
80      }
81  
82      private Long getGameModelId() {
83          return new Long(this.getPathParam("gameModelId"));
84      }
85  
86      /**
87       *
88       * @return
89       */
90      @Override
91      protected VariableDescriptorFacade getFacade() {
92          return this.variableDescriptorFacade;
93      }
