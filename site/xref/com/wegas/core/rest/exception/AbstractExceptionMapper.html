10  package com.wegas.core.rest.exception;

12  import com.wegas.core.ejb.RequestFacade;
13  import com.wegas.core.ejb.exception.ConstraintViolationException;
14  import com.wegas.exception.WegasException;
15  import java.sql.SQLException;
16  import java.util.Iterator;
17  import javax.ejb.EJBException;
18  import javax.enterprise.event.ObserverException;
19  import javax.transaction.RollbackException;
20  import javax.transaction.TransactionRolledbackException;
21  import javax.ws.rs.core.Response;
22  import org.eclipse.persistence.exceptions.DatabaseException;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  
26  /**
27   *
28   * @author Francois-Xavier Aeberhard <fx@red-agent.com>
29   */
30  public abstract class AbstractExceptionMapper {
31  
32      final static private Logger logger = LoggerFactory.getLogger(AbstractExceptionMapper.class);
33  
34      /**
35       *
36       * @param exception
37       * @return
38       */
39      public static Response processException(Throwable exception) {
40  
41          if (exception instanceof RollbackException
42                  || exception instanceof TransactionRolledbackException) {
43              return processException(exception.getCause());
44  
45          } else if (exception instanceof EJBException) {
46              return processException(((EJBException) exception).getCausedByException());
47  
48          } else if (exception instanceof org.omg.CORBA.TRANSACTION_ROLLEDBACK) {
49              return processException(exception.getCause());
50  
51          } else if (exception instanceof DatabaseException) {
52              DatabaseException dbe = (DatabaseException) exception;
53              return processException(dbe.getInternalException());
54  
55          } else if (exception instanceof ObserverException) {
56              return processException(exception.getCause());
57  
58          } else if (exception instanceof javax.script.ScriptException) {
59              logger.error(exception.getLocalizedMessage());
60              javax.script.ScriptException scriptException = (javax.script.ScriptException) exception;
61  
62              if (scriptException.getCause() instanceof ConstraintViolationException) {
63                  return processException(scriptException.getCause());
64              } else {
65                  return Response.status(Response.Status.BAD_REQUEST).
66                          entity(new ExceptionWrapper("400", scriptException.getClass(), scriptException.getLocalizedMessage())).
67                          build();
68              }
69  
70          } else if (exception instanceof SQLException) {
71              logger.error(exception.getLocalizedMessage());
72              SQLException sqlException = (SQLException) exception;
73              return Response.status(Response.Status.BAD_REQUEST).
74                      entity(new ExceptionWrapper("400", sqlException.getClass(), sqlException.getLocalizedMessage())).build();
75  
76          } else if (exception instanceof WegasException) {
77              logger.error(exception.getLocalizedMessage());
78              WegasException wegasException = (WegasException) exception;
79              return Response.status(Response.Status.BAD_REQUEST).
80                      entity(new ExceptionWrapper("400", wegasException.getClass(), wegasException.getLocalizedMessage())).build();
81  
82          } else if (exception instanceof javax.validation.ConstraintViolationException) {
83              javax.validation.ConstraintViolationException constraintViolationException = (javax.validation.ConstraintViolationException) exception;
84  
85              String msg = RequestFacade.lookup().getBundle("com.wegas.app.errors").getString("constraint"); //internationalised error (sample)
86              Iterator it = constraintViolationException.getConstraintViolations().iterator();
87              while (it.hasNext()) {
88                  javax.validation.ConstraintViolation violation = (javax.validation.ConstraintViolation) it.next();
89                  msg += "\n" + violation.getLeafBean() + ":" + violation.getRootBean() + violation.getPropertyPath();
90              }
91              logger.error(msg);
92              // constraintViolationException.getMessage()
93              return Response.status(Response.Status.BAD_REQUEST).
94                      entity(new ExceptionWrapper("400", exception.getClass(), constraintViolationException.getLocalizedMessage())).build();
95  
96          } else {
97              logger.error(RequestFacade.lookup().getBundle("com.wegas.app.errors").getString("unexpected"), exception); //internationalised error (sample)
98              return Response.status(Response.Status.BAD_REQUEST).
99                      entity(new ExceptionWrapper("400", exception.getClass(), exception.getLocalizedMessage())).build();
100         }
101     }
