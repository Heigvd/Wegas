28  @Stateless
29  @Path("GameModel/{gameModelId : [1-9][0-9]*}/VariableDescriptor/Trigger/")
30  public class TriggerController extends AbstractRestController<TriggerDescriptorFacade> {
31  
32      @EJB
33      private TriggerDescriptorFacade triggerDescriptorFacade;
34  
35      @Override
36      protected TriggerDescriptorFacade getFacade() {
37          return this.triggerDescriptorFacade;
38      }
39  
40      @Override
41      public TriggerDescriptorEntity create(AbstractEntity entity) {
42          Long gameModelId = new Long(this.getPathParam("gameModelId"));
43          TriggerDescriptorEntity trigger = (TriggerDescriptorEntity) entity;
44          State initialState = new State(), finalState = new State();
45          Transition transition = new Transition();
46          transition.setTriggerCondition(trigger.getTriggerEvent());
47          List<Transition> transitions = initialState.getTransitions();
48          transitions.add(transition);
49          initialState.setTransitions(transitions);
50          HashMap<Long, State> states = new HashMap<>();
51          if (trigger.isOneShot()) {
52              trigger.setOpposedTrigger(false);
53              transition.setNextStateId(2L);
54              finalState.setOnEnterEvent(trigger.getPostTriggerEvent());
55              states.put(2L, finalState);
56          } else if (trigger.isOpposedTrigger()) {
57              transition.setNextStateId(2L);
58              finalState.setOnEnterEvent(trigger.getPostTriggerEvent());
59              Transition returnTransition = new Transition();
60              returnTransition.setNextStateId(1L);
61              //TODO : Not(triggerEvent)
62              returnTransition.setTriggerCondition(trigger.getTriggerEvent());
63              returnTransition.setNextStateId(1L);
64              List<Transition> returnTransitions = finalState.getTransitions();
65              returnTransitions.add(returnTransition);
66              finalState.setTransitions(returnTransitions);
67              states.put(2L, finalState);
68          } else {
69              transition.setNextStateId(1L);
70              initialState.setOnEnterEvent(trigger.getPostTriggerEvent());
71          }
72          states.put(1L, initialState);
73          trigger.setInitialStateId(1L);
74          trigger.setStates(states);
75          this.triggerDescriptorFacade.create(gameModelId, trigger);
76          return trigger;
77      }
78  }
