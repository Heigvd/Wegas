80      /**
81       * This method process a string token. It checks if the given token
82       * corresponds to a game and then to a team, and return the corresponding
83       * result.
84       *
85       * @param token
86       * @return
87       */
88      @GET
89      @Path("/JoinGame/{token : .*}/")
90      @Produces(MediaType.APPLICATION_JSON)
91      public Object joinGame(@PathParam("token") String token) throws Exception {
92          Game game = null;
93          Team team = null;
94          try {
95              team = teamFacade.findByToken(token);                               // we try to lookup for a team entity.
96              game = team.getGame();
97          }
98          catch (Exception e2) {
99              try {
100                 game = gameFacade.findByToken(token);                           // We check if there is game with given token
101             }
102             catch (Exception e) {
103                 throw new Exception("Could not find any game associated with this token.");
104             }
105         }
106         try {                                                                   // We check if logged user is already registered in the target game
107             playerFacade.findByGameIdAndUserId(
108                     game.getId(), userFacade.getCurrentUser().getId());
109             throw new Exception("You are already registered to this game.");    // There user is already registered to target game
110         }
111         catch (PersistenceException e) {                                        // If there is no NoResultException, everything is ok, we can return the game
112             return ( team != null ) ? team : game;
113         }
114     }
