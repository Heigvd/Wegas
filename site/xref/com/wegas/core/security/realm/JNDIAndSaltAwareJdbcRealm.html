116     @Override
117     protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
118         //identify account to log to
119         UsernamePasswordToken userPassToken = (UsernamePasswordToken) token;
120         String username = userPassToken.getUsername();
121 
122         if (username == null) {
123             log.debug("Username is null.");
124             return null;
125         }
126 
127         // read password hash and salt from db 
128         PasswdSalt passwdSalt = getPasswordForUser(username);
129 
130         if (passwdSalt == null) {
131             log.debug("No account found for user [" + username + "]");
132             return null;
133         }
134 
135         // return salted credentials
136         SimpleAuthenticationInfo info = new SimpleAuthenticationInfo(username, passwdSalt.password, getName());
137         info.setCredentialsSalt(new SimpleByteSource(passwdSalt.salt));
138 
139         return info;
140     }

142     private PasswdSalt getPasswordForUser(String username) {
143         PreparedStatement statement = null;
144         ResultSet resultSet = null;
145         Connection conn = null;
146         try {
147             conn = dataSource.getConnection();
148             statement = conn.prepareStatement(authenticationQuery);
149             statement.setString(1, username);
150 
151             resultSet = statement.executeQuery();
152 
153             boolean hasAccount = resultSet.next();
154             if (!hasAccount) {
155                 return null;
156             }
157 
158             String salt = null;
159             String password = resultSet.getString(1);
160             if (resultSet.getMetaData().getColumnCount() > 1) {
161                 salt = resultSet.getString(2);
162             }
163 
164             if (resultSet.next()) {
165                 throw new AuthenticationException("More than one user row found for user [" + username + "]. Usernames must be unique.");
166             }
167 
168             return new PasswdSalt(password, salt);
169         }
170         catch (SQLException e) {
171             final String message = "There was a SQL error while authenticating user [" + username + "]";
172             if (log.isErrorEnabled()) {
173                 log.error(message, e);
174             }
175             throw new AuthenticationException(message, e);
176 
177         }
178         finally {
179             JdbcUtils.closeResultSet(resultSet);
180             JdbcUtils.closeStatement(statement);
181             JdbcUtils.closeConnection(conn);
182         }
183     }
