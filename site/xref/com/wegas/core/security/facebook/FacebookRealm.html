34      private static final String APP_SECRET = props.getString("facebook.appSecret").toString();
35      private static final String APP_ID = props.getString("facebook.appId").toString();
36      private static final String REDIRECT_URL = props.getString("facebook.redirectUrl").toString();

38      @Override
39      public boolean supports(AuthenticationToken token) {
40          if (token instanceof FacebookToken) {
41              return true;
42          }
43          return false;

48          return new FacebookAuthorizationInfo();

51      @Override
52      protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
53          FacebookToken facebookToken = (FacebookToken) token;
54  
55  // do all the facebook gubbins
56          if (facebookToken.getCode() != null && facebookToken.getCode().trim().length() > 0) {
57              URL authUrl;
58              try {
59                  authUrl = new URL("https://graph.facebook.com/oauth/access_token?" + "client_id=" + APP_ID
60                          + "&redirect_uri=" + REDIRECT_URL + "&client_secret=" + APP_SECRET + "&code="
61                          + facebookToken.getCode());
62  
63                  String authResponse = readURL(authUrl);
64                  System.out.println(authResponse);
65                  String accessToken = getPropsMap(authResponse).get("access_token");
66                  URL url = new URL("https://graph.facebook.com/me?access_token=" + accessToken);
67                  String fbResponse = readURL(url);
68                  FacebookUserDetails fud = new FacebookUserDetails(fbResponse);
69                  return new FacebookAuthenticationInfo(fud, this.getName());
70              }
71              catch (MalformedURLException e1) {
72                  e1.printStackTrace();
73                  throw new AuthenticationException(e1);
74              }
75              catch (IOException ioe) {
76                  ioe.printStackTrace();
77                  throw new AuthenticationException(ioe);
78              }
79              catch (Throwable e) {
80                  e.printStackTrace();
81              }
82          }
83          return null;
84      }

89      private String readURL(URL url) throws IOException {
90          ByteArrayOutputStream baos = new ByteArrayOutputStream();
91          InputStream is = url.openStream();
92          int r;
93          while (( r = is.read() ) != -1) {
94              baos.write(r);
95          }
96          return new String(baos.toByteArray());
97      }

99      private Map<String, String> getPropsMap(String someString) {
100         String[] pairs = someString.split("&");
101         Map<String, String> props = new HashMap<String, String>();
102         for (String propPair : pairs) {
103             String[] pair = propPair.split("=");
104             props.put(pair[0], pair[1]);
105         }
106         return props;
107     }
