33      public FacebookAuthenticationInfo(FacebookUserDetails facebookUserDetails, String realmName) {

36          principals.add(facebookUserDetails.getFirstName() + " " + facebookUserDetails.getLastName()); // Is this appropriate is the name not really a Principal ?
37          this.principalCollection = new SimplePrincipalCollection(principals, realmName);
38      }

43      }
