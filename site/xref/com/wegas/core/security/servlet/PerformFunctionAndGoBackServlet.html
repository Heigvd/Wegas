32      @Override
33      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
34          String action = request.getParameter("action");
35          String actionResult = performAction(action);
36          request.setAttribute("actionResultMessage", actionResult);
37  
38          // forward the request and response back to original page
39          String originalPage = request.getParameter("originalPage");
40          RequestDispatcher dispatcher = getServletContext().getRequestDispatcher(originalPage);
41          dispatcher.forward(request, response);
42      }
43  
44      private String performAction(String actionName) {
45          try {
46              Actions action = findAction(actionName);
47              String result = action == null ? null : action.doIt();
48              log.debug("Performed function with result: " + result);
49              return result;
50          }
51          catch (ShiroException ex) {
52              log.debug("Function failed with " + ex.getMessage() + " message.");
53              return "Error: " + ex.getMessage();
54          }
55      }
56  
57      private Actions findAction(String actionName) {
58          if (actionName == null) {
59              return null;
60          }
61  
62          Actions[] values = Actions.values();
63  
64          for (Actions action : values) {
65              if (actionName.equals(action.getName())) {
66                  return action;
67              }
68          }
69          return null;
70      }
