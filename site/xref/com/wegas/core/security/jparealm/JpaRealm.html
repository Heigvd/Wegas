16  import com.wegas.core.ejb.Helper;
17  import com.wegas.core.security.ejb.AccountFacade;
18  import com.wegas.core.security.persistence.AbstractAccount;
19  import com.wegas.core.security.persistence.Role;
20  import javax.ejb.EJBException;
21  import javax.naming.NamingException;
22  import org.apache.shiro.authc.*;
23  import org.apache.shiro.authz.AuthorizationInfo;
24  import org.apache.shiro.authz.SimpleAuthorizationInfo;
25  import org.apache.shiro.realm.AuthorizingRealm;
26  import org.apache.shiro.subject.PrincipalCollection;
27  import org.apache.shiro.util.SimpleByteSource;
28  import org.slf4j.LoggerFactory;
29  
30  public class JpaRealm extends AuthorizingRealm {
31  
32      private static final org.slf4j.Logger logger = LoggerFactory.getLogger(JpaRealm.class);
33  
34      public JpaRealm() {
35          setName("JpaRealm");                                                    //This name must match the name in the User class's getPrincipals() method
36      }
37  
38      @Override
39      protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authcToken) throws AuthenticationException {
40          UsernamePasswordToken token = (UsernamePasswordToken) authcToken;
41          try {
42              JpaAccount account = (JpaAccount) accountFacade().findByEmail(token.getUsername());
43              SimpleAuthenticationInfo info = new SimpleAuthenticationInfo(account.getId(), account.getPasswordHex(), getName());
44              info.setCredentialsSalt(new SimpleByteSource(account.getSalt()));
45              return info;
46          }
47          catch (EJBException e) {
48              return null;
49          }
50          catch (NamingException ex) {
51              logger.error("Unable to find AocountFacade EJB", ex);
52              return null;
53          }
54      }
55  
56      @Override
57      protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
58          try {
59              Long accountId = (Long) principals.fromRealm(getName()).iterator().next();
60  
61              AbstractAccount account = accountFacade().find(accountId);
62              SimpleAuthorizationInfo info = new SimpleAuthorizationInfo();
63              for (Role role : account.getRoles()) {
64                  info.addRole(role.getName());
65                  info.addStringPermissions(role.getPermissions());
66              }
67              return info;
68          }
69          catch (EJBException e) {
70              return null;
71          }
72          catch (NamingException ex) {
73              logger.error("Unable to find AocountFacade EJB", ex);
74              return null;
75          }
76      }
77  
78      public AccountFacade accountFacade() throws NamingException {
79          return Helper.lookupBy(AccountFacade.class);
80      }
81  }
