123      * Allows to login using a post request
124      *
125      * @param email
126      * @param password
127      * @param rememberMe
128      * @param request
129      */
130     @POST
131     @Path("Authenticate")
132     @Produces(MediaType.APPLICATION_JSON)
133     public void login(@QueryParam("email") String email,
134             @QueryParam("password") String password,
135             @QueryParam("remember") @DefaultValue("false") boolean remember,
136             @Context HttpServletRequest request) {
137 
138         Subject currentUser = SecurityUtils.getSubject();
139 
140         if (!currentUser.isAuthenticated()) {
141             UsernamePasswordToken token = new UsernamePasswordToken(email, password);
142             token.setRememberMe(remember);
143             currentUser.login(token);
144         }
145     }
