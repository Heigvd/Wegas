 14 YUI.add( "wegas-book-fight", function ( Y ) {
 15     "use strict";
 16     
 17     var CONTENTBOX = "contentBox", Fight;
 18     
 19     Fight = Y.Base.create( "wegas-book-fight", Y.Widget, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable ], {
 20         
 21         handlers: new Array(),
 22         dice:null,
 23         opponentStamina:0,
 24         opponentCombatSkill:0,
 25         success:null,
 26         failure:null,
 27         alternative:null,
 28         
 29         doFight: function(e){
 30             var combatSkill = Y.Wegas.VariableDescriptorFacade.rest.find("name", "combatSkill"),
 31             stamina = Y.Wegas.VariableDescriptorFacade.rest.find("name", "stamina").getInstance().get("value"),
 32             damageGiven, damageTaken, handicap, diceValue = e.target.result;
 33             handicap = combatSkill.getInstance().get("value")-this.opponentCombatSkill;
 34             if(handicap < -10) handicap = -10;
 35             if(handicap > 10) handicap = 10;
 36             switch(diceValue){
 37                 case 1 :
 38                     damageGiven = 0;
 39                     damageTaken = ((handicap<0)?Math.abs(handicap):0)+2;
 40                     break;
 41                 case 2 :
 42                     damageGiven = Math.floor(((handicap>0)?handicap:0)/5);
 43                     damageTaken = Math.ceil(((handicap<0)?Math.abs(handicap):0)/2)+1;
 44                     break;
 45                 case 3 :
 46                     damageGiven = Math.floor(((handicap>0)?handicap:0)/4);
 47                     damageTaken = Math.ceil(((handicap<0)?Math.abs(handicap):0)/3)+1;
 48                     break;
 49                 case 4 :
 50                     damageGiven = Math.ceil(((handicap>0)?handicap:0)/3)+1;
 51                     damageTaken = Math.floor(((handicap<0)?Math.abs(handicap):0)/4);
 52                     break;
 53                 case 5 :
 54                     damageGiven = Math.ceil(((handicap>0)?handicap:0)/2)+1;
 55                     damageTaken = Math.floor(((handicap<0)?Math.abs(handicap):0)/5);
 56                     break;
 57                 case 6 :
 58                     damageGiven = ((handicap>0)?handicap:0)+2;
 59                     damageTaken = 0;
 60                     break;
 61             }
 62             this.opponentStamina -= damageGiven;
 63             stamina -= damageTaken;
 64             if(this.opponentStamina<=0){
 65                 this.opponentStamina = 0; 
 66                 this.doBattleResult(true);
 67             } else if(stamina<=0){
 68                 stamina=0;
 69                 this.doBattleResult(false);
 70             }else{
 71                 this.dice.rollButton.enable();
 72             }
 73             this.setStamina(stamina);
 74             this.syncUI();
 75         },
 76         
 77         setStamina:function(stamina){
 78             if(typeof stamina !== "number") return;
 79             Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
 80                 request: "/Script/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
 81                 headers:{
 82                     'Content-Type': 'application/json; charset=ISO-8859-1',
 83                     'Managed-Mode':'true'
 84                 },
 85                 cfg: {
 86                     method: "POST",
 87                     data: Y.JSON.stringify({
 88                         "@class": "Script",
 89                         "language": "JavaScript",
 90                         "content": "importPackage(com.wegas.core.script);\nstamina.value ="+stamina+";"
 91                     })
 92                 }
 93             });
 94         },
 95         
 96         doBattleResult: function(success){
 97             var cb = this.get(CONTENTBOX);
 98             if(success){
 99                 if(this.success)this.success.render(cb.one(".result"));
100             } else{
101                 if(this.failure)this.failure.render(cb.one(".result"));
102             }
103         },
104         
105         displayOpponentState: function(cb){
106             cb.one(".opponent .stamina .value").setHTML(this.opponentStamina);
107             cb.one(".opponent .combatSkill .value").setHTML(this.opponentCombatSkill);
108         },
109         
110         initializer: function(){
111             this.dice = new Y.Wegas.Dice({
112                 label:"Combattre",
113                 animated:"true"
114             });
115             if(this.get("success")){
116                 this.success = new Y.Wegas.List({
117                     "label":"success",
118                     "cssClass":"success-list",
119                     "children":this.get("success")
120                 });
121             }
122             if(this.get("failure")){
123                 this.failure = new Y.Wegas.List({
124                     "label":"failure",
125                     "cssClass":"failure-list",
126                     "children":this.get("failure")
127                 });
128             }
129             if(this.get("alternative")){
130                 this.alternative = new Y.Wegas.List({
131                     "label":"alternative",
132                     "cssClass":"alternative-list",
133                     "children":this.get("alternative")
134                 });
135             }
136             this.opponentStamina = this.get("stamina");
137             this.opponentCombatSkill = this.get("combatSkill");
138         },
139         
140         renderUI: function(){
141             var cb = this.get(CONTENTBOX), opponement;
142             opponement = Y.Node.create("<div class='opponent'></div>");
143             opponement.append("<div class='name'></div>").append("<div class='stamina'></div>").append("<div class='combatSkill'></div>");
144             opponement.one(".stamina").append("<div class='label'></div>").append("<div class='value'></div>");
145             opponement.one(".combatSkill").append("<div class='label'></div>").append("<div class='value'></div>");
146             opponement.one(".stamina .label").setHTML(this.get("staminaLabel"));
147             opponement.one(".combatSkill .label").setHTML(this.get("combatSkillLabel"));
148             opponement.one(".name").setHTML(this.get("name"));
149             cb.append(opponement);
150             cb.append("<div class='dice'></div>");
151             cb.append("<div class='result'></div>");
152             cb.append("<div class='alternative'></div>");
153             this.dice.render(cb.one(".dice"));
154             if(this.alternative)this.alternative.render(cb.one(".alternative"));
155         },
156         
157         bindUI: function(){
158             this.dice.after("diceRolling",function(){
159                 this.dice.rollButton.disable();
160             }, this);
161             this.dice.after("diceRolled", this.doFight, this);
162         },
163         
164         syncUI: function(){
165             var cb = this.get(CONTENTBOX);
166             this.displayOpponentState(cb);
167         },
168         
169         destructor: function(){
170             var i;
171             for (i=0; i<this.handlers.length;i++) {
172                 this.handlers[i].detach();
173             }
174             if(this.dice)this.dice.destroy();
175             if(this.success)this.success.destroy();
176             if(this.failure)this.failure.destroy();
177             if(this.alternative)this.alternative.destroy();
178         }  
179     
180     }, {
181         ATTRS : {
182             name:{
183                 type: "String",
184                 value: "unknown"
185             },
186             staminaLabel : {
187                 type: "String",
188                 value: "Stamina: "
189             },
190             stamina : {
191                 type: "Integer",
192                 value: 1
193             },
194             combatSkillLabel : {
195                 type: "String",
196                 value: "Combat skill: "
197             },
198             combatSkill : {
199                 type: "Integer",
200                 value: 1
201             },
202             success : {
203                 validator: Y.Lang.isArray
204             },
205             failure : {
206                 validator: Y.Lang.isArray
207             },
208             alternative : {
209                 validator: Y.Lang.isArray
210             }
211         }
212     });
213     
214     Y.namespace( "Wegas" ).Fight = Fight;
215 });
