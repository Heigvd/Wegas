 15 YUI.add( 'wegas-mcqtabview', function ( Y ) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19     MCQTabView;
 20 
 21     MCQTabView = Y.Base.create( "wegas-mcqtabview", Y.Widget, [ Y.WidgetChild, Y.Wegas.Widget ], {
 22 
 23         // *** Private fields *** //
 24         tabView: null,
 25         dataSource: null,
 26         handlers: null,
 27 
 28         // *** Lifecycle Methods *** //
 29         renderUI: function () {
 30             var cb = this.get( CONTENTBOX );
 31             this.dataSource = Y.Wegas.VariableDescriptorFacade;
 32             this.tabView = new Y.TabView();
 33             this.tabView.render( cb );
 34             cb.append( "<div style='clear:both'></div>" );
 35             this.handlers = {};
 36         },
 37 
 38         bindUI: function () {
 39             this.get(CONTENTBOX).delegate("click", function (e) {
 40                 this.showOverlay();
 41                 this.dataSource.rest.sendRequest({
 42                     request: "/QuestionDescriptor/SelectChoice/" + e.target.get('id')
 43                     + "/Player/" + Y.Wegas.app.get('currentPlayer')
 44                 });
 45             }, "input[type=submit]", this);
 46 
 47             this.tabView.after("selectionChange", this.onTabSelected, this)
 48 
 49             this.handlers.response = this.dataSource.after( "response", this.syncUI, this);
 50             this.handlers.playerChange = Y.Wegas.app.after( 'currentPlayerChange', this.syncUI, this);
 51         },
 52 
 53         syncUI: function () {
 54             var i, j, cReplyLabel, cQuestion, ret, firstChild, cQuestionInstance, cQuestionLabel, tab, cChoices, choiceDescriptor, reply,
 55             questions = this.get( "variableDesc" ),
 56             selectedTab = this.tabView.get( 'selection' ),
 57             lastSelection = ( selectedTab ) ? selectedTab.get('index') : 0;
 58 
 59             if ( !questions ) {
 60                 return;
 61             }
 62             questions = questions.get( "items" );
 63 
 64             this.tabView.removeAll();
 65 
 66             for ( i = 0; i < questions.length; i += 1) {
 67                 cQuestion = questions[i];
 68                 cQuestionLabel = cQuestion.getPublicLabel() || "undefined";
 69                 ret = [//'<div class="title">Details</div>',
 70                 '<div class="content">',
 71                 '<div class="title">', cQuestionLabel, '</div>',
 72                 '<div class="description">',
 73                 ( cQuestion.get( "description" ) || "<em>No description</em>" ),'</div>'];
 74                 cQuestionInstance = cQuestion.getInstance();
 75                 firstChild = "first-child";
 76                 cReplyLabel = null,
 77                 cChoices = cQuestion.get("items");
 78 
 79                 if ( cQuestionInstance.get( "active" ) ) {
 80                     if ( cQuestionInstance.get( "replies" ).length === 0        // If the question is not replied, we display its reply set
 81                         || cQuestion.get( "allowMultipleReplies" ) ) {
 82 
 83                         ret.push('<div class="subtitle">Possible replies</div><div class="replies">');
 84 
 85                         for (j = 0; j < cChoices.length; j += 1) {
 86                             if ( cChoices[j].getInstance().get( "active" ) ) {
 87                                 ret.push('<div class="reply ', firstChild, '">',
 88                                     '<div class="name">', cChoices[j].get( "label" ), '</div>',
 89                                     '<div class="content">', cChoices[j].get( "description" ), '</div>',
 90                                     '<input type="submit" id="', cChoices[j].get( "id" ),'" value="Submit"></input>',
 91                                     '<div style="clear:both"></div>',
 92                                     '</div>');
 93                                 firstChild = "";
 94                             }
 95                         }
 96                         ret.push('</div>');
 97                     }
 98 
 99                     if ( cQuestionInstance.get( "replies" ).length > 0 ){       // Display the selected replies
100                         ret.push('<div class="subtitle">Selected replies</div><div class="replies">');
101                         for (j = 0; j < cQuestionInstance.get( "replies" ).length; j += 1) {
102                             reply = cQuestionInstance.get( "replies" )[j];
103                             choiceDescriptor = reply.getChoiceDescriptor();
104                             ret.push('<div class="reply"><div class="name">', choiceDescriptor.get( "label" ), '</div>',
105                                 '<div>', choiceDescriptor.get( "description" ), '</div>',
106                                 '<div style="clear:both"></div></div>');
107 
108                             ret.push('<div class="subtitle">Result</div>',
109                                 '<div class="replies"><div class="reply first-child">', reply.get( "result" ).get( "answer" ), '</div></div>');
110 
111                             if (!cReplyLabel) {
112                                 cReplyLabel = choiceDescriptor.getPublicLabel().substr( 0, 15 ) + "...";
113                             }
114                         }
115                         ret.push( "</div>" );
116                     }
117 
118                     ret.push( "</div>" );
119                     tab = new Y.Tab({
120                         label: '<div class="' + ( cQuestionInstance.get( "replies" ).length === 0 ? "unread" : "" ) + '"><div class="label">' + cQuestionLabel + '</div>'
121                         +'<div class="status">' + (cReplyLabel || "unanswered") + '</div></div>',
122                         content: ret.join('')
123                     });
124                     tab.questionInstance = cQuestionInstance;
125                     this.tabView.add(tab);
126                 }
127             }
128 
129             this.tabView.selectChild(lastSelection);
130             this.hideOverlay();
131         },
132         /**
133          *
134          */
135         destructor: function () {
136             for (var i in this.handlers) {
137                 this.handlers[i].detach();
138             }
139         },
140 
141         // *** Private Methods *** //
142         // *** Callbacks *** //
143         onTabSelected: function (e) {
144 
145             return;                                                             // This logic is not executed, since we do not manage read count on questions
146 
147             if (this.timer) {                                                   // If there is an active unread
148                 this.timer.cancel();                                            // question timer, we cancel it.
149             }
150 
151             if (e.newVal && e.newVal.questionInstance.get( "unread" ) ) {       // If the question is currently unread,
152                 Y.log("Sending question read update", "info",  "MCQTabView");
153                 this.questionInstance = e.newVal.questionInstance;
154                 this.timer = Y.later(2000, this, function () {
155                     this.questionInstance.set( "unread" ) = false;
156                     this.dataSource.rest.put(this.questionInstance);
157                 });
158             }
159         }
160     }, {
161         ATTRS: {
162             variable: {},
163             expr: {},
164             /**
165              * The target variable, returned either based on the variableName attribute,
166              * and if absent by evaluating the expr attribute.
167              */
168             variableDesc: {
169                 getter: function () {
170                     if ( this.get( "variable" ) ) {
171                         return this.dataSource.rest.find( 'name', this.get( "variable" ) )
172                     } else {
173                         return this.dataSource.rest.findById(
174                             Y.Wegas.VariableDescriptorFacade.script.scopedEval( this.get( "expr" ) ) );
175                     }
176                 }
177             }
178         }
179     });
180 
181     Y.namespace('Wegas').MCQTabView = MCQTabView;
182 });
