 19 /* 
 20  * the library specific functions, such as find offset, get id, get attribute, extend etc.  
 21  * the full list is:
 22  * 
 23  * addClass				adds a class to the given element
 24  * animate				calls the underlying library's animate functionality
 25  * appendElement		appends a child element to a parent element.
 26  * bind					binds some event to an element
 27  * dragEvents			a dictionary of event names
 28  * extend				extend some js object with another.  probably not overly necessary; jsPlumb could just do this internally.
 29  * getAttribute			gets some attribute from an element
 30  * getDragObject		gets the object that is being dragged, by extracting it from the arguments passed to a drag callback
 31  * getDragScope			gets the drag scope for a given element.
 32  * getDropScope			gets the drop scope for a given element.
 33  * getElementObject		turns an id or dom element into an element object of the underlying library's type.
 34  * getOffset			gets an element's offset
 35  * getPageXY			gets the page event's xy location.
 36  * getParent			gets the parent of some element.
 37  * getScrollLeft		gets an element's scroll left.  TODO: is this actually used?  will it be?
 38  * getScrollTop			gets an element's scroll top.  TODO: is this actually used?  will it be?
 39  * getSize				gets an element's size.
 40  * getUIPosition		gets the position of some element that is currently being dragged, by extracting it from the arguments passed to a drag callback.
 41  * hasClass				returns whether or not the given element has the given class.
 42  * initDraggable		initializes an element to be draggable 
 43  * initDroppable		initializes an element to be droppable
 44  * isDragSupported		returns whether or not drag is supported for some element.
 45  * isDropSupported		returns whether or not drop is supported for some element.
 46  * removeClass			removes a class from a given element.
 47  * removeElement		removes some element completely from the DOM.
 48  * setAttribute			sets an attribute on some element.
 49  * setDraggable			sets whether or not some element should be draggable.
 50  * setDragScope			sets the drag scope for a given element.
 51  * setOffset			sets the offset of some element.
 52  * trigger				triggers some event on an element.
 53  * unbind				unbinds some listener from some element.
 54  */
 55 (function($) {	
 56 	
 57 	//var getBoundingClientRectSupported = "getBoundingClientRect" in document.documentElement;
 58 
 59 	jsPlumb.CurrentLibrary = {					        
 60 		
 61 		/**
 62 		 * adds the given class to the element object.
 63 		 */
 64 		addClass : function(el, clazz) {
 65 			el = jsPlumb.CurrentLibrary.getElementObject(el);
 66 			try {
 67 				if (el[0].className.constructor == SVGAnimatedString) {
 68 					jsPlumb.util.svg.addClass(el[0], clazz);
 69 				}
 70 			}
 71 			catch (e) {
 72 				// SVGAnimatedString not supported; no problem.
 73 			}
 74 			el.addClass(clazz);
 75 		},
 76 		
 77 		/**
 78 		 * animates the given element.
 79 		 */
 80 		animate : function(el, properties, options) {
 81 			el.animate(properties, options);
 82 		},				
 83 		
 84 		/**
 85 		 * appends the given child to the given parent.
 86 		 */
 87 		appendElement : function(child, parent) {
 88 			jsPlumb.CurrentLibrary.getElementObject(parent).append(child);			
 89 		},   
 90 
 91 		/**
 92 		* executes ana ajax call.
 93 		*/
 94 		ajax : function(params) {
 95 			params = params || {};
 96 			params.type = params.type || "get";
 97 			$.ajax(params);
 98 		},
 99 		
100 		/**
101 		 * event binding wrapper.  it just so happens that jQuery uses 'bind' also.  yui3, for example,
102 		 * uses 'on'.
103 		 */
104 		bind : function(el, event, callback) {
105 			el = jsPlumb.CurrentLibrary.getElementObject(el);
106 			el.bind(event, callback);
107 		},
108 		
109 		/**
110          * mapping of drag events for jQuery
111          */
112 		dragEvents : {
113 			'start':'start', 'stop':'stop', 'drag':'drag', 'step':'step',
114 			'over':'over', 'out':'out', 'drop':'drop', 'complete':'complete'
115 		},
116 				
117 		/**
118 		 * wrapper around the library's 'extend' functionality (which it hopefully has.
119 		 * otherwise you'll have to do it yourself). perhaps jsPlumb could do this for you
120 		 * instead.  it's not like its hard.
121 		 */
122 		extend : function(o1, o2) {
123 			return $.extend(o1, o2);
124 		},
125 		
126 		/**
127 		 * gets the named attribute from the given element object.  
128 		 */
129 		getAttribute : function(el, attName) {
130 			return el.attr(attName);
131 		},
132 		
133 		getClientXY : function(eventObject) {
134 			return [eventObject.clientX, eventObject.clientY];
135 		},
136 		
137 		getDocumentElement : function() { return document; },
138 		
139 		/**
140 		 * takes the args passed to an event function and returns you an object representing that which is being dragged.
141 		 */
142 		getDragObject : function(eventArgs) {
143 			return eventArgs[1].draggable;
144 		},
145 		
146 		getDragScope : function(el) {
147 			return el.draggable("option", "scope");
148 		},
149 
150 		getDropEvent : function(args) {
151 			return args[0];
152 		},
153 		
154 		getDropScope : function(el) {
155 			return el.droppable("option", "scope");		
156 		},
157 
158 		/**
159 		* gets a DOM element from the given input, which might be a string (in which case we just do document.getElementById),
160 		* a selector (in which case we return el[0]), or a DOM element already (we assume this if it's not either of the other
161 		* two cases).  this is the opposite of getElementObject below.
162 		*/
163 		getDOMElement : function(el) {
164 			if (typeof(el) == "string") return document.getElementById(el);
165 			else if (el.context) return el[0];
166 			else return el;
167 		},
168 	
169 		/**
170 		 * gets an "element object" from the given input.  this means an object that is used by the
171 		 * underlying library on which jsPlumb is running.  'el' may already be one of these objects,
172 		 * in which case it is returned as-is.  otherwise, 'el' is a String, the library's lookup 
173 		 * function is used to find the element, using the given String as the element's id.
174 		 * 
175 		 */		
176 		getElementObject : function(el) {			
177 			return typeof(el) == "string" ? $("#" + el) : $(el);
178 		},
179 		
180 		/**
181 		  * gets the offset for the element object.  this should return a js object like this:
182 		  *
183 		  * { left:xxx, top: xxx }
184 		 */
185 		getOffset : function(el) {
186 			return el.offset();
187 		},
188 		
189 		getPageXY : function(eventObject) {
190 			return [eventObject.pageX, eventObject.pageY];
191 		},
192 		
193 		getParent : function(el) {
194 			return jsPlumb.CurrentLibrary.getElementObject(el).parent();
195 		},
196 														
197 		getScrollLeft : function(el) {
198 			return el.scrollLeft();
199 		},
200 		
201 		getScrollTop : function(el) {
202 			return el.scrollTop();
203 		},
204 		
205 		getSelector : function(spec) {
206 			return $(spec);
207 		},
208 		
209 		/**
210 		 * gets the size for the element object, in an array : [ width, height ].
211 		 */
212 		getSize : function(el) {
213 			return [el.outerWidth(), el.outerHeight()];
214 		},
215 
216         getTagName : function(el) {
217             var e = jsPlumb.CurrentLibrary.getElementObject(el);
218             return e.length > 0 ? e[0].tagName : null;
219         },
220 		
221 		/**
222 		 * takes the args passed to an event function and returns you an object that gives the
223 		 * position of the object being moved, as a js object with the same params as the result of
224 		 * getOffset, ie: { left: xxx, top: xxx }.
225 		 * 
226 		 * different libraries have different signatures for their event callbacks.  
227 		 * see getDragObject as well
228 		 */
229 		getUIPosition : function(eventArgs) {
230 			
231 			// this code is a workaround for the case that the element being dragged has a margin set on it. jquery UI passes
232 			// in the wrong offset if the element has a margin (it doesn't take the margin into account).  the getBoundingClientRect
233 			// method, which is in pretty much all browsers now, reports the right numbers.  but it introduces a noticeable lag, which
234 			// i don't like.
235             
236 			/*if ( getBoundingClientRectSupported ) {
237 				var r = eventArgs[1].helper[0].getBoundingClientRect();
238 				return { left : r.left, top: r.top };
239 			} else {*/
240 			if (eventArgs.length == 1) {
241 				ret = { left: eventArgs[0].pageX, top:eventArgs[0].pageY };
242 			}
243 			else {
244 				var ui = eventArgs[1], _offset = ui.offset;			
245 				ret = _offset || ui.absolutePosition;
246 			}
247             return ret;
248 		},		
249 		
250 		hasClass : function(el, clazz) {
251 			return el.hasClass(clazz);
252 		},
253 		
254 		/**
255 		 * initialises the given element to be draggable.
256 		 */
257 		initDraggable : function(el, options, isPlumbedComponent) {
258 			options = options || {};
259 			// remove helper directive if present.  
260 			options.helper = null;
261 			if (isPlumbedComponent)
262 				options['scope'] = options['scope'] || jsPlumb.Defaults.Scope;
263 			el.draggable(options);
264 		},
265 		
266 		/**
267 		 * initialises the given element to be droppable.
268 		 */
269 		initDroppable : function(el, options) {
270 			options['scope'] = options['scope'] || jsPlumb.Defaults.Scope;
271 			el.droppable(options);
272 		},
273 		
274 		isAlreadyDraggable : function(el) {
275 			el = jsPlumb.CurrentLibrary.getElementObject(el);
276 			return el.hasClass("ui-draggable");
277 		},
278 		
279 		/**
280 		 * returns whether or not drag is supported (by the library, not whether or not it is disabled) for the given element.
281 		 */
282 		isDragSupported : function(el, options) {
283 			return el.draggable;
284 		},				
285 						
286 		/**
287 		 * returns whether or not drop is supported (by the library, not whether or not it is disabled) for the given element.
288 		 */
289 		isDropSupported : function(el, options) {
290 			return el.droppable;
291 		},							
292 		
293 		/**
294 		 * removes the given class from the element object.
295 		 */
296 		removeClass : function(el, clazz) {
297 			el = jsPlumb.CurrentLibrary.getElementObject(el);
298 			try {
299 				if (el[0].className.constructor == SVGAnimatedString) {
300 					jsPlumb.util.svg.removeClass(el[0], clazz);
301 				}
302 			}
303 			catch (e) {
304 				// SVGAnimatedString not supported; no problem.
305 			}
306 			el.removeClass(clazz);
307 		},
308 		
309 		removeElement : function(element, parent) {			
310 			jsPlumb.CurrentLibrary.getElementObject(element).remove();
311 		},
312 		
313 		/**
314 		 * sets the named attribute on the given element object.  
315 		 */
316 		setAttribute : function(el, attName, attValue) {
317 			el.attr(attName, attValue);
318 		},
319 		
320 		/**
321 		 * sets the draggable state for the given element
322 		 */
323 		setDraggable : function(el, draggable) {
324 			el.draggable("option", "disabled", !draggable);
325 		},
326 		
327 		/**
328 		 * sets the drag scope.  probably time for a setDragOption method (roll this and the one above together)
329 		 * @param el
330 		 * @param scope
331 		 */
332 		setDragScope : function(el, scope) {
333 			el.draggable("option", "scope", scope);
334 		},
335 		
336 		setOffset : function(el, o) {
337 			jsPlumb.CurrentLibrary.getElementObject(el).offset(o);
338 		},
339 		
340 		/**
341 		 * note that jquery ignores the name of the event you wanted to trigger, and figures it out for itself.
342 		 * the other libraries do not.  yui, in fact, cannot even pass an original event.  we have to pull out stuff
343 		 * from the originalEvent to put in an options object for YUI. 
344 		 * @param el
345 		 * @param event
346 		 * @param originalEvent
347 		 */
348 		trigger : function(el, event, originalEvent) {
349 			//originalEvent.stopPropagation();
350 			//jsPlumb.CurrentLibrary.getElementObject(el).trigger(originalEvent);
351             var h = jQuery._data(jsPlumb.CurrentLibrary.getElementObject(el)[0], "handle");
352             h(originalEvent);
353             //originalEvent.stopPropagation();
354 		},
355 		
356 		/**
357 		 * event unbinding wrapper.  it just so happens that jQuery uses 'unbind' also.  yui3, for example,
358 		 * uses..something else.
359 		 */
360 		unbind : function(el, event, callback) {
361 			el = jsPlumb.CurrentLibrary.getElementObject(el);
362 			el.unbind(event, callback);
363 		}
364 	};
365 	
366 	$(document).ready(jsPlumb.init);
