 15 YUI.add( 'wegas-tabview', function ( Y ) {
 16     "use strict";
 17 
 18     var TabView, Tab;
 19 
 20     TabView = Y.Base.create( "tabview", Y.TabView, [Y.WidgetChild, Y.Wegas.Widget], {
 21 
 22         initializer: function () {
 23             TabView.superclass.initializer.apply(this, arguments);
 24         //this.plug( Removeable );
 25         },
 26 
 27         bindUI: function () {
 28             TabView.superclass.bindUI.apply(this, arguments);
 29 
 30             // @fixme we notify the editor for any change, so widget can be updated
 31             // this should be done through wiget-parent, widget-child event bubbling
 32             this.after( "selectionChange", function () {
 33                 Y.Wegas.app.fire( "layout:resize" );
 34             });
 35         }
 36     }, {
 37 
 38         CSS_PREFIX: "yui3-tabview",
 39 
 40         tabs: {},
 41 
 42         getTab: function (id) {
 43             return TabView.tabs[ id ];
 44         },
 45         createTab: function ( id, tabViewSelector, tabCfg ) {
 46             if ( !TabView.tabs[ id ] ) {                                        // If the tab does not exist,
 47                 var tabView = Y.Widget.getByNode( tabViewSelector );            // Look for the parent
 48                 tabCfg = tabCfg || {};
 49                 Y.mix(tabCfg, {
 50                     type: "Tab",
 51                     label: id,
 52                     id: id
 53                 });
 54                 var tabs = tabView.add( tabCfg );                                // Instantiate a new tab
 55                 return tabs.item(0);
 56             } else {                                                            // Otherwise,
 57                 TabView.tabs[id].setAttrs( tabCfg )                             // update the tab config
 58             }
 59             return TabView.tabs[ id ];
 60         },
 61         destroyTab: function( id ){                                             //FIX destroy config
 62             if ( TabView.tabs[ id ] ) {
 63                 TabView.tabs[id].remove();
 64             }
 65         },
 66         /**
 67          *  Helper function
 68          */
 69         findTabAndLoadWidget: function (id, tabViewSelector, tabCfg, widgetCfg, fn) {
 70             var tab = TabView.getTab( id );                                     // Look for the tab
 71             if ( !tab ) {                                                       // If it does not exist,
 72                 tab = TabView.createTab( id, tabViewSelector, tabCfg);          // create a new one
 73                 tab.load(widgetCfg, fn);                                        // and load the widget in it.
 74             } else {                                                            // Otherwise,
 75                 tab.setAttrs(tabCfg);                                           // update the tab config
 76                 if (fn) {
 77                     fn( tab.item(0) );                                          // and trigger the callback
 78                 }
 79             }
 80             tab.set("selected", 2);
 81             tab.plug( Removeable );
 82         }
 83     });
 84     Y.namespace('Wegas').TabView = TabView;
 85 
 86     /**
 87      * Extension enabling a Tab to be a parent of another Widget.
 88      *
 89      * @modified from original WidgetParent module
 90      *
 91      * @module widget-parent
 92      */
 93     function Parent(config) {
 94         //Y.WidgetParent.call(this, config);
 95         this.publish("addChild", {
 96             defaultTargetOnly: true,
 97             defaultFn: this._defAddChildFn
 98         });
 99         this.publish("removeChild", {
100             defaultTargetOnly: true,
101             defaultFn: this._defRemoveChildFn
102         });
103 
104         this._items = [];
105 
106         var children,
107         handle;
108 
109         if (config && config.children) {
110 
111             children = config.children;
112 
113             handle = this.after("initializedChange", function (e) {
114                 this._add(children);
115                 handle.detach();
116             });
117 
118         }
119 
120         //  Widget method overlap
121         Y.after(this._renderChildren, this, "renderUI");
122         Y.after(this._bindUIParent, this, "bindUI");
123 
124         //this.after("selectionChange", this._afterSelectionChange);
125         //this.after("selectedChange", this._afterParentSelectedChange);
126         //this.after("activeDescendantChange", this._afterActiveDescendantChange);
127 
128         this._hDestroyChild = this.after("*:destroy", this._afterDestroyChild);
129         this.after("*:focusedChange", this._updateActiveDescendant);
130     }
131 
132     //Y.extend(Parent, Y.WidgetParent);
133     Y.mix(Parent, Y.WidgetParent);
134     Y.augment(Parent, Y.WidgetParent);
135     Parent.ATTRS = {};
136     Y.mix(Parent.ATTRS, Y.WidgetParent.ATTRS);
137     delete Parent.ATTRS.selected;
138 
139     /**
140      * Custom Tab implementation
141      */
142     Tab = Y.Base.create("tab", Y.Tab, [ Y.Wegas.Widget, Parent, Y.WidgetChild ], {
143 
144         // *** Private Fields *** //
145 
146         // *** Lifecycle Methods *** //
147         initializer: function(cfg) {
148             Tab.superclass.initializer.apply( this, arguments );
149             TabView.tabs[ cfg.id ] = this;
150             this._witems = [];
151 
152         //this.plug( Closable );
153         },
154 
155         renderUI: function () {
156             Tab.superclass.renderUI.apply( this, arguments );
157         },
158 
159         syncUI: function () {
160             Tab.superclass.syncUI.apply( this, arguments );
161         },
162 
163         destructor: function () {
164             delete TabView.tabs[ this.get( "id" ) ];
165             Tab.superclass.destructor.apply( this, arguments );
166         },
167 
168         // *** Private Methods *** //
169         /**
170          * Retrieves the given widget configuration and add it to the tab
171          *
172          * @function load
173          *
174          */
175         load: function ( cfg, callback ) {
176             Y.Wegas.Widget.use(cfg,  Y.bind(function (cfg, callback) {          // Load the subpage dependencies
177                 var widgets = this.add(cfg);                                    // Render the subpage
178                 if (callback) {
179                     callback(widgets.item(0));                                  // Trigger the callback
180                 }
181             }, this, cfg, callback));
182         },
183 
184         /**
185          *  Override Y.WidgetParent to render children in the panel node;
186          */
187         _renderChildren: function () {
188             var renderTo = this._childrenContainer || this.get("panelNode");    // @modified
189 
190             this._childrenContainer = renderTo;
191 
192             this.each(function (child) {
193                 this._witems.push( child );                                     // @modified
194                 child.render( renderTo )
195             }, this);
196         },
197         witem: function ( index ) {
198             return this._witems[index];
199         }
200     }, {
201 
202         CSS_PREFIX: "yui3-tab",
203 
204         ATTRS : {
205             content: {
206                 setter: function () { }                                         // Overrides the panelNode management
207             }
208         }
209     });
210     Y.namespace( 'Wegas' ).Tab = Tab;
