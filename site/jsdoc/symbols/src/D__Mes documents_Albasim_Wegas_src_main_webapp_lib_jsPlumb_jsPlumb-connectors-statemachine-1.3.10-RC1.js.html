 24 	var Line = function(x1, y1, x2, y2) {
 25 
 26 		this.m = (y2 - y1) / (x2 - x1);
 27 		this.b = -1 * ((this.m * x1) - y1);

 29 		this.rectIntersect = function(x,y,w,h) {
 30 			var results = [];
 31 		
 32 			// 	try top face
 33 			// 	the equation of the top face is y = (0 * x) + b; y = b.
 34 			var xInt = (y - this.b) / this.m;
 35 			// test that the X value is in the line's range.
 36 			if (xInt >= x && xInt <= (x + w)) results.push([ xInt, (this.m * xInt) + this.b ]);
 37 		
 38 			// try right face
 39 			var yInt = (this.m * (x + w)) + this.b;
 40 			if (yInt >= y && yInt <= (y + h)) results.push([ (yInt - this.b) / this.m, yInt ]);
 41 		
 42 			// 	bottom face
 43 			var xInt = ((y + h) - this.b) / this.m;
 44 			// test that the X value is in the line's range.
 45 			if (xInt >= x && xInt <= (x + w)) results.push([ xInt, (this.m * xInt) + this.b ]);
 46 		
 47 			// try left face
 48 			var yInt = (this.m * x) + this.b;
 49 			if (yInt >= y && yInt <= (y + h)) results.push([ (yInt - this.b) / this.m, yInt ]);
 50 
 51 			if (results.length == 2) {
 52 				var midx = (results[0][0] + results[1][0]) / 2, midy = (results[0][1] + results[1][1]) / 2;
 53 				results.push([ midx,midy ]);
 54 				// now calculate the segment inside the rectangle where the midpoint lies.
 55 				var xseg = midx <= x + (w / 2) ? -1 : 1,
 56 					yseg = midy <= y + (h / 2) ? -1 : 1;
 57 				results.push([xseg, yseg]);
 58 				return results;
 59 			}
 60 		
 61 			return null;
 62 
 63 		};

 65 	_segment = function(x1, y1, x2, y2) {
 66 		if (x1 <= x2 && y2 <= y1) return 1;
 67 		else if (x1 <= x2 && y1 <= y2) return 2;
 68 		else if (x2 <= x1 && y2 >= y1) return 3;
 69 		return 4;
 70 	},
 71 		
 72 		// the control point we will use depends on the faces to which each end of the connection is assigned, specifically whether or not the
 73 		// two faces are parallel or perpendicular.  if they are parallel then the control point lies on the midpoint of the axis in which they
 74 		// are parellel and varies only in the other axis; this variation is proportional to the distance that the anchor points lie from the
 75 		// center of that face.  if the two faces are perpendicular then the control point is at some distance from both the midpoints; the amount and
 76 		// direction are dependent on the orientation of the two elements. 'seg', passed in to this method, tells you which segment the target element
 77 		// lies in with respect to the source: 1 is top right, 2 is bottom right, 3 is bottom left, 4 is top left.
 78 		//
 79 		// sourcePos and targetPos are arrays of info about where on the source and target each anchor is located.  their contents are:
 80 		//
 81 		// 0 - absolute x
 82 		// 1 - absolute y
 83 		// 2 - proportional x in element (0 is left edge, 1 is right edge)
 84 		// 3 - proportional y in element (0 is top edge, 1 is bottom edge)
 85 		// 	
 86 	_findControlPoint = function(midx, midy, segment, sourceEdge, targetEdge, dx, dy, distance, proximityLimit) {
 87 
 88         // TODO (maybe)
 89         // - if anchor pos is 0.5, make the control point take into account the relative position of the elements.
 90         if (distance <= proximityLimit) return [midx, midy];
 91 
 92         if (segment == 1) {
 93             if (sourceEdge[3] <= 0 && targetEdge[3] >= 1) return [ midx + (sourceEdge[2] < 0.5 ? -1 * dx : dx), midy ];
 94             else if (sourceEdge[2] >= 1 && targetEdge[2] <= 0) return [ midx, midy + (sourceEdge[3] < 0.5 ? -1 * dy : dy) ];
 95             else return [ midx + (-1 * dx) , midy + (-1 * dy) ];
 96         }
 97         else if (segment == 2) {
 98             if (sourceEdge[3] >= 1 && targetEdge[3] <= 0) return [ midx + (sourceEdge[2] < 0.5 ? -1 * dx : dx), midy ];
 99             else if (sourceEdge[2] >= 1 && targetEdge[2] <= 0) return [ midx, midy + (sourceEdge[3] < 0.5 ? -1 * dy : dy) ];
100             else return [ midx + (1 * dx) , midy + (-1 * dy) ];
101         }
102         else if (segment == 3) {
103             if (sourceEdge[3] >= 1 && targetEdge[3] <= 0) return [ midx + (sourceEdge[2] < 0.5 ? -1 * dx : dx), midy ];
104             else if (sourceEdge[2] <= 0 && targetEdge[2] >= 1) return [ midx, midy + (sourceEdge[3] < 0.5 ? -1 * dy : dy) ];
105             else return [ midx + (-1 * dx) , midy + (-1 * dy) ];
106         }
107         else if (segment == 4) {
108             if (sourceEdge[3] <= 0 && targetEdge[3] >= 1) return [ midx + (sourceEdge[2] < 0.5 ? -1 * dx : dx), midy ];
109             else if (sourceEdge[2] <= 0 && targetEdge[2] >= 1) return [ midx, midy + (sourceEdge[3] < 0.5 ? -1 * dy : dy) ];
110             else return [ midx + (1 * dx) , midy + (-1 * dy) ];
111         }

114 	/*
115 		Function: StateMachine constructor
116 		
117 		Allowed parameters:
118 			curviness 		-	measure of how "curvy" the connectors will be.  this is translated as the distance that the
119 								Bezier curve's control point is from the midpoint of the straight line connecting the two
120 								endpoints, and does not mean that the connector is this wide.  The Bezier curve never reaches
121 								its control points; they act as gravitational masses. defaults to 10.
122 			margin			-	distance from element to start and end connectors, in pixels.  defaults to 5.
123 			proximityLimit  -   sets the distance beneath which the elements are consider too close together to bother with fancy
124 			                    curves. by default this is 80 pixels.
125 			loopbackRadius	-	the radius of a loopback connector.  optional; defaults to 25.
126 	*/
127 	jsPlumb.Connectors.StateMachine = function(params) {
128 		var self = this,
129 		currentPoints = null,
130 		_sx, _sy, _tx, _ty, _controlPoint = [],
131 		curviness = params.curviness || 10,
132 		margin = params.margin || 5,
133 		proximityLimit = params.proximityLimit || 80,
134 		clockwise = params.orientation && params.orientation == "clockwise",
135 		loopbackRadius = params.loopbackRadius || 25,
136 		isLoopback = false;
137 
138 		this.type = "StateMachine";
139 		params = params || {};

141 		this.compute = function(sourcePos, targetPos, sourceEndpoint, targetEndpoint, sourceAnchor, targetAnchor, lineWidth, minWidth) {
142 
143 			var w = Math.abs(sourcePos[0] - targetPos[0]),
144    	 	       	h = Math.abs(sourcePos[1] - targetPos[1]),
145 	   	     	// these are padding to ensure the whole connector line appears
146    	   	   		xo = 0.45 * w, yo = 0.45 * h;
147    		   		// these are padding to ensure the whole connector line appears
148             	w *= 1.9; h *= 1.9;
149                 //ensure at least one pixel width
150                 lineWidth = lineWidth || 1;
151             	var x = Math.min(sourcePos[0], targetPos[0]) - xo,
152         		    y = Math.min(sourcePos[1], targetPos[1]) - yo;
153 		
154 			if (sourceEndpoint.elementId != targetEndpoint.elementId) {
155                             
156                 isLoopback = false;
157                             
158         		_sx = sourcePos[0] < targetPos[0] ?  xo : w-xo;
159             	_sy = sourcePos[1] < targetPos[1] ? yo:h-yo;
160             	_tx = sourcePos[0] < targetPos[0] ? w-xo : xo;
161             	_ty = sourcePos[1] < targetPos[1] ? h-yo : yo;
162             
163             	// now adjust for the margin
164             	if (sourcePos[2] == 0) _sx -= margin;
165             	if (sourcePos[2] == 1) _sx += margin;
166             	if (sourcePos[3] == 0) _sy -= margin;
167             	if (sourcePos[3] == 1) _sy += margin;
168             	if (targetPos[2] == 0) _tx -= margin;
169             	if (targetPos[2] == 1) _tx += margin;
170             	if (targetPos[3] == 0) _ty -= margin;
171             	if (targetPos[3] == 1) _ty += margin;
172 
173             	//
174 	            // these connectors are quadratic bezier curves, having a single control point. if both anchors 
175     	        // are located at 0.5 on their respective faces, the control point is set to the midpoint and you
176         	    // get a straight line.  this is also the case if the two anchors are within 'proximityLimit', since
177            	 	// it seems to make good aesthetic sense to do that. outside of that, the control point is positioned 
178            	 	// at 'curviness' pixels away along the normal to the straight line connecting the two anchors.
179 	            // 
180    	        	// there may be two improvements to this.  firstly, we might actually support the notion of avoiding nodes
181             	// in the UI, or at least making a good effort at doing so.  if a connection would pass underneath some node,
182             	// for example, we might increase the distance the control point is away from the midpoint in a bid to
183             	// steer it around that node.  this will work within limits, but i think those limits would also be the likely
184             	// limits for, once again, aesthetic good sense in the layout of a chart using these connectors.
185             	//
186             	// the second possible change is actually two possible changes: firstly, it is possible we should gradually
187             	// decrease the 'curviness' as the distance between the anchors decreases; start tailing it off to 0 at some
188             	// point (which should be configurable).  secondly, we might slightly increase the 'curviness' for connectors
189             	// with respect to how far their anchor is from the center of its respective face. this could either look cool,
190             	// or stupid, and may indeed work only in a way that is so subtle as to have been a waste of time.
191             	//
192 
193 				var _midx = (_sx + _tx) / 2, _midy = (_sy + _ty) / 2, 
194             	    m2 = (-1 * _midx) / _midy, theta2 = Math.atan(m2),
195             	    dy =  (m2 == Infinity || m2 == -Infinity) ? 0 : Math.abs(curviness / 2 * Math.sin(theta2)),
196 				    dx =  (m2 == Infinity || m2 == -Infinity) ? 0 : Math.abs(curviness / 2 * Math.cos(theta2)),
197 				    segment = _segment(_sx, _sy, _tx, _ty),
198 				    distance = Math.sqrt(Math.pow(_tx - _sx, 2) + Math.pow(_ty - _sy, 2));
199 			
200             	// calculate the control point.  this code will be where we'll put in a rudimentary element avoidance scheme; it
201             	// will work by extending the control point to force the curve to be, um, curvier.
202 				_controlPoint = _findControlPoint(_midx,
203                                                   _midy,
204                                                   segment,
205                                                   sourcePos,
206                                                   targetPos,
207                                                   curviness, curviness,
208                                                   distance,
209                                                   proximityLimit);
210 
211 	            	
212             	var requiredWidth = Math.max(Math.abs(_controlPoint[0] - _sx) * 3, Math.abs(_controlPoint[0] - _tx) * 3, Math.abs(_tx-_sx), 2 * lineWidth, minWidth),
213             		requiredHeight = Math.max(Math.abs(_controlPoint[1] - _sy) * 3, Math.abs(_controlPoint[1] - _ty) * 3, Math.abs(_ty-_sy), 2 * lineWidth, minWidth);
214 
215             	if (w < requiredWidth) {      	
216             		var dw = requiredWidth - w;            		
217             		x -= (dw / 2);
218             		_sx += (dw / 2);
219             		_tx  += (dw / 2);
220             		w = requiredWidth;
221                     _controlPoint[0] += (dw / 2);
222             	}
223             	
224             	if (h < requiredHeight) {
225             		var dh = requiredHeight - h;
226             		y -= (dh / 2);
227             		_sy += (dh / 2);
228             		_ty += (dh / 2);
229             		h = requiredHeight;
230                     _controlPoint[1] += (dh / 2);
231             	}
232             	currentPoints = [ x, y, w, h, _sx, _sy, _tx, _ty, _controlPoint[0], _controlPoint[1] ];                                        
233             }
234             else {
235             	isLoopback = true;
236             	// a loopback connector.  draw an arc from one anchor to the other.
237             	// i guess we'll do this the same way as the others.  just the control point will be a fair distance away.
238         		var x1 = sourcePos[0], x2 = sourcePos[0], y1 = sourcePos[1] - margin, y2 = sourcePos[1] - margin, 				
239 					cx = x1, cy = y1 - loopbackRadius;
240 				
241 				// canvas sizing stuff, to ensure the whole painted area is visible.
242 				w = ((2 * lineWidth) + (4 * loopbackRadius)), h = ((2 * lineWidth) + (4 * loopbackRadius));
243 				x = cx - loopbackRadius - lineWidth - loopbackRadius, y = cy - loopbackRadius - lineWidth - loopbackRadius;
244 				currentPoints = [ x, y, w, h, cx-x, cy-y, loopbackRadius, clockwise, x1-x, y1-y, x2-x, y2-y];
245             }
246                 
247             return currentPoints;
248         };

250         var _makeCurve = function() {
251         	return [	
252 	        	{ x:_tx, y:_ty },
253 	        	{ x:_controlPoint[0], y:_controlPoint[1] },
254 	        	{ x:_controlPoint[0] + 1, y:_controlPoint[1] + 1},	        	
255 	        	{ x:_sx, y:_sy }
256          	];

259         /**
260          * returns the point on the connector's path that is 'location' along the length of the path, where 'location' is a decimal from
261          * 0 to 1 inclusive. for the straight line connector this is simple maths.  for Bezier, not so much.
262          */
263         this.pointOnPath = function(location) {   
264 			if (isLoopback) {
265 
266                 if (location > 0 && location < 1) location = 1- location;
267                 
268 // current points are [ x, y, width, height, center x, center y, radius, clockwise, startx, starty, endx, endy ]				
269 				// so the path length is the circumference of the circle
270 				//var len = 2 * Math.PI * currentPoints[6],
271 				// map 'location' to an angle. 0 is PI/2 when the connector is on the top face; if we
272 				// support other faces it will have to be calculated for each one. 1 is also PI/2.
273 				// 0.5 is -PI/2.
274 				var startAngle = (location * 2 * Math.PI) + (Math.PI / 2),
275 					startX = currentPoints[4] + (currentPoints[6] * Math.cos(startAngle)),
276 					startY = currentPoints[5] + (currentPoints[6] * Math.sin(startAngle));					
277 
278                 return {x:startX, y:startY};
279 					
280 			}
281         	else return jsBezier.pointOnCurve(_makeCurve(), location);
282         };

284         /**
285          * returns the gradient of the connector at the given point.
286          */
287         this.gradientAtPoint = function(location) {
288 			if (isLoopback)
289 				return Math.atan(location * 2 * Math.PI);
290         	else
291                 return jsBezier.gradientAtPoint(_makeCurve(), location);
292         };

294         /**
295          * for Bezier curves this method is a little tricky, cos calculating path distance algebraically is notoriously difficult.
296          * this method is iterative, jumping forward .05% of the path at a time and summing the distance between this point and the previous
297          * one, until the sum reaches 'distance'. the method may turn out to be computationally expensive; we'll see.
298          * another drawback of this method is that if the connector gets quite long, .05% of the length of it is not necessarily smaller
299          * than the desired distance, in which case the loop returns immediately and the arrow is mis-shapen. so a better strategy might be to
300          * calculate the step as a function of distance/distance between endpoints.  
301          */
302         this.pointAlongPathFrom = function(location, distance) {        	
303 			if (isLoopback) {
304 
305                 if (location > 0 && location < 1) location = 1- location;
306 
307 				var circumference = 2 * Math.PI * currentPoints[6],
308 					arcSpan = distance / circumference * 2 * Math.PI,
309 					startAngle = (location * 2 * Math.PI) - arcSpan + (Math.PI / 2),	
310 					
311 					startX = currentPoints[4] + (currentPoints[6] * Math.cos(startAngle)),
312 					startY = currentPoints[5] + (currentPoints[6] * Math.sin(startAngle));	
313 
314 				return {x:startX, y:startY};
315 			}
316         	return jsBezier.pointAlongCurveFrom(_makeCurve(), location, distance);
317         };
