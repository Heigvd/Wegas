 14 YUI.add("wegas-pmg-treebletasklist", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Tasklist;
 18 
 19     Tasklist = Y.Base.create("wegas-pmg-treebletasklist", Y.Wegas.PmgDatatable, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 20         handlers: null,
 21         root: null,
 22         /*** lifecycle methods ***/
 23         initializer: function () {
 24             var i, treeble_config, schema_plugin_config, schema, resultFields, dataSource;
 25             this.handlers = {};
 26 
 27             //Create and add datasource to Treeble Datatable
 28             //Add formatter to the "name column" (for indentation)
 29             for (i = 0; i < this.datatable.get('columns').length; i += 1) {
 30                 if (this.datatable.get('columns')[i].key === 'Name') {
 31                     this.datatable.get('columns')[i].formatter = Y.Treeble.treeValueFormatter;
 32                     this.datatable.get('columns')[i].allowHTML = true;
 33                 }
 34             }
 35 
 36             //Add the column wich will contain the "open node"
 37             this.datatable.addColumn({
 38                 key: 'treeblenub',
 39                 label: ' ',
 40                 nodeFormatter: Y.Treeble.buildTwistdownFormatter(Y.bind(this.treebleTwistdown, this))
 41             }, 1, this);
 42 
 43             //Create the schema of the table's datas
 44             resultFields = this.get('columnTitles').slice(0);
 45             resultFields.unshift('_id');
 46             resultFields.push('_open');
 47             resultFields.push({
 48                 key: 'kiddies',
 49                 parser: 'treebledatasource'
 50             });
 51             schema = {
 52                 resultFields: resultFields
 53             };
 54 
 55             //Create the schema plugin for the datasource's root 
 56             schema_plugin_config = {
 57                 fn: Y.Plugin.DataSourceArraySchema,
 58                 cfg: {
 59                     schema: schema
 60                 }
 61             };
 62 
 63             //Create the config object for the datasource's root 
 64             treeble_config = {
 65                 generateRequest: function () {
 66                 },
 67                 schemaPluginConfig: schema_plugin_config,
 68                 childNodesKey: 'kiddies',
 69                 nodeOpenKey: '_open',
 70                 totalRecordsReturnExpr: '.meta.totalRecords'
 71             };
 72 
 73             //Create the root object for the datasource
 74             this.root = new Y.DataSource.Local({
 75                 source: null
 76             });
 77             this.root.treeble_config = Y.clone(treeble_config, true);
 78             this.root.plug(schema_plugin_config);
 79 
 80             //Create the datasource of the table
 81             dataSource = new Y.DataSource.Treeble({
 82                 root: this.root,
 83                 paginateChildren: false,
 84                 uniqueIdKey: '_id'
 85             }, this);
 86 
 87             //plug datasource to the Treeble Datatable
 88             this.datatable.plug(Y.Plugin.DataTableDataSource, {
 89                 datasource: dataSource
 90             }, this);
 91         },
 92         renderUI: function () {
 93             Tasklist.superclass.renderUI.apply(this);
 94         },
 95         bindUI: function () {
 96             Tasklist.superclass.bindUI.apply(this);
 97             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 98             
 99             this.handlers.sort = this.datatable.after('sort', this.syncUI, this);
100             
101             this.handlers.displayDescription = this.datatable.delegate('click', function (e) {
102                 this.displayDescription(e);
103             }, '.yui3-datatable-data td', this);
104         
105             this.handlers.removeDescription = this.datatable.delegate('mouseout', function (e) {
106                 this.removeDescription(e);
107             }, '.yui3-datatable-data tr', this);
108         },
109         syncUI: function () {
110             Tasklist.superclass.syncUI.apply(this);
111             Y.bind(this.treebleTwistdown(), this);
112             this.checkRealization();
113         },
114         destructor: function () {
115             var k;
116             for (k in this.handlers) {
117                 this.handlers[k].detach();
118             }
119         },
120         //*** Private Methods ***/
121         checkRealization: function () {
122             var i, cb = this.get(CONTENTBOX), tasks, taskDesc, taskInst, realized, allRow;
123             if (this.data == null
124                     || this.data.length == 0
125                     || this.get("columnValues").indexOf('realized') <= -1) {
126                 return;
127             }
128             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
129             allRow = cb.all(".yui3-datatable-data tr");
130             allRow.removeClass("notstarted").removeClass("started").removeClass("completed");
131             allRow.each(function (node) {
132                 for (i = 0; i < tasks.get('items').length; i += 1) {
133                     taskDesc = tasks.get('items')[i];
134                     taskInst = taskDesc.getInstance();
135                     realized = (taskInst.get('properties').realized) ? taskInst.get('properties').realized : null;
136                     if (realized) {
137                         if (node.one("*").getContent() == taskDesc.get('id')) {
138                             if (realized >= 100) {
139                                 node.addClass("completed");
140                             } else if (realized > 0) {
141                                 node.addClass("started");
142                             } else {
143                                 node.addClass("notstarted");
144                             }
145                             break;
146                         }
147                     }
148                 }
149             });
150         },
151         displayDescription: function (e) {
152             var i, id, label, tasks, node, divDesc, taskDesc, description;
153             node = e.currentTarget;
154             if (this.get("viewDescription") == "false"
155                     || node.one(".description")
156                     || node.get("className").indexOf("cell-gantt") > -1) {
157                 return;
158             }
159             id = node.ancestor().one("*").getContent();
160             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
161             if (!id || !tasks) {
162                 return;
163             }
164             for (i = 0; i < tasks.get('items').length; i += 1) {
165                 taskDesc = tasks.get('items')[i];
166                 if (taskDesc.get('id') == id) {
167                     label = (taskDesc.get("label") || taskDesc.get("name"));
168                     description = taskDesc.get("description");
169                     break;
170                 }
171             }
172             divDesc = Y.Node.create("<div class='description'></div>");
173             divDesc.append("<p class='task_name'>" + label + "</p>").append("<p class='content'>" + description + "</p>");
174             node.append(divDesc);
175         },
176         removeDescription: function (e) {
177             var disappearAnim, node;
178             node = this.get(CONTENTBOX).one('.description');
179             if (!node) {
180                 return;
181             }
182             disappearAnim = new Y.Anim({
183                 node: node,
184                 to: {
185                     opacity: 0
186                 },
187                 duration: 0.2
188             });
189             disappearAnim.run();
190             disappearAnim.on('end', function () {
191                 node.remove();
192             });
193         },
194         getTreebleDatas: function () {
195             var i, data = this.data.slice(0), tasks, description;
196             if (!data) {
197                 return null;
198             }
199             //get data of the Treeble Datatable and add "kiddies" with description of the task
200             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
201             for (i = 0; i < data.length; i += 1) {
202                 description = tasks.get('items')[i].get('description');
203                 data[i].kiddies = [{
204                         Name: description
205                     }];
206             }
207             return data;
208         },
209         treebleTwistdown: function () {
210             this.root.set('source', this.getTreebleDatas());                    //Get currents datas and set datasource
211             this.datatable.datasource.load({//Request max 10 rows per trebble's level
212                 request: {
213                     startIndex: 0,
214                     resultCount: 10
215                 }
216             });
217             this.checkRealization();
218         }
219 
220     }, {
221         ATTRS: {
222             viewDescription: {
223                 value: true,
224                 validator: function (b) {
225                     return b == "false" || b == "true";
226                 }
227             },
228             viewAssignements: {//todo
229                 value: true,
230                 validator: function (b) {
231                     return b == "false" || b == "true";
232                 }
233             }
234         }
235     });
236 
237     Y.namespace("Wegas").PmgTreebleTasklist = Tasklist;
238 });
