 12 YUI.add('wegas-editor-pagetreeview', function(Y) {
 13     var PageEditor, CONTENT_BOX = "contentBox", BOUNDING_BOX = "boundingBox";
 14 
 15     PageEditor = Y.Base.create("wegas-editor-page", Y.Widget, [Y.WidgetChild], {
 16         initializer: function() {
 17             this.dataSource = Y.Wegas.PageFacade.rest;
 18             this.plug(Y.Plugin.WidgetToolbar);
 19         },
 20         renderUI: function() {
 21             this.tw = new Y.TreeView({
 22                 render: this.get(CONTENT_BOX)
 23             });
 24             this.toolbar.add(new Y.Button({
 25                 label: "New Page",
 26                 on: {
 27                     click: function() {
 28                         this.fire("newPage");
 29                     }
 30                 }
 31             }));
 32         },
 33         syncUI: function() {
 34             this.dataSource.getIndex(Y.bind(this.buildIndex, this));
 35         },
 36         bindUI: function() {
 37             this.on("*:newPage", function(e) {
 38                 this.dataSource.post({});
 39             });
 40             this.tw.on("treenode:click", function(e) {
 41                 var node = e.node;
 42                 if (!node.get("data")) {
 43                     return;
 44                 }
 45                 if (node.get("data")["page"]) {
 46                     this.get("pageLoader").set("pageId", node.get("data")["page"]);
 47                 }
 48             }, this);
 49             this.tw.on("treenode:nodeExpanded", function(e) {
 50                 var node = e.node;
 51                 if (!node.get("data")) {
 52                     return;
 53                 }
 54                 if (node.get("data")["page"]) {
 55                     this.get("pageLoader").set("pageId", node.get("data")["page"]);
 56                 }
 57             }, this);
 58             this.tw.get(CONTENT_BOX).delegate("mouseenter", function(e) {
 59                 var node = Y.Widget.getByNode(e.target);
 60                 e.halt(true);
 61                 if (node.get("data.widget")) {
 62                     if (this.get("pageLoader").pageeditor) {
 63                         this.get("pageLoader").pageeditor.showOverlay(node.get("data.widget"));
 64                     } else {
 65                         node.get("data.widget").get(BOUNDING_BOX).addClass("wegas-focused-widget");
 66                     }
 67                 }
 68             }, ".content-header", this);
 69             this.tw.get(CONTENT_BOX).delegate("mouseleave", function(e) {
 70                 var node = Y.Widget.getByNode(e.target);
 71                 e.halt(true);
 72                 if (node.get("data.widget")) {
 73                     if (this.get("pageLoader").pageeditor) {
 74                         this.get("pageLoader").pageeditor.hideOverlay();
 75                     } else {
 76                         node.get("data.widget").get(BOUNDING_BOX).removeClass("wegas-focused-widget");
 77                     }
 78                 }
 79             }, ".content-header", this);
 80 
 81 
 82             this.dataSource.after("pageUpdated", this.syncUI, this);
 83             this.get("pageLoader").after("pageIdChange", this.syncUI, this);
 84             //this.get("pageLoader").get("widget").after("*:destroy", this.syncUI, this);
 85         },
 86         buildWidgetTree: function(node) {
 87             var widget = this.get("pageLoader").get("widget");
 88             this.cleanAllChildren();
 89             node.removeAll();
 90             this.buildSubTree(node, widget);
 91             node.expand(false);
 92         },
 93         buildSubTree: function(node, widget) {
 94             var treeNode, button = new Y.Button({label: "edit"});
 95 
 96             if (widget instanceof Y.Wegas.List) {
 97                 treeNode = new Y.TreeNode({
 98                     label: "List",
 99                     data: {
100                         widget: widget
101                     }
102                 });
103 
104                 widget.each(function(item, index) {
105                     this.buildSubTree(treeNode, item);
106                 }, this);
107 //                for (var i in widget["children"]) {
108 //                    this.buildSubTree(treeNode, widget["children"][i]);
109 //                }
110             } else {
111                 treeNode = new Y.TreeLeaf({
112                     label: "Widget: " + widget.constructor.NAME,
113                     data: {
114                         widget: widget
115                     }
116                 });
117             }
118             button.plug(Y.Plugin.WidgetMenu, {
119                 children: [{
120                         type: "Button",
121                         label: "Edit",
122                         on: {
123                             click: Y.bind(function() {                             // Display the edit form
124                                 Y.Plugin.EditEntityAction.showEditForm(widget, Y.bind(function(targetWidget, val, e, f) {
125                                     Y.Plugin.EditEntityAction.hideEditFormOverlay();
126                                     targetWidget.setAttrs(val);
127                                     targetWidget.syncUI();
128                                     Y.Wegas.PageFacade.rest.patch(targetWidget.get("root").toObject());
129                                 }, this, widget));
130                             }, this)
131                         }
132                     }, {
133                         type: "Button",
134                         label: "Delete",
135                         on: {
136                             click: Y.bind(function() {
137                                 var root = widget.get("root");
138                                 /* dont delete root ! */
139                                 if (root !== widget) {
140                                     widget.destroy();
141                                     this.destroy();
142                                 } else if (widget.item && widget.item(0)) {
143                                     widget.removeAll();
144                                     this.removeAll();
145                                 }
146                                 Y.Wegas.PageFacade.rest.patch(root.toObject());
147 
148                             }, treeNode, widget)
149                         }
150                     }],
151                 event: "click"
152             });
153 
154             treeNode.set("rightWidget", button);
155             node.add(treeNode);
156         },
157         buildIndex: function(index) {
158             var i, page = this.get("pageLoader").get("pageId"),
159                     node,
160                     widget = this.get("pageLoader").get("widget");
161             this.tw.removeAll();
162             for (i in index) {
163                 node = new Y.TreeNode({
164                     label: "Page: " + (Y.Lang.isString(index[i]) ? index[i] : "") + " (" + i + ")",
165                     data: {
166                         page: i
167                     }
168                 });
169                 this.tw.add(node);
170                 if (+i === +page) {
171                     this.buildSubTree(node, widget);
172                     node.expand(false);
173                 }
174             }
175         },
176         cleanAllChildren: function() {
177             for (var i in this.tw._items) {
178                 this.tw._items[i].collapse(false);
179                 //this.tw._items[i].removeAll();
180             }
181         },
182         destructor: function() {
183             this.tw.destroy();
184         }
185     }, {
186         CSS_PREFIX: "wegas-page-editor",
187         ATTRS: {
188             pageLoader: {
189                 value: "previewPageLoader",
190                 getter: function(v) {
191                     return Y.Wegas.PageLoader.find(v);
192                 }
193             }
194         }
195     });
196     Y.namespace("Wegas").PageTreeview = PageEditor;
197 });
