  3 /**
  4  * terminate the current week.
  5  * if value of week <= max value of week :
  6  * - Increment the value of week
  7  * - Check the end of all task (checkTasksEnd)
  8  * - Check the end of all absent (checkAbsencesEnd)
  9  * - Remove all deactivated assignments (removeDeactivatedAssignements)
 10  * - Active or Deactive some tasks (checkTasksState)
 11  * - Pay all active resources (payResources)
 12  * - Check the moral of all active resources (checkMoral)
 13  * - Check the 'LeadershipLevel' of all active resources (checkLeadershipLevel)
 14  */
 15 function finishCurrentWeek(){
 16     var gm = self.getGameModel(),
 17     weekDescriptor = VariableDescriptorFacade.findByName(gm, 'week'),
 18     actionsDescriptor = VariableDescriptorFacade.findByName(gm, 'actions'),
 19     weekInstance = weekDescriptor.getInstance(self);
 20     if(weekInstance.getValue() <= weekDescriptor.getMaxValue()){
 21         weekInstance.setValue(weekInstance.getValue()+1);
 22         actionsDescriptor.getInstance(self).setValue(actionsDescriptor.getMaxValue());
 23         this.checkTasksEnd();
 24         this.checkAbsencesEnd();
 25         this.removeDeactivatedAssignements();
 26         this.checkTasksState();
 27         this.payResources();
 28         this.checkMoral();
 29         this.checkLeadershipLevel();
 30         this.sendScore();
 31         this.resetDialogueValues(); //specific to this scenario.
 32     }
 33 }

 35 /**
 36  * @return an Array of all active and not absent resourceDescriptor.
 37  */
 38 function getValideResources(){
 39     var i, j, k, gm = self.getGameModel(), valideResources = new Array(), isValid = false,
 40     listResources = VariableDescriptorFacade.findByName(gm, 'resources'), resourceDescriptor, resourceInstance,
 41     listAbsences = VariableDescriptorFacade.findByName(gm, 'absences'), absencesInstance, assignment;
 42     for(i=0; i<listResources.items.size();i++){
 43         resourceDescriptor = listResources.items.get(i);
 44         resourceInstance = resourceDescriptor.getInstance(self);
 45         isValid = false;
 46         if(resourceInstance.getActive() == true){
 47             isValid = true;
 48             for(j=0; j<resourceInstance.getAssignments().size(); j++){
 49                 assignment = resourceInstance.getAssignments().get(j);
 50                 for(k=0; k<listAbsences.items.size();k++){
 51                     absencesInstance = listAbsences.items.get(k).getInstance(self);
 52                     if(assignment.getTaskDescriptorId() == absencesInstance.getDescriptorId() && absencesInstance.getActive() == true){
 53                         isValid = false;
 54                     }
 55                 }
 56             }
 57         }
 58         if(isValid) valideResources.push(resourceDescriptor);
 59     }
 60     return valideResources;

 90 /**
 91  * Decreases the duration of all worked task by the number of workers on this task.
 92  * for each worked task that have a duration <=0, call the fonction 'doTaskEnd'
 93  * with the task and the corresponding workers for args.
 94  */
 95 function checkTasksEnd(){
 96     var i, j, k, assignment, taskWorkers = new Array(),
 97     listValidResource = this.getValideResources(), resourceInstance,
 98     listWorkedTasks = this.getWorkedTasks(listValidResource), taskInstance;
 99     for(i=0; i<listWorkedTasks.length;i++){
100         taskWorkers.length = 0;
101         taskInstance = listWorkedTasks[i].getInstance(self);
102         for(j=0; j<listValidResource.length; j++){
103             resourceInstance = listValidResource[j].getInstance(self);
104             for(k=0; k<resourceInstance.getAssignments().size(); k++){
105                 assignment = resourceInstance.getAssignments().get(k);
106                 if(assignment.getTaskDescriptorId() == taskInstance.getDescriptorId()){
107                     taskWorkers.push(listValidResource[j]);
108                 }
109             }
110         }
111         taskInstance.setDuration(taskInstance.getDuration()-taskWorkers.length);
112         if(taskInstance.getDuration() <= 0){
113             this.doTaskEnd(taskWorkers, listWorkedTasks[i]);
114         }
115     }

118 /**
119  * Calculate experience (general gained and experience in skillsets needed by
120  * the finished task) for all workers on a finished tasks.
121  * can remove the 'wish' or the 'hate' values of these workers.
122  * add the remuneration of the task at the general budget.
123  * send a rapport-email to the player
124  * Deactivate the task
125  * @param Array of resourceDescriptor workersDescriptor, all the resources that worked on the task.
126  * @param TaskDescriptor taskDescriptor, the task ended.
127  */
128 function doTaskEnd(workersDescriptor, taskDescriptor){
129     if(workersDescriptor.length <= 0 || taskDescriptor == null) return;
130     var i,j, gm = self.getGameModel(), remuneration, taskInstance = taskDescriptor.getInstance(self),
131     budgetInstance = VariableDescriptorFacade.findByName(gm, 'budget').getInstance(self), from = new Array(), content = new Array(),
132     clientsSatisfaction = VariableDescriptorFacade.findByName(gm, 'clientsSatisfaction').getInstance(self), taskSkillKey, wish, hate,
133     taskSkillValue, listTaskSkill = taskInstance.getSkillset(), taskDuration = taskInstance.getDescriptor().defaultInstance.getDuration(), workerInstance,
134     workQuality = 0, workPartsQuality = new Array(), workPartSkillsQuality = new Array(), sumWorkPartSkills = 0, averageWorkPartSkills = 0,
135     sumWorkParts = 0, workerSkillsetValue, totalExperience = 0, skillExperience, moral, confidence, learningCoefficient, randomNumber, existingSkills = new Array(),
136     punderationMoral = 0.1,
137     punderationConfidence = 0.2,
138     punderationSkillsets = 0.7,
139     punderationQualityWeight = 0.5
140 
141     //get existing skills
142     for(i=0; i<workersDescriptor[0].getInstance(self).getSkillset().size(); i++){
143         existingSkills.push(workersDescriptor[0].getInstance(self).getSkillset().keySet().toArray()[i]);
144     }
145 
146     for(i=0; i<workersDescriptor.length; i++){
147         workPartSkillsQuality.length = 0;
148         sumWorkPartSkills = 0;
149         workerInstance = workersDescriptor[i].getInstance(self);
150         moral = parseInt(workerInstance.getMoral());
151         confidence = parseInt(workerInstance.getConfidence());
152         learningCoefficient = parseInt(workerInstance.getProperty('learningCoefficient'));
153         totalExperience = 0;
154         from.push(workerInstance.getProperty('surname'));//for e-mail
155         for (j=0; j<listTaskSkill.size(); j++){
156             taskSkillKey = listTaskSkill.keySet().toArray()[j];
157             taskSkillValue = parseInt(listTaskSkill.get(taskSkillKey));
158             workerSkillsetValue = parseInt(workerInstance.getSkillset().get(taskSkillKey));
159             //calculate experience g�n�ral (totalExperienceGained) part 1/2
160             if(workerSkillsetValue<taskSkillValue){
161                 totalExperience += (taskSkillValue - workerSkillsetValue) * taskDuration / workersDescriptor.length;
162             }
163             //calculate experience in current skill
164             if(workerSkillsetValue<taskSkillValue){
165                 skillExperience = Math.ceil((1-workerSkillsetValue/100)*10*learningCoefficient*taskDuration/workersDescriptor.length);
166                 workerInstance.setSkillset(taskSkillKey, parseInt(workerInstance.getSkillset(taskSkillKey)) + skillExperience);
167             }
168             //calculate work Quality part 1/4
169             workPartSkillsQuality.push((workerSkillsetValue - taskSkillValue +100)/2);
170         }
171         //calculate experience g�n�ral (totalExperienceGained) part 2/2
172         workerInstance.setProperty('totalExperienceGained',  parseInt(workerInstance.getProperty('totalExperienceGained')) + totalExperience);
173         //calculate work Quality part 2/4
174         for(j=0; j<workPartSkillsQuality.length; j++){
175             sumWorkPartSkills += workPartSkillsQuality[j];
176         }
177         averageWorkPartSkills = (sumWorkPartSkills/workPartSkillsQuality.length) * punderationSkillsets + moral * punderationMoral + confidence * punderationConfidence
178         // check wish and hate
179         for (j=0; j<listTaskSkill.size(); j++){
180             taskSkillKey = listTaskSkill.keySet().toArray()[j];
181             taskSkillValue = parseInt(listTaskSkill.get(taskSkillKey));
182             workerSkillsetValue = parseInt(workerInstance.getSkillset().get(taskSkillKey));
183             //calculate work Quality part 3/4
184             if(taskSkillKey == workerInstance.getProperty('wish')){
185                 randomNumber = Math.floor(Math.random()*6)+5;
186                 averageWorkPartSkills += randomNumber;
187                 if(averageWorkPartSkills>100) averageWorkPartSkills = 100;
188             }
189             if(taskSkillKey == workerInstance.getProperty('hate')){
190                 randomNumber = Math.floor(Math.random()*6)+5;
191                 averageWorkPartSkills -= randomNumber;
192                 if(averageWorkPartSkills<0) averageWorkPartSkills = 0;
193             }
194             //check wish
195             randomNumber = Math.random();
196             wish = workerInstance.getProperty('wish');
197             if((averageWorkPartSkills <= 50 || randomNumber < 0.33) && taskSkillKey == wish){
198                 wish = '';
199             }
200             while(!wish){
201                 wish = existingSkills[Math.floor(Math.random() * existingSkills.length)];
202                 if(wish == workerInstance.getProperty('wish')) wish = '';
203             }
204             workerInstance.setProperty('wishIsKnow', false);
205             workerInstance.setProperty('wish', wish);
206             //check hate
207             randomNumber = Math.random();
208             hate = workerInstance.getProperty('hate');
209             if((averageWorkPartSkills > 50 && randomNumber < 0.33) && taskSkillKey == hate){
210                 hate = '';
211             }
212             while(!hate){
213                 hate = existingSkills[Math.floor(Math.random() * existingSkills.length)];
214                 if(hate == workerInstance.getProperty('hate')) hate = '';
215             }
216             workerInstance.setProperty('hateIsKnow', false);
217             workerInstance.setProperty('hate', hate);
218         }
219         averageWorkPartSkills = Math.round(averageWorkPartSkills);
220         workPartsQuality.push(averageWorkPartSkills);
221         workerInstance.setProperty('lastWorkQuality', averageWorkPartSkills);
222         //set moral and confidence depending of averageWorkPartSkills.
223         if(averageWorkPartSkills<50){
224             workerInstance.setMoral(workerInstance.getMoral()-25);
225             workerInstance.setConfidence(workerInstance.getConfidence()-15);
226         }
227         else{
228             workerInstance.setMoral(workerInstance.getMoral()+15);
229             workerInstance.setConfidence(workerInstance.getConfidence()+10);
230         }
231         //check 'workWithLeader'
232         if(taskInstance.getProperty('workWithLeader') == 'true'){
233             workerInstance.setProperty('numberOfWorkWithLeader', parseInt(workerInstance.getProperty('numberOfWorkWithLeader'))+1);
234         }
235     }
236     //calculate work Quality part 4/4
237     for(i=0; i<workPartsQuality.length; i++){
238         sumWorkParts += (workPartsQuality[i]);
239     }
240     workQuality = (sumWorkParts/workPartSkillsQuality.length);
241     clientsSatisfaction.setValue(Math.round(clientsSatisfaction.getValue()+(workQuality-50)*punderationQualityWeight));
242     if(clientsSatisfaction.getValue()>100) clientsSatisfaction.setValue(100);
243     if(clientsSatisfaction.getValue()<0) clientsSatisfaction.getValue(0);
244     //calculate remuneration
245     remuneration = parseInt(taskInstance.getProperty('salary'));
246     if(workQuality >= parseInt(taskInstance.getProperty('workQualityMinForBonus'))){
247         remuneration+=parseInt(taskInstance.getProperty('bonus'));
248     }
249     budgetInstance.setValue(budgetInstance.getValue()+remuneration);
250     //e-mail
251     content.push('Boujour, <br />Le mandat <<');
252     content.push(taskDescriptor.getName());
253     content.push('>> vient de se terminer. Le client ');
254     switch(true){
255         case workQuality<20 :
256             content.push('est totalement instatisfait par notre travail. Aucune chance de d�crocher un autre mandate aupr�s de ce client. ');
257             break;
258         case workQuality<40 :
259             content.push('a d�clar� �tre instatisfait de notre travail. Il est vrai que certaine erreurs ont �t� commises. ');
260             break;
261         case workQuality<60 :
262             content.push('est moyennement satisfait. La qualit� est faible mais le projet est satisfaisant. ');
263             break;
264         case workQuality<80 :
265             content.push('est content du travail r�alis�. Il pense nous recontacter pour de futurs mandats. ');
266             break;
267         default :
268             content.push('est ravi par le travail r�alis�. La qualit� est au-del� de ses esp�rences et sera fid�le � notre entreprise. ');
269             break;
270     }
271     content.push('<br />');
272     if( parseInt(taskInstance.getProperty('workQualityMinForBonus')) > 0 && workQuality >= parseInt(taskInstance.getProperty('workQualityMinForBonus') && parseInt(taskInstance.getProperty('bonus')) > 0)){
273         content.push('Il nous remercier par un bonus de ');
274         content.push(taskInstance.getProperty('bonus'));
275         content.push('.-');
276     }
277     content.push('<br /> Bonne journ�e. <br />');
278     content.push(from.join(', '));
279     this.sendMessage('Fin de mandat', content.join(''), workersDescriptor[0].getInstance(self).getProperty('surname'));
280     //desactivate Task
281     taskInstance.setActive(false);
282 }

284 /**
285  * Decrease absence duration by switch the current absenceTask with a other absenceTask wich have a duration 1 time smaller.
286  * If the duration must reach 0, remove this assignation, set the resource 'moral' to 40 and recalculate the teamMotivation's value.
287  * If duration is smaller than 0, deactivate the task.
288  */
289 function checkAbsencesEnd(){
290     var i, j, k, l, gm = self.getGameModel(), assignment, duration,
291     listAbsences = VariableDescriptorFacade.findByName(gm, 'absences'), absenceInstance,
292     listResources = VariableDescriptorFacade.findByName(gm, 'resources'), resourceInstance,
293     assignmentToRemove = new Array(), assignmentToAdd = new Array();
294     for(i=0; i<listResources.items.size();i++){
295         assignmentToRemove.length=0;
296         assignmentToAdd.length=0;
297         resourceInstance = listResources.items.get(i).getInstance(self);
298         for(j=0; j<resourceInstance.getAssignments().size();j++){
299             assignment = resourceInstance.getAssignments().get(j);
300             for(k=0; k<listAbsences.items.size();k++){
301                 absenceInstance = listAbsences.items.get(k).getInstance(self);
302                 if(assignment.getTaskDescriptorId() == absenceInstance.getDescriptorId() && absenceInstance.getActive() == true){
303                     duration = absenceInstance.getDuration();
304                     if(duration <= 0){
305                         absenceInstance.setActive(false);
306                     } else{
307                         assignmentToRemove.push(assignment);
308                         resourceInstance.setMoral(40);
309                         if (duration > 1){
310                             for(l=0; l<listAbsences.items.size();l++){
311                                 if(listAbsences.items.get(l).getInstance(self).getDuration() == duration-1){
312                                     assignmentToAdd.push(listAbsences.items.get(l));
313                                 }
314                             }
315                         }
316                     }
317                 }
318             }
319         }
320         for(j=assignmentToRemove.length-1;j>=0;j--){
321             resourceInstance.getAssignments().remove(assignmentToRemove[j]);
322         }
323         for(j=0;j<assignmentToAdd.length;j++){
324             resourceInstance.assign(0, assignmentToAdd[j]);
325         }
326         this.calculateTeamMotivation()
327     }
328 }

482 /**
483  * Calculate the current team motivation
484  * firste, do the average of all moral value of resources.
485  * then make an average between this value and the worst moral value.
486  * set the 'teamMotivation' value with this new value.
487  */
488 function calculateTeamMotivation(){
489     var i, sumMotivation = 0, gm = self.getGameModel(), activeResources = 0, moral,
490     listResources = VariableDescriptorFacade.findByName(gm, 'resources'), worstMoralValue = 100,
491     teamMotivation = VariableDescriptorFacade.findByName(gm, 'teamMotivation').getInstance(self);

493         if(listResources.items.get(i).getInstance(self).getActive() == true){
494             moral = parseInt(listResources.items.get(i).getInstance(self).getMoral())
495             sumMotivation += moral;
496             activeResources++;
497             if(moral<worstMoralValue) worstMoralValue = moral;
498         }
499     }
500     teamMotivation.value = Math.round(((Math.round(sumMotivation/activeResources)+worstMoralValue)/2));

504 /**
505  * Set the 'LeadershipLevel' value for each active resource.
506  */
507 function checkLeadershipLevel(){
508     var i, gm = self.getGameModel(), resourceInstance, newLeadershipLevel, leadershipPoints,
509     listResources = VariableDescriptorFacade.findByName(gm, 'resources'),
510     weekMaxValue = parseInt(VariableDescriptorFacade.findByName(gm, 'week').getMaxValue()),
511     pointsMinToLlvl2 = 10*(weekMaxValue-1) + 75,
512     pointsMinToLlvl3 = 15*(weekMaxValue-1) + 85;
513     for(i=0; i<listResources.items.size(); i++){
514         resourceInstance = listResources.items.get(i).getInstance(self);
515         if(resourceInstance.getActive() == true){
516             leadershipPoints = parseInt(resourceInstance.getConfidence()) + parseInt(resourceInstance.getProperty('totalExperienceGained'));
517             switch(true){

602 /**
603  * assign the given task at the given resource
604  * if the given resource have already a assigned task, this new assignement will replace the old one.
605  * @param Integer resourceDescriptorId, the id of resourceDescriptors to assign
606  * @param Integer taskDescriptorId, the id of task to assign
607  */
608 function assignTask(resourceDescriptorId, taskDescriptorId){
609     var i, j, resInstance, taskDescriptor, gm=self.getGameModel(),
610     listResources = VariableDescriptorFacade.findByName(gm, 'resources'),
611     listTasks = VariableDescriptorFacade.findByName(gm, 'tasks');
612     //Search resource
613     for(i=0; i<listResources.items.size(); i++){
614         if(resourceDescriptorId == listResources.items.get(i).getId()){
615             resInstance = listResources.items.get(i).getInstance(self);
616         }
617     }
618     if(resInstance == null) return;
619     //Search task
620     for(i=0; i<listTasks.items.size(); i++){
621         if(taskDescriptorId == listTasks.items.get(i).getId()){
622             taskDescriptor = listTasks.items.get(i);
623         }
624         //remove old previous assigned task
625         for(j=0; j<resInstance.getAssignments().size(); j++){
626             if(resInstance.getAssignments().get(j).getTaskDescriptorId() == listTasks.items.get(i).getId() && listTasks.items.get(i).getInstance(self).getActive() == true){
627                 resInstance.getAssignments().remove(j);
628             }
629         }
630     }
631     if(taskDescriptor == null) return;
632     // assign task to resource
633     resInstance.assign(0, taskDescriptor);
634 }
