 15 YUI.add('wegas-tooltip', function (Y) {
 16     "use strict";
 17 
 18     /**
 19      *  @class NewEntityAction
 20      *  @module Wegas
 21      *  @constructor
 22      */
 23     var TooltipPlg = Y.Base.create("wegas-tooltipplugin", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
 24         initializer: function () {
 25             var tt = Tooltip.getInstance();
 26             tt.addTriggerNode(this.get("host").get("boundingBox"),
 27                 this.get("content"));
 28         }
 29     }, {
 30         NS: "tooltip",
 31         NAME: "tooltip",
 32         ATTRS: {
 33             content: {
 34                 type: "string",
 35                 format: "html",
 36                 _inputex: {
 37                     label: "Tooltip content"
 38                 }
 39             }
 40         }
 41     });
 42 
 43     Y.namespace("Plugin").Tooltip = TooltipPlg;
 44 
 45     var Lang = Y.Lang,
 46     Node = Y.Node,
 47     OX = -10000,
 48     OY = -10000;
 49 
 50     var Tooltip = Y.Base.create("tooltip", Y.Widget, [Y.WidgetPosition, Y.WidgetStack, Y.WidgetPositionConstrain ], {
 51 
 52         // PROTOTYPE METHODS/PROPERTIES
 53 
 54         /*
 55          * Initialization Code: Sets up privately used state
 56          * properties, and publishes the events Tooltip introduces
 57          */
 58         initializer : function (config) {
 59 
 60             // Currently bound trigger node information
 61             this._currTrigger = {
 62                 node: null,
 63                 title: null,
 64                 mouseX: Tooltip.OFFSCREEN_X,
 65                 mouseY: Tooltip.OFFSCREEN_Y
 66             };
 67 
 68             // Event handles - mouse over is set on the delegate
 69             // element, mousemove and mouseleave are set on the trigger node
 70             this._eventHandles = {
 71                 delegate: null,
 72                 trigger: {
 73                     mouseMove : null,
 74                     mouseOut: null
 75                 }
 76             };
 77 
 78             // Show/hide timers
 79             this._timers = {
 80                 show: null,
 81                 hide: null
 82             };
 83 
 84             // Publish events introduced by Tooltip. Note the triggerEnter event is preventable,
 85             // with the default behavior defined in the _defTriggerEnterFn method
 86             this.publish("triggerEnter", {
 87                 defaultFn: this._defTriggerEnterFn,
 88                 preventable: true
 89             });
 90             this.publish("triggerLeave", {
 91                 preventable: false
 92             });
 93         },
 94 
 95         /*
 96          * Destruction Code: Clears event handles, timers,
 97          * and current trigger information
 98          */
 99         destructor : function () {
100             this._clearCurrentTrigger();
101             this._clearTimers();
102             this._clearHandles();
103         },
104 
105         /*
106          * bindUI is used to bind attribute change and dom event
107          * listeners
108          */
109         bindUI : function () {
110             this.after("delegateChange", this._afterSetDelegate);
111             this.after("nodesChange", this._afterSetNodes);
112 
113             this._bindDelegate();
114         },
115 
116         /*
117          * syncUI is used to update the rendered DOM, based on the current
118          * Tooltip state
119          */
120         syncUI : function () {
121             this._uiSetNodes(this.get("triggerNodes"));
122         },
123 
124         /*
125          * Public method, which can be used by triggerEvent event listeners
126          * to set the content of the tooltip for the current trigger node
127          */
128         setTriggerContent : function (content) {
129             var i, contentBox = this.get("contentBox");
130             contentBox.set("innerHTML", "");
131 
132             if (content) {
133                 if (content instanceof Node) {
134                     for (i = 0; i < content.size(); ++i) {
135                         contentBox.appendChild(content.item(i));
136                     }
137                 } else if (Lang.isString(content)) {
138                     contentBox.set("innerHTML", content);
139                     contentBox.all("img").once("load", function () {
140                         this.constrain();
141                     }, this);
142                 }
143             }
144         },
145 
146         addTriggerNode: function (node, content) {
147             this.get("content")[node.get("id")] = content;
148             this.get("triggerNodes").push(node);
149             this.syncUI();
150         },
151 
152         /*
153          * Default attribute change listener for
154          * the triggerNodes attribute
155          */
156         _afterSetNodes : function (e) {
157             this._uiSetNodes(e.newVal);
158         },
159 
160         /*
161          * Default attribute change listener for
162          * the delegate attribute
163          */
164         _afterSetDelegate : function (e) {
165             this._bindDelegate(e.newVal);
166         },
167 
168         /*
169          * Updates the rendered DOM to reflect the
170          * set of trigger nodes passed in
171          */
172         _uiSetNodes : function (nodes) {
173             if (this._triggerNodes) {
174                 this._triggerNodes.removeClass(this.getClassName("trigger"));
175             }
176 
177             if (nodes) {
178                 this._triggerNodes = nodes;
179                 this._triggerNodes.addClass(this.getClassName("trigger"));
180             }
181         },
182 
183         /*
184          * Attaches the default mouseover DOM listener to the
185          * current delegate node
186          */
187         _bindDelegate : function () {
188             var eventHandles = this._eventHandles;
189 
190             if (eventHandles.delegate) {
191                 eventHandles.delegate.detach();
192                 eventHandles.delegate = null;
193             }
194             eventHandles.delegate = Y.delegate("mouseenter", Y.bind(this._onNodeMouseEnter, this), this.get("delegate"), this.get("delegateSelect"));
195         },
196 
197         /*
198          * Default mouse enter DOM event listener.
199          *
200          * Delegates to the _enterTrigger method,
201          * if the mouseover enters a trigger node.
202          */
203         _onNodeMouseEnter : function (e) {
204             var node = e.currentTarget;
205             if (node && (!this._currTrigger.node || !node.compareTo(this._currTrigger.node))) {
206                 this._enterTrigger(node, e.pageX, e.pageY);
207             }
208         },
209 
210         /*
211          * Default mouse leave DOM event listener
212          *
213          * Delegates to _leaveTrigger if the mouse
214          * leaves the current trigger node
215          */
216         _onNodeMouseLeave : function (e) {
217             this._leaveTrigger(e.currentTarget);
218         },
219 
220         /*
221          * Default mouse move DOM event listener
222          */
223         _onNodeMouseMove: function (e) {
224             this._overTrigger(e.pageX, e.pageY);
225         },
226 
227         /*
228          * Default handler invoked when the mouse enters
229          * a trigger node. Fires the triggerEnter
230          * event which can be prevented by listeners to
231          * show the tooltip from being displayed.
232          */
233         _enterTrigger: function (node, x, y) {
234             this._clearCurrentTrigger();
235             this._setCurrentTrigger(node, x, y);
236             this.fire("triggerEnter", {
237                 node: node,
238                 pageX: x,
239                 pageY: y
240             });
241         },
242 
243         /*
244          * Default handler for the triggerEvent event,
245          * which will setup the timer to display the tooltip,
246          * if the default handler has not been prevented.
247          */
248         _defTriggerEnterFn: function (e) {
249             var delay, node = e.node;
250             if (!this.get("disabled")) {
251                 this._clearTimers();
252                 delay = (this.get("visible")) ? 0 : this.get("showDelay");
253                 this._timers.show = Y.later(delay, this, this._showTooltip, [node]);
254             }
255         },
256 
257         /*
258          * Default handler invoked when the mouse leaves
259          * the current trigger node. Fires the triggerLeave
260          * event and sets up the hide timer
261          */
262         _leaveTrigger: function (node) {
263             this.fire("triggerLeave");
264 
265             this._clearCurrentTrigger();
266             this._clearTimers();
267 
268             this._timers.hide = Y.later(this.get("hideDelay"), this, this._hideTooltip);
269         },
270 
271         /*
272          * Default handler invoked for mousemove events
273          * on the trigger node. Stores the current mouse
274          * x, y positions
275          */
276         _overTrigger: function (x, y) {
277             this._currTrigger.mouseX = x;
278             this._currTrigger.mouseY = y;
279         },
280 
281         /*
282          * Shows the tooltip, after moving it to the current mouse
283          * position.
284          */
285         _showTooltip: function (node) {
286             var x = this._currTrigger.mouseX, y = this._currTrigger.mouseY;
287 
288             this.move(x + Tooltip.OFFSET_X, y + Tooltip.OFFSET_Y);
289 
290             this.show();
291             this._clearTimers();
292 
293             this._timers.hide = Y.later(this.get("autoHideDelay"), this, this._hideTooltip);
294         },
295 
296         /*
297          * Hides the tooltip, after clearing existing timers.
298          */
299         _hideTooltip: function () {
300             this._clearTimers();
301             this.hide();
302         },
303 
304         /*
305          * Set the rendered content of the tooltip for the current
306          * trigger, based on (in order of precedence):
307          *
308          * a). The string/node content attribute value
309          * b). From the content lookup map if it is set, or
310          * c). From the title attribute if set.
311          */
312         _setTriggerContent: function (node) {
313             var content = this.get("content");
314             if (content && !(content instanceof Node || Lang.isString(content))) {
315                 content = content[node.get("id")] || unescape(node.getAttribute("title"));
316             }
317             this.setTriggerContent(content);
318         },
319 
320         /*
321          * Set the currently bound trigger node information, clearing
322          * out the title attribute if set and setting up mousemove/out
323          * listeners.
324          */
325         _setCurrentTrigger : function (node, x, y) {
326 
327             var title, currTrigger = this._currTrigger,
328             triggerHandles = this._eventHandles.trigger;
329 
330             this._setTriggerContent(node);
331 
332             triggerHandles.mouseMove = Y.on("mousemove", Y.bind(this._onNodeMouseMove, this), node);
333             triggerHandles.mouseOut = Y.on("mouseleave", Y.bind(this._onNodeMouseLeave, this), node);
334 
335             title = node.getAttribute("title");
336             node.setAttribute("title", "");
337 
338             currTrigger.mouseX = x;
339             currTrigger.mouseY = y;
340             currTrigger.node = node;
341             currTrigger.title = title;
342         },
343 
344         /*
345          * Clear out the current trigger state, restoring
346          * the title attribute on the trigger node,
347          * if it was originally set.
348          */
349         _clearCurrentTrigger: function () {
350 
351             var currTrigger = this._currTrigger,
352             triggerHandles = this._eventHandles.trigger;
353 
354             if (currTrigger.node) {
355                 var node = currTrigger.node,
356                 title = currTrigger.title || "";
357 
358                 currTrigger.node = null;
359                 currTrigger.title = "";
360 
361                 triggerHandles.mouseMove.detach();
362                 triggerHandles.mouseOut.detach();
363                 triggerHandles.mouseMove = null;
364                 triggerHandles.mouseOut = null;
365 
366                 node.setAttribute("title", title);
367             }
368         },
369 
370         /*
371          * Cancel any existing show/hide timers
372          */
373         _clearTimers: function () {
374             var timers = this._timers;
375             if (timers.hide) {
376                 timers.hide.cancel();
377                 timers.hide = null;
378             }
379             if (timers.show) {
380                 timers.show.cancel();
381                 timers.show = null;
382             }
383         },
384 
385         /*
386          * Detach any stored event handles
387          */
388         _clearHandles: function () {
389             var eventHandles = this._eventHandles;
390 
391             if (eventHandles.delegate) {
392                 this._eventHandles.delegate.detach();
393             }
394             if (eventHandles.trigger.mouseOut) {
395                 eventHandles.trigger.mouseOut.detach();
396             }
397             if (eventHandles.trigger.mouseMove) {
398                 eventHandles.trigger.mouseMove.detach();
399             }
400         }
401     }, {
402 
403         // STATIC METHODS/PROPERTIES
404 
405         OFFSET_X: 15,
406         OFFSET_Y: 15,
407         OFFSCREEN_X: OX,
408         OFFSCREEN_Y: OY,
409 
410         CSS_PREFIX: "wegas-tooltip",
411 
412         /**
413          * Retrieves a singleton of the tt instance.
414          */
415         getInstance: function () {
416             if (!Tooltip.tt) {
417                 Tooltip.tt = new Tooltip({
418                     triggerNodes: new Y.NodeList(),
419                     delegate: "body",
420                     content: {},
421                     render: true
422                 });
423             }
424             return Tooltip.tt;
425         },
426         ATTRS : {
427 
428             constrain: {
429                 value: true
430             },
431 
432             shim: {
433                 value: false
434             },
435 
436             zIndex: {
437                 value: 26
438             },
439             /*
440              * The tooltip content. This can either be a fixed content value,
441              * or a map of id-to-values, designed to be used when a single
442              * tooltip is mapped to multiple trigger elements.
443              */
444             content : {
445                 value: null
446             },
447 
448             /*
449              * The set of nodes to bind to the tooltip instance. Can be a string,
450              * or a node instance.
451              */
452             triggerNodes : {
453                 value: null,
454                 setter: function (val) {
455                     if (val && Lang.isString(val)) {
456                         val = Node.all(val);
457                     }
458                     return val;
459                 }
460             },
461 
462             /*
463              * The delegate node to which event listeners should be attached.
464              * This node should be an ancestor of all trigger nodes bound
465              * to the instance. By default the document is used.
466              */
467             delegate : {
468                 value: null,
469                 setter: function (val) {
470                     return Y.one(val) || Y.one("document");
471                 }
472             },
473 
474             /*
475              * The time to wait, after the mouse enters the trigger node,
476              * to display the tooltip
477              */
478             showDelay : {
479                 value: 250
480             },
481 
482             /*
483              * The time to wait, after the mouse leaves the trigger node,
484              * to hide the tooltip
485              */
486             hideDelay : {
487                 value: 10
488             },
489 
490             /*
491              * The time to wait, after the tooltip is first displayed for
492              * a trigger node, to hide it, if the mouse has not left the
493              * trigger node
494              */
495             autoHideDelay : {
496                 value: 2000
497             },
498 
499             /*
500              * Override the default visibility set by the widget base class
501              */
502             visible : {
503                 value: false
504             },
505 
506             /*
507              * Override the default XY value set by the widget base class,
508              * to position the tooltip offscreen
509              */
510             xy: {
511                 value: [OX, OY]
512             },
513             delegateSelect: {
514                 value: null,
515                 getter: function (val) {
516 
517                     return val || ("." + this.getClassName("trigger"));
518                 }
519             }
520         }
521     });
522 
523     Y.namespace("Wegas").Tooltip = Tooltip;
524 });
