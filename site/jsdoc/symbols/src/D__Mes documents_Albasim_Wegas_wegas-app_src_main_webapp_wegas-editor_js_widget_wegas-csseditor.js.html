  5 YUI.add('wegas-csseditor', function (Y) {
  6     var CONTENTBOX = 'contentBox',
  7     CSSEditor;
  8 
  9     CSSEditor = Y.Base.create("wegas-csseditor", Y.Widget, [Y.WidgetChild,  Y.Wegas.Widget], {
 10 
 11         // *** Lifecycle Methods *** //
 12 
 13         renderUI: function () {
 14             this.plug(Y.Plugin.WidgetToolbar);
 15 
 16             var form, cb = this.get(CONTENTBOX),
 17             value = Y.Plugin.CSSLoader.customCSSText || '',
 18             el = this.toolbar.get('header');
 19 
 20             this.form = new Y.inputEx.AceField({
 21                 parentEl: cb._node,
 22                 name: 'text',
 23                 type: 'ace',
 24                 height: "100%",
 25                 language: "css",
 26                 value: value
 27             });
 28 
 29             this.previewButton = new Y.Button({
 30                 label: "<span class=\"wegas-icon wegas-icon-preview\"></span>Preview",
 31                 on: {
 32                     click: Y.bind(function () {
 33                         Y.Plugin.CSSLoader.customCSSStyleSheet.disable();
 34                         Y.Plugin.CSSLoader.customCSSStyleSheet = new Y.StyleSheet(form.getValue());
 35                         // showFormMessage('success', 'CSS has been updated.');
 36                     }, this)
 37                 }
 38             }).render(el);
 39 
 40             this.saveButton = new Y.Button({
 41                 label: "<span class=\"wegas-icon wegas-icon-save\"></span>Save",
 42                 disabled: true
 43             }).render(el);
 44         },
 45         destructor: function () {
 46             this.previewButton.destroy();
 47             this.saveButton.destroy();
 48             this.form.destroy();
 49         }
 50     });
 51 
 52 
 53     Y.namespace('Wegas').CSSEditor = CSSEditor;
 54 });
