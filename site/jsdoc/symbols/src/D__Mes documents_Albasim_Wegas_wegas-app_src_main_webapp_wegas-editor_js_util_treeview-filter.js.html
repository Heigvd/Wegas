 12 YUI.add("treeview-filter", function(Y) {
 13     "use strict";
 14 
 15     var TreeViewFilter = Y.Base.create("treeview-filter", Y.Plugin.Base, [], {
 16         initializer: function() {
 17             if (!(this.get("host") instanceof Y.TreeView)) {
 18                 Y.log("TreeView filter host must be a TreeView", "warn", "TreeViewFilter");
 19                 return;
 20             }
 21             this.after("searchValChange", function(e) {
 22                 if (e.prevVal !== e.newVal) {
 23                     this.filter(this.get("host"), e.newVal);
 24                 }
 25             });
 26 
 27             this.afterHostEvent("*:addChild", function(e) {
 28                 this.filter(this.get("host"), this.get("searchVal"));
 29             });
 30         },
 31         filter: function(item, match) {
 32             var matches = false, i,
 33                     matcher;
 34             if (item instanceof Y.TreeView) {
 35                 matches = false;
 36             } else {
 37                 if (this.get("regExp")) {
 38                     try {
 39                         matcher = new RegExp(match);
 40                     } catch (e) {
 41                         return;
 42                     }
 43                     for (i in this.get("searchAttrs")) {
 44                         try {
 45                             matches = matches || matcher.test(item.get(this.get("searchAttrs")[i]));
 46                         } catch (e) {
 47                             //SearchAttrs fail
 48                         }
 49                     }
 50                 } else {
 51                     for (i in this.get("searchAttrs")) {
 52                         try {
 53                             matches = matches || (item.get(this.get("searchAttrs")[i]).toLowerCase().indexOf(match.toLowerCase()) > -1);
 54                         } catch (e) {
 55                             //SearchAttrs fail
 56                         }
 57                     }
 58                 }
 59 
 60             }
 61             if (item.each) {
 62                 item.each(function(node) {
 63                     if (this.filter(node, match) && item.expand) {
 64                         item.expand(false);
 65                     }
 66                 }, this);
 67             }
 68             if (matches || item.get("boundingBox").one(".filter-match")) {
 69                 item.get("boundingBox").addClass("filter-match");
 70                 item.get("boundingBox").removeClass("filter-no-match");
 71             } else {
 72                 item.get("boundingBox").addClass("filter-no-match");
 73                 item.get("boundingBox").removeClass("filter-match");
 74             }
 75             return matches;
 76 
 77         },
 78         destructor: function() {
 79 
 80         }
 81 
 82     }, {
 83         NAME: "TreeViewFilter",
 84         NS: "filter",
 85         ATTRS: {
 86             searchVal: {
 87                 value: ""
 88             },
 89             searchAttrs: {
 90                 value: ["label"]
 91             },
 92             regExp: {
 93                 value: false,
 94                 validator: Y.Lang.isBoolean
 95             }
 96         }
 97     });
 98 
 99     Y.namespace("Plugin").TreeViewFilter = TreeViewFilter;
100 });
