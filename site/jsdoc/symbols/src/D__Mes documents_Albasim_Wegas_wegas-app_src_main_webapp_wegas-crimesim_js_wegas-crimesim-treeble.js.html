 15 YUI.add('wegas-crimesim-treeble', function(Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', Treeble;
 19 
 20     Treeble = Y.Base.create("wegas-crimesim-treeble", Y.Widget, [], {
 21         // *** Fields *** /
 22         datatable: null,
 23         root: null,
 24         handlers: null,
 25         data: null,
 26         dataSource: null,
 27         descriptionColumn: null,
 28         // *** Lifecycle Methods *** //
 29         initializer: function() {
 30             this.handlers = {};
 31             this.data = [];
 32             this.descriptionColumn = this.get('descriptionColumn');
 33         },
 34         renderUI: function() {
 35             var columns = this.get('columns');
 36             if (this.get('isTreeble')) {
 37                 this.datatable = new Y.Treeble({
 38                     columns: columns
 39                 });
 40                 this.initTreeble(columns);
 41             } else {
 42                 this.datatable = new Y.DataTable({
 43                     columns: columns
 44                 });
 45             }
 46             this.datatable.render(this.get(CONTENTBOX));
 47         },
 48         bindUI: function() {
 49             if (this.dataSource) {
 50                 this.handlers.openRow = this.dataSource.on('toggled', this.mergeColumns, this);
 51             }
 52         },
 53         syncUI: function(data) {
 54             this.data = data || [];
 55             if (this.get('isTreeble')) {
 56                 Y.bind(this.treebleTwistdown(), this);
 57             } else {
 58                 this.datatable.addRows(this.data);
 59             }
 60         },
 61         destructor: function() {
 62             this.datatable.destroy();
 63             for (var i in this.handlers) {
 64                 this.handlers[i].detach();
 65             }
 66         },
 67         /*** private functions ***/
 68         initTreeble: function(columns) {
 69             var i, treeble_config, schema_plugin_config, schema, resultFields, dataSource;
 70             this.handlers = [];
 71 
 72             //Create and add datasource to Treeble Datatable
 73             //Add formatter to the "name column" (for indentation)
 74             for (i = 0; i < this.datatable.get('columns').length; i += 1) {
 75                 if (this.datatable.get('columns')[i].key === this.descriptionColumn) {
 76                     this.datatable.get('columns')[i].formatter = Y.Treeble.treeValueFormatter;
 77                     this.datatable.get('columns')[i].allowHTML = true;
 78                 }
 79             }
 80 
 81             //Add the column wich will contain the "open node"
 82             this.datatable.addColumn({
 83                 key: 'treeblenub',
 84                 label: ' ',
 85                 nodeFormatter: Y.Treeble.buildTwistdownFormatter(Y.bind(this.treebleTwistdown, this))
 86             }, 0, this);
 87 
 88             //Create the schema of the table's datas
 89             resultFields = columns.slice(0);
 90             resultFields.push('_open');
 91             resultFields.push({
 92                 key: 'kiddies',
 93                 parser: 'treebledatasource'
 94             });
 95             schema = {
 96                 resultFields: resultFields
 97             };
 98 
 99             //Create the schema plugin for the datasource's root
100             schema_plugin_config = {
101                 fn: Y.Plugin.DataSourceArraySchema,
102                 cfg: {
103                     schema: schema
104                 }
105             };
106 
107             //Create the config object for the datasource's root
108             treeble_config = {
109                 generateRequest: function() {
110                 },
111                 schemaPluginConfig: schema_plugin_config,
112                 childNodesKey: 'kiddies',
113                 nodeOpenKey: '_open',
114                 totalRecordsReturnExpr: '.meta.totalRecords'
115             };
116 
117             //Create the root object for the datasource
118             this.root = new Y.DataSource.Local({
119                 source: null                                                     //because dynamic change of datas
120             });
121             this.root.treeble_config = Y.clone(treeble_config, true);
122             this.root.plug(schema_plugin_config);
123 
124             //Create the datasource of the table
125             this.dataSource = new Y.DataSource.Treeble({
126                 root: this.root,
127                 paginateChildren: false,
128                 uniqueIdKey: 'startTime'
129             }, this);
130 
131             //plug datasource to the Treeble Datatable
132             this.datatable.plug(Y.Plugin.DataTableDataSource, {
133                 datasource: this.dataSource
134             }, this);
135         },
136         getTreebleDatas: function() {
137             var i, j, data = this.data.slice(0), reply,
138                     description, kiddies;
139             if (!data) {
140                 return null;
141             }
142             for (i = 0; i < data.length; i += 1) {
143                 reply = Y.Wegas.VariableDescriptorFacade.rest.findById(data[i].choiceDescriptorId);
144                 if (reply && reply.get('description')) {
145                     description = reply.get('description');
146                     kiddies = [{}];
147                     kiddies[0][this.descriptionColumn] = description;
148                     data[i].kiddies = kiddies;
149                 }
150             }
151             return data;
152         },
153         treebleTwistdown: function() {
154             this.root.set('source', this.getTreebleDatas());                    //Get currents datas and set datasource
155             this.datatable.datasource.load({//Request max 100 rows per trebble's level
156                 request: {
157                     startIndex: 0,
158                     resultCount: 100
159                 }
160             });
161         },
162         mergeColumns: function() {                                             //add a colspan to description to merge columns of description's row.
163             var colName = 'yui3-datatable-col-' + this.descriptionColumn,
164                     nbCols = this.get('columns').length,
165                     isAfterCol = false;
166             this.get(CONTENTBOX).all('tr').each(function(row, i) {
167                 if (row.get('className').indexOf('treeble-depth-') > -1) {           //is a treeble row
168                     if (row.get('className').indexOf('treeble-depth-0') <= -1) {     //and isn't a firste-level treeble row
169                         if (row.get('className').indexOf(colName) <= -1) {           //and if the column exist
170                             row.one('.' + colName).setAttribute('colspan', nbCols);  //add colspan value
171                             row.all('td').each(function(cell, i) {                  //hide overflowed columns
172                                 if (isAfterCol) {
173                                     cell.addClass('hidden');
174                                 }
175                                 if (cell.get('className').indexOf(colName) > -1) {
176                                     isAfterCol = true;
177                                 }
178                             });
179                         }
180                     }
181                 }
182             });
183         }
184     }, {
185     CSS_PREFIX: "wegas-crimesim-treeble",
186         ATTRS: {
187             columns: {
188                 validator: Y.Lang.isArray
189             },
190             data: {
191                 validator: Y.Lang.isArray
192             },
193             isTreeble: {
194                 value: false,
195                 validator: function(b) {
196                     return b || b === "true";
197                 }
198             },
199             descriptionColumn: {
200                 value: 'startTime',
201                 type: 'string'
202             }
203         }
204     });
205 
206     //Below : Hack because current verion of TreebleDataSource isn't on YUI (this is the worked version from Guithub).
207     Y.TreebleDataSource.prototype.toggle = function(path, request, completion) {
208         var searchOpen = function(
209                 /* array */list,
210                 /* int */nodeIndex) {
211             for (var i = 0; i < list.length; i++) {
212                 if (+list[i].index === +nodeIndex) {
213                     return list[i];
214                 }
215             }
216 
217             return false;
218         };
219 
220         function toggleSuccess(e, node, completion, path) {
221             if (node.ds.treeble_config.totalRecordsExpr) {
222                 eval('node.childTotal=e.response' + node.ds.treeble_config.totalRecordsExpr);
223             }
224             else if (node.ds.treeble_config.totalRecordsReturnExpr) {
225                 node.childTotal = e.response.results.length;
226             }
227 
228             node.open = true;
229             node.children = [];
230             complete(completion);
231 
232             this.fire('toggled',
233                     {
234                         path: path,
235                         open: node.open
236                     });
237         }
238         ;
239 
240         function toggleFailure(e, node, completion, path) {
241             node.childTotal = 0;
242 
243             node.open = true;
244             node.children = [];
245             complete(completion);
246 
247             this.fire('toggled',
248                     {
249                         path: path,
250                         open: node.open
251                     });
252         }
253 
254         function complete(f) {
255             if (Y.Lang.isFunction(f)) {
256                 f();
257             }
258             else if (f && f.fn) {
259                 f.fn.apply(f.scope || window, Y.Lang.isUndefined(f.args) ? [] : f.args);
260             }
261         }
262 
263         var list = this._open;
264         for (var i = 0; i < path.length; i++) {
265             var node = searchOpen.call(this, list, path[i]);
266             if (!node) {
267                 return false;
268             }
269             list = node.children;
270         }
271 
272         if (node.open === null) {
273             request.startIndex = 0;
274             request.resultCount = 0;
275             node.ds.sendRequest(
276                     {
277                         request: node.ds.treeble_config.generateRequest(request, path),
278                         cfg: node.ds.treeble_config.requestCfg,
279                         callback:
280                                 {
281                                     success: Y.rbind(toggleSuccess, this, node, completion, path),
282                                     failure: Y.rbind(toggleFailure, this, node, completion, path)
283                                 }
284                     });
285         } else {
286             node.open = !node.open;
287             complete(completion);
288 
289             this.fire('toggled',
290                     {
291                         path: path,
292                         open: node.open
293                     });
294         }
295         return true;
296     };
297 
298     Y.namespace('Wegas').CrimeSimTreeble = Treeble;
299 });
