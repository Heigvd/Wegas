 12 YUI.add( "wegas-injector", function ( Y ) {
 13     "use strict";
 14     var Injector,
 15     parser = function(element){
 16         var dico = {
 17             UserName : Y.Wegas.app.get("currentUser").name,
 18             PlayerName : Y.Wegas.GameFacade.rest.getCurrentPlayer().get("name"),
 19             TeamName : Y.Wegas.GameFacade.rest.getCurrentTeam().get("name"),
 20             ContentPath : Y.Wegas.app.get("dataSources").File.source + "read"
 21         },
 22         regExp = /\${([^}]*)}/g,
 23         html = element.getHTML();
 24         if(regExp.test(html)){
 25             element.setHTML(html.replace(regExp, function(gr0, gr1){
 26                 return dico[gr1];
 27             }));
 28         }
 29     };
 30 
 31     Injector = Y.Base.create("wegas-injector", Y.Base, [], {
 32 
 33         initializer: function(){
 34             this.eventHandler = Y.delegate("DOMSubtreeModified", function (e){
 35                 parser(e.currentTarget);
 36             }, this.get("observe"), "*");
 37         },
 38 
 39         destructor: function(){
 40             this.eventHandler.detach();
 41         }
 42         
 43     }, {
 44         ATTRS:{
 45             observe:{
 46                 value: ".body"
 47             }
 48         }
 49     });
 50 
 51     Y.namespace("Wegas").Injector = Injector;
 52 });
