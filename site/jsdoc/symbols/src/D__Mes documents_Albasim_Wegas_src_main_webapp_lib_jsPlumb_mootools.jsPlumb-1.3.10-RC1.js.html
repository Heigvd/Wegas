 30 	var jsPlumbMorph = new Class({
 31 		Extends:Fx.Morph,
 32 		onStep : null,
 33 		initialize : function(el, options) {
 34 			this.parent(el, options);
 35 			if (options['onStep']) {
 36 				this.onStep = options['onStep'];
 37 			}
 38 		},
 39 		step : function(now) {
 40 			this.parent(now);
 41 			if (this.onStep) { 
 42 				try { this.onStep(); } 
 43 				catch(e) { } 
 44 			}
 45 		}

 56 	var _executeDroppableOption = function(el, dr, event, originalEvent) {
 57 		if (dr) {
 58 			var id = dr.get("id");
 59 			if (id) {
 60 				var options = _droppableOptions[id];
 61 				if (options) {
 62 					if (options[event]) {
 63 						options[event](el, dr, originalEvent);
 64 					}
 65 				}
 66 			}
 67 		}
 68 	};

 70 	var _checkHover = function(el, entering) {
 71 		if (el) {
 72 			var id = el.get("id");
 73 			if (id) {
 74 				var options = _droppableOptions[id];
 75 				if (options) {
 76 					if (options['hoverClass']) {
 77 						if (entering) el.addClass(options['hoverClass']);
 78 						else el.removeClass(options['hoverClass']);
 79 					}
 80 				}
 81 			}
 82 		}
 83 	};

 91 		var l = list[_scope];
 92 		if (!l) {
 93 			l = [];
 94 			list[_scope] = l;
 95 		}

273 		initDraggable : function(el, options, isPlumbedComponent) {
274 			var id = jsPlumb.getId(el);
275 			var drag = _draggablesById[id];
276 			if (!drag) {
277 				var originalZIndex = 0,
278                     originalCursor = null,
279 				    dragZIndex = jsPlumb.Defaults.DragOptions.zIndex || 2000;
280                 
281 				options['onStart'] = jsPlumb.wrap(options['onStart'], function() {
282                     originalZIndex = this.element.getStyle('z-index');
283 					this.element.setStyle('z-index', dragZIndex);
284                     drag.originalZIndex = originalZIndex;
285 					if (jsPlumb.Defaults.DragOptions.cursor) {
286 						originalCursor = this.element.getStyle('cursor');
287 						this.element.setStyle('cursor', jsPlumb.Defaults.DragOptions.cursor);
288 					}
289 				});
290 				
291 				options['onComplete'] = jsPlumb.wrap(options['onComplete'], function() {
292 					this.element.setStyle('z-index', originalZIndex);
293 					if (originalCursor) {
294 						this.element.setStyle('cursor', originalCursor);
295 					}
296 
297                     _removeNonPermanentDroppables(drag);                                        
298 				});
299 				
300 				// DROPPABLES - only relevant if this is a plumbed component, ie. not just the result of the user making some DOM element
301                 // draggable.  this is the only library adapter that has to care about this parameter.
302 				var scope = "" + (options["scope"] || jsPlumb.Defaults.Scope),
303 				    filterFunc = function(entry) {
304 					    return entry.get("id") != el.get("id");
305 				    },
306 				    droppables = _droppables[scope] ? _droppables[scope].filter(filterFunc) : [];
307 
308                 if (isPlumbedComponent) {
309 
310 				    options['droppables'] = droppables;
311 				    options['onLeave'] = jsPlumb.wrap(options['onLeave'], function(el, dr) {
312 		    			if (dr) {
313 			    			_checkHover(dr, false);
314 				    		_executeDroppableOption(el, dr, 'onLeave');
315 					    }
316 				    });
317 				    options['onEnter'] = jsPlumb.wrap(options['onEnter'], function(el, dr) {
318 					    if (dr) {
319 						    _checkHover(dr, true);
320 						    _executeDroppableOption(el, dr, 'onEnter');
321 					    }
322 				    });
323 				    options['onDrop'] = function(el, dr, event) {
324 					    if (dr) {
325 						    _checkHover(dr, false);
326 						    _executeDroppableOption(el, dr, 'onDrop', event);
327 					    }
328 				    };
329                 }
330                 else
331                     options["droppables"] = [];
332 				
333 				drag = new Drag.Move(el, options);
334 				drag.scope = scope;
335                 drag.originalZIndex = originalZIndex;
336                 _add(_draggablesById, el.get("id"), drag);
337 				// again, only keep a record of this for scope stuff if this is a plumbed component (an endpoint)
338                 if (isPlumbedComponent) {
339 				    _add(_draggablesByScope, scope, drag);
340                 }
341 				// test for disabled.
342 				if (options.disabled) drag.detach();
343 			}
344 			return drag;
345 		},

347 		initDroppable : function(el, options, isPlumbedComponent, isPermanent) {
348 			var scope = options["scope"];
349             _add(_droppables, scope, el);
350 			var id = jsPlumb.getId(el);
351 
352             el.setAttribute("_isPermanentDroppable", isPermanent);  // we use this to clear out droppables on drag complete.
353 			_droppableOptions[id] = options;
354 			_droppableScopesById[id] = scope;
355 			var filterFunc = function(entry) { return entry.element != el; },
356 			    draggables = _draggablesByScope[scope] ? _draggablesByScope[scope].filter(filterFunc) : [];
357 			for (var i = 0; i < draggables.length; i++) {
358 				draggables[i].droppables.push(el);
359 			}
360 		},
