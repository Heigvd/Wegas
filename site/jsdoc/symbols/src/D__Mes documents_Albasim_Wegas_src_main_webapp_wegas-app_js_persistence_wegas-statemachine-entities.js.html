394     Y.Wegas.persistence.DialogueDescriptor = Y.Base.create("DialogueDescriptor", Y.Wegas.persistence.FSMDescriptor, [], {
395 
396         /**
397         * Triggers a Dialogue Transition programmatically
398         * @param {DialogueTransition} transition - the transition object to trigger.
399         * @param {Object} callbacks - {success:Function|String, failure:Function|String} - the callback functions to execute.
400         */
401         doTransition: function(transition, callbacks){
402             var request;
403             if(transition instanceof Y.Wegas.persistence.DialogueTransition){
404                 if(!this.get("id") || !transition.get("id")){
405                     console.warn("Transition and Dialogue not persisted");
406                     return false;
407                 }
408                 if(transition.get("triggerCondition") == null || transition.get("triggerCondition").isEmpty()){
409                     request = "/StateMachine/" + this.get("id")
410                     + "/Player/" + Y.Wegas.app.get("currentPlayer")
411                     + "/Do/" + transition.get("id");
412                     try{
413                         Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
414                             request: request,
415                             cfg: {
416                                 method: "GET",
417                                 headers:{
418                                     'Content-Type': 'application/json; charset=iso-8859-1',
419                                     'Managed-Mode':'true'
420                                 }
421                             },
422                             callback: callbacks
423                         });
424                     }catch(e){
425                         //TODO : that
426                         console.error("REST plugin failed");
427                     }
428                     return true;
429                 }else{
430                     console.warn("Transition Condition : false");
431                     return false;
432                 }
433             }else{
434                 return false;
435             }
436         }
437     }, {
438         ATTRS: {
439             "@class": {
440                 value: "DialogueDescriptor"
441             }
442         }

478     Y.Wegas.persistence.DialogueState = Y.Base.create("DialogueState", Y.Wegas.persistence.State, [], {
479         initializer: function(){
480             this.publish("actionsAvailable");
481         },
482         /*
483          * Listen to "DialogueState:actionsAvailable" to get the result
484          * event.actionsAvailable contains available transitions
485          */
486         getAvailableActions: function(){
487             var i, transitions = this.get("transitions"),
488             ctrlObj = {};
489             ctrlObj.availableActions = [];
490             ctrlObj.toEval = 0;
491             ctrlObj.evaluatedCount = 0;
492             for (i in transitions){
493                 if(transitions[i] instanceof Y.Wegas.persistence.DialogueTransition){
494                     if(!transitions[i].get("triggerCondition")){
495                         ctrlObj.availableActions.push(transitions[i]);
496                     }else{
497                         transitions[i].get("triggerCondition").once("Script:evaluated", function(e, o, obj){
498                             var ctrlObj = obj[0], transition = obj[1];
499                             ctrlObj.evaluatedCount +=1;
500                             if(o === true){
501                                 ctrlObj.availableActions.push(transition);
502                             }
503                             if(ctrlObj.toEval === ctrlObj.evaluatedCount){
504                                 this.fire("actionsAvailable", {
505                                     actionsAvailable:ctrlObj.availableActions
506                                 });
507                             }
508                         }, this, [ctrlObj, transitions[i]]);
509                         ctrlObj.toEval +=1;
510                         transitions[i].get("triggerCondition").localEval();
511 
512                     }
513                 }
514             }
515             if(ctrlObj.toEval === ctrlObj.evaluatedCount){
516                 this.fire("actionsAvailable", {
517                     actionsAvailable:ctrlObj.availableActions
518                 });
519             }
520         },
521 
522         /**
523         * Get an array of texts from the state's text, split by a token
524         * @param {String} The token to split by
525         */
526         getTexts: function ( token ) {
