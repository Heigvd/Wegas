 34     Y.extend(EntityAction, Action, {}, {
 35         NS: "entityaction",
 36         NAME: "EntityAction",
 37         ATTRS: {
 38             entity: {
 39                 getter: function (val) {
 40                     if (val === "currentGameModel") {
 41                         return Y.Wegas.GameModelFacade.rest.getCurrentGameModel();
 42                     }
 43                     return val;
 44                 }
 45             },
 46             dataSource: {
 47                 getter: function (val) {
 48                     if (Y.Lang.isString(val)) {
 49                         return Y.Wegas.app.dataSources[val];
 50                     }
 51                     return val;
 52                 }
 53             }
 54         }
 55     });
 56     Y.namespace("Plugin").EntityAction = EntityAction;

 60     /**
 61      *  @class EditEntityAction
 62      *  @module Wegas
 63      *  @constructor
 64      */
 65     var EditEntityAction = function () {
 66         EditEntityAction.superclass.constructor.apply(this, arguments);
 67     };
 68 
 69     Y.extend(EditEntityAction, EntityAction, {
 70         execute: function () {
 71             EditEntityAction.showUpdateForm(this.get("entity"), this.get("dataSource"));
 72         }
 73     }, {
 74         NS: "editentity",
 75         NAME: "EditEntityAction",
 76         /**
 77          *
 78          */
 79         tab: null,
 80 
 81         /**
 82          *
 83          */
 84         form: null,
 85 
 86         /**
 87          * Show edition form in the target div
 88          */
 89         showEditForm: function (entity, callback) {
 90             EditEntityAction.callback = callback;
 91             EditEntityAction.currentEntity = entity;
 92 
 93             if (!EditEntityAction.tab) {                                      // First make sure the edit tab exists
 94                 EditEntityAction.tab = Y.Wegas.TabView.createTab("Edit", '#rightTabView');
 95                 //EditEntityAction.tab = Y.Wegas.TabView.createTab("Edit", '#centerTabView');
 96                 EditEntityAction.form = new Y.Wegas.FormWidget();
 97 
 98                 EditEntityAction.form.on("submit", function (e) {
 99                     this.form.showOverlay();
100                     this.callback(e.value, this.currentEntity);
101                 }, EditEntityAction);
102 
103                 EditEntityAction.form.on("cancel", function () {
104                     EditEntityAction.tab.remove();
105                     EditEntityAction.tab.destroy();
106                     delete EditEntityAction.tab;
107 
108                 //Y.Wegas.app.widget.hidePosition("right");                   // Hide the right layout
109                 });
110                 EditEntityAction.tab.add(EditEntityAction.form);
111             }
112 
113             EditEntityAction.tab.set("selected", 2);
114             EditEntityAction.form.emptyMessage();
115             EditEntityAction.form.set("values",  entity.toObject());
116             EditEntityAction.form.set("cfg", entity.getFormCfg());
117         },
118 
119         /**
120          *
121          */
122         hideFormFields: function () {
123             EditEntityAction.form.destroyForm();
124         },
125 
126         /**
127          *
128          */
129         showEditFormOverlay: function () {
130             EditEntityAction.form.showOverlay();
131         },
132 
133         /**
134          *
135          */
136         hideEditFormOverlay: function () {
137             EditEntityAction.form.hideOverlay();
138         },
139 
140         /**
141          *
142          */
143         showFormMessage: function (level, msg, timeout) {
144             EditEntityAction.form.showMessage(level, msg);
145         },
146 
147         /**
148          *
149          */
150         showUpdateForm: function (entity, dataSource) {
151             EditEntityAction.showEditForm(entity, function (cfg) {           // Display the edit form
152                 // entity.setAttrs(cfg);
153                 dataSource.rest.put(cfg, {
154                     success: function () {
155                         EditEntityAction.showFormMessage("success", "Item has been updated");
156                         EditEntityAction.hideEditFormOverlay();
157                     },
158                     failure: function (e) {
159                         EditEntityAction.showFormMessage("error", e.response.message || "Error while update item");
160                         EditEntityAction.hideEditFormOverlay();
161                     }
162                 });
163             });
164         },
165 
166         showAddForm: function (entity, parentData, dataSource) {
167             EditEntityAction.showEditForm(entity, function (newVal) {
168                 dataSource.rest.post(newVal, (parentData) ? parentData.toObject() : parentData, {
169                     success: function (e) {
170                         EditEntityAction.hideEditFormOverlay();
171                         EditEntityAction.showUpdateForm(e.response.entity, dataSource);
172                         EditEntityAction.showFormMessage("success", "Item has been added");
173                     },
174                     failure: function (e) {
175                         EditEntityAction.hideEditFormOverlay();
176                         EditEntityAction.showFormMessage("error", e.response.results.message || "Error while adding item");
177                     }
178                 });
179             });
180         }
181     });
182 
183     Y.namespace("Plugin").EditEntityAction = EditEntityAction;

189      */
190     var NewEntityAction = function () {
191         NewEntityAction.superclass.constructor.apply(this, arguments);
192     };
193 
194     Y.extend(NewEntityAction, EntityAction, {
195         execute: function () {
196             Y.Wegas.Editable.useAndRevive({                                     // Load target class dependencies
197                 "@class": this.get("targetClass")
198             }, Y.bind(function (entity) {
199                 EditEntityAction.showAddForm(entity, null,this.get("dataSource"));           // and display the edition form
200             }, this));
201         }
202     }, {
203         NS: "NewEntityAction",
204         NAME: "NewEntityAction",
205         ATTRS: {
206             targetClass: { },
207             dataSource: {
208                 getter: function (val) {
209                     if (!val) {
210                         return Y.Wegas.VariableDescriptorFacade;
211                     }
212                     if (Y.Lang.isString(val)) {
213                         return Y.Wegas.app.dataSources[val];
214                     }
215                     return val;
216                 }
217             }
218         }
219     });
220     Y.namespace("Plugin").NewEntityAction = NewEntityAction;

224      *  @module Wegas
225      *  @constructor
226      */
227     var EditEntityArrayFieldAction = function () {
228         EditEntityArrayFieldAction.superclass.constructor.apply(this, arguments);
229     };
230 
231     Y.extend(EditEntityArrayFieldAction, EntityAction, {
232         execute: function () {
233             var entity = this.get("entity"),
234             dataSource = this.get("dataSource"),
235             parentEntity = this.get("parentEntity"),
236             newEntity, targetArray;
237 
238             switch (this.get("method")) {
239                 case "put":
240                     EditEntityAction.showEditForm(entity, function (newVal) {
241 
242                         entity.setAttrs(newVal);
243 
244                         dataSource.rest.put(parentEntity.toObject(), {
245                             success: function () {
246                                 EditEntityAction.hideEditFormOverlay();
247                                 EditEntityAction.showFormMessage("success", "Item has been updated");
248                             },
249                             failure: function (e) {
250                                 EditEntityAction.hideEditFormOverlay();
251                                 EditEntityAction.showFormMessage("error", e.response.message || "Error while update item");
252                             }
253                         });
254                     });
255                     break;
256 
257                 case "post":
258                     newEntity = Y.Wegas.Editable.revive({
259                         "@class": this.get("targetClass")
260                     });
261                     EditEntityAction.showEditForm(newEntity, Y.bind(function (newVal) {
262                         newEntity.setAttrs(newVal);
263                         entity.get(this.get("attributeKey")).push(newEntity);
264 
265                         dataSource.rest.put(entity.toObject(), {
266                             success: function () {
267                                 EditEntityAction.hideEditFormOverlay();
268                                 EditEntityAction.showFormMessage("success", "Item has been added");
269                                 EditEntityAction.hideFormFields();
270                             },
271                             failure: function (e) {
272                                 EditEntityAction.hideEditFormOverlay();
273                                 EditEntityAction.showFormMessage("error", e.response.message || "Error while update item");
274                             }
275                         });
276                     }, this));
277                     break;
278 
279                 case "delete":
280                     if (confirm("Are your sure your want to delete this item ?")) {
281                         targetArray = parentEntity.get(this.get("attributeKey"));
282                         Y.Array.find(targetArray, function (e, i, a) {
283                             if (e.get("id") === entity.get("id")) {
284                                 a.splice(i, 1);
285                                 return true;
286                             }
287                             return false;
288                         });
289                         dataSource.rest.put(parentEntity.toObject());
290                     } else {
291                         return;
292                     }
293                     break;
294             }
295         }
296     }, {
297         NS: "editentitarrayfieldaction",
298         NAME: "EditEntityArrayFieldAction",
299         ATTRS: {
300             /**
301              * Can be put, post or delete
302              */
303             method: {
304                 value: "put"
305             },
306             parentEntity: { },
307             targetClass: { },
308             attributeKey: { }
309         }
310     });
311 
312     Y.namespace("Plugin").EditEntityArrayFieldAction = EditEntityArrayFieldAction;
