1 YUI.add('wegas-menu', function (Y) {
  2     'use strict';
  3 
  4     var WegasMenu,
  5     CONTENT_BOX="contentBox",
  6     tooltipTrigger = "wegas-tooltip-trigger";
  7 
  8 
  9     WegasMenu = Y.Base.create("wegas-menu", Y.Widget, [],{
 10         BOUNDING_TEMPLATE: "<div></div>",
 11         CONTENT_TEMPLATE:null,
 12         nodeInstances: null,
 13         eventInstances: null,
 14         clickHandler: null,
 15 
 16         initializer: function () {
 17             this.nodeInstances = [];
 18             this.eventInstances = [];
 19             this.publish("itemClick", {
 20                 emitFacade: true,
 21                 bubbles: true
 22             });
 23         },
 24         renderUI: function () {
 25             this.buildMenu(this.get("items"), this.get(CONTENT_BOX));
 26 
 27         },
 28         bindUI: function () {
 29             this.clickHandler = this.get(CONTENT_BOX).delegate('click', function(e) {					// Listen for click events on the table
 30                 e.stopImmediatePropagation();
 31                 this.fire("itemClick", {
 32                     data:  e.currentTarget.item.data,
 33                     params: this.get("params")
 34                 });
 35             }, 'li', this);
 36             if(this.get("eventTarget")){
 37                 this.addTarget(this.get("eventTarget"));
 38             }
 39         },
 40 
 41         destructor: function () {
 42             this.clickHandler.detach();
 43             for(var n in this.nodeInstances){
 44                 this.nodeInstances[n].destroy();
 45             }
 46         },
 47         buildMenu: function(items, node){
 48             var listItem, item, content = Y.Node.create("<ul></ul>");
 49             for (var i in items){
 50                 item = items[i];
 51 
 52                 listItem = this.itemCreator(item);
 53                 if(item.items){
 54                     this.buildMenu(item.items, listItem);
 55                 }
 56                 content.append(listItem);
 57                 this.nodeInstances.push(listItem);
 58             }
 59             node.append(content);
 60         },
 61         itemCreator: function (item) {
 62             //TODO: Tooltip
 63             var node = Y.Node.create("<li><div>" + (item.cssClass ? "<span class='menu-icon "+ item.cssClass +"'></span>" : "") + "<span>"+(item.label ? item.label : "")+"</span></div></li>"),
 64             divNode = node.one("div");
 65             node.item = item;
 66             node.addClass(this.getClassName("itemlist", this.get("horizontal") ? "horizontal" : "vertical"));
 67             if(item.tooltip){
 68                 divNode.addClass(tooltipTrigger);
 69                 divNode.setAttribute("title", item.tooltip);
 70             }
 71             return node;
 72         }
 73     },{
 74         NAME:"wegas-menu",
 75         CSS_PREFIX: "wegas-menu",
 76         ATTRS:{
 77             eventTarget:{
 78                 value:null
 79             },
 80             items:{
 81                 validator: Y.Lang.isArray
 82             },
 83             horizontal:{
 84                 value:false
 85             },
 86             params:{                                                            // Given input params returned with the click event, a reference for instance
 87                 value: null
 88             }
 89         }
 90     });
 91 
 92     Y.namespace('Wegas').WegasMenu = WegasMenu;
 93 });
