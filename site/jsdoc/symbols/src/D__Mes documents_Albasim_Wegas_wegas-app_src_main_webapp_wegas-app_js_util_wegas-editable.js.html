 15 YUI.add('wegas-editable', function (Y) {
 16     "use strict";
 17 
 18     var Lang = Y.Lang;
 19 
 20     /**
 21      *  Add custom attributes to be used in ATTR param in static cfg.
 22      */
 23     Y.Base._ATTR_CFG.push("type", "properties", "_inputex", "optional", "format", "choices", "items", "enum", "default", "transient");
 24     Y.Base._ATTR_CFG_HASH = Y.Array.hash(Y.Base._ATTR_CFG);
 25 
 26      /**
 27      * Extension allowing edition on wegas entities.
 28      *
 29      * @class Y.Wegas.Editable
 30      */
 31     function Editable () {
 32     }
 33 
 34     Y.mix(Editable.prototype, {
 35         /** @lends Y.Wegas.Editable# */
 36         /**
 37          * Serialize to a json object. Method used <b>recursively</b> by JSON.stringify
 38          *
 39          * @method toJSON
 40          * @return {object}
 41          */
 42         toJSON: function () {
 43             var k, ret = this.getAttrs(),
 44             attrCfgs = this.getAttrCfgs();
 45 
 46             for (k in ret) {
 47                 if (attrCfgs[k] && attrCfgs[k]["transient"]) {                           // Remove any transient attribute
 48                     delete ret[k];
 49                 }
 50             }
 51             return ret;                                                         // Return a copy of this's fields.
 52         },
 53         /**
 54          * Create a new JSON Object from this entity, filtered out by mask
 55          *
 56          * @fixme This method will parse the field and the clone it
 57          *
 58          * @method toObject
 59          * @param {Array} mask or {String}* a list of params
 60          * @return {Object} a filtered out clone
 61          */
 62         toObject: function (mask) {
 63             var e = JSON.parse(JSON.stringify(this));
 64             mask = Lang.isArray(mask) ? mask : Array.prototype.slice.call(arguments);
 65             return mask.length > 0 ? Y.clone(e, true, function (value, key, output, input) {
 66                 if (mask.indexOf(key) !== -1) {
 67                     return false;
 68                 } else {
 69                     return true;
 70                 }
 71             }) : e;
 72 
 73         },
 74         /**
 75          * Create a new Object from this entity
 76          * may be used by revive
 77          * @method clone
 78          * @return {Object} a clone
 79          */
 80         clone: function () {
 81             return this.toObject(["id", "variableInstances"]);
 82         },
 83         /**
 84          * @param {Array} fieldsToIgnore (optional), don't create these inputs.
 85          * Returns the form configuration associated to this object, to be used a an inputex object.
 86          */
 87         getFormCfg: function (fieldsToIgnore) {
 88             var i, form, schemaMap, attrCfgs, builder;
 89             fieldsToIgnore = (fieldsToIgnore || []);
 90             // forms = Y.Wegas.app.get('editorForms'),                          // Select first server defined forms, based on the @class or the type attribute
 91             // form = forms[this.get('@class')] || forms[this.get("type")]
 92 
 93             form = form || this.constructor.EDITFORM;                          // And if no form is defined we check if there is a default one defined in the entity
 94 
 95             if (!form) {                                                        // If no edit form could be found, we generate one based on the ATTRS parameter.
 96                 attrCfgs = this.getAttrCfgs();
 97 
 98                 for (i in attrCfgs) {
 99                     attrCfgs[i]["default"] = attrCfgs[i].value;                 // Use the value as default (useful form json object serialization)
100 
101                     if (attrCfgs[i]["transient"] || fieldsToIgnore.indexOf(i) > -1) {
102                         delete attrCfgs[i];
103                     }
104                 }
105 
106                 schemaMap = {
107                     Entity: {
108                         properties: attrCfgs
109                     }
110                 };
111 
112                 builder = new Y.inputEx.JsonSchema.Builder({
113                     'schemaIdentifierMap': schemaMap,
114                     'defaultOptions': {
115                         'showMsg': true
116                     }
117                 });
118                 form = builder.schemaToInputEx(schemaMap.Entity);
119             }
120             return form || [];
121         }
122         ,
123         /**
124          * Returns the edition menu associated to this object, to be used a an inputex object.
125          */
126         getMenuCfg: function (data) {
127             var menus = Y.Wegas.app.get('editorMenus'),
128             //    staticMenus =
129             menu;
130 
131             if (menus) {
132                 menu = menus[ this.get('@class')] || menus[this.get("type")];   // Select first server defined forms, based on the @class or the type attribute
133             }
134             menu = menu || this.getStatic("EDITMENU")[0] || [];                 // And if no form is defined we return the default one defined in the entity
135 
136 
137             function mixMenuCfg (elts, data) {
138                 var i, j;
139                 for (i = 0; i < elts.length; i += 1) {
140                     Y.mix(elts[i], data, true);                                 // Attach self and the provided datasource to the menu items, to allow them to know which entity to update
141 
142                     if (elts[i].plugins) {
143                         for (j = 0; j < elts[i].plugins.length; j = j + 1) {
144                             elts[i].plugins[j].cfg = elts[i].plugins[j].cfg || {};
145                             Y.mix(elts[i].plugins[j].cfg, data, true);
146                             if (elts[i].plugins[j].cfg.children) {
147                                 mixMenuCfg(elts[i].plugins[j].cfg.children, data);
148                             }
149                         }
150                     }
151                 }
152 
153             }
154 
155             mixMenuCfg(menu, data);
156             return menu;
157         },
158         /**
159          * Returns the edition menu associated to this object, to be used a an wysiwyg editor.
160          */
161         getMethodCfgs: function (data) {
162             var menu = this.getStatic("METHODS")[0] || {};
163             return menu;
164         },
165         /**
166          *  Helper function that walks the class hierarchy and returns it's attributes
167          *  cfg (ATTRS), used in Y.Wegas.Entity.getFormCfg().
168          */
169         getAttrCfgs: function () {
170             return this._aggregateAttrs(this.getStatic("ATTRS"));
171         },
172         getStatic: function (key) {
173             var c = this.constructor, ret = [];
174 
175             while (c) {
176                 if (c[key]) {                                                  // Add to attributes
177                     ret[ret.length] = c[key];
178                 }
179                 c = c.superclass ? c.superclass.constructor : null;
180             }
181             return ret;
182         }
183     });
184     Y.mix(Editable, {
185         /**
186          * Load the modules from an Wegas widget definition
187          */
188         use: function (cfg, cb) {
189             var modules = Editable.getModulesFromDefinition(cfg);
190             modules.push(cb);
191             Y.use.apply(Y, modules);
192         },
193         /**
194          * Return recursively the inputex modules from their 'type' property using (modulesByType from loader.js)
195          */
196         getRawModulesFromDefinition: function (cfg) {
197             var i, props, type = cfg.type || cfg["@class"],
198             module = YUI_config.groups.wegas.modulesByType[type],
199             modules = [],
200             pushFn = function (field) {
201                 if (field) {
202                     modules = modules.concat(Editable.getModulesFromDefinition(field));
203                 }
204             };
205 
206             if (module) {
207                 modules.push(module);
208             }
209 
210             props = ["children", "entities", "items"];                          // Revive array attributes
211             for (i = 0; i < props.length; i += 1) {
212                 if (cfg[props[i]]) {                                            // Get definitions from children (for Y.WidgetParent widgets)
213                     Y.Array.each(cfg[props[i]], pushFn);
214                 }
215             }
216             if (cfg.plugins) {                                                  // Plugins must be revived in the proper way
217                 Y.Array.each(cfg.plugins, function (field) {
218                     field.cfg = field.cfg || {};
219                     field.cfg.type = field.fn;
220                     modules = modules.concat(Editable.getModulesFromDefinition(field.cfg));
221                 });
222             }
223 
224             props = ["left", "right", "center", "top", "bottom"];               // Revive  objects attributes
225             for (i = 0; i < props.length; i = i + 1) {
226                 if (cfg[props[i]]) {
227                     modules = modules.concat(Editable.getModulesFromDefinition(cfg[props[i]]));
228                 }
229             }
230 
231             return modules;
232         },
233         /**
234          * Return unique modules definitions
235          */
236         getModulesFromDefinition: function (cfg) {
237             var modules = Editable.getRawModulesFromDefinition(cfg);
238             return Y.Object.keys(Y.Array.hash(modules));
239         },
240         /**
241          *  This method takes a js object and recuresively instantiate it based on
242          *  on their @class attribute. Target class are found in namespace
243          *  Y.Wegas.persistence.
244          *
245          *  @methodOf Y.Wegas.Editable
246          *  @static
247          *  @augments o Object the object to revive
248          *  @return Y.Wegas.Widget the resulting entity
249          */
250         revive: function (data) {
251             var walk = function (o, key) {
252                 var k, v, value = o[key];
253                 if (value && typeof value === "object") {
254                     for (k in value) {
255                         if (value.hasOwnProperty(k)) {
256                             v = walk(value, k);
257                             if (!Lang.isUndefined(v)) {
258                                 value[k] = v;
259                             }
260                         }
261                     }
262                     if (!Lang.isUndefined(value["@class"]) || !Lang.isUndefined(value.type)) {
263                         return Editable.reviver(value);
264                     }
265                 }
266                 return value;                                                   // If no value was returned before, return raw original object
267             };
268 
269             //return typeof reviver === 'function' ? walk({'':data},'') : data;
270             return walk({
271                 '': data
272             }, '');
273         },
274         /**
275          *  Takes an js object and lookup the corresponding entity in the
276          *  Y.Wegas.persistence package, based on its @class or type attributes.
277          *
278          *  @methodOf Y.Wegas.Editable
279          *  @static
280          *  @augments o Object the object to revive
281          *  @return Y.Wegas.Widget the resulting entity
282          */
283         reviver: function (o) {
284             var classDef = Y.Wegas.persistence.DefaultEntity;
285 
286             if (o["@class"]) {
287                 classDef = Y.Wegas.persistence[o["@class"]] || Y.Wegas.persistence.DefaultEntity;
288 
289             } else if (o.type) {
290                 classDef = Y.Wegas.persistence[o.type] || Y.Wegas.persistence.WidgetEntity;
291 
292             }
293             return new classDef(o);
294         },
295         useAndRevive: function (cfg, cb) {
296             Editable.use(cfg, Y.bind(function (cb) {                            // Load target class dependencies
297                 cb(Editable.revive(this));
298             }, cfg, cb));
299         }
300     });
301     Y.namespace("Wegas").Editable = Editable;
302 
303 });
