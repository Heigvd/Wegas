  5 YUI.add('wegas-cep-folder', function (Y) {
  6     "use strict";
  7     var CONTENTBOX = 'contentBox', Folder;
  8     Folder = Y.Base.create("wegas-cep-folder", Y.Wegas.ItemSelector, [Y.Wegas.Widget], {
  9         handlers: null,
 10         initializer: function () {
 11             this.handlers = {};
 12         },
 13         bindUI: function () {
 14             Folder.superclass.bindUI.apply(this);
 15             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 16         },
 17         syncUI: function () {
 18             var i, cb = this.get(CONTENTBOX), nodeToRemove = [];
 19             Folder.superclass.syncUI.apply(this);
 20             
 21             //remove all "previous" node with "leadershiplevel < 1"
 22             cb.all('.nodeformatter-position').each(function (node) {
 23                 node.all('*').each(function (n) {
 24                     if (n.getAttribute('data-position') < 0 && n.get('className') === 'previous') {
 25                         nodeToRemove.push(n);
 26                     }
 27                 });
 28             });
 29             for (i = 0; i < nodeToRemove.length; i += 1) {
 30                 nodeToRemove[i].remove();
 31             }
 32         },
 33         destructor: function () {
 34             var k;
 35             for (k in this.handlers) {
 36                 this.handlers[k].detach();
 37             }
 38         }
 39     }, {
 40         ATTRS: {
 41         }
 42     });
 43     Y.namespace('Wegas').CEPFolder = Folder;
 44 });
