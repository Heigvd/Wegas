 15 YUI.add('wegas-action', function(Y) {
 16     "use strict";
 17 
 18     /**
 19      *  Extension that adds editable capacities to plugins
 20      */
 21     function Plugin() {
 22     }
 23 
 24     Y.mix(Plugin.prototype, {});
 25     Y.mix(Plugin, {
 26         ATTRS: {
 27             host: {
 28                 "transient": true
 29             },
 30             initialized: {
 31                 "transient": true
 32             },
 33             destroyed: {
 34                 "transient": true
 35             }
 36         }
 37     });
 38 
 39     Y.namespace("Wegas").Plugin = Plugin;
 40 
 41     var Action = Y.Base.create("wegas-actionplugin", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
 42         initializer: function() {
 43             this.onHostEvent(this.get("targetEvent"), function() {
 44                // this.setAttrs(this.get("host").get("data"));                  // Pass the action data from the host to the plug
 45                 this.execute();
 46             }, this);
 47         },
 48         execute: function() {
 49             Y.error("Y.Plugin.Action.execute() is abstract, should be overriddent");
 50         }
 51     }, {
 52         NS: "wegas",
 53         NAME: "Action",
 54         ATTRS: {
 55             targetEvent: {
 56                 value: "click"
 57             }
 58         }
 59     });
 60     Y.namespace("Plugin").Action = Action;
 61 
 62     /**
 63      *  @class OpenGameAction
 64      *  @module Wegas
 65      *  @constructor
 66      */
 67     var OpenUrlAction = function() {
 68         OpenUrlAction.superclass.constructor.apply(this, arguments);
 69     };
 70 
 71     Y.extend(OpenUrlAction, Action, {
 72         execute: function() {
 73             var targetUrl = Y.Wegas.app.get("base") + this.get("url");
 74 
 75             if (this.get("target") === "blank") {
 76                 window.open(targetUrl);
 77             } else {
 78                 window.location.href = targetUrl;
 79             }
 80         }
 81     }, {
 82         NS: "openurlaction",
 83         NAME: "OpenUrlAction",
 84         ATTRS: {
 85             url: {},
 86             /**
 87              * Can be "self" or "blank"
 88              */
 89             target: {
 90                 value: "blank"
 91             }
 92         }
 93     });
 94 
 95     Y.namespace("Plugin").OpenUrlAction = OpenUrlAction;
 96 
 97 
 98     /**
 99      *  @class OpenPageAction
100      *  @module Wegas
101      *  @constructor
102      */
103     var OpenPageAction = function() {
104         OpenPageAction.superclass.constructor.apply(this, arguments);
105     };
106 
107     Y.extend(OpenPageAction, Action, {
108         initializer: function() {
109             OpenPageAction.superclass.initializer.apply(this, arguments);
110             this.afterHostEvent("render", function() {
111                 var targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
112                 if (targetPageLoader.get("pageId") === this.get("subpageId")) {
113                     this.get("host").set("selected", 1);
114                 }
115             }, this);
116         },
117         execute: function() {
118             var targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
119             targetPageLoader.set("pageId", this.get("subpageId"));
120             this.get("host").set("selected", 1);
121         }
122     }, {
123         NS: "OpenPageAction",
124         NAME: "OpenPageAction",
125         ATTRS: {
126             subpageId: {
127                 type: "string",
128                 _inputex: {
129                     label: "Page to display",
130                     _type: "pageselect",
131                     required: false
132                 }
133             },
134             targetPageLoaderId: {
135                 type: "string",
136                 _inputex: {
137                     label: "Target page loader"
138                 }
139             }
140         }
141     });
142 
143     Y.namespace("Plugin").OpenPageAction = OpenPageAction;
144 
145     /**
146      *  @class ExecuteScriptAction
147      *  @module Wegas
148      *  @constructor
149      */
150     var ExecuteScriptAction = function() {
151         ExecuteScriptAction.superclass.constructor.apply(this, arguments);
152     };
153 
154     Y.extend(ExecuteScriptAction, Action, {
155         execute: function() {
156             var host = this.get("host"), overlayGuest, guest = host.get("root");
157             if (guest.showOverlay && guest.hideOverlay) {
158                 overlayGuest = guest;
159                 overlayGuest.showOverlay();
160             }
161 
162             Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
163                 request: "/Script/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
164                 cfg: {
165                     method: "POST",
166                     data: Y.JSON.stringify(this.get("onClick"))
167                 },
168                 on: {
169                     success: function(r) {
170                         if (overlayGuest) {
171                             overlayGuest.hideOverlay();
172                         }
173                     },
174                     failure: function(r) {
175                         if (overlayGuest) {
176                             overlayGuest.hideOverlay();
177                         }
178                         Y.bind(host.defaultExceptionHandler, host, r);
179                     }
180                 }
181             });
182         }
183     }, {
184         NS: "ExecuteScriptAction",
185         NAME: "ExecuteScriptAction",
186         ATTRS: {
187             onClick: {
188                 _inputex: {
189                     _type: "script"
190                 }
191             }
192         }
193     });
194 
195     Y.namespace("Plugin").ExecuteScriptAction = ExecuteScriptAction;
196 
197 
198     var PopupPlg = function() {
199         PopupPlg.superclass.constructor.apply(this, arguments);
200     };
201     Y.extend(PopupPlg, Y.Plugin.Base, {
202         initializer: function() {
203             this.afterHostEvent("render", function() {
204                 this.get("host").showMessage("info", this.get("content"));
205             });
206         }
207     }, {
208         NS: "PopupPlg",
209         NAME: "PopupPlg",
210         ATTRS: {
211             content: {
212                 type: "string",
213                 format: "text"
214             }
215         }
216     });
217     Y.namespace("Plugin").PopupPlg = PopupPlg;
218 
219 });
