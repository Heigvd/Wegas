 15 YUI.add('wegas-variabledisplay', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', VariableDisplay;
 19 
 20     VariableDisplay = Y.Base.create("wegas-variabledisplay", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 21 
 22         // ** Lifecycle Methods ** //
 23         initializer: function () {
 24             this.handlers = [];
 25         },
 26 
 27         destructor: function () {
 28             var i;
 29             for (i = 0; i < this.handlers.length; i += 1) {
 30                 this.handlers[i].detach();
 31             }
 32         },
 33 
 34         bindUI: function () {
 35             this.handlers.push(Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this));
 36             this.handlers.push(Y.Wegas.app.after('currentPlayerChange', this.syncUI, this));
 37         },
 38 
 39         syncUI: function () {
 40             var variableDescriptor = this.get("variable.evaluated");
 41 
 42             if (!variableDescriptor) {
 43                 return;
 44             }
 45 
 46             if (variableDescriptor instanceof Y.Wegas.persistence.ListDescriptor) {   // If the widget is a list,
 47                 variableDescriptor = variableDescriptor.get("currentItem");     // display it with the current list and the current element
 48             }
 49 
 50             this.get(CONTENTBOX).setHTML(this.genMarkup(variableDescriptor));   // Display the variable
 51 
 52         },
 53 
 54         genMarkup: function (variableDescriptor) {
 55             var acc, i, maxVal = variableDescriptor.get("maxValue"),
 56             minVal = variableDescriptor.get("minValue"),
 57             value = variableDescriptor.getInstance().get("value"),
 58             label = this.get("label") || variableDescriptor.getPublicLabel();
 59 
 60             switch (this.get('view')) {
 61                 case 'text':
 62                     return '<span class="wegas-label wegas-variabledisplay-text-label">' + label + '</span>'
 63                     + ' <span class="wegas-value wegas-variabledisplay-text-value">' + value + '</span>';
 64                     break;
 65 
 66                 case 'title':
 67                     return '<span class="wegas-label wegas-variabledisplay-text-label">' + label + '</span>'
 68                     + ' <span class="wegas-value wegas-variabledisplay-text-value">' + variableDescriptor.getPublicLabel() + '</span>';
 69                     break;
 70 
 71                 case 'box':
 72                     acc = [];
 73                     for (i = 0; i < value; i += 1) {
 74                         acc.push('<div class="wegas-variabledisplay-box-unit"></div>');
 75                     }
 76                     return '<div class="wegas-variabledisplay-box-label">' + label + '</div>'
 77                     + '<span class="wegas-variabledisplay-box-units">' + acc.join('') + '</span>'
 78                     + '<span class="wegas-variabledisplay-box-value">(' + value + '<span class="wegas-variabledisplay-box-valueMax">/' + maxVal + '</span>)</span>';
 79                     break;
 80 
 81                 case 'fraction':
 82                     return '<span class="wegas-variabledisplay-fraction-label">' + label + '</span>'
 83                     + '<span class="wegas-variabledisplay-fraction-minValue">' + minVal + '</span>'
 84                     + '<span class="wegas-variabledisplay-fraction-minSeparator"> / </span>'
 85                     + '<span class="wegas-variabledisplay-fraction-value">' + value +'</span>'
 86                     + '<span class="wegas-variabledisplay-fraction-maxSeparator"> / </span>'
 87                     + '<span class="wegas-variabledisplay-fraction-maxValue">' + maxVal + '</span>';
 88                     break;
 89 
 90                 case 'valuebox':
 91                     acc = [];
 92                     if (variableDescriptor) {
 93                         for (i = variableDescriptor.get("minValue"); i <= variableDescriptor.get("maxValue"); i += 1) {
 94                             acc.push('<div class="wegas-valuebox-unit '
 95                                 + ((i === value) ? "wegas-valuebox-selected" : "")
 96                                 + ((i < value) ? "wegas-valuebox-previous" : "")
 97                                 + ((i > value) ? "wegas-valuebox-next" : "")
 98                                 + '">' + i + '</div>');
 99                         }
100                     }
101                     return '<span class="wegas-label wegas-variabledisplay-valuebox-label">' + label + '</span>'
102                     + '<div class="wegas-value wegas-variabledisplay-valuebox-units">' + acc.join('') + "</div>";
103                     break;
104             }
105 
106         }
107     }, {
108         ATTRS : {
109             /**
110              * The target variable, returned either based on the variableName attribute,
111              * and if absent by evaluating the expr attribute.
112              */
113             variable: {
114                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
115                 _inputex: {
116                     _type: "variableselect",
117                     label: "variable"
118                 }
119             },
120             label : {
121                 type: "string",
122                 optional: true,
123                 validator: Y.Lang.isString
124             },
125             view: {
126                 type: "string",
127                 value: "text",
128                 choices: [{
129                     value: "text"
130                 }, {
131                     value: "box"
132                 }, {
133                     value: "valuebox"
134                 }, {
135                     value: "fraction"
136                 }]
137             }
138         }
139     });
140 
141     Y.namespace('Wegas').VariableDisplay = VariableDisplay;
142 });
