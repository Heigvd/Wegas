 15 YUI.add("wegas-inputex-wysiwygscript", function (Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx;
 19 
 20     inputEx.WysiwygScript = function (options) {
 21         inputEx.WysiwygScript.superclass.constructor.call(this, options);
 22     };
 23 
 24     Y.extend(inputEx.WysiwygScript, inputEx.Script, {
 25 
 26         /**
 27          *
 28          */
 29         destroy: function () {
 30             this.exprList.destroy();
 31             this.viewSrc.destroy();
 32             inputEx.WysiwygScript.superclass.destroy.call(this);
 33         },
 34         /**
 35          *
 36          */
 37         setOptions: function (options) {
 38             inputEx.WysiwygScript.superclass.setOptions.call(this, options);
 39             this.options.className = options.className || 'inputEx-Field inputEx-WysiwigScript';
 40             this.options.mode = options.mode || "wysiwyg";
 41             this.options.expects = options.expects || "expression";             // condition or expression
 42         },
 43 
 44         /**
 45          *
 46          */
 47         getValue: function () {
 48             if (this.options.mode === "wysiwyg") {
 49                 var ct = "";
 50                 if (this.exprList.getArray().length > 0) {
 51                     if (this.options.expects === "condition") {
 52                         ct = this.exprList.getArray().join(" && ");
 53                     } else {
 54                         ct = this.exprList.getArray().join(";\n") + ";";
 55                     }
 56                 }
 57                 return {
 58                     '@class': "Script",
 59                     language: "JavaScript",
 60                     content: ct
 61                 };
 62             } else {
 63                 return inputEx.WysiwygScript.superclass.getValue.apply(this, arguments);
 64             }
 65         },
 66 
 67 
 68         // *** Private Methods *** //
 69         /**
 70          *
 71          */
 72         renderComponent: function () {
 73             inputEx.Script.superclass.renderComponent.call(this);
 74 
 75             this.viewSrc = new Y.Wegas.Button({                                 // Add the "view src" button
 76                 label: "<span class=\"wegas-icon wegas-icon-viewsrc\"></span>"
 77             });
 78             this.viewSrc.after("click", function () {
 79                 if (!this.viewSrc.get("disabled")) {
 80                     if (this.options.mode === "wysiwyg") {                      // If current mode is wysiwyg
 81                         this.updateTextarea();                                  // update textatea content
 82                     } else {
 83                         this.updateExpressionList();
 84                     }
 85                     this.setMode((this.options.mode === "wysiwyg") ? "text" : "wysiwyg");
 86                 }
 87             }, this);
 88             this.viewSrc.render(this.fieldContainer);
 89 
 90             var container = new Y.Node(this.fieldContainer);                    // Render a div where the wysiwyg list will be rendered
 91             container.prepend(this.viewSrc.get("boundingBox"));
 92             container.append("<em class=\"msg\"></em>");
 93 
 94             this.on("updated", this.updateExpressionList, this);                // Whenever the value is updated, we synchronize the UI
 95 
 96             this.updateExpressionList();
 97             this.setMode(this.options.mode);
 98         },
 99 
100         /**
101          *
102          */
103         setMode: function (mode) {
104             var wysiwygmode = (mode === "wysiwyg");
105 
106             this.options.mode = mode;
107             this.viewSrc.set("selected", wysiwygmode ? 0 : 1);
108             this.wrapEl.style.display = (wysiwygmode) ? "none" : "block";
109 
110             if (wysiwygmode) {
111                 this.exprList.show();
112             } else {
113                 this.exprList.hide();
114             }
115 
116         },
117 
118         /**
119          *
120          */
121         updateTextarea: function () {
122             if (this.options.mode === "wysiwyg") {                              // If current mode is wysiwyg
123                 this.el.value =  this.getValue().content;                       // update textatea content
124             }
125         },
126 
127         updateExpressionList: function () {
128             var i, tree,
129             container = new Y.Node(this.fieldContainer),
130             fields = [];
131 
132             container.one(".msg").setContent("");                               // Reset layout
133 
134             try {
135                 tree = window.esprima.parse(this.el.value, {                    // Generate the syntaxic tree using esprima
136                     raw: true
137                 });
138 
139                 for (i = 0; i < tree.body.length; i = i + 1) {
140                     fields = fields.concat(this.generateExpression(tree.body[i].expression));
141                 }
142             } catch (e) {
143                 //Y.error("Error evaluating line: " + window.escodegen.generate(tree.body[i].expression, {indent: true}));
144                 this.setMode("text");
145                 this.viewSrc.set("disabled", true);
146                 container.one(".msg").setContent("Unable to read this impact, displaying source only.");
147                 return;
148             }
149 
150             this.viewSrc.set("disabled", false);
151             if (this.exprList) {
152                 this.exprList.destroy();
153             }
154             if (this.options.expects === "condition") {
155                 for (i = 0; i < fields.length; i += 1) {
156                     fields[i].type = "variabledescriptorcondition";
157                 }
158             }
159             this.exprList = Y.inputEx({                                         // Render the expression as a Y.inputEx.Wegas.ListField
160                 type: "listfield",
161                 fields: fields,
162                 useButtons: true,
163                 parentEl: this.fieldContainer,
164                 addType: (this.options.expects === "condition") ? "variabledescriptorcondition" : "variabledescriptorsetter"
165             });
166 
167             if (this.options.mode !== "wysiwyg") {
168                 this.exprList.hide();
169             }
170         },
171 
172         /**
173          *
174          */
175         generateExpression: function (expression) {
176             Y.log("generateExpression(" + expression.type + ")");
177             switch (expression.type) {
178 
179                 case "Identifier":
180                     return expression.name;
181 
182                 case "Literal":
183                     return expression.value;
184                 //return unesacapeJSString(expression.raw);
185 
186                 case "UnaryExpression":
187                     return expression.operator + this.generateExpression(expression.argument);
188 
189                 case "ArrayExpression":
190                     var args = [];
191                     Y.Array.each(expression.elements, function(i) {
192                         args.push(this.generateExpression(i));
193                     }, this);
194                     return args;
195 
196                 case "BinaryExpression":
197                     var vdSelect = this.generateExpression(expression.left)[0], args = [];
198                     vdSelect.type = "variabledescriptorcondition";
199                     vdSelect.operator = expression.operator;
200                     vdSelect.rightValue = this.generateExpression(expression.right);
201                     return [vdSelect];
202 
203                 case "LogicalExpression":
204                     //return [{
205                     //    type: "inputlist",
206                     //    fields: this.generateExpression(expression.left).concat(this.generateExpression(expression.right),
207                     //    useButtons: true,
208                     //    addType: "variabledescriptorcondition"
209                     //}]
210                     return this.generateExpression(expression.left).concat(this.generateExpression(expression.right));
211 
212                 case "CallExpression":
213                     switch (expression.callee.object.type) {
214                         case "Identifier":
215                             switch (expression.callee.object.name) {
216                                 case "VariableDescriptorFacade":
217                                     return {
218                                         type: "variabledescriptorsetter",
219                                         value: expression.arguments[0].value
220                                     };
221                             }
222                             break;
223                         default:
224                             //return new MethodSelect({
225                             //    object: this.generateExpression(expression.callee.object),
226                             //    name: expression.callee.property.value,
227                             //    arguments: expression.callee.arguments
228                             var vdSelect = this.generateExpression(expression.callee.object), args = [];
229 
230                             Y.Array.each(expression.arguments, function (i) {
231                                 args.push(this.generateExpression(i));
232                             }, this);
233                             Y.mix(vdSelect, {
234                                 //type: "variabledescriptormethodselect",
235                                 //object: this.generateExpression(expression.callee.object),
236                                 //fields: [],
237                                 method: expression.callee.property.name,
238                                 arguments: args
239 
240                             });
241                             return [vdSelect];
242                     }
243                     break;
244             }
245             throw new Error("Unable to parse expression.");
246         }
247     });
248 
249     inputEx.registerType('script', inputEx.WysiwygScript);                      // Register this class as "script" type
250 });
