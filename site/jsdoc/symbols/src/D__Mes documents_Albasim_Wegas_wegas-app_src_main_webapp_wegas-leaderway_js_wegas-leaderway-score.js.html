  5 YUI.add('wegas-leaderway-score', function (Y) {
  6     "use strict";
  7 
  8     var CONTENTBOX = 'contentBox', Score;
  9 
 10     Score = Y.Base.create("wegas-score", Y.Widget, [Y.Wegas.Widget], {
 11         // *** Fields *** /
 12         table: null,
 13         data: null,
 14         // *** Lifecycle Methods *** //
 15         initializer: function () {
 16             this.data = [];
 17         },
 18         /**
 19          * Render the widget.
 20          * Create the child widget "table"
 21          */
 22         renderUI: function () {
 23             var cb = this.get(CONTENTBOX);
 24             this.table = new Y.DataTable({
 25                 columns: [
 26                     {
 27                         key: "number",
 28                         label: "#"
 29                     },
 30                     {
 31                         key: "team",
 32                         label: "Entreprises"
 33                     },
 34                     {
 35                         key: "score",
 36                         label: "Score"
 37                     }
 38                 ]
 39             });
 40             cb.setContent(
 41                     '<div class="scoreTitle">' + this.get('title') + '</div>\n\
 42                 <div class="datatable"></div>');
 43             this.table.render(cb.one(".datatable"));
 44         },
 45         /**
 46          * Synchronise the content of this widget.
 47          */
 48         syncUI: function () {
 49             this.data.length = 0;
 50             this.getTeamScore(this.get('maxRows'));
 51             this.table.addRows(this.data);
 52             if (!this.data[0]) {
 53                 this.table.showMessage("Aucun score n'est disponible.");
 54             } else {
 55                 this.table.hideMessage();
 56             }
 57             this.goToFinalPage();// ! hack function
 58         },
 59         /*
 60          * Destroy all child widget
 61          */
 62         destructor: function () {
 63             this.table.destroy();
 64         },
 65         //*** Particular Methods ***/
 66         /**
 67          * Add rows to the datatable. Create the hall of fame from team of all time.
 68          * @param Integer rows, number of wanted rows.
 69          */
 70         getTeamScore: function (maxRows) {
 71             var i, j, k, allScore, team = [], score = [], sortedScore = [], sortedTeam = [], exist = false;
 72             allScore = Y.Wegas.VariableDescriptorFacade.rest.find("name", "score").get("scope").get("variableInstances");
 73             for (i in allScore) {
 74                 team.push(Y.Wegas.GameFacade.rest.findById(i).get('name'));
 75                 score.push(allScore[i].get('value'));
 76             }
 77             if (score.length > 0) {
 78                 sortedScore = score.slice(0);
 79                 sortedScore.sort();
 80                 sortedScore.reverse();
 81                 for (i = 0; i < sortedScore.length; i++) {
 82                     for (j = 0; j < score.length; j++) {
 83                         if (sortedScore[i] === score[j]) {
 84                             exist = false;
 85                             for (k = 0; k < sortedTeam.length; k++) {
 86                                 if (sortedTeam[k] === team[j]) {
 87                                     exist = true;
 88                                     break;
 89                                 }
 90                             }
 91                             if (!exist)
 92                                 sortedTeam.push(team[j]);
 93                         }
 94                     }
 95                 }
 96                 if (score.length <= maxRows) {
 97                     maxRows = score.length;
 98                 }
 99                 for (i = 0; i < maxRows; i++) {
100                     this.data.push({
101                         number: i + 1,
102                         team: sortedTeam[i],
103                         score: sortedScore[i]
104                     });
105                 }
106             }
107         },
108         // *** hack Methods *** //
109         /**
110          * if current week > max value of week value, then
111          * change the current widget to go on the "dialogue" widget.
112          */
113         goToFinalPage: function () {
114             var currentWeek = Y.Wegas.VariableDescriptorFacade.rest.find("name", "week"),
115                     targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
116             if (parseInt(currentWeek.getInstance().get('value')) > currentWeek.get('maxValue')) {
117                 targetPageLoader.once("widgetChange", function (e) {
118                     e.newVal.setCurrentDialogue("dialogueFinal");
119                 });
120                 targetPageLoader.set("pageId", this.get('dialoguePageId'));
121             }
122         }
123     }, {
124         ATTRS: {
125             title: {
126                 value: 'Top 5 des meilleurs entreprises.',
127                 validator: function (s) {
128                     return s === null || Y.Lang.isString(s);
129                 }
130             },
131             maxRows: {
132                 value: 5,
133                 validator: function (i) {
134                     return i === null || Y.Lang.isNumber(i);
135                 }
136             },
137             dialoguePageId: {
138                 value: null,
139                 validator: function (s) {
140                     return s === null || Y.Lang.isString(s);
141                 }
142             },
143             targetPageLoaderId: {
144                 value: null,
145                 validator: function (s) {
146                     return s === null || Y.Lang.isString(s);
147                 }
148             }
149         }
150     });
151 
152     Y.namespace('Wegas').Score = Score;
153 });
