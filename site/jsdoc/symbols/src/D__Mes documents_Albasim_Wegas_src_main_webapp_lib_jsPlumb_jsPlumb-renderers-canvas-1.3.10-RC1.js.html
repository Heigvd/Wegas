 34 	 * Provides support for mouse events on canvases.

 36 	var CanvasMouseAdapter = function() {
 37 		var self = this;
 38 		self.overlayPlacements = [];
 39 		jsPlumb.jsPlumbUIComponent.apply(this, arguments);
 40 		jsPlumbUtil.EventGenerator.apply(this, arguments);
 41 		/**
 42 		 * returns whether or not the given event is ojver a painted area of the canvas. 
 43 		 */
 44 	    this._over = function(e) {		    			  		    	
 45 	    	var o = _getOffset(_getElementObject(self.canvas)),
 46 	    	pageXY = _pageXY(e),
 47 	    	x = pageXY[0] - o.left, y = pageXY[1] - o.top;
 48 	    	if (x > 0 && y > 0 && x < self.canvas.width && y < self.canvas.height) {
 49 		    	// first check overlays
 50 		    	for ( var i = 0; i < self.overlayPlacements.length; i++) {
 51 		    		var p = self.overlayPlacements[i];
 52 		    		if (p && (p[0] <= x && p[1] >= x && p[2] <= y && p[3] >= y))
 53 		    			return true;
 54 		    	}
 55 		    	
 56 		    	// then the canvas
 57 		    	var d = self.canvas.getContext("2d").getImageData(parseInt(x), parseInt(y), 1, 1);
 58 		    	return d.data[0] != 0 || d.data[1] != 0 || d.data[2] != 0 || d.data[3] != 0;		  
 59 	    	}
 60 	    	return false;
 61 	    };
 62 	    
 63 	    var _mouseover = false, _mouseDown = false, _posWhenMouseDown = null, _mouseWasDown = false,
 64 	    _nullSafeHasClass = function(el, clazz) {
 65 	    	return el != null && _hasClass(el, clazz);
 66 	    };
 67 	    this.mousemove = function(e) {		    
 68 	    	var pageXY = _pageXY(e), clientXY = _clientXY(e),	   
 69 	    	ee = document.elementFromPoint(clientXY[0], clientXY[1]),
 70 	    	eventSourceWasOverlay = _nullSafeHasClass(ee, "_jsPlumb_overlay");	    	
 71 			var _continue = _connectionBeingDragged == null && (_nullSafeHasClass(ee, "_jsPlumb_endpoint") || _nullSafeHasClass(ee, "_jsPlumb_connector"));
 72 			if (!_mouseover && _continue && self._over(e)) {
 73 				_mouseover = true;
 74 				self.fire("mouseenter", self, e);		
 75 				return true;
 76 			}
 77 			// TODO here there is a remote chance that the overlay the mouse moved onto
 78 			// is actually not an overlay for the current component. a more thorough check would
 79 			// be to ensure the overlay belonged to the current component.  
 80 			else if (_mouseover && (!self._over(e) || !_continue) && !eventSourceWasOverlay) {
 81 				_mouseover = false;
 82 				self.fire("mouseexit", self, e);				
 83 			}
 84 			self.fire("mousemove", self, e);
 85 	    };
 86 	    		    		    
 87 	    this.click = function(e) {	    		
 88 	    	if (_mouseover && self._over(e) && !_mouseWasDown) 
 89 	    		self.fire("click", self, e);		    	
 90 	    	_mouseWasDown = false;
 91 	    };
 92 	    
 93 	    this.dblclick = function(e) {
 94 	    	if (_mouseover && self._over(e) && !_mouseWasDown) 
 95 	    		self.fire("dblclick", self, e);		    	
 96 	    	_mouseWasDown = false;
 97 	    };
 98 	    
 99 	    this.mousedown = function(e) {
100 	    	if(self._over(e) && !_mouseDown) {
101 	    		_mouseDown = true;	    		
102 	    		_posWhenMouseDown = _getOffset(_getElementObject(self.canvas));
103 	    		self.fire("mousedown", self, e);
104 	    	}
105 	    };
106 	    
107 	    this.mouseup = function(e) {
108 	    	_mouseDown = false;
109 	    	self.fire("mouseup", self, e);
110 	    };
111 
112         this.contextmenu = function(e) {
113           if (_mouseover && self._over(e) && !_mouseWasDown)
114             self.fire("contextmenu", self, e);
115           _mouseWasDown = false;
116         };
117 	};

119 	var _newCanvas = function(params) {
120 		var canvas = document.createElement("canvas");
121 		params["_jsPlumb"].appendElement(canvas, params.parent);
122 		canvas.style.position = "absolute";
123 		if (params["class"]) canvas.className = params["class"];
124 		// set an id. if no id on the element and if uuid was supplied it
125 		// will be used, otherwise we'll create one.
126 		params["_jsPlumb"].getId(canvas, params.uuid);
127 		if (params.tooltip) canvas.setAttribute("title", params.tooltip);
128 
129 		return canvas;
130 	};

148 		var _paintOneStyle = function(dim, aStyle) {
149 			self.ctx.save();
150 			jsPlumb.extend(self.ctx, aStyle);
151 			if (aStyle.gradient) {
152 				var g = self.createGradient(dim, self.ctx);
153 				for ( var i = 0; i < aStyle.gradient.stops.length; i++)
154 					g.addColorStop(aStyle.gradient.stops[i][0], aStyle.gradient.stops[i][1]);
155 				self.ctx.strokeStyle = g;
156 			}
157 			self._paint(dim, aStyle);
158 			self.ctx.restore();
159 		};
160 
161 		var self = this,
162 		clazz = self._jsPlumb.connectorClass + " " + (params.cssClass || "");
163 		self.canvas = _newCanvas({ 
164 			"class":clazz, 
165 			_jsPlumb:self._jsPlumb,
166 			parent:params.parent,
167 			tooltip:params.tooltip
168 		});	
169 		self.ctx = self.canvas.getContext("2d");
170 		
171 		self.appendDisplayElement(self.canvas);
172 		
173 		self.paint = function(dim, style) {						
174 			if (style != null) {																				
175 				jsPlumb.sizeCanvas(self.canvas, dim[0], dim[1], dim[2], dim[3]);				
176 				if (style.outlineColor != null) {
177 					var outlineWidth = style.outlineWidth || 1,
178 					outlineStrokeWidth = style.lineWidth + (2 * outlineWidth),
179 					outlineStyle = {
180 						strokeStyle:style.outlineColor,
181 						lineWidth:outlineStrokeWidth
182 					};
183 					_paintOneStyle(dim, outlineStyle);
184 				}
185 				_paintOneStyle(dim, style);
186 			}
187 		};

224 	jsPlumb.Endpoints.canvas.Dot = function(params) {		
225 		jsPlumb.Endpoints.Dot.apply(this, arguments);
226 		CanvasEndpoint.apply(this, arguments);
227 		var self = this,		
228 		parseValue = function(value) {
229 			try { return parseInt(value); }
230 			catch(e) {
231 				if (value.substring(value.length - 1) == '%')
232 					return parseInt(value.substring(0, value - 1));
233 			}
234 		},					    	
235 		calculateAdjustments = function(gradient) {
236 			var offsetAdjustment = self.defaultOffset, innerRadius = self.defaultInnerRadius;
237 			gradient.offset && (offsetAdjustment = parseValue(gradient.offset));
238         	gradient.innerRadius && (innerRadius = parseValue(gradient.innerRadius));
239         	return [offsetAdjustment, innerRadius];
240 		};
241 		this._paint = function(d, style, anchor) {
242 			if (style != null) {			
243 				var ctx = self.canvas.getContext('2d'), orientation = anchor.getOrientation(self);
244 				jsPlumb.extend(ctx, style);							
245 	            if (style.gradient) {            	
246 	            	var adjustments = calculateAdjustments(style.gradient), 
247 	            	yAdjust = orientation[1] == 1 ? adjustments[0] * -1 : adjustments[0],
248 	            	xAdjust = orientation[0] == 1 ? adjustments[0] * -1:  adjustments[0],
249 	            	g = ctx.createRadialGradient(d[4], d[4], d[4], d[4] + xAdjust, d[4] + yAdjust, adjustments[1]);
250 		            for (var i = 0; i < style.gradient.stops.length; i++)
251 		            	g.addColorStop(style.gradient.stops[i][0], style.gradient.stops[i][1]);
252 		            ctx.fillStyle = g;
253 	            }				
254 				ctx.beginPath();    		
255 				ctx.arc(d[4], d[4], d[4], 0, Math.PI*2, true);
256 				ctx.closePath();				
257 				if (style.fillStyle || style.gradient) ctx.fill();
258 				if (style.strokeStyle) ctx.stroke();
259 			}
260     	};

264 		
265 		var self = this;
266 		jsPlumb.Endpoints.Rectangle.apply(this, arguments);
267 		CanvasEndpoint.apply(this, arguments);				
268 		
269     	this._paint = function(d, style, anchor) {
270 				
271 			var ctx = self.canvas.getContext("2d"), orientation = anchor.getOrientation(self);
272 			jsPlumb.extend(ctx, style);
273 			
274 			/* canvas gradient */
275 		    if (style.gradient) {
276 		    	// first figure out which direction to run the gradient in (it depends on the orientation of the anchors)
277 		    	var y1 = orientation[1] == 1 ? d[3] : orientation[1] == 0 ? d[3] / 2 : 0;
278 				var y2 = orientation[1] == -1 ? d[3] : orientation[1] == 0 ? d[3] / 2 : 0;
279 				var x1 = orientation[0] == 1 ? d[2] : orientation[0] == 0 ? d[2] / 2 : 0;
280 				var x2 = orientation[0] == -1 ? d[2] : orientation[0] == 0 ? d[2] / 2 : 0;
281 			    var g = ctx.createLinearGradient(x1,y1,x2,y2);
282 			    for (var i = 0; i < style.gradient.stops.length; i++)
283 	            	g.addColorStop(style.gradient.stops[i][0], style.gradient.stops[i][1]);
284 	            ctx.fillStyle = g;
285 		    }
286 			
287 			ctx.beginPath();
288 			ctx.rect(0, 0, d[2], d[3]);
289 			ctx.closePath();				
290 			if (style.fillStyle || style.gradient) ctx.fill();
291 			if (style.strokeStyle) ctx.stroke();
292     	};

296 	        			
297 		var self = this;
298 		jsPlumb.Endpoints.Triangle.apply(this, arguments);
299 		CanvasEndpoint.apply(this, arguments);			
300 		
301     	this._paint = function(d, style, anchor)
302 		{    		
303 			var width = d[2], height = d[3], x = d[0], y = d[1],			
304 			ctx = self.canvas.getContext('2d'),
305 			offsetX = 0, offsetY = 0, angle = 0,
306 			orientation = anchor.getOrientation(self);
307 			
308 			if( orientation[0] == 1 ) {
309 				offsetX = width;
310 				offsetY = height;
311 				angle = 180;
312 			}
313 			if( orientation[1] == -1 ) {
314 				offsetX = width;
315 				angle = 90;
316 			}
317 			if( orientation[1] == 1 ) {
318 				offsetY = height;
319 				angle = -90;
320 			}
321 			
322 			ctx.fillStyle = style.fillStyle;
323 			
324 			ctx.translate(offsetX, offsetY);
325 			ctx.rotate(angle * Math.PI/180);
326 
327 			ctx.beginPath();
328 			ctx.moveTo(0, 0);
329 			ctx.lineTo(width/2, height/2);
330 			ctx.lineTo(0, height);
331 			ctx.closePath();
332 			if (style.fillStyle || style.gradient) ctx.fill();
333 			if (style.strokeStyle) ctx.stroke();				
334     	};

368      * Canvas straight line Connector. Draws a straight line onto a Canvas element.
369      */
370     jsPlumb.Connectors.canvas.Straight = function() {   	 
371 		var self = this,
372 			segmentMultipliers = [null, [1, -1], [1, 1], [-1, 1], [-1, -1] ];
373 
374 		jsPlumb.Connectors.Straight.apply(this, arguments);
375 		CanvasConnector.apply(this, arguments);
376 		this._paint = function(dimensions, style) {
377 
378 			self.ctx.beginPath();
379 
380 			if (style.dashstyle && style.dashstyle.split(" ").length == 2) {			
381 				// only a very simple dashed style is supported - having two values, which define the stroke length 
382 				// (as a multiple of the stroke width) and then the space length (also as a multiple of stroke width). 
383 				var ds = style.dashstyle.split(" ");
384 				if (ds.length != 2) ds = [2, 2];
385 				var dss = [ ds[0] * style.lineWidth, ds[1] * style.lineWidth ],
386 					m = (dimensions[6] - dimensions[4]) / (dimensions[7] - dimensions[5]),
387 					s = jsPlumbUtil.segment([dimensions[4], dimensions[5]], [ dimensions[6], dimensions[7] ]),
388 					sm = segmentMultipliers[s],
389 					theta = Math.atan(m),
390 					l = Math.sqrt(Math.pow(dimensions[6] - dimensions[4], 2) + Math.pow(dimensions[7] - dimensions[5], 2)),
391 					repeats = Math.floor(l / (dss[0] + dss[1])),
392 					curPos = [dimensions[4], dimensions[5]];
393 
394 				
395 				// TODO: the question here is why could we not support this in all connector types? it's really
396 				// just a case of going along and asking jsPlumb for the next point on the path a few times, until it
397 				// reaches the end. every type of connector supports that method, after all.  but right now its only the
398 				// bezier connector that gives you back the new location on the path along with the x,y coordinates, which
399 				// we would need. we'd start out at loc=0 and ask for the point along the path that is dss[0] pixels away.
400 				// we then ask for the point that is (dss[0] + dss[1]) pixels away; and from that one we need not just the
401 				// x,y but the location, cos we're gonna plug that location back in in order to find where that dash ends.
402 				//
403 				// it also strikes me that it should be trivial to support arbitrary dash styles (having more or less than two
404 				// entries). you'd just iterate that array using a step size of 2, and generify the (rss[0] + rss[1])
405 				// computation to be sum(rss[0]..rss[n]).
406 
407 				for (var i = 0; i < repeats; i++) {
408 					self.ctx.moveTo(curPos[0], curPos[1]);
409 
410 					var nextEndX = curPos[0] + (Math.abs(Math.sin(theta) * dss[0]) * sm[0]),
411 						nextEndY = curPos[1] + (Math.abs(Math.cos(theta) * dss[0]) * sm[1]),
412 						nextStartX = curPos[0] + (Math.abs(Math.sin(theta) * (dss[0] + dss[1]))  * sm[0]),
413 						nextStartY = curPos[1] + (Math.abs(Math.cos(theta) * (dss[0] + dss[1])) * sm[1])
414 
415 					self.ctx.lineTo(nextEndX, nextEndY);
416 					curPos = [nextStartX, nextStartY];					
417 				}
418 
419 				// now draw the last bit
420 				self.ctx.moveTo(curPos[0], curPos[1]);
421 				self.ctx.lineTo(dimensions[6], dimensions[7]);		
422 
423 			}	        
424 	        else {
425 		        self.ctx.moveTo(dimensions[4], dimensions[5]);
426 	        	self.ctx.lineTo(dimensions[6], dimensions[7]);
427 	        }
428 	        	      
429 	        self.ctx.stroke();            
430 	    };
431 	    
432 	    // TODO this does not handle the case that src and target are swapped.
433 	    this.createGradient = function(dim, ctx) {
434         	return ctx.createLinearGradient(dim[4], dim[5], dim[6], dim[7]);
435         };
436     };

486 				ctx.strokeStyle = strokeStyle;
487 				ctx.stroke();
488 			}
489 			if (fillStyle) {
