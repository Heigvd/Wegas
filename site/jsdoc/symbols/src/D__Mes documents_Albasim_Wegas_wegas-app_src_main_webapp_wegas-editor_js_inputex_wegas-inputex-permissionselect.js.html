  8     /**
  9      * Adds an url regexp, and display the favicon at this url
 10      *
 11      * @class Y.inputEx.Wegas.PermissionSelect
 12      * @extends Y.inputEx.StringField
 13      * @constructor
 14      * @param {Object} options inputEx.Field options object
 15      * <ul>
 16      *   <li>favicon: boolean whether the domain favicon.ico should be displayed or not (default is true, except for https)</li>
 17      * </ul>
 18      */
 19     Y.namespace("inputEx.Wegas").PermissionSelect = function (options) {
 20         inputEx.Wegas.PermissionSelect.superclass.constructor.call(this, options);
 21     };
 22     Y.extend(inputEx.Wegas.PermissionSelect, inputEx.Field,  {
 23         /** @lends Y.inputEx.Wegas.PermissionSelect# */
 24         setOptions: function (options) {
 25             inputEx.Wegas.PermissionSelect.superclass.setOptions.call(this, options);
 26             this.options.permissions = options.permissionsChoices;
 27             this.options.roles = options.roles;
 28             this.options.targetEntityId = options.targetEntityId;
 29             this.value = {};
 30         },
 31 
 32         renderComponent: function () {
 33             this.fireUpdatedEvt();
 34 
 35             this.roleSelect = new Y.inputEx.Wegas.RoleSelect({
 36                 parentEl: this.fieldContainer
 37             });
 38 
 39             this.roleSelect.on("updated", function (val) {
 40 
 41                 if (this.value.id !== val.id) {
 42                     Y.Wegas.UserFacade.rest.deleteAllRolePermissions(this.value.id, this.options.targetEntityId);
 43                 }
 44 
 45                 this.value.id = val.id;
 46                 this.value.permissions = [];
 47                 this.checkboxValue();
 48                 this.fireUpdatedEvt();
 49             }, this);
 50 
 51             var logDiv = Y.Node.create('<div class="permissionList"></div>');
 52             this.fieldContainer.div = this.fieldContainer.appendChild(logDiv.getDOMNode());
 53             this.permissionsCheckBoxes = [];
 54             Y.Array.forEach(this.options.permissions, function (item, i) {
 55                 var splitedPermissions = item.name.split(":"),
 56                 box = new Y.inputEx.CheckBox({
 57                     rightLabel: splitedPermissions[1],
 58                     name: splitedPermissions[0] + ":" + splitedPermissions[1],
 59                     value: false,
 60                     parentEl: this.fieldContainer.div,
 61                     className: "eachPermissions"
 62                 });
 63 
 64                 box.on("updated", function (e, field) {
 65                     if (field.getValue()) {
 66                         Y.Wegas.UserFacade.rest.sendRequest({
 67                             request: "/AddPermission/" + this.roleSelect.getValue().id + "/" + field.options.name +":" + this.options.targetEntityId,
 68                             cfg: {
 69                                 method: "POST"
 70                             }
 71                         });
 72                     } else {
 73                         Y.Wegas.UserFacade.rest.sendRequest({
 74                             request: "/DeletePermission/" + this.roleSelect.getValue().id + "/" + field.options.name +":" + this.options.targetEntityId,
 75                             cfg: {
 76                                 method: "POST"
 77                             }
 78                         });
 79                     }
 80                     this.fireUpdatedEvt();
 81                 }, this);
 82                 this.permissionsCheckBoxes.push(box);
 83             }, this);
 84 
 85             this.checkboxValue();
 86         },
 87         getValue: function () {
 88             this.value.id = this.roleSelect.getValue().id;
 89             return this.value;
 90         },
 91         setValue: function (val, sendUpdatedEvent) {
 92             inputEx.Wegas.PermissionSelect.superclass.setValue.call(this, val, sendUpdatedEvent);
 93             this.roleSelect.setValue(val, false);
 94             this.value = val;
 95             this.checkboxValue();
 96         },
 97         checkboxValue: function () {
 98             Y.Array.forEach(this.permissionsCheckBoxes, function(box, i) {
 99                 box.setValue(false, false);
100             });
101 
102             //this.roleSelect.setValue(this.getValue(), false);
103             Y.Array.forEach(this.getValue().permissions, function (perm, i) {
104                 var splitedPermissions = perm.split(":");
105                 Y.Array.forEach(this.permissionsCheckBoxes, function (box, i) {
106                     if (box.options.rightLabel === splitedPermissions[1]) {
107                         box.setValue(true, false);
108                     }
109                 }, this);
110             }, this);
111         },
112 
113         destroy: function () {
114             inputEx.Wegas.PermissionSelect.superclass.destroy.call(this);
115             this.roleSelect.destroy();
116             var i;
117             for (i = 0; i < this.permissionsCheckBoxes.length; i += 1) {
118                 this.permissionsCheckBoxes[i].destroy();
119             }
120         }
121 
122     });

132     var CONTENTBOX = "contentBox", RolePermissionList;

134     RolePermissionList = Y.Base.create("wegas-permissionlist", Y.Widget, [Y.WidgetChild, Y.Wegas.Editable, Y.Wegas.Widget], {
135         /** @lends Y.Wegas.RolePermissionList# */
136         renderUI: function () {
137             this.plug(Y.Plugin.WidgetToolbar);
138             this.bNew = this.toolbar.add({
139                 type: "Button",
140                 label: "<span class=\"wegas-icon wegas-icon-new\"></span>New"
141             });
142         },
143 
144         bindUI: function () {
145             this.bNew.on('click', function (e) {
146                 this.permsField.onAddButton(e);
147             }, this);
148         },
149 
150         syncUI: function () {
151             if (this.permsField) {
152                 this.permsField.destroy();
153             }
154 
155             var e = this.get("entity"),
156             gmId = e.get("id");
157             if (e instanceof Y.Wegas.persistence.GameModel) {
158                 this.targetEntityId = "gm" + gmId;
159             } else {
160                 this.targetEntityId = "g" + gmId;
161             }
162             Y.Wegas.UserFacade.rest.sendRequest({
163                 request: "/GameModelPermissions/" + this.targetEntityId,
164                 cfg: {
165                     method: "GET"
166                 },
167                 on: {
168                     success: Y.bind(function (e) {
169                         var data = e.response.results.entities,
170                         acc = [];
171 
172                         Y.Array.forEach(data, function (role) {
173                             acc.push(role.get("val"));
174                         }, this);
175 
176                         this.messageDiv = Y.Node.create('<div class="wegas-smallmessage">Click new to add rights</div>');
177                         this.get(CONTENTBOX).getDOMNode().appendChild(this.messageDiv.getDOMNode());
178 
179                         this.permsField = new PermissionList({
180                             //listLabel: 'Websites',
181                             elementType: {
182                                 type: 'permissionsselect',
183                                 permissionsChoices: this.get("permsList"),
184                                 targetEntityId: this.targetEntityId,
185                                 roles: acc,
186                                 className: "role-permissions"
187                             },
188                             useButtons: true,
189                             value: acc,
190                             parentEl: this.get(CONTENTBOX).getDOMNode(),
191                             className: "roleBox"
192                         });
193                         this.permsField.on("updated", this.sync, this);
194 
195                         this.get(CONTENTBOX).one(".roleBox img").hide();
196                     }, this),
197                     failure: Y.bind(function (e) {
198                         this.fire("exception", e.response.results);
199                     }, this)
200                 }
201             });
202         },
203 
204         destructor: function () {
205             if (this.permsField) {
206                 this.permsField.destroy();
207             }
208         },
209 
210         sync: function() {
211             var list = this.permsField.getRoleIds();
212 
213             Y.Array.forEach(this.permsField.subFields, function(eachSubfield) {
214                 Y.Array.forEach(eachSubfield.roleSelect.choicesList, function(role) {
215                     if (eachSubfield.roleSelect.getValue().id === role.value || list.indexOf(role.value) === -1) {
216                         eachSubfield.roleSelect.showChoice(role);
217                     } else {
218                         eachSubfield.roleSelect.hideChoice(role);
219                     }
220                 }, this);
221             }, this);
222 
223             if (this.permsField.subFields.length < 1){
224                 this.messageDiv.show();
225             } else {
226                 this.messageDiv.hide();
227             }
228         }
229     }, {
230         /** @lends Y.Wegas.RolePermissionList */
231         ATTRS: {
232             permsList: {
233                 value: []
234             },
235             entity: {}
236         }
237     });

242     /**
243      * @fixme @hack override to had event
244      *
245      * @class Y.inputEx.Wegas.PermissionList
246      * @extends Y.inputEx.ListField
247      * @constructor
248      * @param {Object} options
249      */
250     var PermissionList = function(options) {
251         PermissionList.superclass.constructor.call(this, options);
252     };
253 
254     Y.extend(PermissionList, inputEx.ListField,  {
255 
256         getRoleIds: function () {
257             var list = [], roleWithPermission;
258             Y.Array.forEach(this.subFields, function(field) {
259                 roleWithPermission = field.getValue();
260                 list.push(roleWithPermission.id);
261             });
262             return list;
263         },
264 
265         onAddButton: function() {
266             PermissionList.superclass.onAddButton.apply(this, arguments);
267             var newField = this.subFields[this.subFields.length - 1],
268             filter = this.getRoleIds(),
269             i= 0;
270 
271             while (filter.indexOf(newField.getValue().id) > -1
272                 && i < newField.roleSelect.choicesList.length) {
273                 newField.setValue({
274                     id:  newField.roleSelect.choicesList[i].value
275                 });
276                 i++;
277             }
278         },
279 
280         onDelete: function(e) {
281             var elementDiv = e.target._node.parentNode,
282             i, subFieldEl = elementDiv.childNodes[this.options.useButtons ? 1 : 0];
283             for (i = 0; i < this.subFields.length; i++) {
284                 if (this.subFields[i].getEl() == subFieldEl) {
285                     Y.Wegas.UserFacade.rest.deleteAllRolePermissions(this.subFields[i].roleSelect.getValue().id, this.subFields[i].options.targetEntityId);
286                     break;
287                 }
288             }
289 
290             PermissionList.superclass.onDelete.apply(this, arguments);
291         }
292     });
