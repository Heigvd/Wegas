 13 YUI.add('wegas-cssloader', function(Y) {
 14     "use strict";
 15 
 16     var CSSLoader = Y.Base.create("wegas-cssloader", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
 17         initializer: function() {
 18             var i, css = Y.Wegas.app.get('cssStylesheets'),
 19                 cfg = {
 20                     context: this,
 21                     on : {
 22                         success : function (tId, e) {
 23                             CSSLoader.customCSSText = e.responseText;
 24                             CSSLoader.customCSSStyleSheet = new Y.StyleSheet(e.responseText);
 25                         },
 26                         failure : function (id, o) {
 27                             Y.error("initCSS(): Page CSS loading async call failed!", new Error("Loading failed"), "Y.Wegas.App");
 28                         }
 29                     }
 30                 };
 31 
 32             for (i = 0; i < css.length; i += 1) {
 33                 Y.io(Y.Wegas.app.get('base') + css[i] + '?id=' + Y.Wegas.App.genId(), cfg);  // Load the page css
 34             }
 35         }
 36     }, {
 37         NS: "CSSLoader",
 38         NAME: "CSSLoader"
 39     });
 40     Y.namespace("Plugin").CSSLoader = CSSLoader;
 41 
 42 });
