 15 YUI.add('wegas-inputex-ace', function(Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx;
 19 
 20     /**
 21      * Ace code editor field
 22      */
 23     inputEx.AceField = function(options) {
 24         inputEx.AceField.superclass.constructor.call(this,options);
 25     };
 26     Y.extend(inputEx.AceField, inputEx.Field, {
 27         /**
 28          * Set the default values of the options
 29          * @param {Object} options Options object as passed to the constructor
 30          */
 31         setOptions: function (options) {
 32             inputEx.AceField.superclass.setOptions.call(this, options);
 33 
 34             this.options.language = options.language || "javascript";
 35             this.options.height = options.height || "150px";
 36         },
 37 
 38         /**
 39 	 * Render the field using the YUI Editor widget
 40 	 */
 41         renderComponent: function () {
 42             this.el = Y.Node.create( '<div style="">'
 43                 + ( this.options.value ? this.options.value : "" ) + '</div>' );
 44             this.fieldContainer.appendChild(this.el.getDOMNode());
 45             //this.fieldContainer.style[ "position" ] = "relative";
 46 
 47             this.editor = ace.edit( this.el.getDOMNode() );
 48             this.editor.setHighlightActiveLine(false);
 49             this.editor.renderer.setHScrollBarAlwaysVisible(false);
 50             this.session = this.editor.getSession();
 51 
 52             var Mode = require("ace/mode/" + this.options.language).Mode;
 53             this.session.setMode(new Mode());
 54 
 55             Y.Wegas.app.after( "layout:resize", function() {
 56                 Y.once( 'domready', this.resize, this );
 57             }, this.editor );
 58 
 59             Y.after( 'windowresize', Y.bind( this.editor.resize, this.editor ));
 60 
 61         //this.session.addEventListener("tokenizerUpdate", Y.bind(function(e) {
 62         //    var i, token,
 63         //    tokens = this.session.getTokens(e.data.first, e.data.last);
 64         //
 65         //    for (i = 0; i > tokens.length; i += 1) {
 66         //        token = tokens[i];                                          //identifier
 67         //    }
 68         //}, this));
 69         },
 70 
 71         genTree: function(token) {
 72 
 73         },
 74 
 75         /**
 76 	 * Set the html content
 77 	 * @param {String} value The html string
 78 	 * @param {boolean} [sendUpdatedEvt] (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
 79 	 */
 80         setValue: function(value, sendUpdatedEvt) {
 81             this.session.setValue(value);
 82             if(sendUpdatedEvt !== false) {
 83                 // fire update event
 84                 this.fireUpdatedEvt();
 85             }
 86         },
 87 
 88         /**
 89 	 * Get the ace content
 90 	 * @return {String} the ace area content string
 91 	 */
 92         getValue: function() {
 93             return this.session.getValue();
 94         }
 95     });
 96 
 97     // Register this class as "html" type
 98     inputEx.registerType("ace", inputEx.AceField, []);
 99 });
