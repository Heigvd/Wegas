 15 YUI.add('wegas-form', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19     FormWidget;
 20 
 21     FormWidget = Y.Base.create( "wegas-form", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 22 
 23         // ** Private Fields ** //
 24 
 25         // ** Lifecycle Methods ** //
 26         initializer: function () {
 27             this.plug( Y.Plugin.WidgetToolbar );
 28             this.publish( "submit", {
 29                 emitFacade: true
 30             });
 31         },
 32 
 33         renderUI: function () {
 34             this.renderToolbar();
 35         },
 36 
 37         // ** Private Methods ** //
 38 
 39         renderToolbar: function () {
 40             var toolbarNode = this.toolbar.get( 'header' );
 41 
 42             this.saveButton = new Y.Button({
 43                 label: "<span class=\"wegas-icon wegas-icon-save\" ></span>Save",
 44                 on: {
 45                     click: Y.bind( function () {
 46                         var form = this.get( "form" ),
 47                         val = form.getValue();
 48 
 49                         if ( !form.validate() ) {
 50                             return;
 51                         }
 52                         form.fire( "afterValidation");
 53                         if (val.valueselector) {
 54                             val = val.valueselector;
 55                         }
 56                         this.fire( "submit", {
 57                             value: val
 58                         });
 59                     }, this)
 60                 }
 61             }).render(toolbarNode);
 62 
 63             this.cancelButton = new Y.Button( {
 64                 label: "<span class=\"wegas-icon wegas-icon-cancel\" ></span>Cancel",
 65                 on: {
 66                     click: Y.bind(function () {
 67                         this.fire( "cancel" );
 68                     }, this)
 69                 }
 70             } ).render(toolbarNode);
 71         },
 72         setForm: function (values, formCfg) {
 73             this.set( "values", values);
 74             this.set( "formCfg", formCfg)
 75         },
 76 
 77         destroyForm: function () {
 78             this.set( "form", null );
 79         }
 80 
 81     }, {
 82         ATTRS: {
 83             values: {
 84                 value: {}
 85             },
 86             form: {
 87                 setter: function ( val ) {
 88                     if ( this.get( "form" ) ) {                                 // If there is alread a form instantiated, we destroy it
 89                         this.get("form").destroy();
 90                     }
 91                     return val;
 92                 }
 93             },
 94             formCfg: {
 95                 setter: function (val) {
 96                     val.parentEl = this.get(CONTENTBOX);                        //  Set up the form parentEl attribute, so it knows where to render
 97 
 98                     Y.inputEx.use(val, Y.bind(function (formCfg) {              // Load form dependencies
 99                         var form = Y.inputEx(formCfg);                          // Initialize and render form
100                         form.setValue(this.get("values"));                      // Sync form with "values" ATTR
101                         this.set("form", form);
102                     }, this, val));
103                     return val;
104                 }
105             }
106         }
107     });
108 
109     Y.namespace("Wegas").FormWidget = FormWidget;
110 });
