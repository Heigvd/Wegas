 14 YUI.add('wegas-proggame-level', function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = 'contentBox',
 18     GRIDSIZE = 31,
 19     ProgGameLevel;
 20 
 21     /**
 22      *  The level display class, with script input, ia, debugger and
 23      *  terrain display.
 24      *
 25      */
 26     ProgGameLevel = Y.Base.create( "wegas-proggame-level", Y.Widget, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable ], {
 27 
 28         // *** Fields *** //
 29         CONTENT_TEMPLATE: '<div class="yui3-g">'
 30 
 31         + '<div class="yui3-u left">'
 32         + '<div class="inventory"><h1>Inventory</h1><i><center style="padding-top:40px;">empty</center></i></div>'
 33         + '<div class="api"><h1>Api</h1></div>'
 34         + '</div>'
 35 
 36         + '<div class="yui3-u right">'
 37         + '<div class="yui3-g topright">'
 38         + '<div class="yui3-u topcenter"><h1></h1><div class="terrain"></div></div>'
 39         + '<div class="yui3-u toptopright">'
 40         + '<div class="buttons"></div>'
 41         + '<div class="ai"><h1>Enemy A.I.</h1></div>'
 42         + '<div class="debugger"><h1>Log</h1></div>'
 43         + '</div>'
 44         + '</div>'
 45 
 46         + '<div class="code"><h1>Your code</h1></div>'
 47         + '</div>'
 48         + '<div style="clear:both"></div>',
 49 
 50         // *** Lifecycle Methods *** //
 51         renderUI: function () {
 52             var cb = this.get( CONTENTBOX );
 53 
 54             cb.one( ".ai" ).append( Y.Wegas.App.nl2br(this.get( "ai" )) );
 55             cb.one( ".topcenter h1" ).setHTML(this.get( "label" ));
 56 
 57             cb.one(".api").append(this.get("api")+"*");
 58 
 59             this.aceField = new Y.inputEx.AceField({
 60                 parentEl: cb.one( ".code" ),
 61                 name: 'text',
 62                 type: 'ace',
 63                 height: "300px",
 64                 language: "javascript",
 65                 value: "move();fire();"
 66             });
 67 
 68             this.display = new ProgGameDisplay(this.toObject());
 69             this.display.render( cb.one( ".terrain" ));
 70 
 71             this.runButton = new Y.Wegas.Button( {
 72                 label: "RUN SCRIPT"
 73             } );
 74             this.runButton.render( cb.one( ".buttons" ));
 75 
 76         },
 77         bindUI: function () {
 78             Y.Wegas.app.VariableDescriptorFacade.after( "response",             // If data changes, refresh
 79                 this.syncUI, this);
 80 
 81             Y.Wegas.app.after( 'currentPlayerChange', this.syncUI, this);       // If current user changes, refresh (editor only)
 82 
 83             this.runButton.on( "click" , function () {
 84 
 85                 this.display.set( "objects", this.get( "objects" ));            // Reset the display to default
 86                 this.display.syncUI();
 87                 this.get( CONTENTBOX ).one( ".debugger" ).setHTML( "<h1>Debugger</h1>" );
 88                 this.runButton.set( "label", "RUNNING..." );
 89                 this.runButton.set( "disabled", true );
 90 
 91                 Y.Wegas.app.VariableDescriptorFacade.rest.sendRequest({
 92                     request: "/ProgGame/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
 93                     cfg: {
 94                         method: "POST",
 95                         data: "var ret = [], objects = " + Y.JSON.stringify(this.get( "objects" ))
 96                         + ",cObject,"
 97                         + "winingCondition=function(){return "+this.get( "winningCondition" ) +";};"
 98                         + "for (var i =0;i<2;i++) {"
 99                         + "cObject = 'player';"
100                         + "sendCommand({type:'log', 'text': 'Player turn.'});"
101                         + this.aceField.getValue()
102                         + "cObject='enemy';"
103                         + "sendCommand({type:'log', 'text': 'Enemy turn.'});"
104                         + this.get( "ai" )
105                         + "}"
106                         + "sendCommand({type:'log', 'text': 'Max turn reached, match is a draw.'});"
107                         + "JSON.stringify(ret)"
108                     },
109                     on: {
110                         success: Y.bind(this.onServerReply, this ),
111                         failure: Y.bind( function () {
112                             this.runButton.set( "label", "RUN SCRIPT" );
113                             this.runButton.set( "disabled", false );
114                             alert( "Your script contains an error." );
115                         }, this )
116                     }
117 
118                 });
119             }, this );
120 
121         },
122         syncUI: function () {
123             this.display.syncUI();
124         },
125 
126         onServerReply: function ( e ) {
127             this.commandsStack = Y.JSON.parse( e.response.entity );
128             this.consumeServerCommand();
129         },
130 
131         doNextLevel: function () {
132             Y.Wegas.app.VariableDescriptorFacade.rest.sendRequest({
133                 request: "/ProgGame/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
134                 cfg: {
135                     method: "POST",
136                     data: this.get( "onWin" )
137                 }
138             });
139         },
140 
141         consumeServerCommand: function () {
142             if (this.commandsStack.length > 0 ) {
143                 var command = this.commandsStack.shift();
144 
145                 this.display.execute( command );
146 
147                 switch ( command.type ) {
148 
149                     case "gameWon":
150                         this.runButton.set( "label", "NEXT LEVEL" );
151                         this.runButton.set( "disabled", false );
152                         this.runButton.detachAll( "click" );
153                         this.runButton.on( "click", this.doNextLevel, this );
154                         break;
155 
156                     case "log":
157                         this.get( "contentBox" ).one( ".debugger" ).append( command.text + "<br />" );
158 
159                     default:
160                         Y.later( 500, this, this.consumeServerCommand );
161                         break;
162 
163                 }
164 
165             } else {
166                 this.runButton.set( "label", "RUN SCRIPT");
167                 this.runButton.set( "disabled", false );
168             }
169         }
170 
171     }, {
172         ATTRS : {
173             label: {
174                 type: "string"
175             },
176             objects: {
177                 _inputex: {
178                     _type: "object"
179                 }
180             },
181             api: {
182                 value: []
183             },
184             ai: {
185                 type: "string",
186                 format: "text"
187 //                _inputex: {
188 //                    _type: "ace"
189 //                }
190             },
191             winningCondition: {
192                 type: "string",
193                 format: "text"
194 //                _inputex: {
195 //                    _type: "ace"
196 //                }
197             },
198             onWin: {
199                 type: "string",
200                 _inputex: {
201                     _type: "ace"
202                 }
203             }
204         }
205     });
206 
207     Y.namespace('Wegas').ProgGameLevel = ProgGameLevel;
208 
209 
210     /**
211      * Level display, should handle canvas, for now renders the level as a
212      * table element.
213      *
214      */
215     var ProgGameDisplay = Y.Base.create( "wegas-proggame-display", Y.Widget, [], {
216         CONTENT_TEMPLATE: '<div><div class="object-layer"></div></div>',
217 
218         execute: function ( command ) {
219             var cb = this.get( CONTENTBOX );
220             switch ( command.type ) {
221                 case "move":
222                     this.set( "objects", command.objects );
223                     this.syncUI();
224                     break;
225 
226                 case "fire":
227                     var object = this.findObjectByType( command.object ),
228                     p = Y.Node.create( '<div class="missile"></div>' ),     // Create a missile
229                     source = cb.one( "." + object.id ),                     // Retrieve the source node (the object which fired)
230                     to = source.getXY();
231                     
232                     switch ( object.direction ) {
233                         case 1:
234                             to[1] -= GRIDSIZE * 3;
235                             break;
236                         case 2:
237                             to[0] += GRIDSIZE * 3;
238                             break;
239                         case 3:
240                             to[1] += GRIDSIZE * 3;
241                             break;
242                         case 4:
243                             to[0] -= GRIDSIZE * 3;
244                             break;
245                     }
246 
247                     var anim = new Y.Anim({
248                         node: p,
249                         from: {
250                             xy: source.getXY()
251                         },
252                         to: {
253                             xy: to
254                         },
255                         duration: 0.5
256                     });
257                     cb.one( ".object-layer" ).append( p );
258                     anim.run();
259                     break;
260 
261                 case "die":
262                     var source2 = cb.one( "." + command.object ),
263                     anim2 = new Y.Anim({
264                         node: source2,
265                         from: {
266                             opacity: 1
267                         },
268                         to: {
269                             opacity: 0
270                         },
271                         duration: 1,
272                         iterations: 2
273                     });
274                     anim2.run();
275 
276                     break;
277 
278             }
279         },
280 
281         findObjectByType: function ( type ) {
282             var i, objs = this.get( "objects" );
283             for ( i = 0; i < objs.length; i = i + 1 ) {
284                 if ( objs[i].id === type ) {
285                     return objs[i];
286                 }
287             }
288             return null;
289         },
290 
291         renderUI: function () {
292             var i, j,
293             cb = this.get( "contentBox" ),
294             acc = [ "<table>" ];
295 
296             for ( i = this.get( "gridH" ) - 1 ; i >= 0 ; i -= 1 ) {          // Render table elements
297                 acc.push( "<tr>" );
298                 for ( j = 0; j < this.get( "gridW" ); j += 1 ) {
299                     acc.push( "<td></td>" );
300                 }
301                 acc.push( "</tr>" );
302             }
303             acc.push( "</table>" );
304             cb.append( acc.join( "" ));
305         },
306 
307         syncUI: function () {
308             var i,
309             playerLayer = this.get( "contentBox" ).one( ".object-layer" ),
310             objs = this.get( "objects" );
311             playerLayer.setHTML( "" );
312 
313             for ( i = 0; i < objs.length; i = i + 1) {
314                 var node = Y.Node.create( '<div class="' + objs[i].id + ' dir-' + objs[i].direction + '"></div>');
315                 node.setXY( [ objs[i].x * GRIDSIZE , (this.get( "gridH" ) - objs[i].y - 1 ) * GRIDSIZE ]);
316                 playerLayer.append( node );
317             }
318 
319         }
320 
321     }, {
322         ATTRS: {
323             gridW: {
324                 value: 8
325             },
326             gridH: {
327                 value: 8
328             },
329             objects: {
330 
331         }
332         }
333     });
334     Y.namespace('Wegas').ProgGameDisplay = ProgGameDisplay;
335 
336 });
