113         /**
114          *
115          */
116         syncUI: function () {
117             var i, tree = window.esprima.parse( this.el.value, {                // Generate the form
118                 raw: true
119             }),
120             container = new Y.Node( this.fieldContainer ),
121             fields = [];
122 
123             container.one( ".msg" ).setContent( "");
124 
125             for (i = 0; i < tree.body.length; i = i + 1) {
126                 try {
127                     fields.push( this.generateExpression( tree.body[i].expression ) );
128                 } catch( e ) {
129                     //Y.error( "Error evaluating line: " +
130                     //    window.escodegen.generate(tree.body[i].expression, {
131                     //        indent: true
132                     //    }));
133                     this.setMode( "text" );
134                     this.viewSrc.set( "disabled", true );
135                     container.one( ".msg" ).setContent( "Unable to read this impact, displaying source only.");
136                     return;
137                 }
138             }
139 
140             if ( this.form ) {
141                 this.form.destroy();
142             }
143 
144             this.form = Y.inputEx({
145                 type: "inputlist",
146                 fields: fields,
147                 useButtons: true,
148                 parentEl: this.fieldContainer
149             });
150 
151             this.form.addButton.get( "boundingBox" ).setStyle( "display", "inline-block" );
152 
153             this.form.on( "updated", this.updateTextareaValue, this );
154 
155             this.setMode( this.options.mode );
156             this.viewSrc.set( "disabled", false );
157         },

159         updateTextareaValue: function () {
160             if ( this.options.mode === "wysiwyg" ) {
161                 this.el.value =  this.form.getArray().join(";\n") + ";";
162             }
163         },
164         /**
165          *
166          */
167         generateExpression: function ( expression ) {
168             switch ( expression.type ) {
169                 case "CallExpression":
170                     switch ( expression.callee.object.type ) {
171                         case "Identifier":
172                             switch ( expression.callee.object.name ) {
173                                 case "VariableDescriptorFacade":
174                                     return {
175                                         type: "variabledescriptorselect",
176                                         value: expression.arguments[0].value
177                                     };
178                             }
179                             break;
180                         default:
181                             //return new MethodSelect({
182                             //    object: this.generateExpression( expression.callee.object ),
183                             //    name: expression.callee.property.value,
184                             //    arguments: expression.callee.arguments
185                             var vdSelect = this.generateExpression( expression.callee.object ), args = [];
186 
187                             Y.Array.each( expression.arguments, function ( i ) {
188                                 args.push( i.value || i.name );
189                             });
190                             Y.mix(vdSelect, {
191                                 //type: "variabledescriptormethodselect",
192                                 //object: this.generateExpression( expression.callee.object ),
193                                 //fields: [  ],
194                                 method: expression.callee.property.name,
195                                 arguments: args
196 
197                             });
198                             return vdSelect;
199                     }
200             }
201             throw new Exception("Unable to parse expression.");
202         }

231 
232         entityId: null,
233         /**
234 	 * Setup the options.fields from the availableFields option
235 	 */
236         setOptions: function(options) {

238 
239             VariableDescriptorSelect.superclass.setOptions.call(this, options);
240             this.options.method = options.method;
241             this.options.arguments = options.arguments;

245             VariableDescriptorSelect.superclass.render.call( this );
246             this.syncUI();

249         setValue: function ( val ) {
250             // Set value should not ba called directly
251             console.log( "VariableDescriptorSelect.setValue", val );

254         getValue: function () {
255             var i, l = this.inputs.length,
256             args = this.inputs[ l - 1 ].getValue();
257 
258             for ( i = 0; i < args.length; i = i + 1 ) {
259                 if ( this.currentMethod.arguments[i].scriptType === "string" ) {
260                     args[i] = '"' + args[i] + '"';
261                 }
262             }
263             return "VariableDescriptorFacade.find(" + this.inputs[ l - 3].getValue() + ")" +
264                 "." + this.inputs[ l - 2 ].getValue() +

282                 cMethod = Y.Object.values(methods)[0];                          // By default select the first method available
283             }

286 
287             for ( i = 0; i < args.length; i = i + 1 ) {
288                 args[i].entity = this.currentEntity;                            // Adds a reference to the target entity to the argument Fields;

293             this.addField( Y.mix({

313         /**

317         onChange: function( fieldValue, fieldInstance ) {
318 
319             if ( Y.Lang.isNumber( fieldValue ) ) {
320                 this.entityId = fieldValue;
321                 this.options.method = null;
322                 this.options.arguments = null;
323             } else if ( Y.Lang.isString( fieldValue ) ) {
324                 this.entityId = fieldInstance.options.parentEntity.get( "id" );
325                 this.options.method = fieldValue;
326                 this.options.arguments = null;
327             } else {
328                 this.options.arguments = this.inputs[ this.inputs.length - 1 ].getValue() // Saves current args value
329             }
330             this.syncUI();
331             this.fireUpdatedEvt();

340         /**
341          * Generate
342          */
343         generateField: function () {
344             var ret = [],
345             currentId = this.entityId || this.options.value,
346             rootEntities = Y.Wegas.VariableDescriptorFacade.rest.getCache(),
347             currentEntity = Y.Wegas.VariableDescriptorFacade.rest.findById( currentId ) || rootEntities[ 0 ] ;
348 
349             this.currentEntity = currentEntity;                                 // Keeps a reference to the current entity
350 
351             ret.push( this.generateSelectConfig( null,                          // Pushes the current entity to the fields stack
352             currentEntity, currentEntity.get( "items" ) ) );
353 
354             if ( currentEntity.parentDescriptor ) {                             // Add its hierarchy
355                 while ( currentEntity.parentDescriptor ) {
356                     ret.push( this.generateSelectConfig ( currentEntity, currentEntity.parentDescriptor, currentEntity.parentDescriptor.get( "items" ) ) );
357                     currentEntity = currentEntity.parentDescriptor;
358                 }
359             }
360             ret.push( this.generateSelectConfig( currentEntity,                 // And finally the root context (entities that are at the root of the gameModel
361             null, rootEntities ) );
362 
363             return ret.reverse();
364         },
365 
366         /**
367          * Generate
368          */
369         generateSelectConfig: function ( entity, parentEntity, items ) {
370             var i, j, value, choices = [];
371 
372             if ( parentEntity ) {
373                 var methods = parentEntity.getMethodCfgs();
374                 for ( j in methods ) {
375                     choices.push({
376                         value: j,
377                         label: j.label || j
378                     });
379                 }
380             }
381 
382             if ( parentEntity && items ) {
383                 choices.push({
384                     value: "----------"
385                 });
386             }
387 
388             if ( items ) {
389                 //if ( items && ( !parentEntity || (parentEntity instanceof Y.Wegas.persistence.ListDescriptor)) ) {
390                 for ( i = 0 ; i < items.length ; i++ ) {
391                     choices.push({
392                         value: items[i].get( "id" ),
393                         label: items[i].get( "editorLabel" )
394                     });
395                 }
396             }
397             if ( parentEntity === this.currentEntity ) {
398                 value = this.options.method
399             }
400             if ( entity ) {
401                 value = entity.get( "id" );
402             }
403             return {
404                 type: 'select',
405                 choices: choices,
406                 value: value,
407                 parentEntity: parentEntity
408             };
409         },
410 
411         /**
412          *  Overriden to allow adding a list of fields at once
413          */
414         addField: function( fieldOptions ) {
415             if ( Y.Lang.isArray( fieldOptions ) ) {
416                 for ( var i = 0; i < fieldOptions.length; i = i + 1 ) {
417                     this.addField( fieldOptions[i] );
418                 }
419             } else {
420                 VariableDescriptorSelect.superclass.addField.call(this, fieldOptions);
421                 this.inputs[ this.inputs.length - 1].options.parentEntity = fieldOptions.parentEntity;
422             }
423         }
424     });
