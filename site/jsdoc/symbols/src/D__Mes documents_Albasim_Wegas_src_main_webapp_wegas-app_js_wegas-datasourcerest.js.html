 68         /**
 69          * Server requests methods
 70          *
 71          * @method sendRequest
 72          */
 73         sendRequest: function ( requestCfg ) {
 74             requestCfg.callback = requestCfg.callback || {
 75                 success: this._successHandler,
 76                 failure: this._failureHandler
 77             };
 78             if ( requestCfg.cfg && Y.Lang.isObject( requestCfg.cfg.data ) ) {
 79                 requestCfg.cfg.data = Y.JSON.stringify( requestCfg.cfg.data )
 80             }
 81             requestCfg.cfg = requestCfg.cfg || {};
 82             requestCfg.cfg.headers =  requestCfg.cfg.headers || {};
 83             Y.mix( requestCfg.cfg.headers, DEFAULTHEADERS );
 84 
 85             return this.get('host').sendRequest(requestCfg);
 86         },

 88         /**
 89          * @method beforeResponse
 90          * @private
 91          */
 92         beforeResponse: function ( e ) {
 93             var evt, i,
 94             response = Y.Wegas.persistence.Editable.revive( e.response.results );// Transform javascript object litterals to Y.Wegas.persistence.Entity's
 95 
 96             Y.log( "Response received from " + this.get( 'host' ).get( 'source' )/* + e.cfg.request*/, "log", "Wegas.RestDataSource");
 97 
 98             e.response.serverResponse = response
 99             e.response.data = this.getCache();                                  // Provides with a pointer to the datasource current content
100 
101             if (e.error) {                                                      // If there was an server error, do not update the cache
102                 return;
103             }
104 
105             if (Lang.isArray(response)) {                                       // Non-managed response: we apply the operation for each object in the returned array
106                 for (i = 0; i < response.length; i += 1) {
107                     this.updateCache(e.cfg.method, response[i]);
108                 }
109             } else {
110                 for (i = 0; i < response.get("entities").length; i += 1) {   // Update the cache with the Entites in the reply body
111                     e.response.entity = response.get("entities")[i];
112                     if (Lang.isObject(e.response.entity)) {
113                         this.updateCache(e.cfg.method, e.response.entity);
114                     }
115                 }
116 
117                 for (i = 0; i < response.get("events").length; i += 1) {
118                     evt = response.get("events")[i];
119                     if (evt instanceof Y.Wegas.persistence.EntityUpdatedEvent) {// Case 1: EntityUpdatedEvent
120                         for (i = 0; i < evt.get("updatedEntities").length; i += 1) {  // Update the cache with the entites contained in the reply
121                             this.updateCache("POST", evt.get("updatedEntities")[i]);
122                         }
123                     }
124                 }
125             }
126 
127         },

129         /**
130          *  @method updateCache
131          *  @param {String} method Possible values for method are: POST, PUT, DELETE, default being PUT.
132          *  @param {entity} The entity to update in the cache
133          *  @return {Boolean} `true` if object could be located and method applied
134          *  @for DataSourceREST
135          */
136         updateCache: function ( method, entity ) {
137             var ret = null;
138             //Y.log("updateCache(" + method + ", " + entity + ")", "log", "Y.Wegas.DataSourceRest");
139             switch (method) {
140                 case "DELETE":
141                     ret = this.find("id", entity, function(entity, needle, index, stack) {
142                         stack.splice(index, 1);
143                         return true;
144                     });
145                     break;
146                 default:
147                     ret = this.find("id", entity, function(entity, needle) {
148                         entity.setAttrs(needle.getAttrs());
149                         return true;
150                     });
151                     break;
152             }
153             if (ret === null) {
154                 this.addToCache(entity);
155             }
156         },
157 
158         addToCache: function (entity) {
159             this.getCache().push(entity);
160         },

164         },

167             try{
168                 console.error(e.response.results.message);
169             }catch(ex){
170                 Y.error("Datasource reply:", e, 'Y.Wegas.DataSourceRest');
171             }
172         },

211         filter: function (key, val) {
212             var data = this.getCache(), ret = [], i;
213             for (i = 0; i < data.length; i += 1) {
214                 if (this.testEntity(data[i], key, val)) {
215                     ret.push(data[i]);
216                 }
217             }
218             return ret;

220         /**
221          *
222          *  Recuresivly walk the provided stack, looking for an object with an
223          *  id corresponing to needle's and apply an operation based method.
224          *
225          *  @method doFind
226          *  @param {String} method Possible values for method are: POST, PUT, DELETE, default being PUT.
227          *  @param {entity} The entity to update in the cache
228          *  @return {Boolean} `true` if object could be located and method applied
229          *  @for DataSourceREST
230          */
231         doFind: function(key, needle, onFindFn, stack) {
232             //Y.log("doFind(" + needle + ")", 'log', 'Y.Wegas.DataSourceRest');
233             var onWalkFn = Y.bind(this.doFind, this, key, needle, onFindFn);
234 
235             this.ret = null;
236 
237             Y.Array.find(stack, function(item, index, array) {
238                 if (this.testEntity(item, key, needle)) {                       // We check the current element if it's a match
239                     if (onFindFn) {
240                         onFindFn(item, needle, index, array);
241                     }
242                     this.ret = item;
243                     return true;
244                 }
245                 return this.walkEntity(item, onWalkFn);
246             }, this);
247             return this.ret;
248         },

524                 }
525             } else if (entity instanceof Y.Wegas.persistence.Player) {

538             } else {
539                 return "/" + data.id;
540             }
541         },

545                     request: "/" + this.getGameByTeamId( parentData.id ).get( "id" )

713             }
