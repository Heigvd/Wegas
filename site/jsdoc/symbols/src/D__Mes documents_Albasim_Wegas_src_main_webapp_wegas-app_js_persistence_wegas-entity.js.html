 39         /**
 40          * Serialize to a json object. Method used <b>recursively</b> by JSON.stringify
 41          *
 42          * @method toJSON
 43          * @return {object}
 44          */
 45         toJSON: function () {
 46             var k, ret = this.getAttrs(),
 47             attrCfgs = this.getAttrCfgs();
 48 
 49             for (k in ret) {
 50                 if ( attrCfgs[ k ][ "transient" ] ) {                           // Remove any transient attribute
 51                     delete ret[ k ];
 52                 }
 53             }
 54             return ret;                                                         // Return a copy of this's fields.
 55         },
 56         /**
 57          * Create a new JSON Object from this entity, filtered out by mask
 58          *
 59          * @fixme This method will parse the field and the clone it
 60          *
 61          * @method toObject
 62          * @param {Array} mask or {String}* a list of params
 63          * @return {Object} a filtered out clone
 64          */
 65         toObject: function(mask){
 66             var e = JSON.parse( JSON.stringify( this  ) );
 67             mask = Y.Lang.isArray( mask ) ? mask : Array.prototype.slice.call( arguments );
 68             return mask.length > 0 ? Y.clone( e, true, function( value, key, output, input){
 69                 if( mask.indexOf ( key ) != -1) {
 70                     return false;
 71                 }else{
 72                     return true;
 73                 }
 74             }) : e;
 75 
 76         },
 77 
 78         /**
 79          * Create a new Object from this entity
 80          * may be used by revive
 81          * @method clone
 82          * @return {Object} a clone
 83          */
 84         clone: function (){
 85             return this.toObject(["id","variableInstances"]);
 86         },
 87         /**
 88          * Returns the form configuration associated to this object, to be used a an inputex object.
 89          */
 90         getFormCfg: function () {
 91             var i, form;
 92             // forms = Y.Wegas.app.get('editorForms'),                          // Select first server defined forms, based on the @class or the type attribute
 93             // form = forms[this.get('@class')] || forms[this.get("type")]
 94 
 95             form = form ||  this.constructor.EDITFORM;                          // And if no form is defined we check if there is a default one defined in the entity
 96 
 97             if (!form) {                                                        // If no edit form could be found, we generate one based on the ATTRS parameter.
 98                 var schemaMap, attrCfgs = this.getAttrCfgs();
 99 
100                 for (i in attrCfgs) {
101                     attrCfgs[i]["default"] = attrCfgs[i].value;                 // Use the value as default (useful form json object serialization)
102 
103                     if ( attrCfgs[i]["transient"] ) {
104                         delete attrCfgs[i];
105                     }
106                 }
107 
108                 schemaMap = {
109                     Entity: {
110                         properties: attrCfgs
111                     }
112                 };
113 
114                 var builder = new Y.inputEx.JsonSchema.Builder({
115                     'schemaIdentifierMap': schemaMap,
116                     'defaultOptions':{
117                         'showMsg':true
118                     }
119                 });
120                 form = builder.schemaToInputEx(schemaMap.Entity);
121             }
122             return form || [];
123         },

125         /**
126          * Returns the edition menu associated to this object, to be used a an inputex object.
127          */
128         getMenuCfg: function ( data ) {
129             var menus = Y.Wegas.app.get('editorMenus'),
130             //    staticMenus =
131             menu;
132 
133             if ( menus ) {
134                 menu =  menus[ this.get( '@class' ) ] || menus[ this.get( "type" ) ];  // Select first server defined forms, based on the @class or the type attribute
135             }
136             menu = menu || this.getStatic("EDITMENU")[0] || [];                 // And if no form is defined we return the default one defined in the entity
137 
138 
139             function mixMenuCfg ( elts, data ) {
140                 var i, j;
141                 for ( i = 0; i < elts.length; i += 1 ) {
142                     elts[i].data = Y.mix( {}, data );// Attach self and the provided datasource to the menu items, to allow them to know which entity to update
143 
144                     if ( elts[i].plugins ) {
145                         for ( j = 0; j < elts[i].plugins.length; j = j + 1 ) {
146                             if (elts[i].plugins[j].cfg && elts[i].plugins[j].cfg.children)
147                                 mixMenuCfg( elts[i].plugins[j].cfg.children, data );
148                         }
149                     }
150                 }
151 
152             }
153 
154             mixMenuCfg( menu, data );
155             return menu;
156         },
157 
158         /**
159              * Returns the edition menu associated to this object, to be used a an wysiwyg editor.
160              */
161         getMethodCfgs: function ( data ) {
162             var menu = this.getStatic("METHODS")[0] || {};
163             return menu;
164         },

166         /**
167              *  Helper function that walks the class hierarchy and returns it's attributes
168              *  cfg (ATTRS), used in Y.Wegas.Entity.getFormCfg().
169              */
170         getAttrCfgs : function() {
171             return this._aggregateAttrs(this.getStatic("ATTRS"));
172         },
173         getStatic: function(key) {
174             var c = this.constructor, ret = [];
175 
176             while (c) {
177                 if (c[key]) {                                                  // Add to attributes
178                     ret[ret.length] = c[key];
179                 }
180                 c = c.superclass ? c.superclass.constructor : null;
181             }
182             return ret;
183         }

187         /**
188              *  This method takes a parsed json object and instantiate them based
189              *  on their @class attribute. Target class are found in namespace
190              *  Y.Wegas.Data.
191              */
192         revive: function ( data ) {
193             var walk = function (o,key) {
194                 var k,v,value = o[key];
195                 if (value && typeof value === "object" ) {
196                     for (k in value) {
197                         if (value.hasOwnProperty(k)) {
198                             v = walk(value, k);
199                             if (v === undefined) {
200                             //delete value[k];
201                             } else {
202                                 value[k] = v;
203                             }
204                         }
205                     }
206                     if ( !Y.Lang.isArray( value ) &&
207                         ( !Y.Lang.isUndefined( value[ "@class" ] ) || !Y.Lang.isUndefined( value[ "type" ] ) )) {
208                         return Y.Wegas.persistence.Editable.readObject( value );
209                     }
210                 }
211                 return value;                                                   // If no value was returned before, return raw original object
212             };
213 
214             //return typeof reviver === 'function' ? walk({'':data},'') : data;
215             return walk({
216                 '': data
217             }, '');
218         },
219         readObject: function (o) {
220             var classDef = Y.Wegas.persistence.Entity;
221 
222             if ( o["@class"] ) {
223                 classDef = Y.Wegas.persistence[ o[ "@class" ] ] || Y.Wegas.persistence.Entity;
224 
225             } else if ( o[ "type" ] ) {
226                 classDef = Y.Wegas.persistence[ o[ "type" ] ] || Y.Wegas.persistence.WidgetEntity;
227 
228             } else {
229                 if ( o[ "@class" ] && o[ "@class" ].indexOf( "Descriptor" ) !== -1) {// @Hack so VariableDescriptors are instantiated even if they dont have a mapping
230                     classDef = Y.Wegas.persistence.VariableDescriptor;
231                 }
232                 if ( o[ "@class" ] && o[ "@class" ].indexOf( "Instance" ) !== -1) {// @Hack so VariableInstances are instantiated even if they dont have a mapping
233                     classDef = Y.Wegas.persistence.VariableInstance;
234                 }
235             }
236             return new classDef(o);
237         }
