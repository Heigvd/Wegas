 15 YUI.add("wegas-inputex-script", function (Y) {
 16 
 17     var inputEx = Y.inputEx;
 18 
 19     inputEx.Script = function (options) {
 20         inputEx.Script.superclass.constructor.call(this, options);
 21     };
 22 
 23     Y.extend(inputEx.Script, inputEx.Textarea, {
 24 
 25         getValue: function () {
 26             return {
 27                 '@class': "Script",
 28                 language: "JavaScript",
 29                 content: inputEx.Script.superclass.getValue.call(this)
 30             };
 31         },
 32 
 33         setValue: function (val) {
 34             val = val || {
 35                 content: ""
 36             };
 37             inputEx.Script.superclass.setValue.call(this, val.content);
 38         }
 39     });
 40 
 41 
 42     //inputEx.registerType('script', inputEx.Script);                               // Register this class as "script" type
 43 });
