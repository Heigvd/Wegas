 15 YUI.add('wegas-crimesim-resultsdisplay', function(Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19             ResultsDisplay;
 20 
 21     /**
 22      *  The results display class.
 23      */
 24     ResultsDisplay = Y.Base.create("wegas-crimesim-resultsdisplay", Y.Widget,
 25             [Y.Wegas.Widget, Y.WidgetChild, Y.Wegas.Editable], {
 26         // *** Fields *** /
 27         menu: null,
 28         handlers: null,
 29         gallery: null,
 30         datatable: null,
 31         // *** Lifecycle Methods *** //
 32         initializer: function() {
 33             this.handlers = {};
 34         },
 35         renderUI: function() {
 36             this.renderDetailsPanel(this.get(CONTENTBOX));
 37         },
 38         bindUI: function() {
 39             var cb = this.get(CONTENTBOX);
 40             this.handlers.playerChange = // If current user changes, refresh (editor only)
 41                     Y.Wegas.app.after('currentPlayerChange', this.syncUI, this);
 42 
 43             this.handlers.response = // If data changes, refresh
 44                     Y.Wegas.app.dataSources.VariableDescriptor.after("response",
 45                     this.syncUI, this);
 46         },
 47         destructor: function() {
 48             this.datatable.destroy();
 49             for (var i in this.handlers) {
 50                 this.handlers[i].detach();
 51             }
 52         },
 53         syncUI: function() {
 54             var data = this.genData();
 55             this.datatable.syncUI(data);
 56         },
 57         renderDetailsPanel: function(node) {
 58             var columns = [{
 59                     key: "choiceDescriptorId",
 60                     className: "hidden"
 61                 }, {
 62                     sortable: true,
 63                     key: "startTime",
 64                     //className: 'hidden',
 65                     label: "Period",
 66                     className: "period"
 67                 }, {
 68                     sortable: true,
 69                     key: "evidence",
 70                     label: "Evidences"
 71                 }, {
 72                     sortable: true,
 73                     key: "analyis",
 74                     label: "Analyse"
 75                 }, {
 76                     key: "answer",
 77                     label: "Result",
 78                     allowHTML: true
 79                 }, {
 80                     sortable: true,
 81                     key: "fileLinks",
 82                     allowHTML: true,
 83                     label: "Files",
 84                     emptyCellValue: "no files"
 85                 }];
 86             this.datatable = new Y.Wegas.CrimeSimTreeble({
 87                 columns: columns,
 88                 isTreeble: true,
 89                 node: node,
 90                 descriptionColumn: 'evidence'
 91             });
 92             this.datatable.render(this.get(CONTENTBOX));
 93         },
 94         genData: function() {
 95             var i, j, k, questionInstance, reply, replyData, status,
 96                     questions = Y.Wegas.VariableDescriptorFacade.rest.find('name', "evidences").get("items"),
 97                     data = [],
 98                     responsesByStartTime = {},
 99                     period = Y.Wegas.VariableDescriptorFacade.rest.find('name', "period"),
100                     periodInstance = period.getInstance(),
101                     currentTime = periodInstance.get("value") - period.get("minValue");
102 
103             for (i = 0; i < questions.length; i = i + 1) {
104                 questionInstance = questions[i].getInstance();
105                 for (j = 0; j < questionInstance.get("replies").length; j = j + 1) {
106                     reply = questionInstance.get("replies")[j];
107                     replyData = Y.mix(reply.getAttrs(), reply.get("result").getAttrs());
108                     replyData.choiceDescriptorId = reply.get('result').get('choiceDescriptorId');
109                     status = reply.getStatus(currentTime);
110 
111                     replyData.evidence = questions[i].get("name");
112                     replyData.analyis = reply.getChoiceDescriptor().get("name");
113 
114                     replyData.startTime += 1;
115 
116                     if (!replyData.description) {
117                         delete replyData.description;
118                     }
119 
120                     if (status === 1) {
121                         replyData.answer = "analysis in progress";
122                     } else if (status === 2) {
123                         replyData.answer = "analysis planified";
124                     } else {
125                         replyData.fileLinks = "";
126                         for (k = 0; k < replyData.files.length; k = k + 1) {
127                             replyData.fileLinks += '<a target="_blank" href="' +
128                                     Y.Plugin.CRDataSource.getFullpath(replyData.files[k]) + '">' +
129                                     Y.Plugin.CRDataSource.getFilename(replyData.files[k]) + '</a><br />';
130                         }
131                         if (!replyData.fileLinks) {
132                             delete replyData.fileLinks;
133                         }
134                     }
135 
136                     if (!responsesByStartTime[reply.get("startTime")]) {
137                         responsesByStartTime[reply.get("startTime")] = [];
138                     }
139                     responsesByStartTime[reply.get("startTime")].push(replyData);
140                 }
141             }
142             for (i in responsesByStartTime) {
143                 data = data.concat(responsesByStartTime[i]);
144             }
145             return data;
146         }
147     });
148 
149     Y.namespace('Wegas').ResultsDisplay = ResultsDisplay;
150 });
