 15 YUI.add('wegas-pageeditor', function(Y) {
 16     "use strict";
 17 
 18     /**
 19      *  @class WidgetMenu
 20      *  @module Wegas
 21      *  @constructor
 22      */
 23     var BOUNDINGBOX = "boundingBox",
 24             CONTENTBOX = "contentBox",
 25             Alignable = Y.Base.create("wegas-pageeditor-overlay", Y.Widget, [Y.WidgetPosition, Y.WidgetPositionAlign, Y.WidgetStack], {
 26         //CONTENT_TEMPLATE: '<div><span class="wegas-icon wegas-icon-edit"></span><div>'
 27     }, {
 28         CSS_PREFIX: "wegas-pageeditor-overlay"
 29     }),
 30     PageEditor = function() {
 31         PageEditor.superclass.constructor.apply(this, arguments);
 32     };
 33 
 34 
 35     Y.extend(PageEditor, Y.Plugin.Base, {
 36         // *** Lifecycle methods *** //
 37         initializer: function() {
 38             this.afterHostEvent("render", this.render);
 39             this.handlers = [];
 40         },
 41         render: function() {
 42             var el, host = this.get('host'), processSource, processSave;
 43 
 44             if (host.toolbar) {
 45                 el = host.toolbar.get('header');
 46                 this.designButton = new Y.ToggleButton({
 47                     label: "<span class=\"wegas-icon wegas-icon-designmode\"></span>Edit page",
 48                     on: {
 49                         click: Y.bind(function(e) {
 50                             this.get("host").get(BOUNDINGBOX).toggleClass("wegas-pageeditor-designmode", e.target.get("pressed"));
 51                             if (e.target.get("pressed")) {
 52                                 this.bind();
 53                             } else {
 54                                 this.detach();
 55                                 this.highlightOverlay.hide();
 56                             }
 57                         }, this)
 58                     }
 59                 }).render(el);
 60                 /** Source view**/
 61                 this.jsonView = new Y.inputEx.AceField({
 62                     parentEl: this.get("host").get("boundingBox")._node,
 63                     name: 'text',
 64                     type: 'ace',
 65                     height: "100%",
 66                     language: "json",
 67                     value: ''
 68                 });
 69                 this.jsonView.hide();
 70                 processSource = function() {
 71                     if (this.sourceButton.get("pressed")) {
 72                         this.jsonView.setValue(JSON.stringify(host.get("widget").toObject("@pageId"), null, "\t"));
 73                         this.get("host").get("contentBox").hide();
 74                         this.jsonView.show();
 75                         this.jsonView.editor.resize();
 76                         this.jsonView.focus();
 77                         this.designButton.disable();
 78                         this.saveButton.show();
 79                     } else {
 80                         this.get("host").get("contentBox").show();
 81                         this.jsonView.hide();
 82                         this.designButton.enable();
 83                         this.saveButton.hide();
 84                     }
 85                 };
 86                 this.sourceButton = new Y.ToggleButton({
 87                     label: "<span class=\"wegas-icon wegas-icon-viewsrc\"></span>Source",
 88                     on: {
 89                         click: Y.bind(processSource, this)
 90                     }
 91                 }).render(el);
 92                 this.afterHostEvent("widgetChange", processSource);
 93                 
 94                 processSave = function (){
 95                     var page = Y.JSON.parse(this.jsonView.getValue());
 96                     this.sourceButton.set("pressed", false);
 97                     this.get("host").get("contentBox").show();
 98                     this.jsonView.hide();
 99                     this.designButton.enable();
100                     this.saveButton.hide();
101                     //host.get("widget").set("@pageId", host.get("widget")["@pageId"]);
102                     page["@pageId"] = host.get("widget")["@pageId"];
103                     Y.Wegas.PageFacade.rest.patch(page);
104                 };
105                 this.saveButton = new Y.Button({
106                     label: "<span class=\"wegas-icon wegas-icon-save\"></span>Save",
107                     on: {
108                         click: Y.bind(processSave, this)
109                     }
110                 }).render(el);
111                 
112                 
113             }
114 
115             this.highlightOverlay = new Alignable({// Init the highlighting overlay
116                 zIndex: 23,
117                 render: true,
118                 visible: false
119             });
120 
121             this.highlightOverlay.plug(Y.Plugin.WidgetMenu, {
122                 children: [{
123                         type: "Button",
124                         label: "Edit",
125                         on: {
126                             click: Y.bind(function() {                             // Display the edit form
127                                 Y.Plugin.EditEntityAction.showEditForm(this.targetWidget, Y.bind(function(targetWidget, val, e, f) {
128                                     Y.Plugin.EditEntityAction.hideEditFormOverlay();
129                                     targetWidget.setAttrs(val);
130                                     targetWidget.syncUI();
131                                     Y.Wegas.PageFacade.rest.patch(targetWidget.get("root").toObject());
132                                 }, this, this.targetWidget));
133                             }, this)
134                         }
135                     }, {
136                         type: "Button",
137                         label: "Delete",
138                         on: {
139                             click: Y.bind(function() {
140                                 var root = this.targetWidget.get("root");
141                                 if (root !== this.targetWidget) {
142                                     this.targetWidget.destroy();
143                                 } else if (this.targetWidget.item(0)) {
144                                     this.targetWidget.removeAll();
145                                 }
146                                 Y.Wegas.PageFacade.rest.patch(root.toObject());
147                             }, this)
148                         }
149                     }],
150                 event: "click"
151                         /*selector: ".wegas-icon"*/
152             });
153 
154             this.highlightOverlay.menu.on("menuOpen", function() {
155                 this.targetWidget = this.overlayWidget;
156             }, this);
157         },
158         bind: function() {
159             var cb = this.get('host').get(CONTENTBOX);
160 
161             //            this.handlers.push(cb.delegate("mouseup", function(e) {
162             //            }, '.wegas-widget', this));
163 
164             this.handlers.push(cb.delegate("click", function(e) {
165                 e.halt(true);
166                 this.targetWidget = this.overlayWidget;
167                 this.highlightOverlay.menu.show();
168                 this.highlightOverlay.menu.menu.set("xy", [e.clientX, e.clientY]);
169                 this.showOverlay(this.overlayWidget);
170                 return false;
171             }, '.wegas-widget', this));
172 
173             this.handlers.push(cb.delegate("mouseover", function(e) {
174                 var widget = Y.Widget.getByNode(e.currentTarget);
175                 e.halt();
176                 if (widget) {
177                     this.showOverlay(widget);
178                 } else {
179                     this.hideOverlay();
180                 }
181             }, '.wegas-widget', this));
182             this.handlers.push(cb.delegate("mouseleave", function(e) {
183                 e.halt();
184                 this.hideOverlay();
185             }, '.wegas-widget', this));
186 
187             /*this.handlers.push(cb.delegate("mouseleave", function(e) {
188              //console.log("out", e.currentTarget.get('id'));
189              this.hideOverlay();
190              e.halt();
191 
192              var parentWidget = Y.Widget.getByNode(e.currentTarget.get('parentNode'));
193              if (parentWidget && parentWidget.get('root') !== parentWidget) {
194              this.showOverlay(parentWidget);
195              }
196              }, '.yui3-widget', this));*/
197         },
198         detach: function() {
199             var i;
200             for (i = 0; i < this.handlers.length; i = i + 1) {
201                 this.handlers[i].detach();
202             }
203         },
204         showOverlay: function(widget) {
205             var targetNode = widget.get(BOUNDINGBOX),
206                     fullHeight = parseInt(targetNode.getComputedStyle("height")) +
207                     parseInt(targetNode.getComputedStyle("padding-top")) +
208                     parseInt(targetNode.getComputedStyle("padding-bottom")) +
209                     parseInt(targetNode.getComputedStyle("border-top-width")) +
210                     parseInt(targetNode.getComputedStyle("border-bottom-width")),
211                     fullWidth = parseInt(targetNode.getComputedStyle("width")) +
212                     parseInt(targetNode.getComputedStyle("padding-left")) +
213                     parseInt(targetNode.getComputedStyle("padding-right")) +
214                     parseInt(targetNode.getComputedStyle("border-left-width")) +
215                     parseInt(targetNode.getComputedStyle("border-right-width"));
216 
217             if (!widget.toObject || this.overlayWidget === widget) {
218                 return;
219             }
220 
221             this.overlayWidget = widget;
222 
223             targetNode.prepend(this.highlightOverlay.get(BOUNDINGBOX));
224             this.highlightOverlay.get(CONTENTBOX).setStyle("height", fullHeight);
225             this.highlightOverlay.get(CONTENTBOX).setStyle("width", fullWidth);
226             this.highlightOverlay.get(CONTENTBOX).setContent("<div>" + widget.constructor.NAME + "</div>");
227             this.highlightOverlay.align(targetNode, [Y.WidgetPositionAlign.CC, Y.WidgetPositionAlign.CC]);
228             this.highlightOverlay.show();
229         },
230         hideOverlay: function() {
231             this.overlayWidget = null;
232             this.highlightOverlay.hide();
233         }
234 
235     }, {
236         NS: "pageeditor",
237         NAME: "pageeditor",
238         ATTRS: {}
239     });
240     Y.namespace('Plugin').PageEditor = PageEditor;
241 
242     Y.Node.prototype.getWidth = function() {
243         return parseInt(this.getComputedStyle('width'));
244         //        + parseInt(this.getComputedStyle('margin-left')) + parseInt(this.getComputedStyle('margin-right'))
245         //  + parseInt(this.getComputedStyle('padding-left')) + parseInt(this.getComputedStyle('padding-right'));
246     };
247     Y.Node.prototype.getHeight = function() {
248         return parseInt(this.getComputedStyle('height'));
249         //      + parseInt(this.getComputedStyle('margin-top')) + parseInt(this.getComputedStyle('margin-bottom'))
250         //  + parseInt(this.getComputedStyle('padding-top')) + parseInt(this.getComputedStyle('padding-bottom'));
251     };
252 });
