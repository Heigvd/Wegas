287             /**
288              * Filters out msword html comments, classes, and other junk
289              * (complementary with YAHOO.widget.SimpleEditor.prototype.filter_msword, when filterWord option is true)
290              * @param {String} html The html string
291              * @return {String} The html string
292              */
293             this.editor.filter_msword = function (html) {
294 
295                 html = editorType.prototype.filter_msword.call(this,html);
296 
297                 // if we don't filter ms word junk
298                 if (!this.get('filterWord')) {
299                     return html;
300                 }
301 
302                 html = html.replace( /<!--[^>][\s\S]*-->/gi, ''); // strip (meta-)comments
303                 html = html.replace( /<\/?meta[^>]*>/gi, ''); // strip meta tags
304                 html = html.replace( /<\/?link[^>]*>/gi, ''); // strip link tags
305                 html = html.replace( / class=('|")?MsoNormal('|")?/gi, ''); // strip MS office class
306                 html = Y.Lang.trim(html); // trim spaces
307 
308                 return html;
309             };

601                 var id = /v=[^&]*&/.exec( url )[0].substr( 2 );
602                 return '<object type="application/x-shockwave-flash" style="width:' + width +'px; height:' + height + 'px;" data="http://www.youtube.com/v/' + id + '?version=3">' +
