  5 YUI.add('wegas-list', function (Y) {
  6     "use strict";
  7 
  8     var BOUNDINGBOX = 'boundingBox',
  9     CONTENTBOX = 'contentBox',
 10     List;
 11 
 12     List = Y.Base.create("wegas-list", Y.Widget, [Y.WidgetParent, Y.WidgetChild,  Y.Wegas.Widget ], {
 13 
 14 
 15         // ** Lifecycle Methods ** //
 16 
 17         syncUI: function () {
 18             var cb = this.get(CONTENTBOX);
 19 
 20             if (this.get('direction') === 'vertical') {
 21                 cb.addClass(this.getClassName('vertical'));
 22                 cb.removeClass(this.getClassName('horizontal'));
 23             } else {
 24                 cb.addClass(this.getClassName('horizontal'));
 25                 cb.removeClass(this.getClassName('vertical'));
 26             }
 27             this.get(BOUNDINGBOX).append('<div class="list-clearing" style="clear:both"></div>');
 28         }
 29     }, {
 30         ATTRS : {
 31             defaultChildType: {
 32                 value: "Text"
 33             },
 34             direction: {
 35                 value: 'vertical'
 36             }
 37 
 38             /**
 39              * Prevent widgetchild selection to be propagated through the hierarchy
 40              */
 41             //selected: {
 42             //    value: 2,
 43             //    readonly: true
 44             //}
 45         }
 46     });
 47 
 48     Y.namespace('Wegas').List = List;
 49 });
