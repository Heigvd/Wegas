 15 YUI.add("wegas-inputex-pageselect", function (Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx;
 19 
 20     /**
 21      * Adds an url regexp, and display the favicon at this url
 22      * @class inputEx.UrlField
 23      * @extends inputEx.StringField
 24      * @constructor
 25      * @param {Object} options inputEx.Field options object
 26      * <ul>
 27      *   <li>favicon: boolean whether the domain favicon.ico should be displayed or not (default is true, except for https)</li>
 28      * </ul>
 29      */
 30     Y.namespace("inputEx.Wegas").PageSelect = function (options) {
 31         inputEx.Wegas.PageSelect.superclass.constructor.call(this, options);
 32     };
 33 
 34     Y.extend(inputEx.Wegas.PageSelect, inputEx.SelectField, {
 35         
 36         setOptions: function (options) {
 37             inputEx.Wegas.PageSelect.superclass.setOptions.call(this, options);
 38             Y.Wegas.PageFacade.rest.getIndex(Y.bind(this.buildList, this));
 39             
 40         },
 41         
 42         setValue: function (val) {
 43             inputEx.Wegas.PageSelect.superclass.setValue.apply(this, arguments);
 44             this.options.value = val;
 45         },
 46         
 47         buildList: function(value){
 48             var i;
 49             for(i in value){
 50                 this.addChoice({
 51                     value: i,
 52                     label: "Page : " + i
 53                 });
 54                 if (i == this.options.value){
 55                     this.choicesList[i-1].node.selected = "selected"
 56                 }     
 57             }
 58         }
 59     });
 60 
 61     inputEx.registerType("pageselect", inputEx.Wegas.PageSelect);               // Register this class as "wegasurl" type
 62 });
