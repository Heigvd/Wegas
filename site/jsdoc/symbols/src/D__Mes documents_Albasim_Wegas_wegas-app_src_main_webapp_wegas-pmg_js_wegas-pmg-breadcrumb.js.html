 14 YUI.add("wegas-pmg-breadcrumb", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Breadcrumb;
 18 
 19     Breadcrumb = Y.Base.create("wegas-pmg-breadcrumb", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 20         handlers: null,
 21         initializer: function () {
 22             this.handlers = {};
 23         },
 24         renderUI: function () {
 25             var i, node, cb = this.get(CONTENTBOX), locations = this.get("locations");
 26             if (locations.length == 0)
 27                 return;
 28             node = Y.Node.create("<div class='pmg-breadcrumb'></div>");
 29             for (i = 0; i < locations.length; i++) {
 30                 node.append("<span class='element_" + i + "'>" + locations[i] + "</span>");
 31             }
 32             cb.append(node);
 33         },
 34         bindUI: function () {
 35             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this);
 36         },
 37         syncUI: function () {
 38             var i, cb = this.get(CONTENTBOX), locations = this.get("locations"), varValue,
 39                     varDesc = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variable"));
 40             if (locations.length == 0 || varDesc == null)
 41                 return;
 42             for (i = 0; i < locations.length; i++) {
 43                 cb.one(".pmg-breadcrumb span").removeClass("previous").removeClass("current").removeClass("next");
 44             }
 45             varValue = varDesc.getInstance().get("value") - varDesc.get("minValue");
 46             if (typeof varValue === "string") {
 47                 for (i = 0; i < locations.length; i++) {
 48                     if (locations[i] === varValue)
 49                         cb.one(".pmg-breadcrumb .element_" + i).addClass("current");
 50                     if (locations[i] < varValue)
 51                         cb.one(".pmg-breadcrumb .element_" + i).addClass("previous");
 52                     if (locations[i] > varValue)
 53                         cb.one(".pmg-breadcrumb .element_" + i).addClass("next");
 54                 }
 55             }
 56             else if (typeof varValue === "number") {
 57                 if (varValue > locations.length || varValue < 0)
 58                     return;
 59                 for (i = 0; i < locations.length; i++) {
 60                     if (i === varValue)
 61                         cb.one(".pmg-breadcrumb .element_" + i).addClass("current");
 62                     if (i < varValue)
 63                         cb.one(".pmg-breadcrumb .element_" + i).addClass("previous");
 64                     if (i > varValue)
 65                         cb.one(".pmg-breadcrumb .element_" + i).addClass("next");
 66                 }
 67             }
 68         },
 69         destructor: function () {
 70             var k;
 71             for (k in this.handlers) {
 72                 this.handlers[k].detach();
 73             }
 74         }
 75 
 76     }, {
 77         ATTRS: {
 78             locations: {
 79                 validator: Y.Lang.isArray
 80             },
 81             variable: {
 82                 type: "String",
 83                 value: null
 84             }
 85         }
 86     });
 87 
 88     Y.namespace("Wegas").PmgBreadcrumb = Breadcrumb;
 89 });
