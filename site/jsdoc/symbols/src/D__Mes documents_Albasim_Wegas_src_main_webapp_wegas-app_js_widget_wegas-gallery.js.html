 13 YUI.add("wegas-gallery", function(Y){
 14 
 15     var WegasGallery,
 16     CONTENT_BOX="contentBox",
 17     BOUNDING_BOX="boundingBox";
 18 
 19     /*
 20      * ATTRS:<br/>
 21      * gallery: {Array} an array of picture to display, format : {srcUrl: ...[, description: ...]}
 22      * fullScreen : {Boolean} toggles fullScreen<br/>
 23      * lightGallery : {Boolean} An instance which will read DOM for "light-picture" or "light-gallery" classes<br/>
 24      * <br/>
 25      * light-picture class nodes : "href" attribute (url) , "title" attribute (description)<br/>
 26      * light-gallery class nodes : child nodes with "href" attribute (url), "title" attribute (description)
 27      */
 28     WegasGallery = Y.Base.create("wegas-gallery", Y.Widget, [Y.Wegas.Widget], {
 29         CONTENT_TEMPLATE:"<ul></ul>",
 30 
 31         scrollView: null,
 32         container:null,
 33         fullScreeNode:null,
 34         styleSheet: null,
 35         isFullScreen: false,
 36         eventInstances:[],
 37         images:{},
 38 
 39         initializer : function(){
 40             var prev = Y.Widget.getByNode(".wegas-lightGallery");
 41             if(prev){               // TODO: currently destroying previous one, do a singleton
 42                 prev.destroy();
 43             }
 44             this.images = {};
 45             this.eventInstances = [];
 46             this.isFullScreen = false;
 47             this.fullScreenNode = new Y.Node.create("<span></span>");
 48             this.scrollView = new Y.ScrollView({
 49                 width: (parseInt(this.get("selectedWidth")) + 30),
 50                 srcNode:this.get(BOUNDING_BOX),
 51                 axis: "x",
 52                 flick:{
 53                     minDistance:10,
 54                     minVelocity:0.3,
 55                     axis: "x"
 56                 },
 57                 bounce:0
 58             });
 59 
 60             if(this.get("lightGallery")){
 61                 this.set("render", "body");
 62                 this.set("gallery", []);
 63                 this.render();
 64             }
 65 
 66         },
 67         renderUI : function(){
 68             // LAZY init
 69             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-mask gallery-mask-left'><div>PREVIOUS</div></div>");
 70             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-mask gallery-mask-right'><div>NEXT</div></div>");
 71             this.scrollView.get(BOUNDING_BOX).append("<div class='gallery-toggle'></div>");
 72             this.get(CONTENT_BOX).appendChild("<li class='img-loading'></li>");
 73             if(this.get("lightGallery")){
 74                 this.scrollView.get(BOUNDING_BOX).addClass("wegas-lightGallery");
 75             }else if(this.get("gallery").length > 0){
 76                 this.loadImage(0);
 77             }
 78             this.fullScreenNode.appendTo(Y.one("body"));
 79 
 80         },
 81         syncUI: function(){
 82             if(!this.scrollView.get("rendered")){
 83                 return;
 84             }
 85             var scrollViewId = "#" + this.scrollView.get("id") + " ",           // prefix css with id, allow multiple instance with different style
 86             selW = parseInt(this.get("selectedWidth")),
 87             selH = this.get("selectedHeight"),
 88             smaH,
 89             smaW;
 90 
 91             if(this.styleSheet){
 92                 this.styleSheet.disable();
 93             }
 94             if(this.get("fullScreen") || this.get("lightGallery")){
 95                 selH = Y.one("body").get("winHeight") - 100;
 96                 selW = Y.one("body").get("winWidth") * 0.8;
 97                 if(!this.isFullScreen){
 98                     this.isFullScreen = true;
 99                     this.scrollView.get(BOUNDING_BOX).swap(this.fullScreenNode);
100                 }
101             }else{
102                 if(this.isFullScreen){
103                     this.isFullScreen = false;
104                     this.scrollView.get(BOUNDING_BOX).swap(this.fullScreenNode);
105                 }
106             }
107             smaW = (this.scrollView.get(BOUNDING_BOX).get("parentNode").get("region").width - parseInt(this.scrollView.get(BOUNDING_BOX).get("parentNode").getStyle("padding-right")) - parseInt(this.scrollView.get(BOUNDING_BOX).get("parentNode").getStyle("padding-left")) - selW - 90)/2;
108             smaH = selW;
109             if(this.get("gallery").length > 0){
110                 this.scrollView.show();
111             }else{
112                 this.scrollView.hide();
113                 return;
114             }
115             this.scrollView.set("width", selW + 30);
116             this.styleSheet = new Y.StyleSheet(scrollViewId + ".wegas-gallery li{width:"
117                 + (selW + 30)
118                 + "px;height:"
119                 + (selH + 30)
120                 + "px;}"
121                 + scrollViewId
122                 + ".wegas-gallery img{max-width:"
123                 + (smaW)
124                 + "px;max-height:"
125                 + (smaH)
126                 + "px;}"
127                 + scrollViewId
128                 + ".wegas-gallery .gallery-selected img{max-width:"
129                 + (selW)
130                 + "px;max-height:"
131                 + (selH)
132                 + "px;}"
133                 + scrollViewId
134                 + ".gallery-mask{width:"
135                 + (smaW + 30)
136                 + "px;}"
137                 + scrollViewId
138                 + ".gallery-text{width:"
139                 + (selW + 24)
140                 + "px;max-height:"
141                 + (selH)
142                 + "px;}");
143             this.scrollView.get(BOUNDING_BOX).setStyles({
144                 padding:"0 "+ (smaW+45) + "px 0 " + (smaW+15) + "px"
145             });
146 
147             this.scrollView.syncUI();
148             if(this.scrollView.get("rendered")){
149                 this.scrollView.pages.scrollToIndex(this.scrollView.pages.get("index"));
150             }
151             if(this.get("lightGallery") && !this.get("fullScreen")){
152                 this.scrollView.hide();
153             }
154         },
155         bindUI: function(){
156             this.after("fullScreenChange", function(e){
157                 if(this.get("lightGallery")){
158                     if(this.get("fullScreen")){
159                         this.scrollView.show();
160                     }else{
161                         this.scrollView.hide();
162                     }
163                 }else{
164                     this.syncUI();
165                 }
166             });
167 
168             this.eventInstances.push(this.scrollView.get(BOUNDING_BOX).one('.gallery-mask-left > div').on("click", function(e){
169                 e.halt(true);
170                 this.prev();
171 
172             }, this));
173             this.eventInstances.push(this.scrollView.get(BOUNDING_BOX).one('.gallery-mask-right > div').on("click", function(e){
174                 e.halt(true);
175                 this.next();
176             }, this));
177             this.eventInstances.push(this.scrollView.get(BOUNDING_BOX).one('.gallery-toggle').on("click", function(e){
178                 this.set("fullScreen", !this.get("fullScreen"));
179             }, this));
180             //Prevent scroll on borders
181             this.eventInstances.push(this.scrollView.get(BOUNDING_BOX).one('.gallery-mask-left').on("mousedown", function(e){
182                 e.halt(true);
183             }));
184             this.eventInstances.push(this.scrollView.get(BOUNDING_BOX).one('.gallery-mask-right').on("mousedown", function(e){
185                 e.halt(true);
186             }));
187             this.scrollView.after("scrollEnd", function(e){
188                 this.scrollView.pages.scrollToIndex(this.scrollView.pages.get("index"));              //Constrain
189                 Y.later(600, this, function(){                                  // Let some time before loading next one
190                     var index = this.scrollView.pages.get("index");
191                     if(this.images[index + 1] && !this.images[index + 1].loaded){
192                         this.loadImage(index + 1);
193                     }
194                     if(this.images[index - 1] && !this.images[index - 1].loaded){
195                         this.loadImage(index - 1);
196                     }
197                 });
198 
199             }, this);
200             this.eventInstances.push( Y.on("windowresize", Y.bind(this.windowResizeEvent, this)));
201             this.after("galleryChange", function(e){
202                 if(e.newVal.length > 0){
203                     this.loadImage(0);
204                     if(e.newVal.length > 1){
205                         this.loadImage(1);
206                     }
207                 }
208                 if(!this.scrollView.get("rendered")){
209                     this.scrollView.plug(Y.Plugin.ScrollViewPaginator, {
210                         selector: 'li'
211                     });
212 
213                     this.scrollView.render();
214                     this.scrollView.pages.after("indexChange", function(e){
215                         this.setSelected(e.target.get("index"));
216                     }, this);
217                 }else{
218                     this.scrollView.syncUI();
219 
220                 }
221                 this.setSelected(0);
222                 this.scrollView.pages.set("index", 0);
223                 this.syncUI();
224             });
225             if(this.get("lightGallery")){
226                 this.eventInstances.push(Y.one("body").delegate("click", function(e){
227                     var children, gallery = [];
228                     e.halt(true);
229                     children = e.target.get("children");
230                     children.each(function(){
231                         if(this.getAttribute("href")){
232                             gallery.push({
233                                 srcUrl:this.getAttribute("href"),
234                                 description:this.getAttribute("title")
235                             });
236                         }
237                     });
238                     this.set("fullScreen", true);
239                     this.set("gallery", gallery);
240 
241                 },'.light-gallery', this));
242                 this.eventInstances.push(Y.one("body").delegate("click", function(e){
243                     e.halt(true);
244 
245                     this.set("fullScreen", true);
246                     this.set("gallery", [{
247                         srcUrl:e.target.getAttribute("href"),
248                         description:e.target.getAttribute("title")
249                     }]);
250                 },'.light-picture', this));
251             }
252         },
253         windowResizeEvent:function(){
254             if(this.get("fullScreen")){
255                 this.scrollView.hide();
256                 this.syncUI();
257                 this.scrollView.show();
258             }
259         },
260         next: function(){
261             this.scrollView.pages.next();
262         },
263         prev: function(){
264             this.scrollView.pages.prev();
265         },
266         setSelected: function(index){
267             var list = this.get(CONTENT_BOX).all("li");
268             list.each(function(){
269                 this.removeClass("gallery-selected");
270                 this.removeClass("gallery-before-selected");
271             });
272             if(index > 0){
273                 list.item(index - 1).addClass("gallery-before-selected");
274             }
275             if(list.size()> index){
276                 list.item(index).addClass("gallery-selected");
277             }
278             if(this.images[index] && !this.images[index].loaded){
279                 this.loadImage(index);
280             }
281 
282         },
283         loadImage: function(i){
284             var img;
285             if(this.get(CONTENT_BOX).all("li").item(i).hasChildNodes()){
286                 return;
287             }
288             img = Y.Node.create("<img src='"+this.get("gallery")[i].srcUrl+"' ></img>");
289 
290             img.index = i;
291             img.once("error", function(e){
292                 e.target.get("parentNode").setStyles({
293                     background:"url('../../wegas-app/images/wegas-icon-error-48.png') no-repeat 50%"
294                 });
295             });
296             img.once("load", function(e){
297                 e.target.get("parentNode").removeClass("img-loading");
298                 this.images[e.target.index].loaded = true;
299             }, this);
300             this.get(CONTENT_BOX).all("li").item(i).appendChild(img);
301             if(this.get("gallery")[i].description) {
302                 this.get(CONTENT_BOX).all("li").item(i).appendChild("<div class='gallery-text'>"+this.get("gallery")[i].description+"</div>");
303             }
304         },
305         destructor: function(){
306             if(this.styleSheet){
307                 this.styleSheet.disable();
308             }
309             this.fullScreenNode.destroy();
310             for ( var i = 0; i < this.eventInstances.length; i = i + 1 ){
311                 this.eventInstances[i].detach();
312             }
313         }
314 
315     }, {
316         ATTRS:{
317             gallery:{
318                 value:[],
319                 lazyAdd : true,
320                 setter: function(o){
321                     this.get(CONTENT_BOX).empty();
322                     this.images = {};
323                     for(var i in o){
324                         if ( Y.Lang.isString( o[i] ) ) {
325                             o[i] = {                                            // If the arguent is a string, treat it as an url
326                                 srcUrl: o[i]
327                             }
328                         }
329                         this.images[i] = {
330                             loaded : false
331                         };
332                         this.get(CONTENT_BOX).appendChild("<li class='img-loading'></li>");
333                     }
334                     return o;
335                 }
336             },
337             selectedWidth:{
338                 value:400,
339                 validator: Y.Lang.isNumber,
340                 setter: function(v){
341                     if(this.scrollView){
342                         this.scrollView.set("width", (parseInt(v) + 30))
343                     }
344                     return v;
345                 }
346             },
347             selectedHeight:{
348                 value:250,
349                 validator: Y.Lang.isNumber
350             },
351             fullScreen:{
352                 value:false,
353                 validator:Y.Lang.isBoolean
354             },
355             lightGallery:{
356                 value:false,
357                 writeOnce: "initOnly",
358                 validator:Y.Lang.isBoolean
359             }
360 
361         }
362     });
363 
364     Y.namespace("Wegas").WegasGallery = WegasGallery;
365 
366     var FileExplorerGallery = Y.Base.create("wegas-gallery", WegasGallery, [], {
367 
368         loadImage: function(i) {
369             var img, imgLoader;
370             if(this.get(CONTENT_BOX).all("li").item(i).hasChildNodes()){
371                 return;
372             }
373 
374             img = Y.Node.create("<img />");
375             imgLoader = new Y.Wegas.ImgageLoader({
376                 target: img,
377                 srcUrl: Y.Plugin.CRDataSource.getFullpath( this.get("gallery")[i].srcUrl )
378             });
379             imgLoader.fetch();
380             imgLoader.once( "load", function ( e ) {
381                 if ( e.meta.description ) {
382                     this.get(CONTENT_BOX).all("li").item(i).appendChild("<div class='gallery-text'>" + e.meta.description + "</div>");
383                 }
384             }, this );
385             img.index = i;
386             img.once("error", function(e){
387                 e.target.get("parentNode").setStyles({
388                     background:"url('../../wegas-editor/images/wegas-icon-error-48.png') no-repeat 50%"
389                 });
390             });
391             img.once("load", function(e){
392                 e.target.get("parentNode").removeClass("img-loading");
393                 this.images[e.target.index].loaded = true;
394             }, this);
395             this.get(CONTENT_BOX).all("li").item(i).appendChild(img);
396 
397         }
398     });
399 
400     Y.namespace("Wegas").FileExplorerGallery = FileExplorerGallery;
