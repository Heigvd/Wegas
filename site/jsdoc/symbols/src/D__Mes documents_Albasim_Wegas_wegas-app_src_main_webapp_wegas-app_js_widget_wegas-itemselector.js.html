  5 YUI.add('wegas-itemselector', function (Y) {
  6     "use strict";
  7 
  8     var CONTENTBOX = 'contentBox', ItemSelector;
  9 
 10     ItemSelector = Y.Base.create("wegas-itemselector", Y.Widget, [Y.Wegas.Widget, Y.Wegas.Editable, Y.Wegas.NodeFormatter], {
 11         handlers: null,
 12         currentItem: null,
 13         scrollView: null,
 14         // *** Lifecycle Methods *** //
 15         initializer: function () {
 16             this.handlers = {};
 17         },
 18         /**
 19          * Render the widget.
 20          */
 21         renderUI: function () {
 22             var i, variables, cb = this.get(CONTENTBOX);
 23             cb.append('<div class="selectors"></div>');
 24             cb.append('<div class="informations"></div>');
 25             if (!this.get('listVariables')) {
 26                 return;
 27             }
 28             variables = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get('listVariables'));
 29             if (!variables || !variables.get('items')) {
 30                 return;
 31             }
 32             for (i = 0; i < variables.get('items').length; i++) {
 33                 if (variables.get('items')[i].getInstance().get('active') == null || variables.get('items')[i].getInstance().get('active') == true) {
 34                     this.currentItem = variables.get('items')[i];
 35                     break;
 36                 }
 37             }
 38             this.scrollView = new Y.ScrollView({
 39                 id: 'itemselector-scrollview',
 40                 srcNode: cb.one('.selectors'),
 41                 width: cb.one('.selectors').offsetwidth,
 42                 flick: {
 43                     minDistance: 20,
 44                     minVelocity: 0.6,
 45                     axis: 'x'
 46                 }
 47             });
 48         },
 49         /**
 50          * Bind some function at nodes of this widget
 51          */
 52         bindUI: function () {
 53             var cb = this.get(CONTENTBOX);
 54             this.handlers.itemSelectorUpdate = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 55 
 56             this.handlers.itemSelectorSelect = cb.one('.selectors').delegate('click', function (e) {
 57                 var i, variables, name;
 58                 if (e.target.ancestors('.selector').item(0)) {
 59                     name = e.target.ancestors('.selector').item(0).getAttribute("data-name");
 60                 } else {
 61                     name = e.target.getAttribute("data-name");
 62                 }
 63                 variables = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get('listVariables'));
 64                 if (!variables || !variables.get('items'))
 65                     return;
 66                 for (i = 0; i < variables.get('items').length; i++) {
 67                     if (variables.get('items')[i].get('name') === name) {
 68                         this.currentItem = variables.get('items')[i];
 69                         break;
 70                     }
 71                 }
 72                 this.syncUI();
 73             }, '.selector', this);
 74 
 75             this.handlers.itemSelectorPreventDefault = cb.one('.selectors').delegate('click', function (e) {
 76                 e.preventDefault();
 77             }, '.selector', this);
 78 
 79         },
 80         /**
 81          * Synchronise the content of this widget.
 82          */
 83         syncUI: function () {
 84             var cb = this.get(CONTENTBOX), variables;
 85             if (!this.get('listVariables'))
 86                 return;
 87             variables = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get('listVariables'));
 88             if (!variables || !variables.get('items') || variables.get('items').length <= 0)
 89                 return;
 90             this.createSelector(cb, variables.get('items'));
 91             this.createInformations(cb);
 92         },
 93         /*
 94          * Destroy all child widget and all remanent function
 95          */
 96         destructor: function () {
 97             var k;
 98             for (k in this.handlers) {
 99                 this.handlers[k].detach();
100             }
101         },
102         createSelector: function (cb, variables) {
103             var i, node = cb.one('.selectors'), selector;
104             node.empty();
105             for (i = 0; i < variables.length; i++) {
106                 if (variables[i].getInstance().get('active') == null || variables[i].getInstance().get('active') == true) {
107                     selector = Y.Node.create('<div class="selector" data-name="' + variables[i].get('name') + '"></div>');
108                     if (variables[i] === this.currentItem) {
109                         selector.addClass('current');
110                     }
111                     this.createDOMProperties(selector, variables[i], this.get('selectors'));
112                     node.append(selector);
113                 }
114             }
115             this.scrollView.render();
116         },
117         createInformations: function (cb) {
118             var node = cb.one('.informations');
119             node.empty();
120             this.createDOMProperties(node, this.currentItem, this.get('informations'));
121         },
122         createDOMProperties: function (node, variable, attrs) {
123             var i, type, value, label, className, obj, child;
124             if (!node || !variable || !attrs)
125                 return;
126             for (i = 0; i < attrs.length; i++) {
127                 obj = attrs[i];
128                 if (typeof obj === 'object') {
129                     value = this.getVariableValue(variable, obj['name']);
130                     value = (value != null) ? value : obj['name'];
131                     label = (obj['label'] || null);
132                     type = (obj['type'] || 'undefine');
133                     className = (obj['className'] || null);
134                     switch (type) {
135                         case 'image' :
136                             child = this.makeNodeImage({
137                                 "data-file": value,
138                                 "width": obj['height'],
139                                 "height": obj['width']
140                             }, className);
141                             break;
142                         case 'external-image' :
143                             child = this.makeNodeImage({
144                                 "src": value,
145                                 "width": obj['height'],
146                                 "height": obj['width']
147                             }, className);
148                             break;
149                         case 'position' :
150                             child = this.makeNodePosition(obj['html'], obj['selector'], value, obj['minVal'], obj['invert'], className);
151                             break;
152                         case 'valueBox' :
153                             child = this.makeNodeValueBox('' + value, obj['maxValue'], label, className);
154                             break;
155                         default :
156                             child = this.makeNodeText(value, label, className);
157                             break;
158                     }
159                     child.setAttribute('data-name', obj['name']);
160                     node.append(child);
161                 }
162             }
163         },
164         getVariableValue: function (variable, varName) {
165             var i, prop = this.get('searchInProperties'), value = null;
166             if (!variable || !varName)
167                 return value;
168             if (variable.get(varName) != null) {
169                 value = variable.get(varName);
170             } else if (variable.getInstance().get(varName) != null) {
171                 value = variable.getInstance().get(varName);
172             } else {
173                 for (i = 0; i < prop.length; i++) {
174                     if (variable.getInstance().get(prop[i]) != null && variable.getInstance().get(prop[i])[varName] != null) {
175                         value = variable.getInstance().get(prop[i])[varName];
176                         break;
177                     }
178                 }
179             }
180             if (parseFloat(value)) {
181                 value = parseFloat(value);
182             }
183             return value;
184         }
185 
186     }, {
187         ATTRS: {
188             listVariables: {
189                 value: null,
190                 validator: function (s) {
191                     return s === null || Y.Lang.isString(s);
192                 }
193             },
194             searchInProperties: {
195                 validator: Y.Lang.isArray,
196                 value: []
197             },
198             selectors: {
199                 validator: Y.Lang.isArray,
200                 value: []
201             },
202             informations: {
203                 validator: Y.Lang.isArray,
204                 value: []
205             }
206         }
207     });
208 
209     Y.namespace('Wegas').ItemSelector = ItemSelector;
210 });
