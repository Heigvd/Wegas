 11 /**

 15 YUI.add('wegas-app', function (Y) {
 16     "use strict";
 17      
 18     /**
 19     * @name Y.Wegas.App
 20     * @class  Base class for wegas, handle initialisation of datasources and rendering
 21     * @constructor
 22     * @param Object cfg
 23     * @description create a new wegas-app
 24     */
 25     var App = Y.Base.create("wegas-app", Y.Base, [ ], {
 26 
 27         // ** Private fields ** //
 28         /**
 29          * Holds a reference to all the dataSources used.
 30          */
 31         dataSources: [],
 32 
 33         /** 
 34          * @methodOf Y.Wegas.App#
 35          * @private
 36          * @name initializer
 37          * @description Lifecycle methods
 38          */
 39         initializer: function () {
 40             Y.Wegas.app = this;
 41             this.injector = new Y.Wegas.Injector({
 42                 observe:"#maindisplayarea"
 43             });
 44             /**
 45              * @memberOf Y.Wegas.App#
 46              * @name render
 47              * @event 
 48              * @description render event 
 49              */
 50             this.publish( "render", {});
 51         },
 52 
 53         /** 
 54          * @methodOf Y.Wegas.App#
 55          * @private
 56          * @name destructor
 57          * @description destructor methods.
 58          */
 59         destructor : function () {
 60             for (var i = 0; i < this.dataSources.length; i = i + 1) {
 61                 this.dataSources[i].destroy();
 62             }
 63         },
 64 
 65         /**
 66          * @methodOf Y.Wegas.App#
 67          * @name render
 68          * @description render function
 69          */
 70         render: function () {
 71 
 72             Y.io.header( "Accept-Language", Y.config.lang);                      // Set the language for all requests
 73             this.on( "render", function () {
 74                 Y.one( "body" ).removeClass( "wegas-widget-loading" );
 75             });
 76 
 77             this.initDataSources();
 78             this.initCSS();
 79         },
 80 
 81         // *** Private methods ** //
 82         /**
 83          * @methodOf Y.Wegas.App#
 84          * @private
 85          * @name initDataSources
 86          * @description initilize DataSources
 87          */
 88         initDataSources: function () {
 89             var k, dataSource, dataSources = this.get('dataSources');
 90 
 91             // @todo Shall we use browser native parser ?
 92             // Y.JSON.useNativeParse = true;
 93 
 94             this.requestCounter = 0;
 95 
 96             for (k in dataSources) {
 97                 if (dataSources.hasOwnProperty(k)) {
 98                     dataSources[k].source = this.get("base") + dataSources[k].source;
 99                     dataSource = new Y.Wegas.DataSource(dataSources[k]);
100                     this.dataSources[k] = this[k + "Facade"] = Y.Wegas[k + "Facade"] = dataSource;
101                     dataSource.once( "response", this.onInitialRequest, this );
102                     if ( Y.Lang.isNumber( dataSource.sendInitialRequest() ) ) { // Send an initial request
103                         this.requestCounter += 1;                               // If the request was sent, we update the counter, which is used n the onInitialRequest() callback
104                     }
105                 }
106             }
107 
108             if (this.requestCounter == 0) {                                     // If no request was sent, render directly
109                 this.renderUI();
110             }
111         },
112 
113         /**
114          * @methodOf Y.Wegas.App#
115          * @private
116          * @name onInitialRequest
117          * @param Y.Event A Yui event
118          * @description Listen to the datasources initial requests and run the renderUI()
119          * method when they all arrived.
120          */
121         onInitialRequest: function ( e ) {
122             this.requestCounter -= 1;
123             if (this.requestCounter == 0) {
124                 this.renderUI();
125             }
126         },
127 
128         /**
129          * @methodOf Y.Wegas.App#
130          * @private
131          * @name initCSS
132          * @description Inizilize CSS
133          */
134         initCSS: function () {
135             var i, css = this.get('cssStylesheets'),
136             cfg = {
137                 timeout : 3000,
138                 context: this,
139                 on : {
140                     success : function ( id, o ) {
141                         this._customCSSText = o.responseText;
142                         this._customCSSStyleSheet = new Y.StyleSheet( o.responseText );
143                         //Y.log("RAW JSON DATA: " + o.responseText);
144                         //this.updateCustomCSS(o.responseText);
145                         if (this._customCSSForm) {
146                             this._customCSSForm.setValue( o.responseText );
147                         }
148                     },
149                     failure : function ( id, o ) {
150                         Y.log( "initCSS(): Page CSS loading async call failed!", 'error', 'Wegas.App');
151                     }
152                 }
153             };
154 
155             for ( i = 0; i < css.length; i += 1 ) {
156                 Y.io( this.get('base') + css[ i ] + '?id=' + App.genId(), cfg );// Load the page css
157             }
158         },
159 
160         /**
161          * @methodOf Y.Wegas.App#
162          * @private
163          * @name renderUi
164          * @description renderUI methods
165          */
166         renderUI: function() {
167             Y.io(this.get('base') + this.get('layoutSrc') + '?id=' + App.genId(), {
168                 context: this,
169                 on: {
170                     success: function (id, o, args) {
171                         //Y.log("RedCMS.onWidgetReloadContentReceived():"+  o.responseText, 'log');
172                         var cfg;
173                         try {
174                             cfg = Y.JSON.parse( o.responseText );		// Process the JSON data returned from the server
175                         } catch (e) {
176                             alert( "Wegas.App.initUI(): JSON Parse failed!" );
177                             return;
178                         }
179 
180                         Y.Wegas.Widget.use(cfg, Y.bind( function ( cfg ) {      // Load the subwidget dependencies
181                             this.widget = Y.Wegas.Widget.create( cfg );         // Render the subwidget
182                             this.widget.render();
183                             this.fire( "render" );                              // Fire a render event for some eventual post processing
184                         }, this, cfg));
185 
186                     //this.pageLoader = new Y.Wegas.PageLoader();               // Load the subwidget using pageloader
187                     //this.pageLoader.render();
188                     // cfg.id = -100;
189                     // this.dataSources.Page.data.push(cfg);
190                     //try {
191                     //    this.pageLoader.set("pageId", -100);
192                     //} catch (renderException) {
193                     //    Y.log('initUI(): Error rendering UI: ' + ((renderException.stack) ? renderException.stack : renderException), 'error', 'Wegas.App');
194                     //}
195 
196                     }
197                 }
198             });
199         }
200     }, {
201         /**
202          * @memberOf Y.Wegas.App#
203          * @name attrributes
204          * @description
205          * <p><strong>Method</strong></p>
206          * <ul>
207          *    <li>base() : Base Url for app</li>
208          *    <li>layoutSrc() : xxxxxxxxxxxxxxx</li>
209          *    <li>dataSources() : xxxxxxxxxxxxxxx</li>
210          *    <li>cssStylesheets() : xxxxxxxxxxxxxxx</li>
211          *    <li>currentGameModel() : xxxxxxxxxxxxxxx</li>
212          *    <li>currentGame() : xxxxxxxxxxxxxxx</li>
213          *    <li>currentTeam() : xxxxxxxxxxxxxxx</li>
214          *    <li>currentPlayer() : xxxxxxxxxxxxxx</li>
215          *    <li>currentUser() : Object litteral representing current user</li>
216          * </ul>
217          */
218         ATTRS: {
219             /**
220              * Base url for app
221              */
222             base: {
223                 getter: function () {
224                     return Y.config.groups.wegas.base;
225                 }
226             },
227             layoutSrc: {},
228             dataSources: {
229                 value: {}
230             },
231             cssStylesheets: {
232                 value: []
233             },
234             currentGameModel: {},
235             currentGame: {},
236             currentTeam: {},
237             currentPlayer: {
238                 setter: function (val) {
239                     var cPlayer = this.dataSources.Game.rest.getPlayerById(val);
240                     if (cPlayer) {
241                         // @fixme
242                         this.set('currentTeam', cPlayer.get("teamId"));              // When current player is updated, we also update current team
243                     }
244                     return val;
245                 }
246             },
247             /**
248              * Object litteral representing current user.
249              */
250             currentUser: { }
251         },
252         /**
253          * @methodOf Y.Wegas.App
254          * @static
255          * @name genId
256          * @return {integer} time
257          * @description generate ID
258          */
259         genId: function () {
260             var now = new Date();
261             return now.getHours() + now.getMinutes() + now.getSeconds();
262         },
263         /**
264          * @methodOf Y.Wegas.App
265          * @static
266          * @name htmlEntities
267          * @param str String
268          * @return {String} Escaped string
269          * @description Escape all necessary character
270          */
271         htmlEntities: function ( str ) {
272             return String(str).replace(/&/g, '&').replace(/</g, '<').replace(/>/g, '>').replace(/"/g, '"');
273 
274         }
275     });
276 
277     Y.namespace('Wegas').App = App;
278 });
