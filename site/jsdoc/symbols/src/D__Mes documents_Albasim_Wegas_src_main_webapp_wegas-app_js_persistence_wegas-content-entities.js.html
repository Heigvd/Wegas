 11 Y.add("wegas-content-entities", function(Y){
 12 
 13     Y.Wegas.persistence.Content = Y.Base.create("Content", Y.Wegas.persistence.Entity, [], {}, {
 14         ATTRS:{
 15             "@class":{
 16                 "transient":true
 17             },
 18             id:{
 19                 "transient":true
 20             },
 21             mimeType:{
 22                 type:"string",
 23                 _inputex:{
 24                     _type:"uneditable"
 25                 }
 26             },
 27             name:{
 28                 type:"string",
 29                 _inputex:{
 30                     _type:"uneditable"
 31                 }
 32             },
 33             path:{
 34                 type:"string",
 35                 _inputex:{
 36                     _type:"uneditable"
 37                 }
 38             },
 39             note:{
 40                 type:"string",
 41                 optional:true,
 42                 _inputex:{
 43                     label: "Private notes",
 44                     _type:"text"
 45                 }
 46             },
 47             description:{
 48                 type:"string",
 49                 optional:true,
 50                 _inputex:{
 51                     label: "Public description",
 52                     _type:"text"
 53                 }
 54             }
 55         }
 56     });
 57 
 58     Y.Wegas.persistence.Directory = Y.Base.create("Directory", Y.Wegas.persistence.Content, [], {}, {
 59         ATTRS:{
 60             "@class":{
 61                 value:"Directory"
 62             }
 63         }
 64     });
 65 
 66     Y.Wegas.persistence.File = Y.Base.create("File", Y.Wegas.persistence.Content, [], {}, {
 67         ATTRS:{
 68             "@class":{
 69                 value:"File"
 70             },
 71             bytes:{
 72                 writeOnce:"initOnly",
 73                 setter:function(bytes){
 74                     var precision = 2,
 75                     sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'],
 76                     i = parseInt(Math.floor(Math.log(bytes) / Math.log(1024)));
 77                     return (bytes / Math.pow(1024, i)).toFixed(precision) + ' ' + sizes[i];
 78                 },
 79                 _inputex:{
 80                     label:"Size",
 81                     _type:"uneditable"
 82                 }
 83             },
 84             dataLastModified:{
 85                 writeOnce:"initOnly",
 86                 setter: function(d){
 87                     var date = new Date(d);
 88                     return date.toLocaleString();
 89                 },
 90                 _inputex:{
 91                     label:"File last uploaded",
 92                     _type:"uneditable"
 93                 }
 94             }
 95         }
 96     });
 97 });
