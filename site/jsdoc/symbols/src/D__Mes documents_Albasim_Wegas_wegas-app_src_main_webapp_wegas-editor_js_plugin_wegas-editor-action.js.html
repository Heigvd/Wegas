 14 YUI.add('wegas-editor-action', function (Y) {
 15     "use strict";
 16     /**
 17      *  @class Action
 18      *  @module Wegas
 19      *  @constructor
 20      */
 21     var Action = Y.Plugin.Action;
 22 
 23     /**
 24      *  @class EditFSMAction
 25      *  @module Wegas
 26      *  @constructor
 27      */
 28     var EditFSMAction = function () {
 29         EditFSMAction.superclass.constructor.apply(this, arguments);
 30     };
 31 
 32     Y.extend(EditFSMAction, Y.Plugin.EntityAction, {
 33         execute: function () {
 34             Y.Wegas.TabView.findTabAndLoadWidget("State machine editor",        // Load and display the editor in a new tab
 35                 "#centerTabView", null, {
 36                     type: "StateMachineViewer",
 37                     plugins: [{
 38                         fn: "WidgetToolbar"
 39                     }]
 40                 }, Y.bind(function (entity, widget) {
 41                     widget.set("entity", entity);
 42                 }, this, this.get("entity")));
 43         }
 44     }, {
 45         NS: "wegas",
 46         NAME: "EditFSMAction"
 47     });
 48 
 49     Y.namespace("Plugin").EditFSMAction = EditFSMAction;
 50 
 51     /**

 73     /**
 74      *  @class OpenTabAction
 75      *  @module Wegas
 76      *  @constructor
 77      */
 78     var OpenTabAction = function () {
 79         OpenTabAction.superclass.constructor.apply(this, arguments);
 80     };
 81 
 82     Y.extend(OpenTabAction, Action, {
 83         execute: function () {
 84             var childCfg = this.get("children")[0];                             // @fixme currently we only render the first child
 85             Y.Wegas.TabView.findTabAndLoadWidget(this.get("host").get("label"),
 86                 this.get("tabSelector"), {}, childCfg);                         // Forward plugin data to the target widget
 87         }
 88     }, {
 89         NS: "wegas",
 90         NAME: "OpenTabAction",
 91         ATTRS: {
 92             tabSelector: {
 93                 value: '#centerTabView'
 94             },
 95             children: {
 96                 value: []
 97             }
 98         }
 99     });
100     Y.namespace("Plugin").OpenTabAction = OpenTabAction;
101 
102     /**
103      *  @class OpenGameAction
104      *  @module Wegas
105      *  @constructor
106      */
107     var OpenGameAction = function () {
108         OpenGameAction.superclass.constructor.apply(this, arguments);
109     };
110 
111     Y.extend(OpenGameAction, Y.Plugin.OpenUrlAction, {
112         execute: function () {
113             var params, entity = this.get("entity");
114 
115             if (entity instanceof Y.Wegas.persistence.GameModel) {
116                 params = "gameModelId=" + entity.get("id");
117             } else if (entity instanceof Y.Wegas.persistence.Player) {
118                 params = "id=" + entity.get("id");
119             } else {
120                 params = "gameId=" + entity.get("id");
121             }
122             this.set("url",  this.get("editorUrl") + params);
123             OpenGameAction.superclass.execute.call(this);
124         }
125     }, {
126         NS: "wegas",
127         NAME: "OpenGameAction",
128         ATTRS: {
129             editorUrl: {
130                 value: 'wegas-app/view/editor.html?'
131             },
132             entity: {}
133         }
134     });
135 
136     Y.namespace("Plugin").OpenGameAction = OpenGameAction;
137 
138     /**
139      *  @class LoadTreeviewNodeAction
140      *  @module Wegas
141      *  @constructor
142      */
143     var LoadTreeviewNodeAction = function () {
144         LoadTreeviewNodeAction.superclass.constructor.apply(this, arguments);
145     };
146 
147     Y.extend(LoadTreeviewNodeAction, Action, {
148         execute: function () {
149             var entity = this.get("entity"),
150             tabId = this.get("tabId") || this.get("host").get("label"),
151             tabCfg = {
152                 label: entity.get("name") || "Unnamed"
153             },
154             tab = Y.Wegas.TabView.createTab(tabId, this.get("tabSelector"), tabCfg);
155 
156             tab.set("selected", 2);
157 
158             tab.witem(0).set("emptyMessage", "This game model has no games.");
159             tab.witem(0).toolbar.item(0).set("disabled", false);  // Allow game creation
160 
161             Y.Wegas.GameFacade.set("source",                                    // Change the source attribute on the datasource
162                 Y.Wegas.app.get("base") + "rest/GameModel/" + entity.get("id") + "/Game");
163 
164             Y.Wegas.GameFacade.sendRequest({
165                 request: "/"
166             });
167         }
168     }, {
169         NS: "LoadTreeviewNodeAction",
170         NAME: "LoadTreeviewNodeAction",
171         ATTRS: {
172             tabId: {},
173             tabSelector: {
174                 value: '#centerTabView'
175             },
176             entity: {}
177         }
178     });
179 
180     Y.namespace("Plugin").LoadTreeviewNodeAction = LoadTreeviewNodeAction;

193     /**
194     * @name Y.Wegas.Linkwidget
195     * @extends Y.Widget
196     * @class  class for display the player link in menu's
197     * @constructor
198     * @param Object Will be used to fill attributes field 
199     * @description Allows to display the player link in a menu. 
200     * the link is in a textField. For this field inputEx is used
201     */
202     var CONTENTBOX = 'contentBox',
203     Linkwidget;
204 
205     Linkwidget = Y.Base.create("wegas-playerlink-buttons", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
206 
207         /**
208          * @lends Y.Wegas.Linkwidget#
209          */
210 
211         /**
212          * @function
213          * @private
214          * @description 
215          * 1) Add a <div class="playerlink-label"><p>Player link</p><div> node fordisplay a label in the menu
216          * 2) Add the inputeExStringField
217          * 3) Stop the click event on this contentbox
218          */
219         renderUI: function(){
220             Linkwidget.superclass.renderUI.apply(this);
221 
222             this.p = Y.Node.create('<div class="playerlink-label"><p>Player link</p><div>');
223             this.get(CONTENTBOX).append(this.p);
224 
225             this.textField = new Y.inputEx.StringField({
226                 parentEl: this.get(CONTENTBOX)
227             });
228             this.get(CONTENTBOX).on("click", function (e) {
229                 e.halt(true);
230             });
231         },
232         
233         /**
234          * @function
235          * @private
236          * @description Add the new url
237          */
238         syncUI: function() {
239             var url = Y.Wegas.app.get("base") + "game.html?token=" + this.get("entity").get("token");
240             this.textField.setValue(url);
241         }
242     }, {
243         /**
244          * @lends Y.Wegas.Linkwidget
245          */
246         /**
247          * @field
248          * @static
249          * @description
250          * <p><strong>Method</strong></p>
251          * <ul>
252          *    <li>entity: get the entity</li>
253          * </ul>
254          */
255         ATTRS: {
256             entity: {}
257         }
258     });
259     Y.namespace("Wegas").Linkwidget = Linkwidget;
260 });
