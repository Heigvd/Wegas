 16 YUI.add('wegas-panelwidget', function (Y) {
 17     "use strict";
 18 
 19     var PanelWidget;
 20 
 21     /**
 22     * @name Y.Wegas.PanelWidget
 23     * @extends Y.Panel
 24     * @class  class for creating panel from a JSON file with children
 25     * @constructor
 26     * @param Object Will be used to fill attributes field 
 27     * @description Create a panel with all children
 28     */
 29     PanelWidget = Y.Base.create("wegas-panelwidget", Y.Panel, [Y.WidgetChild, Y.Wegas.Widget], {
 30         /**
 31          * @lends Y.Wegas.PanelWidget#
 32          */
 33         /**
 34          * @function
 35          * @private
 36          * @description Instantiates all children widget
 37          */
 38         syncUI: function () {
 39             PanelWidget.superclass.syncUI.apply(this);
 40             Y.Array.forEach(this.get("children"), function (child, i){
 41                 var widget = Y.Wegas.Widget.create(child);
 42                 widget.render(this.get("contentBox"));
 43                 }, this);
 44         }
 45     },{
 46         /**
 47          * @lends Y.Wegas.PanelWidget
 48          */
 49         /**
 50          * @field
 51          * @static
 52          * @description
 53          * <p><strong>Method</strong></p>
 54          * <ul>
 55          *    <li>chlidren: All children widget used</li>
 56          *    <li>zIndex : define a default zIndex</li>
 57          * </ul>
 58          */
 59         ATTRS: {
 60             children: {
 61                 value: []
 62             },            
 63             zIndex: {
 64                 value: 6
 65             }
 66         }
 67     }
 68     );
 69 
 70     Y.namespace('Wegas').PanelWidget = PanelWidget;
 71 });
