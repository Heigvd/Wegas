 14 YUI.add("wegas-pmg-slidepanel", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", SlidePanel;
 18 
 19     SlidePanel = Y.Base.create("wegas-pmg-slidepanel", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 20         handlers: null,
 21         list: null,
 22         animation: null,
 23         cleaner: null,
 24         initializer: function () {
 25             this.handlers = {};
 26             this.list = new Y.Wegas.List({
 27                 "children": this.get("children")
 28             });
 29         },
 30         renderUI: function () {
 31             var cb = this.get(CONTENTBOX), node;
 32             node = Y.Node.create("<div class='slidepanel' style='position:relative;'></div>")
 33             node.append("<div class='slidepanel-title'><h2>" + this.get('title') + "</h2></div>");
 34             node.append("<div class='slidepanel-content' style='position:absolute;'></div>");
 35             node.append("<div class='slidepanel-cleaner' style='position:relative; z-index:-1;'></div>");
 36             this.list.render(node.one('.slidepanel-content'));
 37 
 38             this.animation = node.one('.slidepanel-content').plug(Y.Plugin.NodeFX, {//slide animation
 39                 from: {
 40                     height: 0
 41                 },
 42                 to: {
 43                     height: function (node) { // dynamic in case of change
 44                         return node.get('scrollHeight'); // get expanded height (offsetHeight may be zero)
 45                     }
 46                 },
 47                 easing: Y.Easing.easeOut,
 48                 duration: 0.5
 49             }, this);
 50 
 51             this.cleaner = node.one('.slidepanel-cleaner').plug(Y.Plugin.NodeFX, {//compensates the non-height of the content's absolute position.
 52                 from: {
 53                     height: 0
 54                 },
 55                 to: {
 56                     height: function (node) {
 57                         return node.ancestor().one('.slidepanel-content').get('scrollHeight');
 58                     }
 59                 },
 60                 easing: Y.Easing.easeOut,
 61                 duration: 0.5
 62             }, this);
 63             cb.append(node);
 64         },
 65         bindUI: function () {
 66             var cb = this.get(CONTENTBOX);
 67             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 68             
 69             this.handlers.anim = cb.one(".slidepanel-title").on('click', function (e) {
 70                 this.animation.fx.set('reverse', !this.animation.fx.get('reverse')); // toggle reverse 
 71                 this.animation.fx.run();
 72                 this.cleaner.fx.set('reverse', !this.cleaner.fx.get('reverse')); // toggle reverse 
 73                 this.cleaner.fx.run();
 74             }, this);
 75         },
 76         syncUI: function () {
 77             var cb = this.get(CONTENTBOX), height;
 78             height = cb.one('.slidepanel-content').get('scrollHeight');
 79             cb.one('.slidepanel-cleaner').setStyle('height', height); //compensates the non-height of the content's absolute position.
 80         },
 81         destructor: function () {
 82             var k;
 83             for (k in this.handlers) {
 84                 this.handlers[k].detach();
 85             }
 86             this.list.destroy();
 87             this.animation.destroy();
 88             this.cleaner.destroy();
 89         }
 90 
 91     }, {
 92         ATTRS: {
 93             title: {
 94                 value: "unnamed",
 95                 validator: function (s) {
 96                     return s === null || Y.Lang.isString(s);
 97                 }
 98             },
 99             children: {
100                 validator: Y.Lang.isArray
101             }
102         }
103     });
104 
105     Y.namespace("Wegas").PmgSlidePanel = SlidePanel;
106 });
