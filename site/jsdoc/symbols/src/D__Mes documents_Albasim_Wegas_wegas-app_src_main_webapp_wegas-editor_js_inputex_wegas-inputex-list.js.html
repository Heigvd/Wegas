 15 YUI.add("wegas-inputex-list", function (Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx;
 19 
 20     /**
 21      *  Adds a method that retrieves the value of each input in the group
 22      *  (unlike Y.inputEx.Group.getValue() that returns an object based on
 23      *  inputs names):
 24      */
 25     Y.inputEx.Group.prototype.getArray = function () {
 26         var i, ret = [];
 27         for (i = 0; i < this.inputs.length; i =  i + 1) {
 28             ret.push(this.inputs[i].getValue());
 29         }
 30         return ret;
 31     };
 32 
 33     /**
 34      * @class ListField
 35      * @constructor
 36      * @extends inputEx.Group
 37      * @param {Object} options InputEx definition object
 38      */
 39     var ListField = function (options) {
 40         ListField.superclass.constructor.call(this, options);
 41 
 42         var parentNode = new Y.Node(this.fieldset);
 43         //parentNode.insert(this.addButton.get("boundingBox").remove(), 1);
 44         this.addButton.render(this.fieldset);
 45         parentNode.prepend(this.addButton.get("boundingBox"));
 46     };
 47     Y.extend(ListField, inputEx.Group, {
 48 
 49         /**
 50 	 * Set the ListField classname
 51 	 * @param {Object} options Options object as passed to the constructor
 52 	 */
 53         setOptions: function (options) {
 54             ListField.superclass.setOptions.call(this, options);
 55             this.options.className = options.className || 'inputEx-Field inputEx-ListField';
 56             this.options.addType = options.addType || "variabledescriptorsetter";
 57         },
 58 
 59         /**
 60 	 * Render the addButton
 61 	 */
 62         render: function () {
 63             ListField.superclass.render.call(this);
 64 
 65             this.addButton = new Y.Wegas.Button({
 66                 label: "<span class=\"wegas-icon wegas-icon-add\"></span>"
 67             });
 68             this.addButton.get("boundingBox").addClass("wegas-addbutton");
 69         },
 70         /**
 71          *
 72          */
 73         destroy: function () {
 74             this.addButton.destroy();
 75             ListField.superclass.destroy.call(this);
 76         },
 77         /**
 78 	 * Handle the click event on the add button
 79 	 */
 80         initEvents: function () {
 81             ListField.superclass.initEvents.call(this);
 82             this.addButton.on("click", this.onAdd, this);
 83         },
 84 
 85         renderField: function (fieldOptions) {
 86             var fieldInstance = ListField.superclass.renderField.call(this, fieldOptions),
 87             removebutton = new Y.Wegas.Button({
 88                 label: '<span class="wegas-icon wegas-icon-remove"></span>'
 89             });
 90 
 91             removebutton.targetField = fieldInstance;
 92             removebutton.get("boundingBox").addClass("wegas-removebutton");
 93             removebutton.render(fieldInstance.divEl);
 94             removebutton.on("click", this.onRemove, this);
 95 
 96             return fieldInstance;
 97         },
 98 
 99         onRemove: function (e) {
100             var i = Y.Array.indexOf(this.inputs, e.target.targetField),
101             d = this.inputs[i];
102             d.destroy();
103             this.inputs.splice(i, 1);
104             this.fireUpdatedEvt();
105         },
106 
107         onAdd: function (e) {
108             this.addField({
109                 type: this.options.addType
110             });
111             this.fireUpdatedEvt();
112         },
113 
114         /**
115          * Override to disable
116          */
117         runInteractions: function () { }
118 
119     });
120 
121     inputEx.registerType("listfield", ListField);                               // Register this class as "list" type
122 
123     /**
124      * @class ListField
125      * @constructor
126      * @extends Y.Wegas.ListField
127      * @param {Object} options InputEx definition object
128      */
129     var EditableList = function (options) {
130         EditableList.superclass.constructor.call(this, options);
131     };
132     Y.extend(EditableList, ListField, {
133 
134         /**
135 	 * Set the ListField classname
136 	 * @param {Object} options Options object as passed to the constructor
137 	 */
138         setOptions: function (options) {
139             EditableList.superclass.setOptions.call(this, options);
140             this.options.fields = options.fields || [];
141             this.options.items = options.items || [];
142         },
143 
144         setValue: function (value, fireUpdatedEvent) {
145             //EditableList.superclass.setValue.apply(this, arguments);
146             this.clear();
147             var i;
148             for (i = 0; i < value.length; i += 1) {
149                 this.addPluginField(value[i].fn, value[i].cfg);
150             }
151 
152             if (fireUpdatedEvent) {
153             //this.fireUpdatedEvent();
154             }
155         },
156 
157         addPluginField: function(fn, value) {
158             var cfg, targetPlg = Y.Plugin[fn],
159             w = new Y.Wegas.Text();                                              // Use this hack to retrieve a plugin config
160             w.plug(targetPlg);
161             cfg = w[targetPlg.NS].getFormCfg();
162             cfg.value = value;
163             Y.inputEx.use(w[targetPlg.NS].getFormCfg(), Y.bind(function (cfg) {
164                 this.addField(cfg);
165                 this.fireUpdatedEvt();
166             }, this, cfg, value));
167         },
168 
169         /**
170 	 * Handle the click event on the add button
171 	 */
172         initEvents: function () {
173             EditableList.superclass.initEvents.call(this);
174             /*var ttplugin = new Y.Plugin.Tooltip;*/
175 
176             this.addButton.plug(Y.Plugin.WidgetMenu, {
177                 children: this.options.items
178             });
179             this.addButton.menu.on("button:click", function (e) {
180                 this.addPluginField(e.target.get("data"));
181             }, this);
182 
183         },
184         /**
185          * Override to prevent field creation on click
186          */
187         onAdd: function (e) {},
188 
189         /**
190          * Override to disable
191          */
192         runInteractions: function () {}
193 
194     });
195 
196     inputEx.registerType("editablelist", EditableList);                               // Register this class as "list" type
197 });
