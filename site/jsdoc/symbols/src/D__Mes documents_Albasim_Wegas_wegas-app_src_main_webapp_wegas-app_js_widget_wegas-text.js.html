 14 YUI.add("wegas-text", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Text;
 18 
 19     Text = Y.Base.create("wegas-text", Y.Widget, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable ], {
 20         syncUI: function () {
 21             this.set("content", this.get("content"));
 22         }
 23     }, {
 24         ATTRS : {
 25             content: {
 26                 type: "string",
 27                 format: "html",
 28                 setter: function (val) {
 29                     this.get(CONTENTBOX).setContent(val);
 30                     return val;
 31                 }
 32             }
 33         }
 34     });
 35 
 36     Y.namespace("Wegas").Text = Text;
 37 });
