129 		}

131

134 			var fillColor = jsPlumbUtil.convertStyle(style.fillStyle, true);
135 			styleToWrite["fillcolor"] = fillColor;
136 			_maybeSetOpacity(styleToWrite, fillColor, "fill", component);
137 		}
138 		else styleToWrite["filled"] = "false";
139 		
140 		if(style["dashstyle"]) {
141 			if (component.strokeNode == null) {
142 				component.strokeNode = _node("stroke", [0,0,0,0], { dashstyle:style["dashstyle"] }, node, _jsPlumb);				
143 			}
144 			else
145 				component.strokeNode.dashstyle = style["dashstyle"];
146 		}					
147 		else if (style["stroke-dasharray"] && style["lineWidth"]) {
148 			var sep = style["stroke-dasharray"].indexOf(",") == -1 ? " " : ",",
149 			parts = style["stroke-dasharray"].split(sep),
150 			styleToUse = "";
151 			for(var i = 0; i < parts.length; i++) {
152 				styleToUse += (Math.floor(parts[i] / style.lineWidth) + sep);
153 			}
154 			if (component.strokeNode == null) {
155 				component.strokeNode = _node("stroke", [0,0,0,0], { dashstyle:styleToUse }, node, _jsPlumb);

160 		}

368 	var AbstractVmlArrowOverlay = function(superclass, originalArgs) {
369     	superclass.apply(this, originalArgs);
370     	VmlComponent.apply(this, originalArgs);
371     	var self = this, path = null;
372     	self.canvas = null; 
373     	self.isAppendedAtTopLevel = true;
374     	var getPath = function(d, connectorDimensions) {    		
375     		return "m " + _conv(d.hxy.x) + "," + _conv(d.hxy.y) +
376     		       " l " + _conv(d.tail[0].x) + "," + _conv(d.tail[0].y) + 
377     		       " " + _conv(d.cxy.x) + "," + _conv(d.cxy.y) + 
378     		       " " + _conv(d.tail[1].x) + "," + _conv(d.tail[1].y) + 
379     		       " x e";
380     	};
381     	this.paint = function(connector, d, lineWidth, strokeStyle, fillStyle, connectorDimensions) {
382     		var p = {};
383 			if (strokeStyle) {
384 				p["stroked"] = "true";
385 				p["strokecolor"] = jsPlumbUtil.convertStyle(strokeStyle, true);    				
386 			}
387 			if (lineWidth) p["strokeweight"] = lineWidth + "px";
388 			if (fillStyle) {
389 				p["filled"] = "true";
390 				p["fillcolor"] = fillStyle;
391 			}
392 			var xmin = Math.min(d.hxy.x, d.tail[0].x, d.tail[1].x, d.cxy.x),
393 			ymin = Math.min(d.hxy.y, d.tail[0].y, d.tail[1].y, d.cxy.y),
394 			xmax = Math.max(d.hxy.x, d.tail[0].x, d.tail[1].x, d.cxy.x),
395 			ymax = Math.max(d.hxy.y, d.tail[0].y, d.tail[1].y, d.cxy.y),
396 			w = Math.abs(xmax - xmin),
397 			h = Math.abs(ymax - ymin),
398 			dim = [xmin, ymin, w, h];
399 			
400 			// for VML, we create overlays using shapes that have the same dimensions and
401 			// coordsize as their connector - overlays calculate themselves relative to the
402 			// connector (it's how it's been done since the original canvas implementation, because
403 			// for canvas that makes sense).
404 			p["path"] = getPath(d, connectorDimensions);
405 			p["coordsize"] = (connectorDimensions[2] * scale) + "," + (connectorDimensions[3] * scale);
406 			
407 			dim[0] = connectorDimensions[0];
408 			dim[1] = connectorDimensions[1];
409 			dim[2] = connectorDimensions[2];
410 			dim[3] = connectorDimensions[3];
411 			
412     		if (self.canvas == null) {
413     			//p["class"] = jsPlumb.overlayClass; // TODO currentInstance?
414 				self.canvas = _node("shape", dim, p, connector.canvas.parentNode, connector._jsPlumb);								
415 				connector.appendDisplayElement(self.canvas, true);
416 				self.attachListeners(self.canvas, connector);
417 				self.attachListeners(self.canvas, self);
418 			}
419 			else {				
420 				_pos(self.canvas, dim);
421 				_atts(self.canvas, p);
422 			}    		
423     	};
424     	
425     	this.reattachListeners = function() {
426 			if (self.canvas) self.reattachListenersForElement(self.canvas, self);
427 		};
