 14 YUI.add("wegas-pmg-tasklist", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Tasklist;
 18 
 19     Tasklist = Y.Base.create("wegas-pmg-tasklist", Y.Wegas.PmgDatatable, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 20         handlers: null,
 21         /*** Lifecycle methods ***/
 22         initializer: function () {
 23             this.handlers = {};
 24         },
 25         renderUI: function () {
 26             Tasklist.superclass.renderUI.apply(this);
 27         },
 28         bindUI: function () {
 29             Tasklist.superclass.bindUI.apply(this);
 30             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 31 
 32             this.handlers.sort = this.datatable.after('sort', this.syncUI, this);
 33 
 34             this.handlers.displayDescription = this.datatable.delegate('click', function (e) {
 35                 this.displayDescription(e);
 36             }, '.yui3-datatable-data td', this);
 37 
 38             this.handlers.removeDescription = this.datatable.delegate('mouseout', function (e) {
 39                 this.removeDescription(e);
 40             }, '.yui3-datatable-data tr', this);
 41         },
 42         syncUI: function () {
 43             Tasklist.superclass.syncUI.apply(this);
 44             this.checkRealization();
 45         },
 46         destructor: function () {
 47             var k;
 48             for (k in this.handlers) {
 49                 this.handlers[k].detach();
 50             }
 51         },
 52         //*** Private Methods ***/
 53         checkRealization: function () {
 54             var i, cb = this.get(CONTENTBOX), tasks, taskDesc, taskInst, realized, allRow;
 55             if (this.data == null
 56                     || this.data.length == 0
 57                     || this.get("columnValues").indexOf('realized') <= -1) {
 58                 return;
 59             }
 60             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
 61             allRow = cb.all(".yui3-datatable-data tr");
 62             allRow.removeClass("notstarted").removeClass("started").removeClass("completed");
 63             allRow.each(function (node) {
 64                 for (i = 0; i < tasks.get('items').length; i += 1) {
 65                     taskDesc = tasks.get('items')[i];
 66                     taskInst = taskDesc.getInstance();
 67                     realized = (taskInst.get('properties').realized) ? taskInst.get('properties').realized : null;
 68                     if (realized) {
 69                         if (node.one("*").getContent() == taskDesc.get('id')) {
 70                             if (realized >= 100) {
 71                                 node.addClass("completed");
 72                             } else if (realized > 0) {
 73                                 node.addClass("started");
 74                             } else {
 75                                 node.addClass("notstarted");
 76                             }
 77                             break;
 78                         }
 79                     }
 80                 }
 81             });
 82         },
 83         displayDescription: function (e) {
 84             var i, id, label, tasks, node, divDesc, taskDesc, description;
 85             node = e.currentTarget;
 86             if (this.get("viewDescription") == "false"
 87                     || node.one(".description")
 88                     || node.get("className").indexOf("cell-gantt") > -1) {
 89                 return;
 90             }
 91             id = node.ancestor().one("*").getContent();
 92             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
 93             if (!id || !tasks) {
 94                 return;
 95             }
 96             for (i = 0; i < tasks.get('items').length; i += 1) {
 97                 taskDesc = tasks.get('items')[i];
 98                 if (taskDesc.get('id') == id) {
 99                     label = (taskDesc.get("label") || taskDesc.get("name"));
100                     description = taskDesc.get("description");
101                     break;
102                 }
103             }
104             divDesc = Y.Node.create("<div class='description'></div>");
105             divDesc.append("<p class='task_name'>" + label + "</p>").append("<p class='content'>" + description + "</p>");
106             node.append(divDesc);
107         },
108         removeDescription: function (e) {
109             var disappearAnim, node;
110             node = this.get(CONTENTBOX).one('.description');
111             if (!node) {
112                 return;
113             }
114             disappearAnim = new Y.Anim({
115                 node: node,
116                 to: {
117                     opacity: 0
118                 },
119                 duration: 0.2
120             });
121             disappearAnim.run();
122             disappearAnim.on('end', function () {
123                 node.remove();
124             });
125         }
126     }, {
127         ATTRS: {
128             viewDescription: {
129                 value: true,
130                 validator: function (b) {
131                     return b == "false" || b == "true";
132                 }
133             },
134             viewAssignements: {//todo
135                 value: true,
136                 validator: function (b) {
137                     return b == "false" || b == "true";
138                 }
139             }
140         }
141     });
142 
143     Y.namespace("Wegas").PmgTasklist = Tasklist;
144 });
