  5 YUI.add('wegas-cep-nodeformatter', function (Y) {
  6     "use strict";
  7 
  8     var CONTENTBOX = 'contentBox', NodeFormatter;
  9 
 10     NodeFormatter = Y.Base.create("wegas-cep-nodeformatter", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 11         
 12         makeNodeText: function(value, label, className){
 13             var node = Y.Node.create('<div class="nodeformatter-properties"></div>');
 14             value = (value || 'undefine');
 15             if(className) node.addClass(className);
 16             if (label) node.append('<span class="label">'+label+'</span>');
 17             node.append('<span class="value">'+value+'</span>');
 18             return node;
 19         },
 20         
 21         makeNodeImage: function(attrs, className){
 22             var k, node = new Y.Node.create('<img class="nodeformatter-img"></img>');
 23             if(className) node.addClass(className);
 24             if(typeof attrs !== 'object') return node;
 25             for(k in attrs){
 26                 node.set(k, attrs[k])
 27             }
 28             return node;
 29         },
 30         
 31         makeNodeValueBox: function(value, maxVal, label, className){
 32             var i, acc = [], node = new Y.Node.create('<div class="nodeformatter-valuebox"></div>');
 33             value = (value || 'undefine');
 34             maxVal = (maxVal || 'undefine');
 35             label = (label || 'undefine');
 36             if(className) node.addClass(className);
 37             for (i=0; i<value; i++) {
 38                 acc.push('<div class="box-unit"></div>');
 39             }
 40             node.append('<div class="label">'+label+'</div>');
 41             node.append('<span class="box-units">'+acc.join('')+'</span>');
 42             node.append('<span class="box-value">('+value+'<span class="box-valueMax">/'+maxVal+'</span>)</span>');
 43             return node
 44         },
 45         
 46         makeNodePosition: function(html, selector, value, invert, className){
 47             var node = new Y.Node.create('<div class="nodeformatter-position"></div>');
 48             value = (typeof value === 'number' || value == 0)? value : -1;
 49             invert = (invert == 'true')? true : false;
 50             if(className) node.addClass(className);
 51             node.append(html);
 52             node.all(selector).each(function(n, i, q){
 53                 i = (invert)? q.size()-1-i : i;
 54                 if(i<value){
 55                     n.addClass('previous')
 56                 } else if(i==value){
 57                     n.addClass('current')
 58                 } else{
 59                     n.addClass('next');
 60                 }
 61             });
 62             return node
 63         }
 64         
 65     }, {
 66         ATTRS : {
 67             
 68     }
 69     });
 70 
 71     Y.namespace('Wegas').CepNodeFormatter = NodeFormatter;
 72 });
