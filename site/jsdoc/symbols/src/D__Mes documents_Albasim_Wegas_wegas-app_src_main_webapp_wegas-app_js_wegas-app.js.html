 11 /**

 15 YUI.add('wegas-app', function (Y) {
 16     "use strict";
 17 
 18     /**
 19     * @name Y.Wegas.App
 20     * @class  Base class for wegas, handle initialisation of datasources and rendering
 21     * @constructor
 22     * @param Object Will be used to fill attributes field
 23     * @description create a new wegas-app
 24     */
 25     var App = Y.Base.create("wegas-app", Y.Base, [], {
 26 
 27         /**
 28          * @lends Y.Wegas.App#
 29          */
 30         // ** Private fields ** //
 31         /**
 32          * Holds a reference to all the dataSources used.
 33          */
 34         dataSources: [],
 35 
 36         /**
 37          * @function
 38          * @private
 39          * @description Lifecycle methods
 40          */
 41         initializer: function () {
 42             Y.Wegas.app = this;
 43             /**
 44              * @memberOf Y.Wegas.App#
 45              * @event
 46              * @name render
 47              * @description render event
 48              */
 49             this.publish("render", {});
 50         },
 51 
 52         /**
 53          * @function
 54          * @private
 55          * @description destructor methods.
 56          */
 57         destructor : function () {
 58             var i;
 59             for (i = 0; i < this.dataSources.length; i = i + 1) {
 60                 this.dataSources[i].destroy();
 61             }
 62         },
 63 
 64         /**
 65          * @function
 66          * @description render function
 67          */
 68         render: function () {
 69             Y.io.header("Accept-Language", Y.config.lang);                      // Set the language for all requests
 70             Y.JSON.useNativeParse = true;                                       // @todo Shall we use browser native parser ?
 71 
 72             this.on("render", function () {                                     // Remove loading overlay on render
 73                 Y.one("body").removeClass("wegas-loading-overlay");
 74             });
 75 
 76             Y.on("io:failure", function (tId, e) {
 77                 if (!e) {
 78                     return;
 79                 }
 80                 var exception = e.responseText.substring(e.responseText.indexOf('"exception'), e.responseText.length);
 81                 exception = exception.split(",");
 82                 if (e.status === 400 && exception[0] === '"exception":"org.apache.shiro.authz.UnauthorizedException"' ||
 83                     exception[0] === '"exception":"org.apache.shiro.authz.UnauthenticatedException"') {
 84                     // Y.config.win.location.href = Y.Wegas.app.get("base") + 'wegas-app/view/login.html';   //Redirect to login
 85                     alert("You have been logged out or does not have permissions");
 86                 }
 87             }, this);
 88 
 89             this.initDataSources();
 90         },
 91 
 92         // *** Private methods ** //
 93         /**
 94          * @function
 95          * @private
 96          * @description initilize DataSources
 97          */
 98         initDataSources: function () {
 99             var k, dataSource, dataSources = this.get('dataSources'),
100             onInitialRequest = function () {
101                 this.requestCounter -= 1;
102                 if (this.requestCounter === 0) {
103                     this.initPage();                                            // Run the initPage() method when they all arrived.
104                 }
105             };
106 
107             this.requestCounter = 0;
108 
109             for (k in dataSources) {
110                 if (dataSources.hasOwnProperty(k)) {
111                     dataSources[k].source = this.get("base") + dataSources[k].source; // Set up datasource path
112                     dataSource = new Y.Wegas.DataSource(dataSources[k]);        // Instantiate the datasource
113                     this.dataSources[k] = Y.Wegas[k + "Facade"] = this[k + "Facade"] = dataSource; // Set up global references
114                     dataSource.once("response", onInitialRequest, this);        // Listen to the datasources initial requests
115                     if (Y.Lang.isNumber(dataSource.sendInitialRequest())) {     // Send the initial request
116                         this.requestCounter += 1;                               // If the request was sent, update the counter
117                     }
118                 }
119             }
120 
121             if (this.requestCounter === 0) {                                    // If no request was sent, render directly
122                 this.initPage();
123             }
124         },
125 
126         /**
127          * @function
128          * @private
129          * @description initPage methods
130          */
131         initPage: function () {
132             Y.io(this.get('base') + this.get('layoutSrc') + '?id=' + App.genId(), {
133                 context: this,
134                 on: {
135                     success: function (id, o, args) {
136                         var cfg;
137                         try {
138                             cfg = Y.JSON.parse(o.responseText);		// Process the JSON data returned from the server
139                         } catch (e) {
140                             alert("Wegas.App.initUI(): Layout json parse failed failed!");
141                             Y.error("Layout parse failed", e, "Y.Wegas.App");
142                             return;
143                         }
144 
145                         Y.Wegas.Widget.use(cfg, Y.bind(function (cfg) {          // Load the subwidget dependencies
146                             this.widget = Y.Wegas.Widget.create(cfg);           // Render the subwidget
147                             this.widget.render();
148                             this.fire("render");                                // Fire a render event for some eventual post processing
149                         }, this, cfg));
150 
151                     //this.pageLoader = new Y.Wegas.PageLoader();               // Load the subwidget using pageloader
152                     //this.pageLoader.render();
153                     // cfg.id = -100;
154                     // this.dataSources.Page.data.push(cfg);
155                     //try {
156                     //    this.pageLoader.set("pageId", -100);
157                     //} catch (renderException) {
158                     //    Y.log('initUI(): Error rendering UI: ' + ((renderException.stack) ? renderException.stack : renderException), 'error', 'Wegas.App');
159                     //}
160 
161                     }
162                 }
163             });
164         }
165     }, {
166         /**
167          * @lends Y.Wegas.App
168          */
169         /**
170          * @field
171          * @static
172          * @description
173          * <p><strong>Method</strong></p>
174          * <ul>
175          *    <li>base: Base Url for app, <i>default: /Wegas/</i></li>
176          *    <li>layoutSrc : xxxxxxxxxxxxxxx</li>
177          *    <li>dataSources : xxxxxxxxxxxxxxx</li>
178          *    <li>cssStylesheets : xxxxxxxxxxxxxxx<i>default: []</i></li>
179          *    <li>currentGameModel : xxxxxxxxxxxxxxx</li>
180          *    <li>currentGame : xxxxxxxxxxxxxxx</li>
181          *    <li>currentTeam : xxxxxxxxxxxxxxx</li>
182          *    <li>currentPlayer : xxxxxxxxxxxxxx</li>
183          *    <li>currentUser : Object litteral representing current user</li>
184          *    <li>editorMenus :
185          *        This field is used to globally override Entities edition menus.
186          *        Use the target class name as the key.
187          *    </li>
188          *    <li>editorForms :
189          *        This field is used to globally override Entities edition forms.
190          *        Use the target class name as the key.
191          *    </li>
192          * </ul>
193          */
194         ATTRS: {
195             /**
196              * Base url for app
197              */
198             base: {
199                 getter: function () {
200                     return Y.config.groups.wegas.base;
201                 }
202             },
203             layoutSrc: {},
204             dataSources: {
205                 value: {}
206             },
207             cssStylesheets: {
208                 value: []
209             },
210             currentGameModel: {},
211             currentGame: {},
212             currentTeam: {},
213             currentPlayer: {
214                 setter: function (val) {
215                     var cPlayer = this.dataSources.Game.rest.getPlayerById(val);
216                     if (cPlayer) {                                              // @fixme
217                         this.set('currentTeam', cPlayer.get("teamId"));         // When current player is updated, we also update current team
218                     }
219                     return val;
220                 }
221             },
222             /**
223              * Object litteral representing current user.
224              */
225             currentUser: { },
226             /**
227              *
228              */
229             devMode: {
230                 value: false,
231                 setter: function (val) {
232                     if (val) {
233                         Y.one("body").addClass("wegas-devmode");
234                     }
235                 }
236             },
237 
238             /**
239             * This field is used to globally override Entities edition menus.
240             * Use the target class name as the key.
241             */
242             editorMenus: {
243                 value: {}
244             },
245             /**
246             * This field is used to globally override Entities edition forms.
247             * Use the target class name as the key.
248             */
249             editorForms: {
250                 value: {}
251             }
252         },
253         /**
254          * @function
255          * @static
256          * @return {integer} time
257          * @description generate ID
258          */
259         genId: function () {
260             var now = new Date();
261             return now.getHours() + now.getMinutes() + now.getSeconds();
262         },
263         /**
264          * @function
265          * @static
266          * @param str String
267          * @return {String} Escaped string
268          * @description Escape all necessary character
269          */
270         htmlEntities: function (str) {
271             return String(str).replace(/&/g, '&')
272             .replace(/</g, '<')
273             .replace(/>/g, '>')
274             .replace(/"/g, '"');
275         },
276         nl2br: function (str, is_xhtml) {
277             var breakTag = (is_xhtml || typeof is_xhtml === 'undefined') ? '<br />' : '<br>';
278             return (str + '').replace(/([^>\r\n]?)(\r\n|\n\r|\r|\n)/g, '$1' + breakTag + '$2');
279         }
280     });
281 
282     Y.namespace('Wegas').App = App;
283 
284     /**
285      * Shortcut to activate developper mode
286      */
287     Y.devMode = function () {
288         Y.Wegas.app.set("devMode", true);
289     };
290 });
