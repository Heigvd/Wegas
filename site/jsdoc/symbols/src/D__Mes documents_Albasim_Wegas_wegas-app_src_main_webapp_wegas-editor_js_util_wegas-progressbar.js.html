1 YUI.add('wegas-progressbar', function(Y) {
  2     'use strict';
  3     var ProgressBar,
  4             BOUNDING_BOX = "boundingBox",
  5             CONTENT_BOX = "contentBox";
  6 
  7     ProgressBar = Y.Base.create("wegas-progressbar", Y.Widget, [], {
  8         labelNode: null,
  9         valueNode: null,
 10         initializer: function() {
 11             this.labelNode = Y.Node.create("<div></div>");
 12             this.valueNode = Y.Node.create("<div></div>");
 13         },
 14         renderUI: function() {
 15             var bbStyle = this.get(BOUNDING_BOX).getDOMNode().style;
 16             bbStyle.border = "1px solid " + this.get("color");
 17             bbStyle.borderRadius = "5px";
 18             bbStyle.display = "inline-block";
 19             bbStyle.textAlign = "center";
 20             bbStyle.fontSize = this.get("height");
 21             bbStyle.position = "relative";
 22             this.labelNode.setStyles({
 23                 "backgroundColor": "transparent",
 24                 "position": "absolute",
 25                 "top": "0px",
 26                 "left": "0px",
 27                 "width": "100%"
 28             });
 29             this.get(BOUNDING_BOX).append(this.labelNode);
 30 
 31         },
 32         syncUI: function() {
 33             this.set("percent", this.get("percent"));
 34             this.set("color", this.get("color"));
 35             this.set("label", this.get("label"));
 36         }
 37     }, {
 38         ATTRS: {
 39             percent: {
 40                 value: 100,
 41                 setter: function(v) {
 42                     v = (+v).toFixed(1);
 43                     if (this.get(CONTENT_BOX).getDOMNode()) {
 44                         this.get(CONTENT_BOX).getDOMNode().style.width = v + "%";
 45                         this.set("label", this.get("label"));
 46                     }
 47                     return v;
 48                 }
 49             },
 50             color: {
 51                 value: "lightblue",
 52                 validator: Y.Lang.isString,
 53                 setter: function(v) {
 54                     this.get(CONTENT_BOX).getDOMNode().style.backgroundColor = v;
 55                     this.get(BOUNDING_BOX).getDOMNode().style.borderColor = v;
 56                     return v;
 57                 }
 58             },
 59             showValue: {
 60                 value: false,
 61                 validator: Y.Lang.isBoolean,
 62                 setter: function(v) {
 63                     this.set("label", this.get("label"));
 64                     return v;
 65                 }
 66             },
 67             label: {
 68                 value: "",
 69                 validator: Y.Lang.isString,
 70                 setter: function(s) {
 71                     if (this.get("showValue")) {
 72                         this.labelNode.setContent(s + " " + this.get("percent") + "%");
 73                     } else {
 74                         this.labelNode.setContent(s);
 75                     }
 76                     return s;
 77                 }
 78             }
 79         }
 80     });
 81     Y.namespace("Wegas").ProgressBar = ProgressBar;
 82 });
