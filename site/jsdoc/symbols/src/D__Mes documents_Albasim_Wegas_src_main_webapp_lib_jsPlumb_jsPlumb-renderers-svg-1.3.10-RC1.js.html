 76 	_updateGradient = function(parent, node, style, dimensions, uiComponent) {
 77 		var id = JSPLUMB_GRADIENT + uiComponent._jsPlumb.idstamp();
 78 		// first clear out any existing gradient
 79 		_clearGradient(parent);
 80 		// this checks for an 'offset' property in the gradient, and in the absence of it, assumes
 81 		// we want a linear gradient. if it's there, we create a radial gradient.
 82 		// it is possible that a more explicit means of defining the gradient type would be
 83 		// better. relying on 'offset' means that we can never have a radial gradient that uses
 84 		// some default offset, for instance.
 85 		// issue 244 suggested the 'gradientUnits' attribute; without this, straight/flowchart connectors with gradients would
 86 		// not show gradients when the line was perfectly horizontal or vertical.
 87 		if (!style.gradient.offset) {
 88 			var g = _node(LINEAR_GRADIENT, {id:id, gradientUnits:"userSpaceOnUse"});
 89 			parent.appendChild(g);
 90 		}
 91 		else {
 92 			var g = _node(RADIAL_GRADIENT, {
 93 				id:id
 94 			});
 95 			parent.appendChild(g);
 96 		}
 97 		
 98 		// the svg radial gradient seems to treat stops in the reverse 
 99 		// order to how canvas does it.  so we want to keep all the maths the same, but
100 		// iterate the actual style declarations in reverse order, if the x indexes are not in order.
101 		for (var i = 0; i < style.gradient.stops.length; i++) {
102 			// Straight Connectors and Bezier connectors act slightly differently; this code is a bit of a kludge.  but next version of
103 			// jsplumb will be replacing both Straight and Bezier to be generic instances of 'Connector', which has a list of segments.
104 			// so, not too concerned about leaving this in for now.
105 			var styleToUse = i;
106 			if (dimensions.length == 8) 
107 				styleToUse = dimensions[4] < dimensions[6] ? i: style.gradient.stops.length - 1 - i;
108 			else
109 				styleToUse = dimensions[4] < dimensions[6] ? style.gradient.stops.length - 1 - i : i;
110 			var stopColor = jsPlumbUtil.convertStyle(style.gradient.stops[styleToUse][1], true);
111 			var s = _node(STOP, {"offset":Math.floor(style.gradient.stops[i][0] * 100) + "%", "stop-color":stopColor});
112 			g.appendChild(s);
113 		}
114 		var applyGradientTo = style.strokeStyle ? STROKE : FILL;
115 		node.setAttribute(STYLE, applyGradientTo + ":url(#" + id + ")");

117 	_applyStyles = function(parent, node, style, dimensions, uiComponent) {
118 		
119 		if (style.gradient) {
120 			_updateGradient(parent, node, style, dimensions, uiComponent);			
121 		}
122 		else {
123 			// make sure we clear any existing gradient
124 			_clearGradient(parent);
125 			node.setAttribute(STYLE, "");
126 		}
127 		
128 		node.setAttribute(FILL, style.fillStyle ? jsPlumbUtil.convertStyle(style.fillStyle, true) : NONE);
129 		node.setAttribute(STROKE, style.strokeStyle ? jsPlumbUtil.convertStyle(style.strokeStyle, true) : NONE);		
130 		if (style.lineWidth) {
131 			node.setAttribute(STROKE_WIDTH, style.lineWidth);
132 		}
133 	
134 		// in SVG there is a stroke-dasharray attribute we can set, and its syntax looks like
135 		// the syntax in VML but is actually kind of nasty: values are given in the pixel
136 		// coordinate space, whereas in VML they are multiples of the width of the stroked
137 		// line, which makes a lot more sense.  for that reason, jsPlumb is supporting both
138 		// the native svg 'stroke-dasharray' attribute, and also the 'dashstyle' concept from
139 		// VML, which will be the preferred method.  the code below this converts a dashstyle
140 		// attribute given in terms of stroke width into a pixel representation, by using the
141 		// stroke's lineWidth. 
142 		if (style[DASHSTYLE] && style[LINE_WIDTH] && !style[STROKE_DASHARRAY]) {
143 			var sep = style[DASHSTYLE].indexOf(",") == -1 ? " " : ",",
144 			parts = style[DASHSTYLE].split(sep),
145 			styleToUse = "";
146 			parts.forEach(function(p) {
147 				styleToUse += (Math.floor(p * style.lineWidth) + sep);
148 			});
149 			node.setAttribute(STROKE_DASHARRAY, styleToUse);
150 		}		
151 		else if(style[STROKE_DASHARRAY]) {
152 			node.setAttribute(STROKE_DASHARRAY, style[STROKE_DASHARRAY]);
153 		}
154 		
155 		// extra attributes such as join type, dash offset.
156 		for (var i in svgAttributeMap) {
157 			if (style[i]) {
158 				node.setAttribute(svgAttributeMap[i], style[i]);
159 			}
160 		}
