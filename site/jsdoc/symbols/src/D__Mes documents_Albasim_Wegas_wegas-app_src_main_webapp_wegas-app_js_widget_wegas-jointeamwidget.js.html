 16 YUI.add('wegas-jointeamwidget', function (Y) {
 17     "use strict";
 18 
 19     var JoinTeamWidget;
 20 
 21     /**
 22     * @name Y.Wegas.JoinTeamWidget
 23     * @extends Y.Wegas.JoinGameWidget
 24     * @class  class for join a team
 25     * @constructor
 26     * @description Allows just to join a team
 27     */
 28     JoinTeamWidget = Y.Base.create("wegas-jointeamwidget", Y.Wegas.JoinGameWidget, [Y.WidgetChild, Y.Wegas.Widget], {
 29         /**
 30          * @lends Y.Wegas.JoinTeamWidget.prototype
 31          */
 32         /**
 33          * @function
 34          * @private
 35          * @description Difference compared to the parent class:
 36          * 1) token is get from the url
 37          * 2) then call directly the sendJoinGame()
 38          */
 39         renderUI: function () {
 40             JoinTeamWidget.superclass.renderUI.apply(this);
 41             this.tokenField.setValue(document.location.search.substring(7));
 42             this.sendJoinGame();
 43         },
 44 
 45         /**
 46          * @function
 47          * @private
 48          */
 49         bindUI: function () {
 50             JoinTeamWidget.superclass.bindUI.apply(this);
 51         },
 52 
 53         /**
 54          * @function
 55          * @private
 56          * @description Reload the same page
 57          */
 58         joinTeamSuccess: function () {
 59             this.showMessage("success", "Game joined", 10000);
 60             window.location.reload();
 61         }
 62     });
 63 
 64     Y.namespace('Wegas').JoinTeamWidget = JoinTeamWidget;
 65 });
