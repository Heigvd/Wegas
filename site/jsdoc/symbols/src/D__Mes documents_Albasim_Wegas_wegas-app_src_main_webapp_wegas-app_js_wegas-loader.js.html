675     function loadModules(group) {
676         var i, modules = group.modules,
677         moduleName,
678         allModules = [],
679         modulesByType = {};
680         for (moduleName in modules) {                                           // Loop through all modules
681             if (modules.hasOwnProperty(moduleName)) {
682                 allModules.push(moduleName);                                    // Build a list of all modules
683                 if (modules[moduleName].ws_provides) {                          // Build a reverse index on which module provides what type
684                     if (Y.Lang.isArray(modules[moduleName].ws_provides)) {
685                         for (i = 0; i < modules[moduleName].ws_provides.length; i = i + 1) {
686                             modulesByType[modules[moduleName].ws_provides[i]] = moduleName;
687                         }
688                     } else {
689                         modulesByType[modules[moduleName].ws_provides] = moduleName;
690                     }
691                 }
692                 if (modules[moduleName].ix_provides) {                          // Build a reverse index on which module provides what type
693                     if (Y.Lang.isArray(modules[moduleName].ix_provides)) {
694                         for (i = 0; i < modules[moduleName].ix_provides.length; i = i + 1) {
695                             YUI_config.groups.inputex.modulesByType[modules[moduleName].ix_provides[i]] = moduleName;
696                         }
697                     } else {
698                         YUI_config.groups.inputex.modulesByType[modules[moduleName].ix_provides] = moduleName;
699                     }
700                 }
701 
702             }
703         }
704         group.allModules = allModules;
705         group.modulesByType = modulesByType;
706     }

709     loadModules(YUI_config.groups.ace);
