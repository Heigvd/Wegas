 21     escapeJSString = function (str) {
 22         return str.replace(/"/g, '\\"').replace(/\n/g, "\\n");
 23     },
 24     unesacapeJSString = function (str) {
 25         return str.replace(/\\"/g, '"');

 27     /**
 28      * @class Y.Wegas.VariableDescriptorSelect
 29      * @constructor
 30      * @extends
 31      * @param {Object} options InputEx definition object
 32      */
 33     VariableDescriptorSelect = function (options) {
 34         VariableDescriptorSelect.superclass.constructor.call(this, options);
 35     };
 36 
 37     Y.extend(VariableDescriptorSelect, inputEx.Group, {
 38         currentEntityField: null,
 39 
 40         /** @lends Y.Wegas.VariableDescriptorSelect# */
 41         /**
 42 	 * Setup the options.fields from the availableFields option
 43 	 */
 44         setOptions: function (options) {
 45             options.fields = options.fields || [];
 46             VariableDescriptorSelect.superclass.setOptions.call(this, options);
 47         },
 48 
 49         render: function () {
 50             VariableDescriptorSelect.superclass.render.call(this);
 51             this.divEl.classList.add("wegas-inputex-variabledescriptorselect");
 52             this.syncUI();
 53         },
 54 
 55         setValue: function (val) {
 56             // Set value should not ba called directly
 57             //Y.log("VariableDescriptorSelect.setValue", val);
 58             VariableDescriptorSelect.superclass.setValue.apply(this, arguments);
 59             this.options.value = val;
 60             this.syncUI();
 61         },
 62 
 63         getValue: function () {
 64             return "VariableDescriptorFacade.find(" + this.currentEntityField.getValue() + ")";
 65         },
 66 
 67         syncUI: function () {
 68             this.empty();
 69 
 70             var i, ret = [],
 71             rootEntities = Y.Wegas.VariableDescriptorFacade.rest.getCache(),
 72             currentEntity = Y.Wegas.VariableDescriptorFacade.rest.findById(this.options.value) || rootEntities[0];
 73 
 74             this.currentEntity = currentEntity;                                 // Keeps a reference to the current entity
 75 
 76             while (currentEntity.parentDescriptor) {                            // Add the current entity hierarchy
 77                 ret.push(this.generateSelectConfig(currentEntity.parentDescriptor,
 78                     currentEntity, currentEntity.parentDescriptor.get("items")));
 79                 currentEntity = currentEntity.parentDescriptor;
 80             }
 81             ret.push(this.generateSelectConfig(null, currentEntity, rootEntities));// Add the root context (entities that are at the root of the gameModel
 82 
 83             ret = ret.reverse();
 84             for (i = 0; i < ret.length; i += 1) {
 85                 this.addField(ret[i]);
 86             }
 87             this.currentEntityField = this.inputs[this.inputs.length - 1];
 88         },
 89         /**
 90          *
 91          * @overrride Y.inputEx.Group.onChange()
 92          */
 93         onChange: function (fieldValue, fieldInstance) {
 94             if (fieldValue === "----------") {
 95                 return; /* @fixme */
 96             }
 97 
 98             if (Y.Lang.isNumber(fieldValue)) {                                  // value is an number, it is the new current entity's id'
 99                 this.options.value = fieldValue;
100                 this.options.method = null;
101                 this.options.methodCfg = null;
102                 this.options.arguments = null;
103                 this.syncUI();
104             } else if (Y.Lang.isString(fieldValue)) {                           // The id is a string, it's the new current mehtod
105                 this.options.value = fieldInstance.options.parentEntity.get("id");
106                 this.options.method = fieldValue;
107                 this.options.arguments = null;
108                 this.syncUI();
109             } else {                                                            // Otherwise current args value
110                 this.options.arguments = this.inputs[this.inputs.length - 1].getValue();
111             }
112             this.fireUpdatedEvt();
113         },
114 
115         empty: function () {
116             while (this.inputs.length > 0) {
117                 this.inputs.pop().destroy();
118             }
119             (new Y.Node(this.fieldset)).empty();
120         },
121 
122         /**
123          *  Overriden to add reference to parententity
124          */
125         addField: function (fieldOptions) {
126             VariableDescriptorSelect.superclass.addField.call(this, fieldOptions);
127             this.inputs[this.inputs.length - 1].options.parentEntity = fieldOptions.parentEntity;
128         },
129         /**
130          * Generate
131          */
132         generateSelectConfig: function (entity, selectedEntity, items) {
133             var value;
134 
135             if (entity === this.currentEntity) {
136                 value = this.options.method;
137             }
138             if (selectedEntity) {
139                 value = selectedEntity.get("id");
140             }
141             return {
142                 type: 'select',
143                 choices: this.genChoices(entity, items),
144                 value: value,
145                 parentEntity: entity
146             };
147         },
148         genChoices: function (entity, items) {
149             var i, choices = [];
150 
151             if (items) {
152                 for (i = 0; i < items.length; i += 1) {
153                     choices.push({
154                         value: items[i].get("id"),
155                         label: items[i].get("editorLabel")
156                     });
157                 }
158             }
159             return choices;
160         }
161     });

170     Y.extend(VariableDescriptorGetter, VariableDescriptorSelect, {
171         syncUI: function () {
172             VariableDescriptorGetter.superclass.syncUI.call(this);
173 
174             if (this.currentEntity.get("items") && this.currentEntity.get("items").length > 0) {
175                 this.addField(this.generateSelectConfig(null,                   // Pushes the current entity methods and children to the stack
176                     this.currentEntity, this.currentEntity.get("items")));
177             }
178         },
179 
180         genChoices: function (entity, items) {
181             var choices = [];
182 
183             if (items && items.length > 0) {                                    // If required, push separator
184                 choices.push({
185                     value: "----------"
186                 });
187             }
188 
189             return choices.concat(VariableDescriptorGetter.superclass.genChoices.apply(this, arguments));
190         }
191     });

194     /**
195      * @class Y.Wegas.VariableDescriptorMethod
196      * @extends Y.Wegas.VariableDescriptorSelect
197      * @constructor
198      * @extends VariableDescriptorSelect
199      * @param {Object} options InputEx definition object
200      */
201     VariableDescriptorMethod = function (options) {
202         VariableDescriptorMethod.superclass.constructor.call(this, options);
203     };
204 
205     Y.extend(VariableDescriptorMethod, VariableDescriptorSelect, {
206 
207         syncUI: function () {
208             var i, args, methods, cMethod,
209             rootEntities = Y.Wegas.VariableDescriptorFacade.rest.getCache(),
210             currentEntity = Y.Wegas.VariableDescriptorFacade.rest.findById(this.options.value) || rootEntities[0];
211 
212             while (this.getMethods(currentEntity).length === 0                  // If the current entity has no methods,
213                 && currentEntity.get("items") && currentEntity.get("items").length > 0) { // but it has a child
214                 currentEntity = currentEntity.get("items")[0];                  // select its first child
215                 this.options.value = currentEntity.get("id");
216             }
217             cMethod = this.options.methodCfg;                                   //assign cMethod after set this.options.methodCfg by this.getMethods()
218 
219             VariableDescriptorMethod.superclass.syncUI.call(this);
220 
221             this.addField(this.generateSelectConfig(this.currentEntity,         // Pushes the current entity methods and children to the stack
222                 null, this.currentEntity.get("items")));
223 
224             methods  = this.getMethods(this.currentEntity);
225 
226             if (!cMethod && Y.Object.values(methods).length > 0) {
227                 cMethod = Y.Object.values(methods)[0];                          // By default select the first method available
228             }
229 
230             args = (cMethod && cMethod.arguments) ? cMethod.arguments : [];
231 
232             for (i = 0; i < args.length; i = i + 1) {
233                 args[i].entity = this.currentEntity;                            // Adds a reference to the target entity to the argument Fields;
234             }
235 
236             this.currentMethod = cMethod;
237 
238             this.addField(Y.mix({
239                 type: "combine",
240                 fields: args,
241                 value: this.options.arguments,
242                 label: null
243             }, cMethod));
244 
245         // Same as above, but using json object format for method definitions
246         //var schemaMap = {
247         //    Entity: {
248         //        type: "array",
249         //        items: cMethod.arguments
250         //    }
251         //}, builder = new Y.inputEx.JsonSchema.Builder({
252         //    'schemaIdentifierMap': schemaMap,
253         //    'defaultOptions':{
254         //        'showMsg':true
255         //    }
256         //}), field  = builder.schemaToInputEx(schemaMap.Entity);
257         },
258         setOptions: function (options) {
259             VariableDescriptorMethod.superclass.setOptions.call(this, options);
260             this.options.method = options.method;
261             this.options.arguments = options.arguments;
262             this.argsOffset = 1;
263         },
264 
265         getValue: function () {
266             var i, j, l = this.inputs.length,
267             args = this.inputs[l - this.argsOffset].getValue(),
268             method = this.inputs[l - this.argsOffset - 1].getValue();
269 
270             if (!method) {
271                 return "true";
272             }
273 
274             for (i = 0; i < args.length; i = i + 1) {
275                 if (this.currentMethod.arguments[i].scriptType === "string") {
276                     if (Y.Lang.isArray(args[i])) {
277                         for (j = 0; j < args[i].length; j++) {
278                             args[i][j] = escapeJSString(args[i][j]);
279                         }
280                         args[i] = Y.JSON.stringify(args[i]);
281                     } else {
282                         args[i] = '"' + escapeJSString(args[i]) + '"';
283                     }
284                 }
285             }
286             return "VariableDescriptorFacade.find(" + this.inputs[l - this.argsOffset - 2].getValue() + ")"
287             + "." + method + "(" + args.join(", ") + ")";
288         },
289         /**
290          * Generate choices for a given entity: add it's methods and then pass up to parent class
291          * to add children variable descriptors.
292          */
293         genChoices: function (entity, items) {
294             var choices = [];
295             if (entity) {
296                 choices = choices.concat(this.getMethods(entity));              // Push the methods to the select choices
297             }
298 
299             if (items && choices.length > 0) {                                  // If required, push separator
300                 choices.push({
301                     value: "----------"
302                 });
303             }
304             return choices.concat(VariableDescriptorMethod.superclass.genChoices.apply(this, arguments));
305         },
306 
307         getMethods: function (entity) {
308             var i, methods = entity.getMethodCfgs(), ret = [];
309             for (i in methods) {
310                 if (!this.options.returnsFilter                                 // Apply filter on the method return type
311                     || this.options.returnsFilter.indexOf(methods[i].returns || "void") >= 0) {
312 
313 
314                     methods[i].value = i;
315                     methods[i].label =  methods[i].label || i;
316                     ret.push(methods[i]);
317                     if (i === this.options.method) {
318                         this.options.methodCfg = methods[i];
319                     }
320                 }
321             }
322             return ret;
323         }
324     });

350     /**
351      * @class Y.Wegas.VariableDescriptorCondition
352      * @extends Y.Wegas.VariableDescriptorMethod
353      * @constructor
354      * @extends VariableDescriptorMethod
355      * @param {Object} options InputEx definition object
356      */
357     VariableDescriptorCondition = function (options) {
358         VariableDescriptorCondition.superclass.constructor.call(this, options);
359     };
360 
361     Y.extend(VariableDescriptorCondition, VariableDescriptorMethod, {
362 
363         setOptions: function (options) {
364             VariableDescriptorCondition.superclass.setOptions.call(this, options);
365             this.argsOffset = 1;
366             this.options.returnsFilter = ["number", "boolean"];
367             this.options.operator = options.operator;
368             this.options.rightValue = options.rightValue || 0;
369         },
370 
371         getValue: function () {
372             var value = VariableDescriptorCondition.superclass.getValue.call(this);
373 
374             if (this.argsOffset > 1) {
375                 value += this.inputs[this.inputs.length - 1].getValue().join("");
376             }
377             return value;
378         },
379 
380         /**
381          * Override to prevent
382          */
383         onChange: function (fieldValue, fieldInstance) {
384             VariableDescriptorCondition.superclass.onChange.apply(this, arguments);
385         },
386 
387         syncUI: function () {
388             VariableDescriptorCondition.superclass.syncUI.call(this);
389 
390             var methods  = this.getMethods(this.currentEntity),
391             cMethod = methods[this.options.method];
392 
393             if (!cMethod && methods.length > 0) {
394                 cMethod = methods[0];                                           // By default select the first method available
395             }
396 
397             if (cMethod && cMethod.returns === "number") {
398                 this.argsOffset = 2;
399                 this.addField({
400                     type: "combine",
401                     fields: [{
402                         type: "select",
403                         value: this.options.operator,
404                         choices: [{
405                             value: "===",
406                             label: "equals"
407                         }, {
408                             value: ">",
409                             label: "is greater than"
410                         }, {
411                             value: "<",
412                             label: "is smaller than"
413                         }, {
414                             value: ">=",
415                             label: "is greater or equal to"
416                         }, {
417                             value: "<=",
418                             label: "is smaller or equal to"
419                         }]
420                     }, {
421                         value: this.options.rightValue
422                     }]
423                 //                value: this.options.arguments
424                 });
425             } else {
426                 this.argsOffset = 1;
427             }
428 
429         }
430     });
