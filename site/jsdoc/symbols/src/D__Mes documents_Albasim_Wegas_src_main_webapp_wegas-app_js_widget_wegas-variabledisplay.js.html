 15 YUI.add('wegas-variabledisplay', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', VariableDisplay;
 19 
 20     VariableDisplay = Y.Base.create( "wegas-variabledisplay", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable ], {
 21 
 22         // ** Lifecycle Methods ** //
 23 
 24         bindUI: function () {
 25             Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this);
 26             Y.Wegas.app.after('currentPlayerChange', this.syncUI, this);
 27         },
 28 
 29         syncUI: function () {
 30             var variableDescriptor = this.get( "variableDesc" );
 31 
 32             if ( !variableDescriptor ) {
 33                 return;
 34             }
 35 
 36             if ( variableDescriptor instanceof Y.Wegas.persistence.ListDescriptor ) {   // If the widget is a list,
 37                 variableDescriptor = variableDescriptor.get( "currentItem" );   // display it with the current list and the current element
 38             }
 39 
 40             this.get( CONTENTBOX ).setHTML( this.genMarkup( variableDescriptor ) );  // Display the variable
 41 
 42         },
 43 
 44         genMarkup: function ( variableDescriptor ) {
 45             var acc, i, maxVal = variableDescriptor.get( "maxValue" ),
 46             minVal = variableDescriptor.get( "minValue" ),
 47             value = variableDescriptor.getInstance().get( "value" ),
 48             label = this.get( "label" ) || variableDescriptor.getPublicLabel();
 49 
 50             switch ( this.get( 'view' ) ) {
 51                 case 'text':
 52                     return '<span class="wegas-label wegas-variabledisplay-text-label">' + label + '</span>'
 53                     +' <span class="wegas-value wegas-variabledisplay-text-value">'+ value+'</span>';
 54                     break;
 55 
 56                 case 'title':
 57                     return '<span class="wegas-label wegas-variabledisplay-text-label">' + label + '</span>'
 58                     +' <span class="wegas-value wegas-variabledisplay-text-value">'+ variableDescriptor.getPublicLabel() + '</span>';
 59                     break;
 60 
 61                 case 'box':
 62                     acc = [];
 63                     for ( i = 0; i < value; i += 1 ) {
 64                         acc.push('<div class="wegas-variabledisplay-box-unit"></div>');
 65                     }
 66                     return '<div class="wegas-variabledisplay-box-label">' + label + '</div>'
 67                     +'<span class="wegas-variabledisplay-box-units">'+acc.join('')+'</span>'
 68                     +'<span class="wegas-variabledisplay-box-value">(' + value + '<span class="wegas-variabledisplay-box-valueMax">/' + maxVal + '</span>)</span>';
 69 
 70                     break;
 71 
 72                 case 'fraction':
 73                     return '<span class="wegas-variabledisplay-fraction-label">' + label + '</span>'
 74                     +'<span class="wegas-variabledisplay-fraction-minValue">'+ minVal +'</span>'
 75                     +'<span class="wegas-variabledisplay-fraction-minSeparator"> / </span>'
 76                     +'<span class="wegas-variabledisplay-fraction-value">'+ value +'</span>'
 77                     +'<span class="wegas-variabledisplay-fraction-maxSeparator"> / </span>'
 78                     +'<span class="wegas-variabledisplay-fraction-maxValue">'+ maxVal +'</span>';
 79                     break;
 80 
 81                 case 'valuebox':
 82                     acc = [];
 83                     if (variableDescriptor) {
 84                         for (i = variableDescriptor.get("minValue"); i <= variableDescriptor.get("maxValue"); i += 1) {
 85                             acc.push('<div class="wegas-valuebox-unit '
 86                                 + ((i === value) ? "wegas-valuebox-selected" : "")
 87                                 + ((i < value) ? "wegas-valuebox-previous" : "")
 88                                 + ((i > value) ? "wegas-valuebox-next" : "")
 89                                 + '">' + i + '</div>');
 90                         }
 91                     }
 92                     return '<span class="wegas-label wegas-variabledisplay-valuebox-label">' + label + '</span>'
 93                     + '<div class="wegas-value wegas-variabledisplay-valuebox-units">' + acc.join('')+"</div>";
 94                     break;
 95             }
 96 
 97         }
 98     }, {
 99         ATTRS : {
100             variable: {},
101             expr: {},
102             /**
103              * The target variable, returned either based on the variableName attribute,
104              * and if absent by evaluating the expr attribute.
105              */
106             variableDesc: {
107                 getter: function () {
108                     if ( this.get( "variable" ) ) {
109                         return this.get("dataSource").rest.find( 'name', this.get( "variable" ) )
110                     } else {
111                         return this.get("dataSource").rest.findById(
112                             Y.Wegas.VariableDescriptorFacade.script.scopedEval( this.get( "expr" ) ) );
113                     }
114                 }
115             },
116             dataSource: {
117                 "transient": true,
118                 getter: function () {
119                     return Y.Wegas.VariableDescriptorFacade;
120                 }
121             },
122             label : {
123                 type: "string",
124                 optional: true,
125                 validator: Y.Lang.isString
126             },
127             view: {
128                 type: "string",
129                 value: "text",
130                 choices: [{
131                     value: "text"
132                 }, {
133                     value: "box"
134                 }, {
135                     value: "gauge"
136                 }, {
137                     value: "valuebox"
138                 }, {
139                     value: "fraction"
140                 }]
141             }
142         }
143     });
144

182                         +'<span class="wegas-variabledisplay-fraction-minValue">'+ minVal +'</span>'
183                         +'<span class="wegas-variabledisplay-fraction-minSeparator"> / </span>'
184                         +'<span class="wegas-variabledisplay-fraction-value">'+ value +'</span>'

240                     value: "box"
241                 }, {
242                     value: "gauge"
243                 }, {
244                     value: "valuebox"
245                 }, {
246                     value: "fraction"
247                 }]
248             }
249         }
250     });
251 
252     Y.namespace('Wegas').VariableDisplay = VariableDisplay;
253 });
