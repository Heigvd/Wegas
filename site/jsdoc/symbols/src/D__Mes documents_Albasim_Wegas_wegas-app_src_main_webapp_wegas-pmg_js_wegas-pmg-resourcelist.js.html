 14 YUI.add("wegas-pmg-resourcelist", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", ResourceList;
 18 
 19     ResourceList = Y.Base.create("wegas-pmg-resourcelist", Y.Wegas.PmgGantt, [Y.Wegas.Editable], {
 20         handlers: null,
 21         menu: null,
 22         menuDetails: null,
 23         sortable: null,
 24         initializer: function () {
 25             this.handlers = {};
 26             this.menu = new Y.Wegas.Menu();
 27             this.menuDetails = new Y.Wegas.Menu({
 28                 width: "250px"
 29             });
 30             this.sortable = [];
 31         },
 32         renderUI: function () {
 33             var i, columns;
 34             ResourceList.superclass.renderUI.apply(this);
 35             columns = this.datatable.head.columns[0];
 36             for (i = 0; i < columns.length; i++) {
 37                 if (columns[i].key == 'week1') {
 38                     break;
 39                 }
 40             }
 41             this.datatable.addColumn({
 42                 key: 'assignements',
 43                 label: "Assignements"
 44             }, i);
 45         },
 46         bindUI: function () {
 47             ResourceList.superclass.bindUI.apply(this);
 48             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 49 
 50             this.handlers.createMenu = this.datatable.delegate('click', function (e) {            // fill the "add" menu on click
 51                 this.createMenu(e, true);
 52             }, '.yui3-datatable-data .assignement .assign', this);
 53 
 54             this.handlers.moveMenu = this.menu.on("button:mouseenter", function (e) {           // align the menu
 55                 this.menuDetails.set("align", {
 56                     node: this.menu.get("boundingBox"),
 57                     points: (e.details[0].domEvent.clientX > Y.DOM.winWidth() / 2) ?
 58                             ["tr", "tl"] : ["tl", "tr"]
 59                 });
 60                 this.menuDetails.show();
 61                 this.menuDetails.get("contentBox").setHTML('<div style="padding:5px 10px"><i>' + e.target.get("data").description + '</i></div>');
 62             }, this);
 63 
 64             this.handlers.hideMenu = this.menu.on("visibleChange", function (e) {                 // When the menu is hidden, hide the details panel
 65                 if (!e.newVal) {
 66                     this.menuDetails.hide();
 67                 }
 68             }, this);
 69 
 70             this.handlers.assignTask = this.menu.on("button:click", this.onMenuClick, this);     // assign a task to a resource
 71         },
 72         syncUI: function () {
 73             ResourceList.superclass.syncUI.apply(this);
 74             this.addButtonsAssignement();
 75             this.syncSortable();
 76         },
 77         destructor: function () {
 78             var k;
 79             for (k in this.handlers) {
 80                 this.handlers[k].detach();
 81             }
 82             this.menu.destroy();
 83             this.menuDetails.destroy();
 84         },
 85         addButtonsAssignement: function () {
 86             var cb = this.get(CONTENTBOX);
 87             cb.all(".yui3-datatable-data tr .yui3-datatable-col-assignements").each(function (node) {
 88                 node.append("<div class='assignement'></div>");
 89                 node.one('.assignement').append("<span class='assign'></span>");
 90                 node.addClass('noDescription');
 91             });
 92         },
 93         syncSortable: function () {
 94             var i, node, cb = this.get(CONTENTBOX);
 95             //temporary
 96             node = cb.one('.yui3-datatable-data .yui3-datatable-col-assignements');
 97             node.append("<div class='tasks'></div>");
 98             node.one('.tasks').append('<span class="task">t</span>');
 99             node.one('.tasks').append('<span class="task">e</span>');
100             node.one('.tasks').append('<span class="task">s</span>');
101             node.one('.tasks').append('<span class="task">t</span>');
102             node.one('.tasks').append('<span class="task">1</span>');
103             node.one('.tasks').append('<span class="task">2</span>');
104 
105             //the drag and drop feature to reorganise tasks.
106             for (i = 0; i < this.sortable.length; i++) {
107                 this.sortable[i].destroy();
108             }
109 
110             cb.all('.yui3-datatable-data tr .yui3-datatable-col-assignements .tasks').each(function (row, i) {
111                 this.sortable.push(new Y.Sortable({
112                     container: row,
113                     nodes: '.task',
114                     opacity: '.1'
115                 }));
116             }, this);
117 
118             for (i = 0; i < this.sortable.length; i++) {
119                 this.sortable[i].delegate.after('drag:over', function (e) {
120                     var drag = e.drag.get('dragNode');
121                     drag.setStyle('color', 'black');
122                 });
123                 this.sortable[i].delegate.after('drag:exit', function (e) {
124                     var drag = e.drag.get('dragNode');
125                     drag.setStyle('color', 'red');
126                 });
127                 this.sortable[i].delegate.after('drag:dropmiss', function (e) {
128                     var node = e.target.get('node');
129                     this.removeTask(node);
130                 }, this);
131             }
132         },
133         removeTask: function (node) {
134             //todo
135         },
136         createMenu: function (e, add) {
137             var i, tasks, resources, resourceDesc, resourceId;
138             resourceId = e.target.ancestor().ancestor().ancestor().one('*').getContent();
139             resources = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get('variables'));
140             for (i = 0; i < resources.get('items').length; i++) {
141                 if (resources.get('items')[i].get('id') == resourceId) {
142                     resourceDesc = resources.get('items')[i];
143                     break;
144                 }
145             }
146             this.menu.removeAll();
147             tasks = this.getTasks(resourceDesc);
148             if (!tasks || tasks.lenght <= 0) {
149                 return;
150             }
151             this.menu.add(tasks);
152             this.menu.attachTo(e.target);
153         },
154         compareTask: function (a, b) {
155             var sort = (this.get("taskSortBy") || "name");
156             if (a.get(sort)) {
157                 if (a.get(sort) < b.get(sort))
158                     return -1;
159                 if (a.get(sort) > b.get(sort))
160                     return 1;
161             } else if (a.getInstance().get(sort)) {
162                 if (a.getInstance().get(sort) < b.getInstance().get(sort))
163                     return -1;
164                 if (a.getInstance().get(sort) > b.getInstance().get(sort))
165                     return 1;
166             } else {
167                 if (a.getInstance().get('properties')[sort] < b.getInstance().get('properties')[sort])
168                     return -1;
169                 if (a.getInstance().get('properties')[sort] > b.getInstance().get('properties')[sort])
170                     return 1;
171             }
172             return 0;
173         },
174         getTasks: function (resourceDesc) {
175             //add is a boolean to determine if target is remove or add a task
176             //you can only add a task which isn't already added. 
177             //you can only remove a task which is added. 
178             var i, tasks, items, taskDesc, description, label, array = new Array(), no;
179             if (!this.get("taskList")) {
180                 return;
181             }
182             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("taskList"));
183             items = tasks.get('items');
184             items.sort(Y.bind(this.compareTask, this));
185             for (i = 0; i < items.length; i++) {
186                 taskDesc = items[i];
187                 no = (taskDesc.getInstance().get('properties').no || "");
188                 label = (taskDesc.get("label") || taskDesc.get("name") || "undefined");
189                 description = (taskDesc.get("description") || "No description.");
190                 array.push({
191                     type: "Button",
192                     label: no + ". " + label,
193                     data: {
194                         resource: resourceDesc,
195                         task: taskDesc,
196                         description: description
197                     }
198                 });
199             }
200             return array;
201         },
202         onMenuClick: function (e) {
203             var data = e.target.get("data");
204             //console.log('to do...')
205             //console.log(e.target.get("data"), data.resource.get('name'), data.task.get('name'));
206         }
207 
208     }, {
209         ATTRS: {
210             taskList: {
211                 value: null,
212                 validator: function (s) {
213                     return s === null || Y.Lang.isString(s);
214                 }
215             },
216             taskSortBy: {
217                 value: null,
218                 validator: function (s) {
219                     return s === null || Y.Lang.isString(s);
220                 }
221             }
222         }
223     });
224 
225     Y.namespace("Wegas").PmgResourcelist = ResourceList;
226 });
