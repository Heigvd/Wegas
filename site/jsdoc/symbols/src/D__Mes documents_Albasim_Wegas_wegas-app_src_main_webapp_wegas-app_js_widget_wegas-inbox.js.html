 15 YUI.add('wegas-inbox', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', InboxDisplay;
 19 
 20     InboxDisplay = Y.Base.create("wegas-inbox", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 21         // *** Private Fields *** //
 22         tabView: null,
 23         handlers: null,
 24         dataSource: null,
 25         deleteButton: null,
 26         msg: null,
 27         // *** Lifecycle Methods *** //
 28         initializer: function () {
 29             this.dataSource = Y.Wegas.app.VariableDescriptorFacade;
 30             this.handlers = {};
 31             this.plug(Y.Plugin.WidgetToolbar);
 32         },
 33         renderUI: function () {
 34             this.deleteButton = new Y.Wegas.Button({
 35                 label: "<span class='wegas-icon wegas-icon-cancel'></span>Delete"
 36             });
 37             this.toolbar.add(this.deleteButton);
 38             this.tabView = new Y.TabView();
 39             this.tabView.render(this.get(CONTENTBOX));
 40         },
 41         bindUI: function () {
 42             this.tabView.after("selectionChange", this.onTabSelected, this);
 43             this.handlers.dataUpdated = this.dataSource.after("update", this.syncUI, this);
 44             this.handlers.deleteEMail = this.deleteButton.on("click", function (e) {
 45                 if(!this.msg) {
 46                     return;
 47                 }
 48                 if (confirm('The e-mail "' + this.msg.get("subject") + '" will be deleted permanently. Continue ?')) {
 49                     this.deleteEmail();
 50                 }
 51             }, this);
 52         },
 53         syncUI: function () {
 54             var i, msg, tab, from,
 55                     inboxVariable = this.get('variable.evaluated').getInstance(),
 56                     messages = inboxVariable.get("messages"),
 57                     selectedIndex = 0,
 58                     tabs = [];
 59 
 60             this.isSyncing = true;
 61             this.tabView.removeAll();
 62             for (i = messages.length - 1; i >= 0; i -= 1) {
 63                 msg = messages[i];
 64                 from = msg.get("from") || "<i>No sender</i>";
 65                 tab = new Y.Tab({
 66                     label: '<div class="' + (msg.get("unread") ? "unread" : "read") + '"><div class="left">' + from + '</div>'
 67                             + '<div class="right">' + msg.get("subject") + '</div></div>',
 68                     content: '<div class="msg-header">'
 69                             + '<div class="msg-subject">Subject: ' + msg.get("subject") + '</div>'
 70                             + '<div class="msg-from">From: ' + from + '</div>'
 71                             + '</div>'
 72                             + '<div class="msg-body"><center><em><i>Loading</i></center></div>'
 73                 });
 74                 tab.msg = msg;
 75                 tabs.push(tab);
 76 
 77                 if (this.msg && msg.id === this.msg.id) {
 78                     selectedIndex = i;
 79                 }
 80             }
 81             this.tabView.add(tabs);
 82 
 83             if (messages.length === 0) {
 84                 this.tabView.add({
 85                     label: '',
 86                     content: '<center><i>You have no messages</i></center>'
 87                 });
 88             }
 89 
 90             this.isSyncing = false;
 91             this.tabView.selectChild(selectedIndex);
 92         },
 93         destructor: function () {
 94             this.tabView.destroy();
 95             this.handlers.dataUpdated.detach();
 96         },
 97         // *** Private Methods *** //
 98         deleteEmail: function (e) {
 99             if (!this.msg) {
100                 return;
101             }
102             this.dataSource.rest.sendRequest({
103                 request: "/Inbox/Message/" + this.msg.get("id"),
104                 cfg: {
105                     method: "DELETE"
106                 },
107                 on: {
108                     success: this.msg = null
109                 }
110             });
111         },
112         onTabSelected: function (e) {
113             var i, attachement, attachements = [];
114             if (this.isSyncing) {
115                 return;
116             }
117 
118             if (this.timer) {                                                   // If there is an active unread
119                 this.timer.cancel();                                            // message timer, we cancel it.
120             }
121 
122             if (e.newVal && e.newVal.msg) {
123 
124                 this.dataSource.rest.sendRequest({// Retrieve the message body from the server
125                     request: "/Inbox/Message/" + e.newVal.msg.get("id") + "?view=Export",
126                     on: {
127                         success: Y.bind(function (e) {
128                             if (e.response.entity.get("attachements") && e.response.entity.get("attachements").length > 0) {
129                                 for (i = 0; i < e.response.entity.get("attachements").length; i++) {
130                                     attachement = e.response.entity.get("attachements")[i];
131                                     attachements.push("<a href='" + attachement + "'>" + attachement + "</a>");
132                                 }
133                                 if (!this.get("panelNode").one(".msg-header .msg-attachement")) {
134                                     this.get("panelNode").one(".msg-header").append("<div class='msg-attachement'></div>");
135                                 }
136                                 this.get("panelNode").one(".msg-header .msg-attachement").setHTML("Attachements: " + attachements.join("; "));
137                             }
138                             this.get("panelNode").one(".msg-body").setHTML(e.response.entity.get("body") || "<center><em><i>Empty</i></center>");
139                         }, e.newVal)
140                     }
141                 });
142                 this.msg = e.newVal.msg;
143 
144                 if (e.newVal.msg.get("unread")) {                               // If the message is currently unread,
145                     this.timer = Y.later(2000, this, function () {              // Send a request to mark it as read
146                         Y.log("Sending message read update", "info", "InboxDisplay");
147                         this.msg.set("unread", false);
148                         this.dataSource.rest.sendRequest({
149                             request: "/Inbox/Message/Read/" + this.msg.get("id"),
150                             cfg: {
151                                 method: "PUT"
152                             }
153                         });
154                     });
155                 }
156             }
157         }
158 
159     }, {
160         ATTRS: {
161             /**
162              * The target variable, returned either based on the name attribute,
163              * and if absent by evaluating the expr attribute.
164              */
165             variable: {
166                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
167                 _inputex: {
168                     _type: "variableselect",
169                     label: "variable"
170                 }
171             }
172         }
173     });
174 
175     Y.namespace('Wegas').InboxDisplay = InboxDisplay;
176 });
