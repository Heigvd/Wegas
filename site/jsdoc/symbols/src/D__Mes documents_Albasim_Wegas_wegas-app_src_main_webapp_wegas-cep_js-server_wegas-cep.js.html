 35 function setTeamMotivation () {
 36     var i, gm = self.getGameModel(),
 37             listEmployees = VariableDescriptorFacade.findByName(gm, 'employees'),
 38             employeeInstance,
 39             teamMotivation = VariableDescriptorFacade.findByName(gm, 'teamMotivation'),
 40             morals = [],
 41             mSum = 0,
 42             mAverage,
 43             mGap = [],
 44             SumOfSquareOfMGap = 0,
 45             standardDeviation,
 46             newTeamMotivation;
 47     if (!listEmployees || !teamMotivation) {
 48         return;
 49     }
 50 
 51     // calcul arithmetic average of morals (on actives employees only)
 52     for (i = 0; i < listEmployees.items.size(); i++) {
 53         employeeInstance = listEmployees.items.get(i).getInstance(self);
 54         if (employeeInstance.getActive() == true) {
 55             morals.push(parseInt(employeeInstance.getMoral()));
 56             mSum += parseInt(employeeInstance.getMoral());
 57         }
 58     }
 59     mAverage = mSum / morals.length;
 60 
 61     //For each moral calcul gap between moral and average (= moral - average);
 62     //take the sum of each square of gaps (= Sum(n_gaps * n_gaps)). 
 63     for (i = 0; i < morals.length; i++) {
 64         mGap.push(morals[i] - mAverage);
 65         SumOfSquareOfMGap += Math.pow(mGap[i], 2);
 66     }
 67 
 68     // calcul the standard deviation
 69     standardDeviation = Math.sqrt(SumOfSquareOfMGap / morals.length);
 70 
 71     //calcul the new Team Motivation
 72     newTeamMotivation = (mAverage + (mAverage - standardDeviation)) / 2;
 73     if (newTeamMotivation < 0) { //in extrems cases
 74         newTeamMotivation = 0;
 75     }
 76 
 77     //set teamMotivation
 78     teamMotivation.getInstance(self).setValue(Math.round(newTeamMotivation));
 79 }

 81 /**
 82  * Set picture depending of resource's current moral.
 83  */
 84 function changePicture () {
 85     var i, j, valueInst, valueDescr, gm = self.getGameModel(), oldImg, newImg, moral,
 86             listEmployees = VariableDescriptorFacade.findByName(gm, 'employees'),
 87             imgSuffixe = ['Triste', 'Neutre', 'Joie'];
 88     if (!listEmployees) {
 89         return;
 90     }
 91     for (i = 0; i < listEmployees.items.size(); i++) {
 92         valueDescr = listEmployees.items.get(i);
 93         valueInst = valueDescr.getInstance(self);
 94         moral = parseInt(valueInst.getMoral());
 95         oldImg = valueInst.getProperty('picture');
 96         newImg = null;
 97         switch (true) {
 98             case moral < 40 :
 99                 for (j = 0; j < imgSuffixe.length; j++) {
100                     if (oldImg.indexOf(imgSuffixe[j]) > -1) {
101                         newImg = oldImg.replace(imgSuffixe[j], imgSuffixe[0]);
102                         break
103                     }
104                 }
105                 break;
106             case moral < 75 :
107                 for (j = 0; j < imgSuffixe.length; j++) {
108                     if (oldImg.indexOf(imgSuffixe[j]) > -1) {
109                         newImg = oldImg.replace(imgSuffixe[j], imgSuffixe[1]);
110                         break
111                     }
112                 }
113                 break;
114             default :
115                 for (j = 0; j < imgSuffixe.length; j++) {
116                     if (oldImg.indexOf(imgSuffixe[j]) > -1) {
117                         newImg = oldImg.replace(imgSuffixe[j], imgSuffixe[2]);
118                         break
119                     }
120                 }
121                 break;
122         }
123         if (newImg) {
124             valueInst.setProperty('picture', newImg);
125         }
126     }
