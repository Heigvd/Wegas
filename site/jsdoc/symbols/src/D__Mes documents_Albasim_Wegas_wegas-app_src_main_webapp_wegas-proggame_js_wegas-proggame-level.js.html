 14 YUI.add('wegas-proggame-level', function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = 'contentBox', ProgGameLevel;
 18 
 19     /**
 20      *  The level display class, with script input, ia, debugger and
 21      *  terrain display.
 22      *
 23      */
 24     ProgGameLevel = Y.Base.create("wegas-proggame-level", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 25         // *** Fields *** //
 26         handlers: null,
 27         aceField: null,
 28         display: null,
 29         runButton: null,
 30         commandsStack: null,
 31         CONTENT_TEMPLATE: '<div>'
 32         + '<div class="yui3-g top">'
 33 
 34         + '<div class="yui3-u topcenter">'
 35         + '<div class="ui"><h1></h1>'
 36         + '<div class="terrain-ui player-ui"></div>'
 37         + '<div class="terrain-ui enemy-ui"></div>'
 38         + '</div>'
 39         + '<div class="terrain"></div>'
 40         + '</div>'
 41 
 42 
 43         + '<div class="yui3-u topright">'
 44         + '<div class="buttons"></div>'
 45         + '<div class="ai"><h1>Enemy A.I.</h1></div>'
 46         + '<div class="debugger"><h1>Log</h1></div>'
 47         + '</div>'
 48         + '</div>'
 49 
 50         + '<div class="code"><h1>public void main(<span class="arguments"></span>) {</h1><div class="code-content"></div><h1>}</h1></div>'
 51 
 52         + '</div>',
 53 
 54         // *** Lifecycle Methods *** //
 55         initializer: function () {
 56             this.handlers = {};
 57             this.objects = this.get("objects");
 58         },
 59         renderUI: function () {
 60             var i, cb = this.get(CONTENTBOX),
 61             METHODTOTEXT = {
 62                 say: "say(text: String)"
 63             }, api = this.get("api");
 64 
 65             this.aceField = new Y.inputEx.AceField({
 66                 parentEl: cb.one(".code-content"),
 67                 name: 'text',
 68                 type: 'ace',
 69                 height: "300px",
 70                 language: "javascript",
 71                 value: "//Put your code here..."
 72             });
 73 
 74             cb.one(".ai").append(Y.Wegas.App.nl2br(this.findObject("Enemy").ai || "<center><i>empty</i></center>"));
 75             cb.one(".topcenter h1").setHTML(this.get("label"));
 76             cb.one(".arguments").setHTML(this.get("arguments").join(", "));
 77 
 78             var acc = ["<h1>API</h1>"];
 79             for (i = 0; i < api.length; i += 1) {
 80                 acc.push((METHODTOTEXT[api[i].name] || api[i].name + "()") + "<br />");
 81             }
 82             Y.one(".api").setHTML(acc.join(""));
 83 
 84             this.display = new Y.Wegas.ProgGameDisplay(this.toObject());
 85             this.display.render(cb.one(".terrain"));
 86 
 87             this.runButton = new Y.Wegas.Button({
 88                 label: "RUN SCRIPT"
 89             });
 90             this.runButton.render(cb.one(".buttons"));
 91 
 92             this.resetUI();
 93         },
 94         bindUI: function () {
 95             this.handlers.response = Y.Wegas.app.VariableDescriptorFacade.after("response",
 96                 this.syncUI, this);                                             // If data changes, refresh
 97             this.handlers.playerChange = Y.Wegas.app.after('currentPlayerChange',
 98                 this.syncUI, this);                                             // If current user changes, refresh (editor only)
 99 
100             this.runButton.on("click", this.run, this);
101 
102             this.display.after('commandExecuted', this.consumeCommand, this);
103             this.after('commandExecuted', this.consumeCommand, this);
104 
105         },
106         run: function () {
107 
108             if (this.runButton.get("label") === "STOP") {
109                 this.commandsStack = null;
110                 this.runButton.set("label", "RUN SCRIPT");
111                 return;
112             }
113             this.resetUI();
114             this.runButton.set("label", "STOP");
115 
116             Y.Wegas.app.VariableDescriptorFacade.rest.sendRequest({
117                 request: "/ProgGame/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
118                 cfg: {
119                     method: "POST",
120                     data: "JSON.stringify(run("
121                     + "function (name) {" + this.aceField.getValue() + "\n}, "
122                     + Y.JSON.stringify(this.toObject()) + "));"
123                 },
124                 on: {
125                     success: Y.bind(this.onServerReply, this),
126                     failure: Y.bind(function () {
127                         this.runButton.set("label", "RUN SCRIPT");
128                         alert("Your script contains an error.");
129                     }, this)
130                 }
131             });
132         },
133         syncUI: function () {
134             this.display.syncUI();
135             this.syncFrontUI();
136         },
137 
138         destructor: function () {
139             var k;
140             for (k in this.handlers) {
141                 this.handlers[k].detach();
142             }
143             this.aceField.destroy();
144             this.display.destroy();
145             this.runButton.destroy();
146         },
147 
148         resetUI: function () {
149             this.objects = Y.clone(this.get("objects"));
150             this.commandsStack = null;
151 
152             //this.display.set("objects", this.objects);                          // Reset the display to default
153             //this.display.syncUI();
154             this.display.execute({
155                 type:'resetLevel',
156                 objects: this.objects
157             });
158 
159             this.get(CONTENTBOX).one(".debugger").setHTML("<h1>Debugger</h1>");
160             //this.get("contentBox").one(".debugger").empty();
161 
162             this.syncFrontUI();
163         },
164 
165         onServerReply: function (e) {
166             this.commandsStack = Y.JSON.parse(e.response.entity);
167             var i;
168             for (i = 0; i < this.commandsStack.length; i += 1) {
169                Y.log("command: " + this.commandsStack[i].type + ", " + this.commandsStack[i] + ", " + this.commandsStack[i].text);
170             }
171 
172             this.consumeCommand();
173         },
174 
175         findObject: function (id) {
176             return Y.Array.find(this.objects, function (o) {
177                 return o.id === id;
178             });
179         },
180 
181         consumeCommand: function () {
182             if (this.commandsStack && this.commandsStack.length > 0) {
183                 var command = this.commandsStack.shift();
184                 Y.log("consumeCommand" + ", " + command.type + ", " + command);
185 
186                 switch (command.type) {
187 
188                     case "move":
189                     case "fire":
190                         Y.mix(this.findObject(command.object.id),           // Update target object cfg
191                             command.object, true);
192                         this.syncFrontUI();
193                         break;
194 
195                     case "updated":
196                         Y.mix(this.findObject(command.object.id),           // Update target object cfg
197                             command.object, true);
198                         this.syncFrontUI();
199                         this.consumeCommand();
200                         break;
201 
202                     case "gameWon":
203                         this.runButton.set("label", "NEXT LEVEL");
204                         this.runButton.detachAll("click");
205                         this.runButton.on("click", this.doNextLevel, this);
206                         break;
207                     case "log":
208                         this.get("contentBox").one(".debugger").append(command.text + "<br />");
209 
210                         Y.later(500, this, this.consumeCommand);
211                         //this.fire("commandExecuted");
212                         break;
213                     default:
214                         break;
215                 }
216 
217                 this.display.execute(command);                                  // Forware the command to the display
218 
219             } else if (this.commandsStack) {
220                 this.runButton.set("label", "RUN SCRIPT");
221             }
222         },
223 
224 
225         doNextLevel: function () {
226             Y.Wegas.app.VariableDescriptorFacade.rest.sendRequest({
227                 request: "/ProgGame/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
228                 cfg: {
229                     method: "POST",
230                     data: this.get("onWin")
231                 }
232             });
233         },
234 
235         syncFrontUI: function () {
236             var cb = this.get(CONTENTBOX);
237 
238             function updateUI(object, el) {
239                 var i, acc = [];
240                 if (!Y.Lang.isUndefined(object.life)) {
241                     acc.push("Life<div class=\"life\"><span style=\"width:" +object.life + "%;\" ></span></div>")
242                 }
243                 if (!Y.Lang.isUndefined(object.actions)) {
244                     acc.push("Actions<div class=\"actions\">")
245                     for (i = 0; i < object.actions; i += 1) {
246                         acc.push("<span></span>");
247                     }
248                     acc.push("</div>");
249                 }
250                 el.setHTML(acc.join(""));
251                 if (acc.length === 0) {
252                     el.hide();
253                 } else {
254                     el.show();
255                 }
256             }
257             updateUI.call(this, this.findObject("Player"), cb.one(".player-ui"));
258             updateUI.call(this, this.findObject("Enemy"), cb.one(".enemy-ui"));
259         }
260 
261     }, {
262         ATTRS: {
263             label: {
264                 type: "string"
265             },
266             objects: {
267                 _inputex: {
268                     _type: "object"
269                 }
270             },
271             arguments: {
272                 type: "array",
273                 value: []
274             },
275             api: {
276                 type: "array",
277                 value: []
278             },
279             maxTurns: {
280                 type: "string",
281                 format: "Integer",
282                 validator: function (s) {
283                     return (parseInt(s) ? parseInt(s) : 1);
284                 },
285                 _inputex: {
286                     label: "Max turns"
287                 }
288             },
289             winningCondition: {
290                 type: "string",
291                 format: "text"
292             //                _inputex: {
293             //                    _type: "ace"
294             //                }
295             },
296             onStart: {
297                 type: "string",
298                 format: "text"
299             //                _inputex: {
300             //                    _type: "ace"
301             //                }
302             },
303             onTurn: {
304                 type: "string",
305                 format: "text"
306             //                _inputex: {
307             //                    _type: "ace"
308             //                }
309             },
310             onWin: {
311                 type: "string",
312                 _inputex: {
313                     _type: "ace"
314                 }
315             },
316             map: {
317                 validator: Y.Lang.isArray
318             },
319             mapObjects: {
320                 validator: Y.Lang.isArray
321             }
322         }
323     });
324 
325     Y.namespace('Wegas').ProgGameLevel = ProgGameLevel;
326 });
