172 function checkCollision(sourceId, x, y) {
173     var o, k, objects = level.objects;
174     for (k=0; k < objects.length; k++) {
175         o = objects[k];
176         if (o.x === x && o.y === y && o.id !== sourceId &&
177             (o.collides === undefined || o.collides)) {
178             return objects[k];
179         }
180     }
181     return null;

207 Object.prototype.clone = function () {
208     var newObj = (this instanceof Array) ? [] : {};
209     for (var i in this) {
210         if (i == 'clone')
211             continue;
212         if (this[i] && typeof this[i] == "object") {
213             newObj[i] = this[i].clone();
214         } else
215             newObj[i] = this[i]
216     }
217     return newObj;
218 };

220 function dirToVector(dir) {
221     var dirX = 0, dirY = 0;
222     switch (dir) {
223         case 1:
224             dirY = 1;
225             break;
226         case 2:
227             dirX = 1;
228             break;
229         case 3:
230             dirY = -1;
231             break;
232         case 4:
233             dirX = -1;
234             break;
235     }
236     return {
237         x: dirX,
238         y: dirY
239     };
240 }
