 19 jsPlumbUtil = {
 20 	isArray : function(a) {
 21 		return Object.prototype.toString.call(a) === "[object Array]";	
 22 	},
 23 	isString : function(s) {
 24 		return typeof s === "string";
 25 	},
 26 	isObject : function(o) {
 27 		return Object.prototype.toString.call(o) === "[object Object]";	
 28 	},
 29 	convertStyle : function(s, ignoreAlpha) {
 30 		// TODO: jsPlumb should support a separate 'opacity' style member.
 31 		if ("transparent" === s) return s;
 32 		var o = s,
 33 		    pad = function(n) { return n.length == 1 ? "0" + n : n; },
 34 		    hex = function(k) { return pad(Number(k).toString(16)); },
 35 		    pattern = /(rgb[a]?\()(.*)(\))/;
 36 		if (s.match(pattern)) {
 37 			var parts = s.match(pattern)[2].split(",");
 38 			o = "#" + hex(parts[0]) + hex(parts[1]) + hex(parts[2]);
 39 			if (!ignoreAlpha && parts.length == 4) 
 40 				o = o + hex(parts[3]);
 41 		}
 42 		return o;
 43 	},
 44 	gradient : function(p1, p2) {
 45 		p1 = jsPlumbUtil.isArray(p1) ? p1 : [p1.x, p1.y];
 46 		p2 = jsPlumbUtil.isArray(p2) ? p2 : [p2.x, p2.y];			
 47 		return (p2[1] - p1[1]) / (p2[0] - p1[0]);		
 48 	},
 49 	normal : function(p1, p2) {
 50 		return -1 / jsPlumbUtil.gradient(p1,p2);
 51 	},
 52 	lineLength : function(p1, p2) {
 53 		p1 = jsPlumbUtil.isArray(p1) ? p1 : [p1.x, p1.y];
 54 		p2 = jsPlumbUtil.isArray(p2) ? p2 : [p2.x, p2.y];			
 55 		return Math.sqrt(Math.pow(p2[1] - p1[1], 2) + Math.pow(p2[0] - p1[0], 2));			
 56 	},
 57     segment : function(p1, p2) {
 58         p1 = jsPlumbUtil.isArray(p1) ? p1 : [p1.x, p1.y];
 59         p2 = jsPlumbUtil.isArray(p2) ? p2 : [p2.x, p2.y];
 60         if (p2[0] > p1[0]) {
 61             return (p2[1] > p1[1]) ? 2 : 1;
 62         }
 63         else {
 64             return (p2[1] > p1[1]) ? 3 : 4;
 65         }
 66     },
 67     intersects : function(r1, r2) {
 68     	var x1 = r1.x, x2 = r1.x + r1.w, y1 = r1.y, y2 = r1.y + r1.h,
 69     		a1 = r2.x, a2 = r2.x + r2.w, b1 = r2.y, b2 = r2.y + r2.h;
 70     		
 71     	return  ( (x1 <= a1 && a1 <= x2) && (y1 <= b1 && b1 <= y2) ) ||
 72     			( (x1 <= a2 && a2 <= x2) && (y1 <= b1 && b1 <= y2) ) ||
 73     			( (x1 <= a1 && a1 <= x2) && (y1 <= b2 && b2 <= y2) ) ||
 74     			( (x1 <= a2 && a1 <= x2) && (y1 <= b2 && b2 <= y2) ) ||
 75     			
 76     			( (a1 <= x1 && x1 <= a2) && (b1 <= y1 && y1 <= b2) ) ||
 77     			( (a1 <= x2 && x2 <= a2) && (b1 <= y1 && y1 <= b2) ) ||
 78     			( (a1 <= x1 && x1 <= a2) && (b1 <= y2 && y2 <= b2) ) ||
 79     			( (a1 <= x2 && x1 <= a2) && (b1 <= y2 && y2 <= b2) );
 80     },
 81     segmentMultipliers : [null, [1, -1], [1, 1], [-1, 1], [-1, -1] ],
 82     inverseSegmentMultipliers : [null, [-1, -1], [-1, 1], [1, 1], [1, -1] ],
 83 	pointOnLine : function(fromPoint, toPoint, distance) {
 84         var m = jsPlumbUtil.gradient(fromPoint, toPoint),
 85             s = jsPlumbUtil.segment(fromPoint, toPoint),
 86 		    segmentMultiplier = distance > 0 ? jsPlumbUtil.segmentMultipliers[s] : jsPlumbUtil.inverseSegmentMultipliers[s],
 87 			theta = Math.atan(m),
 88     		y = Math.abs(distance * Math.sin(theta)) * segmentMultiplier[1],
 89 			x =  Math.abs(distance * Math.cos(theta)) * segmentMultiplier[0];
 90 		return { x:fromPoint.x + x, y:fromPoint.y + y };
 91 	},
 92     /**
 93      * calculates a perpendicular to the line fromPoint->toPoint, that passes through toPoint and is 'length' long.
 94      * @param fromPoint
 95      * @param toPoint
 96      * @param length
 97      */
 98 	perpendicularLineTo : function(fromPoint, toPoint, length) {
 99 		var m = jsPlumbUtil.gradient(fromPoint, toPoint),
100             theta2 = Math.atan(-1 / m),
101     		y =  length / 2 * Math.sin(theta2),
102 			x =  length / 2 * Math.cos(theta2);
103 		return [{x:toPoint.x + x, y:toPoint.y + y}, {x:toPoint.x - x, y:toPoint.y - y}];
104 	},
105 	findWithFunction : function(a, f) {
106     	if (a)
107   			for (var i = 0; i < a.length; i++) if (f(a[i])) return i;
108 		return -1;
109 	},
110 	indexOf : function(l, v) {
111 		return jsPlumbUtil.findWithFunction(l, function(_v) { return _v == v; });	
112 	},
113     removeWithFunction : function(a, f) {
114         var idx = jsPlumbUtil.findWithFunction(a, f);
115         if (idx > -1) a.splice(idx, 1);
116         return idx != -1;
117     },
118     remove : function(l, v) {
119     	var idx = jsPlumbUtil.indexOf(l, v);	
120     	if (idx > -1) l.splice(idx, 1);
121         return idx != -1;
122     },
123     // TODO support insert index
124     addWithFunction : function(list, item, hashFunction) {
125         if (jsPlumbUtil.findWithFunction(list, hashFunction) == -1) list.push(item);
126     },
127 	addToList : function(map, key, value) {
128 		var l = map[key];
129 		if (l == null) {
130 			l = [], map[key] = l;
131 		}
132 		l.push(value);
133 		return l;
134 	},	
135 	/**
136 	 * EventGenerator
137 	 * Superclass for objects that generate events - jsPlumb extends this, as does jsPlumbUIComponent, which all the UI elements extend.
138 	 */
139 	EventGenerator : function() {
140 		var _listeners = {}, self = this;
141 		
142 		// this is a list of events that should re-throw any errors that occur during their dispatch. as of 1.3.0 this is private to
143 		// jsPlumb, but it seems feasible that people might want to manipulate this list.  the thinking is that we don't want event
144 		// listeners to bring down jsPlumb - or do we.  i can't make up my mind about this, but i know i want to hear about it if the "ready"
145 		// event fails, because then my page has most likely not initialised.  so i have this halfway-house solution.  it will be interesting
146 		// to hear what other people think.
147 		var eventsToDieOn = [ "ready" ];
148 							    
149 		/*
150 		 * Binds a listener to an event.  
151 		 * 
152 		 * Parameters:
153 		 * 	event		-	name of the event to bind to.
154 		 * 	listener	-	function to execute.
155 		 */
156 		this.bind = function(event, listener) {
157 			jsPlumbUtil.addToList(_listeners, event, listener);		
158 			return self;		
159 		};
160 		/*
161 		 * Fires an update for the given event.
162 		 * 
163 		 * Parameters:
164 		 * 	event				-	event to fire
165 		 * 	value				-	value to pass to the event listener(s).
166 		 *  originalEvent	 	- 	the original event from the browser
167 		 */			
168 		this.fire = function(event, value, originalEvent) {
169 			if (_listeners[event]) {
170 				for ( var i = 0; i < _listeners[event].length; i++) {
171 					// doing it this way rather than catching and then possibly re-throwing means that an error propagated by this
172 					// method will have the whole call stack available in the debugger.
173 					if (jsPlumbUtil.findWithFunction(eventsToDieOn, function(e) { return e === event}) != -1)
174 						_listeners[event][i](value, originalEvent);
175 					else {
176 						// for events we don't want to die on, catch and log.
177 						try {
178 							_listeners[event][i](value, originalEvent);
179 						} catch (e) {
180 							jsPlumbUtil.log("jsPlumb: fire failed for event " + event + " : " + e);
181 						}
182 					}
183 				}
184 			}
185 			return self;
186 		};
187 		/*
188 		 * Clears either all listeners, or listeners for some specific event.
189 		 * 
190 		 * Parameters:
191 		 * 	event	-	optional. constrains the clear to just listeners for this event.
192 		 */
193 		this.unbind = function(event) {
194 			if (event)
195 				delete _listeners[event];
196 			else {
197 				delete _listeners;
198 				_listeners = {};
199 			}
200 			return self;
201 		};
202 		
203 		this.getListener = function(forEvent) {
204 			return _listeners[forEvent];
205 		};		
206 	},
