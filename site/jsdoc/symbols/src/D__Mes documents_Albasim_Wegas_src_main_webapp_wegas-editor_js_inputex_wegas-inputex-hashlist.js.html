1 /**
  2  * @module inputex-HashList
  3  */
  4 YUI.add("wegas-inputex-hashlist", function(Y){
  5 
  6     var inputEx = Y.inputEx;
  7 
  8     inputEx.HashList = function(options) {
  9         inputEx.HashList.superclass.constructor.call(this, options);
 10     };
 11 
 12     Y.extend(inputEx.HashList, inputEx.ListField, {
 13 
 14         /**
 15          * Convert the array of 2d elements to an javascript object
 16          */
 17         seq: 2000,
 18         getValue: function() {
 19             var v = inputEx.HashList.superclass.getValue.call(this);
 20             var obj = {}, id;
 21 
 22             for(var i = 0 ; i < v.length ; i++) {
 23                 id = v[i]["id"];
 24                 if (!id) {
 25                     id = this.seq;
 26                     obj[id] = this.seq;
 27                     this.seq++;
 28                 }
 29                 obj[ id] = v[i];
 30             }
 31             return obj;
 32         },
 33 
 34         /**
 35          * Convert the object into a list of pairs
 36          */
 37         setValue: function(v) {
 38             var val = [];
 39             for(var key in v) {
 40                 if( v.hasOwnProperty(key) ) {
 41                     val.push(v[key]);
 42                 }
 43             }
 44             inputEx.HashList.superclass.setValue.call(this,val);
 45         }
 46     });
 47 
 48     // Register this class as "object" type
 49     inputEx.registerType('hashlist', inputEx.HashList);
 50 
 51 });
