 12 YUI.add("wegas-injector", function(Y) {
 13     "use strict";
 14     var MATCHER = "[data-file]",
 15             Injector = function(cfg) {
 16         Injector.superclass.constructor.apply(this, arguments);
 17     },
 18             parser = function(element) {
 19         var nodeName = element.getDOMNode().nodeName;
 20         switch (nodeName) {
 21             case "IMG":
 22                 if (!element.hasAttribute("src")) {
 23                     element.setAttribute("src", Y.Wegas.app.get("dataSources").File.source + "read" + element.getAttribute("data-file"));
 24                     element.removeAttribute("data-file");
 25                 }
 26                 break;
 27             default:
 28                 if (!element.hasAttribute("href")) {
 29                     element.setAttribute("href", Y.Wegas.app.get("dataSources").File.source + "read" + element.getAttribute("data-file"));
 30                     element.removeAttribute("data-file");
 31                 }
 32         }
 33     };
 34 
 35     Y.extend(Injector, Y.Plugin.Base, {
 36         initializer: function() {
 37             Y.mix(Y.Node.DOM_EVENTS, {
 38                 DOMNodeInserted: true,
 39                 DOMNodeRemoved: true,
 40                 DOMCharacterDataModified: true
 41             });
 42             this.gallery = false;
 43             this.insertEvent = this.get("host").get("boundingBox").delegate("DOMNodeInserted", function(e) {
 44                 var list = e.currentTarget.all(MATCHER);
 45                 e.halt(true);
 46                 list.each(function(item) {
 47                     parser(item);
 48                 });
 49                 this.instanciateGallery(e.currentTarget);
 50             }, function(item) {
 51                 return (item.all(MATCHER).size() > 0);
 52             }, this);
 53             this.renderEvent = this.afterHostEvent("*:render", function(e) {
 54                 var list = e.currentTarget.get("boundingBox").all(MATCHER);
 55                 list.each(function(item) {
 56                     parser(item);
 57                 });
 58                 this.instanciateGallery(e.currentTarget.get("boundingBox"));
 59             }, this);
 60         },
 61         instanciateGallery: function(element) {
 62             /* Check for gallery elements and loads it*/
 63             if (!this.gallery && (element.one(".light-picture") || element.one(".light-gallery"))) {
 64                 if (!Injector.GALLERY) {
 65                     Y.use("wegas-gallery", function() {
 66                         Injector.GALLERY = new Y.Wegas.WegasGallery({
 67                             "lightGallery": true
 68                         });
 69                     });
 70                 }
 71                 Injector.GALLERY_COUNTER += 1;
 72                 this.gallery = true;
 73             }
 74         },
 75         destructor: function() {
 76             this.insertEvent.detach();
 77             this.renderEvent.detach();
 78             if (this.gallery && Injector.GALLERY) {
 79                 Injector.GALLERY_COUNTER -= 1;
 80                 if (Injector.GALLERY_COUNTER === 0) {
 81                     Injector.GALLERY.destroy();
 82                 }
 83             }
 84         }
 85     });
 86     Injector.NAME = "Injector";
 87     Injector.NS = "injector";
 88     Injector.GALLERY = null;
 89     Injector.GALLERY_COUNTER = 0;
 90     Injector.ATTRS = {
 91     };
 92 
 93     Y.Plugin.Injector = Injector;
 94 });
