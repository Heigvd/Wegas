 16 YUI.add('wegas-joingamewidget', function (Y) {
 17     "use strict";
 18 
 19     var CONTENTBOX = 'contentBox',
 20     JoinGameWidget;
 21 
 22     /**
 23     * @name Y.Wegas.JoinGameWidget
 24     * @extends Y.Widget
 25     * @class  class for join a game and a team
 26     * @constructor
 27     * @description Allows to join a game by token or a public game. Then you can
 28     * join or create a new team
 29     */
 30     JoinGameWidget = Y.Base.create("wegas-joingamewidget", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 31         /**
 32          * @lends Y.Wegas.JoinGameWidget#
 33          */
 34         // *** Private fields *** //
 35 
 36         /**
 37          * @function
 38          * @private
 39          * @description All button and fields are created.
 40          * For creating the field inputEx libary is used
 41          */
 42         renderUI: function () {
 43             var cb = this.get(CONTENTBOX);
 44 
 45             this.tokenField = new Y.inputEx.StringField({                       // Render
 46                 required: false,
 47                 parentEl: cb,
 48                 label: "Enter a key phrase to join a game",
 49                 typeInvite: "Enter a token"
 50             });
 51 
 52             this.p = Y.Node.create('<div class="lobbyOr"><p>Or</p><div>');
 53             cb.append(this.p);
 54 
 55             this.selectPublicGame = new Y.inputEx.SelectField({                 // Render public games
 56                 required: false,
 57                 parentEl: cb,
 58                 label: "Select a public game"
 59             });
 60 
 61             this.joinGameButton = new Y.Button({
 62                 label: "Join game"
 63             });
 64             this.joinGameButton.render(cb);
 65 
 66             this.teamsField = new Y.inputEx.SelectField({                       // Render team selection
 67                 required: "true",
 68                 parentEl: cb,
 69                 label: "Select the team you want to join"
 70             });
 71             this.teamsField.hide();
 72             this.joinTeamButton = new Y.Button({
 73                 label: "Join team",
 74                 visible: false
 75             });
 76             this.joinTeamButton.render(cb);
 77 
 78             this.createTeamField = new Y.inputEx.StringField({                  // Render team creation
 79                 required: "true",
 80                 parentEl: cb,
 81                 label: "OR Create a new team:"
 82             });
 83             this.createTeamField.hide();
 84             this.createButton = new Y.Button({
 85                 label: "Create team",
 86                 visible: false
 87             });
 88             this.createButton.render(cb);
 89 
 90             this.showPublicGames();
 91         },
 92 
 93         /**
 94          * @function
 95          * @private
 96          * @description All events are added to the buttons
 97          * Create team button call rest url : rest/GameModel/{gameModelID}/Game/{gameID}/CreateTeam/{teamName}
 98          */
 99         bindUI: function () {
100             this.tokenField.on("updated", function (e) {
101                 if (this.tokenField.getValue() !== "") {
102                     this.selectPublicGame.setValue("");
103                 }
104             }, this);
105 
106             this.selectPublicGame.on("updated", function (e) {
107                 if (this.selectPublicGame.getValue() !== "") {
108                     this.tokenField.setValue("");
109                 }
110             }, this);
111 
112             this.joinGameButton.on("click", function (e) {                      // join a game based on a token
113                 if (this.tokenField.getValue() != ""){
114                     this.sendJoinGame();
115                 }else if (this.selectPublicGame.getValue() != ""){
116                     this.currentGame = this.selectPublicGame.getValue();
117                     this.showTeams();
118                     this.selectPublicGame.removeChoice({
119                         value: this.currentGame
120                     });
121                 }else{
122                     this.showMessage("error", "A key phrase or a public Game must be selected");
123                     return;
124                 }
125             }, this);
126 
127             this.joinTeamButton.on("click", function (e) {                      // Join an existing team
128                 if (this.teamsField.validate()) {
129                     this.sendJoinTeamRequest(this.teamsField.getValue());
130                 }
131             }, this);
132             this.createButton.on("click", function (e) {                      // Create a new team
133                 if (this.createTeamField.validate()) {
134                     Y.Wegas.GameFacade.rest.sendRequest({
135                         request: "/" + this.currentGame.get("id") + "/CreateTeam/" + this.createTeamField.getValue(),
136                         cfg: {
137                             method: "POST"
138                         },
139                         on: {
140                             success: Y.bind(function (e) {
141                                 this.sendJoinTeamRequest(e.response.entity.get("id"));
142                             }, this),
143                             failure: Y.bind(function (e) {
144                                 this.showMessage("error", e.response.results.message || "Error creating team", 4000);
145                             }, this)
146                         }
147                     });
148                 }
149             }, this);
150         },
151 
152         /**
153          * @function
154          * @private
155          * @description Method for join a game by token
156          * Call rest request for join the game : rest/GameModel/1/Game/{gameModelID}/JoinGame/{token}
157          */
158         sendJoinGame: function(){
159             Y.Wegas.GameFacade.rest.sendRequest({
160                 request: "/JoinGame/" + this.tokenField.getValue(),
161                 on: {
162                     success: Y.bind(function (e) {
163                         if (e.response.entity                               // If the returned value is a Team enity
164                             instanceof Y.Wegas.persistence.Team) {
165                             this.sendJoinTeamRequest(                       // it means we can join this team directly
166                                 e.response.entity.get("id"));
167                         } else {
168                             this.currentGame = e.response.entity;
169                             this.showTeams();                               // otherwise the player can choose or create its team
170                         }
171                     }, this),
172                     failure: Y.bind(function (e) {
173                         this.showMessage("error", e.response.results.message || "Invalid token", 4000);
174                     }, this)
175                 }
176             });
177         },
178 
179         /**
180          * @function
181          * @private
182          * @description Hide and show all necessary field
183          */
184         showTeams: function () {
185             //this.msg.empty();
186 
187             this.joinGameButton.hide();
188             this.tokenField.hide();
189             this.selectPublicGame.hide();
190             this.p.hide();
191             this.joinTeamButton.show();
192             this.teamsField.show();
193             this.createButton.show();
194             this.createTeamField.show();
195 
196             var i;
197             this.teams = this.currentGame.get("teams");
198             for (i = 0; i < this.teams.length; i = i + 1) {
199                 this.teamsField.addChoice({
200                     label: this.teams[i].get("name"),
201                     value: this.teams[i].get("id")
202                 });
203             }
204             if (this.teams.length == 0) {
205                 this.teamsField.addChoice({
206                     label: "-- No team --",
207                     value: ""
208                 });
209             }
210         },
211 
212         /**
213          * @function
214          * @private
215          * @description Add all public game in a listbox
216          * Use rest request for get public games: rest/PublicGames/Games/{userID}
217          */
218         showPublicGames: function () {
219             this.selectPublicGame.addChoice({
220                 label: "--Select--",
221                 value: ""
222             });
223 
224             Y.Wegas.PublicGamesFacade.rest.sendRequest({
225                 request: "/Games/" + Y.Wegas.app.get('currentUser').id,
226                 on: {
227                     success: Y.bind(function (e) {
228                         var data = e.response.results.entities;
229                         Y.Array.forEach(data, function (game) {
230                             this.selectPublicGame.addChoice({
231                                 label: game.get("name"),
232                                 value: game
233                             });
234                         }, this);
235                     }, this),
236                     failure: Y.bind(function (e) {
237                         this.showMessage("error", "Error");
238                     }, this)
239                 }
240             });
241         },
242 
243         /**
244          * @function
245          * @private
246          * @description User rest request: rest/GameModel/1/Game/{gameModelID}/JoinTeam/{teamID}
247          */
248         sendJoinTeamRequest: function (teamId) {
249             Y.Wegas.GameFacade.rest.sendRequest({
250                 request: "/JoinTeam/" + teamId,
251                 on: {
252                     success: Y.bind(function (e) {
253                         this.joinTeamSuccess();
254                     }, this),
255                     failure: Y.bind(function (e) {
256                         this.showMessage("error", "Error joining team");
257                     }, this)
258                 }
259             });
260         },
261 
262         /**
263          * @function
264          * @private
265          * @description Display the button and field for join a game by token or a public game
266          */
267         joinTeamSuccess : function() {
268             this.showMessage("success", "Game joined, it has been added to your games", 10000);
269 
270             Y.Wegas.RegisteredGamesFacade.rest.clearCache();
271             Y.Wegas.RegisteredGamesFacade.sendInitialRequest();
272 
273             this.joinGameButton.show();
274             this.tokenField.show();
275             this.selectPublicGame.show();
276             this.p.show();
277             this.createButton.hide();
278             this.createTeamField.hide();
279             this.joinTeamButton.hide();
280             this.removeAllTeamsChoices();
281             this.teamsField.hide();
282             this.selectPublicGame.setValue("");
283             this.tokenField.setValue("");
284             this.createTeamField.setValue("");
285         },
286 
287         /**
288          * @function
289          * @private
290          * @description Remove all teams from the listbox
291          */
292         removeAllTeamsChoices: function(){
293             var i;
294             for (i = 0; i < this.teams.length; i = i + 1) {
295                 this.teamsField.removeChoice({
296                     value: this.teams[i].get("id")
297                 });
298             }
299         }
300     });
301 
302     Y.namespace('Wegas').JoinGameWidget = JoinGameWidget;
303 });
