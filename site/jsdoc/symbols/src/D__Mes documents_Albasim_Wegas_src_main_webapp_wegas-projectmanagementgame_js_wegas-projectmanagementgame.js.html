  5 YUI.add('wegas-projectmanagementgame', function (Y) {
  6     "use strict";
  7 
  8     var CONTENTBOX = 'contentBox',
  9         PMGChoiceDisplay;
 10 
 11     PMGChoiceDisplay = Y.Base.create("wegas-pmgchoicedisplay", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 12 
 13         _tabView: null,
 14 
 15         // *** Lifecycle Methods *** //
 16         bindUI: function () {
 17             this.get(CONTENTBOX).delegate("click", function (e) {
 18                 Y.Wegas.app.dataSources.VariableDescriptor.rest.getRequest("MCQVariable/SelectReply/" + e.target.get('id')
 19                     + "/Player/" + Y.Wegas.app.get('currentPlayer'));
 20             }, "input[type=submit]", this);
 21 
 22             Y.Wegas.app.dataSources.VariableDescriptor.after("response", this.syncUI, this);
 23 
 24             Y.Wegas.app.after('currentPlayerChange', this.syncUI, this);
 25         },
 26 
 27         syncUI: function () {
 28             var questions = Y.Wegas.app.dataSources.VariableDescriptor.rest.find('@class', "MCQDescriptor"),
 29                 selectedTab = this._tabView.get('selection'),
 30                 lastSelection = (selectedTab) ? selectedTab.get('index') : 0,
 31                 i,
 32                 j,
 33                 cReplyLabel,
 34                 cQuestion,
 35                 ret,
 36                 cQuestionInstance,
 37                 firstChild;
 38 
 39             this._tabView.removeAll();
 40 
 41             for (i = 0; i < questions.length; i += 1) {
 42                 cQuestion = questions[i];
 43                 ret = [];
 44                 cQuestionInstance = cQuestion.getInstance();
 45                 firstChild = true;
 46 
 47                 if (cQuestionInstance.active) {
 48                     if (cQuestionInstance.replies.length === 0 || cQuestion.allowMultipleReplies) {
 49 
 50                         ret.push('<div class="h4">Answers</div><div class="replies">');
 51 
 52                         for (j = 0; j < cQuestion.replies.length; j += 1) {
 53                             ret.push('<div class="reply ' + ((firstChild) ? "first-child" : "") + '">'
 54                                 + '<bold>' + cQuestion.replies[j].name + '</bold>'
 55                                 + '<div>' + cQuestion.replies[j].description + '</div>'
 56                                 + '<input type="submit" id="' + cQuestion.replies[j].id + '" value="Submit"></input>'
 57                                 + '<div style="clear:both"></div>'
 58                                 + '</div>');
 59                             firstChild = false;
 60                         }
 61                         ret.push('</div>');
 62                     } else {
 63 
 64                         ret.push('<div class="h4">Selected answer</div><div class="replies">');
 65 
 66                         //for (var j=0; j<cQuestion.replies.length; j++) {
 67                         ret.push('<div class="reply first-child">'
 68                             + '<bold>' + cQuestionInstance.replies[0].name + '</bold>'
 69                             + '<div>' + cQuestionInstance.replies[0].description + '</div>'
 70                             + '<div style="clear:both"></div>'
 71                             + '</div>');
 72                         //}
 73                         ret.push('</div><div class="h4">Results</div>',
 74                             '<div class="replies"><div class="reply first-child">', cQuestionInstance.replies[0].answer, '</div></div>');
 75                     }
 76                     cReplyLabel = (cQuestionInstance.replies.length === 0) ? 'unanswered' : cQuestionInstance.replies[0].name;
 77                     this._tabView.add({
 78                         label: '<div class="left">' + cQuestion.label + '</div><div class="right">' + cReplyLabel + '</div>',
 79                         content: '<div class="h2">Details</div>'
 80                             + '<div class="content">'
 81                             + '<div class="h4">' + cQuestion.label + '</div>'
 82                             + '<div class="description">' + cQuestion.description + '</div>'
 83                             + ret.join('') + '</div>'
 84                     });
 85                 }
 86             }
 87             /* @fixme */
 88             Y.later(100, this, function() {
 89                 this._tabView.selectChild(lastSelection);
 90             });
 91         },
 92         renderUI: function () {
 93             this._tabView = new Y.TabView({
 94                 children: []
 95             });
 96 
 97             this._tabView.render(this.get(CONTENTBOX).append('<div></div>'));
 98             this.get(CONTENTBOX).one("ul").prepend(
 99                 '<li class="yui3-tab yui3-widget" role="presentation"><div class="h2">Choices to be made during this week</div></li>'
100                     + '<li class=" yui3-widget pmg-subtitle" role="presentation"><div class="left">Name</div><div class="right">Your choice</div><div style="clear:both"></div></li>'
101             );
102         }
103     });
104 
105     Y.namespace('Wegas').PMGChoiceDisplay = PMGChoiceDisplay;
106 });
