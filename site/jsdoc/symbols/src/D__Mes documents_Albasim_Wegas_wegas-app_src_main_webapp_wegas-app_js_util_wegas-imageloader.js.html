 15 YUI.add('wegas-imageloader', function(Y) {
 16     "use strict";
 17 
 18     /**
 19      * Image objects to be registered with the groups
 20      * @class Y.Wegas.ImgageLoader
 21      * @extends Y.ImgLoadImgObj
 22      * @constructor
 23      */
 24     var BlobBuilder = window.MozBlobBuilder || window.WebKitBlobBuilder || window.BlobBuilder,
 25             ImgageLoader = function() {
 26         Y.Wegas.ImgageLoader.superclass.constructor.apply(this, arguments);
 27     };
 28 
 29     ImgageLoader.NAME = 'ImgageLoader';
 30 
 31     ImgageLoader.ATTRS = {
 32         target: {}
 33     };
 34 
 35     Y.extend(ImgageLoader, Y.ImgLoadImgObj, {
 36         _getImgEl: function() {
 37 
 38             return this.get("target");
 39 
 40             if (this._imgEl === null) {
 41                 this._imgEl = Y.one('#' + this.get('domId'));
 42             }
 43             return this._imgEl;
 44         },
 45         fetch: function(withinY) {
 46             if (this._fetched) {
 47                 return true;
 48             }
 49 
 50             var yPos, el = this._getImgEl();
 51 
 52             if (!el) {
 53                 return false;
 54             }
 55 
 56             if (withinY) {
 57                 // need a distance check
 58                 yPos = this._getYPos();
 59                 if (!yPos || yPos > withinY) {
 60                     return false;
 61                 }
 62                 Y.log('Image with id "' + this.get('domId') + '" is within distance of the fold. Fetching image.', 'info', 'imageloader');
 63             }
 64 
 65             Y.log('Fetching image with id "' + this.get('domId') + '".', 'info', 'imageloader');
 66 
 67             // apply url
 68             if (this.get('bgUrl') !== null) {
 69                 // bg url
 70                 if (this.get('isPng') && Y.UA.ie && Y.UA.ie <= 6) {
 71                     // png for which to apply AlphaImageLoader
 72                     el.setStyle('filter', 'progid:DXImageTransform.Microsoft.AlphaImageLoader(src="' + this.get('bgUrl') + '", sizingMethod="' + this.get('sizingMethod') + '", enabled="' + this.get('enabled') + '")');
 73                 } else {
 74                     el.setStyle('backgroundImage', "url('" + this.get('bgUrl') + "')");// regular bg image
 75                 }
 76             } else if (this.get('srcUrl') !== null) {
 77                 this.loadImage(el, "src");
 78             }
 79 
 80             if (this.get('setVisible')) {                                       // apply attributes
 81                 el.setStyle('visibility', 'visible');
 82             }
 83             if (this.get('width')) {
 84                 el.setAttribute('width', this.get('width'));
 85             }
 86             if (this.get('height')) {
 87                 el.setAttribute('height', this.get('height'));
 88             }
 89 
 90             this._fetched = true;
 91 
 92             return true;
 93         },
 94         loadImage: function(el, attr) {
 95 
 96             // Method 1, using BlobBuilder & XMLHttpRequest
 97             if (window.XMLHttpRequest && window.BlobBuilder) {
 98                 var request = new XMLHttpRequest();
 99                 request.onload = Y.bind(function(loadEvt) {
100                     if (loadEvt.target.status === 200) {
101                         var blob, bb = new BlobBuilder(),
102                                 reader = new FileReader();
103 
104                         bb.append(loadEvt.target.response);                     // Note: not request.responseText
105                         blob = bb.getBlob(loadEvt.target.getResponseHeader("Content-Type"));
106 
107                         reader.onload = Y.bind(function(e) {
108                             el.setAttribute(attr, e.target.result);
109                             this.fire("load", {
110                                 meta: {
111                                     contentType: loadEvt.target.getResponseHeader("Content-Type"),
112                                     description: loadEvt.target.getResponseHeader("Description")
113                                 }
114                             });
115                         }, this);
116                         reader.readAsDataURL(blob);
117                     }
118                 }, this);
119                 request.open("GET", this.get('srcUrl'), true);
120                 request.responseType = "arraybuffer";
121                 request.send(null);
122                 return;
123             } else if (window.GM_xmlhttpRequest) {                              // Method 2, using overrideMimeType
124                 GM_xmlhttpRequest({
125                     method: 'GET',
126                     url: this.get('srcUrl'),
127                     onload: function(respDetails) {
128                         var binResp = customBase64Encode(respDetails.responseText);
129                         el.setAttribute(attr, 'data:image/png;base64,' + binResp);
130                     },
131                     overrideMimeType: 'text/plain; charset=x-user-defined'
132                 });
133                 return;
134             }
135 
136             // Method 3 (fallback), set img src
137             el.setAttribute(attr, this.get('srcUrl'));
138             el.once("load", function(e) {
139                 this.fire("load", {
140                     meta: {}
141                 });
142             }, this);
143         }
144     });
145 
146     Y.namespace("Wegas").ImgageLoader = ImgageLoader;

148     function customBase64Encode(inputStr) {
149         var
150                 bbLen = 3,
151                 enCharLen = 4,
152                 inpLen = inputStr.length,
153                 inx = 0,
154                 jnx,
155                 keyStr = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz"
156                 + "0123456789+/=",
157                 output = "",
158                 paddingBytes = 0;
159         var
160                 bytebuffer = new Array(bbLen),
161                 encodedCharIndexes = new Array(enCharLen);
162 
163         while (inx < inpLen) {
164             for (jnx = 0; jnx < bbLen; ++jnx) {
165                 /*--- Throw away high-order byte, as documented at:
166                  https://developer.mozilla.org/En/Using_XMLHttpRequest#Handling_binary_data
167                  */
168                 if (inx < inpLen)
169                     bytebuffer[jnx] = inputStr.charCodeAt(inx++) & 0xff;
170                 else
171                     bytebuffer[jnx] = 0;
172             }
173 
174             /*--- Get each encoded character, 6 bits at a time.
175              index 0: first  6 bits
176              index 1: second 6 bits
177              (2 least significant bits from inputStr byte 1
178              + 4 most significant bits from byte 2)
179              index 2: third  6 bits
180              (4 least significant bits from inputStr byte 2
181              + 2 most significant bits from byte 3)
182              index 3: forth  6 bits (6 least significant bits from inputStr byte 3)
183              */
184             encodedCharIndexes[0] = bytebuffer[0] >> 2;
185             encodedCharIndexes[1] = ((bytebuffer[0] & 0x3) << 4) | (bytebuffer[1] >> 4);
186             encodedCharIndexes[2] = ((bytebuffer[1] & 0x0f) << 2) | (bytebuffer[2] >> 6);
187             encodedCharIndexes[3] = bytebuffer[2] & 0x3f;
188 
189             //--- Determine whether padding happened, and adjust accordingly.
190             paddingBytes = inx - (inpLen - 1);
191             switch (paddingBytes) {
192                 case 1:
193                     // Set last character to padding char
194                     encodedCharIndexes[3] = 64;
195                     break;
196                 case 2:
197                     // Set last 2 characters to padding char
198                     encodedCharIndexes[3] = 64;
199                     encodedCharIndexes[2] = 64;
200                     break;
201                 default:
202                     break; // No padding - proceed
203             }
204 
205             /*--- Now grab each appropriate character out of our keystring,
206              based on our index array and append it to the output string.
207              */
208             for (jnx = 0; jnx < enCharLen; ++jnx)
209                 output += keyStr.charAt(encodedCharIndexes[jnx]);
210         }
211         return output;
212     }
