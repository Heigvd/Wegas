 15 YUI.add('wegas-layout', function (Y) {
 16     "use strict";
 17 
 18     var Layout;
 19 
 20     Layout = Y.Base.create("wegas-layout", Y.Widget, [ Y.Wegas.Widget, Y.WidgetChild, Y.WidgetStdMod ], {
 21 
 22         // ** Lifecycle Methods ** //
 23 
 24         renderUI: function () {
 25             this.renderPosition('top');
 26             this.renderPosition('left');
 27             this.renderPosition('center');
 28             this.renderPosition('right');
 29             //this.renderPosition('bottom');
 30 
 31             this._syncUIStdMod();
 32         },
 33 
 34         bindUI: function () {
 35             Y.on( "windowresize", Y.bind( this.syncUI, this ) );                // Sync the layout whenever the windows is resized
 36             //this.get("boundingBox").on("resize", this._syncUIStdMod, this);
 37             Y.on( 'domready', this.syncUI, this);
 38         },
 39 
 40         syncUI: function () {
 41             this.syncCenterNode();
 42             this._syncUIStdMod();
 43         },
 44 
 45         // ** Private Methods ** //
 46 
 47         getPositionNode: function ( position ) {
 48             var target;
 49             if (position === "top") {
 50                 target = this.getStdModNode("header");
 51 
 52             } else if (position === "bottom") {
 53                 target = this.getStdModNode("footer");
 54 
 55             } else if (position == "left") {
 56                 target = this.getStdModNode("body").one(".wegas-layout-left");
 57 
 58             } else if (position == "center") {
 59                 target = this.getStdModNode("body").one(".wegas-layout-center");
 60 
 61             } else if (position == "right") {
 62                 target = this.getStdModNode("body").one(".wegas-layout-right");
 63             }
 64             return target;
 65         },
 66 
 67         hidePosition: function ( position ) {
 68             var anim = new Y.Anim({
 69                 node: this.getPositionNode( position ),
 70                 to: {
 71                     width: "0px"
 72                 },
 73                 easing: 'easeIn',
 74                 duration: 0.6
 75             });
 76             anim.on( 'tween', this.syncCenterNode, this );
 77             //anim.on( 'end', this.syncCenterNode, this );
 78             anim.run();
 79         },
 80 
 81         showPosition: function ( position ) {
 82             var anim,
 83             target = this.getPositionNode( position );
 84 
 85             if ( target.getStyle( "width" ) === "0px" ) {                       // Only display if hidden
 86                 anim = new Y.Anim({
 87                     node: this.getPositionNode( position ),
 88                     to: {
 89                         width: "350px"
 90                     },
 91                     easing: 'easeOut',
 92                     duration: 0.6
 93                 });
 94                 anim.on( 'tween', this.syncCenterNode, this );
 95                 //anim.on( 'end ', this.syncCenterNode, this );
 96                 anim.run();
 97             }
 98         },
 99 
100         renderPosition: function ( position ) {
101             var i, cWidget,
102             target = this.getPositionNode( position ),
103             positionCfg = this.get( position );
104 
105 
106             if ( positionCfg ) {                                                // If there is a provided configuration
107                 if (position == "left") {
108                     this.leftResize = new Y.Resize({
109                         node: target,
110                         handles: 'r'
111                     });
112                     this.leftResize.on("resize", this.syncCenterNode, this);
113 
114 
115                 } else if (position == "right") {
116                     this.rightResize = new Y.Resize({
117                         node: target,
118                         handles: 'l'
119                     });
120                     this.rightResize.on("resize", this.syncCenterNode, this);
121 
122                     target.setStyle("width", 0 );
123                 }
124 
125                 for ( i = 0; i < positionCfg.children.length; i = i + 1) {      // ender the children
126                     cWidget = Y.Wegas.Widget.create( positionCfg.children[ i ] );
127                     // cWidget.after( "render", this.syncUI, this );
128                     cWidget.render( target );
129                 }
130             } else {
131                 target.setStyle( "width", "0" );
132             }
133         },
134 
135         /**
136          *
137          */
138         syncCenterNode: function () {
139             var bodyNode = this.getStdModNode("body"),
140             leftNode = bodyNode.one( ".wegas-layout-left" ),
141             rightNode = bodyNode.one( ".wegas-layout-right" );
142 
143             leftNode.setStyles( {
144                 right: "auto",
145                 left: "0px"
146             });
147             bodyNode.one(".wegas-layout-center").setStyles({
148                 "left": leftNode.getStyle("width"),
149                 "right": rightNode.getStyle("width")
150             });
151             rightNode.setStyles( {
152                 right: "0px",
153                 left: "auto"
154             });
155             Y.Wegas.app.fire("layout:resize");
156         },
157 
158         /**
159          * Override yui implementation to prevent section's content sync.
160          */
161         _syncUIStdMod : function() {
162             this._uiSetFillHeight(this.get("fillHeight"));
163         },
164 
165         /**
166          * Override yui implementation to use custom templates.
167          */
168         _getStdModTemplate : function(section) {
169             return Y.Node.create(Layout.TEMPLATES[section], this._stdModNode.get("ownerDocument"));
170         }
171 
172     }, {
173         ATTRS: {
174             left: {},
175             right: {},
176             top: {},
177             bottom: {},
178             center: {},
179             /* Reset default value to force display by default */
180             headerContent: {
181                 value:""
182             },
183             /* Reset default value to force display by default */
184             footerContent: {
185                 value:""
186             },
187             /* Reset default value to force display by default */
188             bodyContent: {
189                 value:''
190             },
191             height: {
192                 value: "100%"
193             }
194         },
195         TEMPLATES: {
196             header : '<div class="yui-widget-hd wegas-layout-top"></div>',
197             body : '<div class="yui-widget-bd wegas-layout-bd"><div class="wegas-layout-left"></div><div class="wegas-layout-center"></div><div class="wegas-layout-right"></div></div>',
198             footer : '<div class="yui-widget-ft wegas-layout-bottom"></div>'
199         }
200     });
201 
202     Y.namespace('Wegas').Layout = Layout;
203 });
