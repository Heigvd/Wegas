 15 YUI.add("wegas-inputex-variableselect", function (Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx;
 19 
 20     /**
 21      * @class inputEx.Variableselect
 22      * @extends inputEx.StringGroup
 23      **/
 24     Y.namespace("inputEx.Wegas").Variableselect = function (options) {
 25         inputEx.Wegas.Variableselect.superclass.constructor.call(this, options);
 26     };
 27     Y.extend(inputEx.Wegas.Variableselect, inputEx.Group,  {
 28         setOptions: function (options) {
 29             inputEx.Wegas.Variableselect.superclass.setOptions.call(this, options);
 30             this.options.mode = options.mode || "wysiwyg";
 31             this.options.label = options.label;
 32             this.options.fields = [{
 33                 type: 'variabledescriptorgetter',
 34                 name: 'id'
 35             }, {
 36                 type: 'text',
 37                 name: 'expr',
 38                 required: true
 39             }];
 40         },
 41         setValue: function (val, fireUpdatedEvent) {
 42             var findVal;
 43             inputEx.Wegas.Variableselect.superclass.setValue.call(this, val, fireUpdatedEvent);
 44             if (val.name) {
 45                 findVal = Y.Wegas.VariableDescriptorFacade.rest.find('name', val.name);
 46             } else if (val.expr) {
 47                 this.setMode("text");
 48                 this.inputs[1].el.value =  val.expr;
 49                 return;
 50             } else if (val.id) {
 51                 findVal = Y.Wegas.VariableDescriptorFacade.rest.findById(val.id);
 52             }
 53             this.inputs[0].setValue(findVal.get("id"));                         // @fixme
 54         },
 55         renderFields: function () {
 56             inputEx.Wegas.Variableselect.superclass.renderFields.apply(this, arguments);
 57             this.options.mode = "text";
 58             this.viewSrc = new Y.Wegas.Button({                                 // Add the "view src" button
 59                 label: "<span class=\"wegas-icon wegas-icon-viewsrc\"></span>"
 60             });
 61             var divLabel = Y.Node.create('<div class="inputEx-label"><label>Variable</label></div>'); //Add label
 62             this.options.mode = "text";
 63             this.setMode((this.options.mode === "wysiwyg") ? "text" : "wysiwyg");
 64             this.viewSrc.after("click", function () {
 65                 if (!this.viewSrc.get("disabled")) {
 66                     if (this.options.mode === "wysiwyg") {                      // If current mode is wysiwyg
 67                         this.updateTextarea();                                  // update textatea content
 68                     } else {
 69                         this.updateExpressionList();
 70                     }
 71                     this.setMode((this.options.mode === "wysiwyg") ? "text" : "wysiwyg");
 72                 }
 73             }, this);
 74             this.viewSrc.render(this.fieldset);
 75             
 76             var container = new Y.Node(this.fieldset);                    // Render a div where the wysiwyg list will be rendered
 77             container.prepend(divLabel.getDOMNode());
 78             container.prepend(this.viewSrc.get("boundingBox"));
 79         },
 80         setMode: function (mode) {
 81             var wysiwygmode = (mode === "wysiwyg");
 82 
 83             this.options.mode = mode;
 84             this.viewSrc.set("selected", wysiwygmode ? 0 : 1);
 85 
 86             if (wysiwygmode) {
 87                 this.inputs[0].show();
 88                 this.inputs[1].hide();
 89             } else {
 90                 this.inputs[0].hide();
 91                 this.inputs[1].show();
 92             }
 93 
 94         },
 95         updateTextarea: function () {
 96             if (this.options.mode === "wysiwyg") {                              // If current mode is wysiwyg
 97                 this.inputs[1].el.value =  this.getValue().id;                  // update textatea content
 98             }
 99         },
100         updateExpressionList: function () {
101             var tree, newVal = null, valObj = {};
102 
103             try {
104                 tree = window.esprima.parse(this.inputs[1].el.value, {                    // Generate the syntaxic tree using esprima
105                     raw: true
106                 });
107                 if (tree.body[0].expression.callee && tree.body[0].expression.arguments) {
108                     if (tree.body[0].expression.callee.object.name === "VariableDescriptorFacade"  &&
109                         tree.body[0].expression.callee.property.name === "find" &&
110                         tree.body[0].expression.arguments[0].value !== null) {
111                         newVal = Y.Wegas.VariableDescriptorFacade.rest.find('id', tree.body[0].expression.arguments[0].value);
112                     }
113                 }
114                 if (newVal !== null) {
115                     valObj.id = newVal.get("id");
116                     this.setValue(valObj);
117                     this.setMode("text");
118                 } else {
119                     this.setMode("wysiwyg");
120                 }
121 
122             } catch (e) {
123                 this.fire("exception", e.response.results);
124             }
125         },
126         destroy: function () {
127             inputEx.Wegas.Variableselect.superclass.destroy.call(this);
128             this.viewSrc.destroy();
129         }
130 
131     });
132 
133     inputEx.registerType("variableselect", inputEx.Wegas.Variableselect);             // Register this class as "wegasurl" type
134 });
