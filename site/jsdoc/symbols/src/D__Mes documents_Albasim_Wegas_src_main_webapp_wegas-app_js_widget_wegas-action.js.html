 15 YUI.add('wegas-action', function (Y) {
 16     "use strict";
 17 
 18     var Action = function () {
 19         Action.superclass.constructor.apply(this, arguments);
 20     };
 21 
 22     Y.mix(Action, {
 23         NS: "wegas",
 24         NAME: "Action"
 25     });
 26 
 27     Y.extend(Action, Y.Plugin.Base, {
 28         initializer: function () {
 29             this.afterHostEvent( this.get( "targetEvent" ), function() {
 30                 this.setAttrs( this.get( "host" ).get( "data" ) );                       // Pass the action data from the host to the plug
 31                 this.execute();
 32             }, this);
 33         },
 34         execute: function () {
 35             Y.error( "Y.Plugin.Action.execute() is abstract, should be overriddent");
 36         }
 37     }, {
 38         ATTRS: {
 39             targetEvent: {
 40                 value: "click"
 41             }
 42         }
 43     });
 44     Y.namespace("Plugin").Action = Action;
 45 
 46     /**
 47      *  @class OpenGameAction
 48      *  @module Wegas
 49      *  @constructor
 50      */
 51     var OpenUrlAction = function () {
 52         OpenUrlAction.superclass.constructor.apply(this, arguments);
 53     };
 54 
 55     Y.mix(OpenUrlAction, {
 56         NS: "openurlaction",
 57         NAME: "OpenUrlAction"
 58     });
 59 
 60     Y.extend(OpenUrlAction, Action, {
 61         execute: function () {
 62             var targetUrl  = Y.Wegas.app.get( "base" ) + this.get( "url" );
 63 
 64             if ( this.get( "target" ) === "blank") {
 65                 window.open( targetUrl );
 66             } else {
 67                 window.location.href = targetUrl;
 68             }
 69         }
 70     }, {
 71         ATTRS: {
 72             url: { },
 73             /**
 74              * Can be "self" or "blank"
 75              */
 76             target: {
 77                 value : "blank"
 78             }
 79         }
 80     });
 81 
 82     Y.namespace( "Plugin" ).OpenUrlAction = OpenUrlAction;
 83 
 84 
 85     /**
 86      *  @class OpenPageAction
 87      *  @module Wegas
 88      *  @constructor
 89      */
 90     var OpenPageAction = function () {
 91         OpenPageAction.superclass.constructor.apply( this, arguments );
 92     };
 93 
 94     Y.extend( OpenPageAction, Action, {
 95 
 96         initializer: function () {
 97             OpenPageAction.superclass.initializer.apply( this, arguments );
 98             this.afterHostEvent( "render", function() {
 99                 var targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
100                 if ( targetPageLoader.get("pageId") === this.get( "subpageId" ) ) {
101                     this.get( "host" ).set( "selected", 1 );
102                 }
103             }, this);
104         },
105 
106         execute: function () {
107             var targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
108             targetPageLoader.set( "pageId", this.get( "subpageId" ) );
109             this.get( "host" ).set( "selected", 1 );
110         }
111     }, {
112         NS: "OpenPageAction",
113         NAME: "OpenPageAction",
114         ATTRS: {
115             subpageId: {},
116             targetPageLoaderId: {}
117         }
118     });
119 
120     Y.namespace("Plugin").OpenPageAction = OpenPageAction;
121 
122     /**
123      *  @class ExecuteScriptAction
124      *  @module Wegas
125      *  @constructor
126      */
127     var ExecuteScriptAction = function () {
128         ExecuteScriptAction.superclass.constructor.apply(this, arguments);
129     };
130 
131     Y.mix(ExecuteScriptAction, {
132         NS: "ExecuteScriptAction",
133         NAME: "ExecuteScriptAction"
134     });
135 
136     Y.extend(ExecuteScriptAction, Action, {
137         execute: function () {
138             Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
139                 request: "/Script/Run/Player/" + Y.Wegas.app.get( 'currentPlayer' ),
140                 cfg: {
141                     method: "POST",
142                     data: Y.JSON.stringify({
143                         "@class": "Script",
144                         "language": "JavaScript",
145                         "content": this.get( "onClick" )
146                     })
147                 }
148             });
149         }
150     }, {
151         ATTRS: {
152             onClick: {}
153         }
154     });
155 
156     Y.namespace("Plugin").ExecuteScriptAction = ExecuteScriptAction;
157 });
