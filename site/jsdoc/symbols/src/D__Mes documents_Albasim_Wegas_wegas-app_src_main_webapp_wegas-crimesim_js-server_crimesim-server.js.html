 16 function EvtTarget() {
 17     this.evts = {};
 18 }
 19 
 20 EvtTarget.prototype = {
 21     on: function (evt, fn, scope) {
 22         this.evts[evt] = this.evts[evt] || [];
 23         this.evts[evt].push({
 24             fn: fn,
 25             scope: scope || this
 26         });
 27     },
 28     fire: function (evt, args) {
 29         var i, evts = this.evts[evt];
 30         for (i = 0; i < evts.length; i += 1) {
 31             evts[i].fn(args);
 32         }
 33     }
 34 };

 36 function lookupBean(name) {
 37     var ctx = new InitialContext();
 38     return ctx.lookup("java:module/" + name);
 39 }

 48         humanResources.value -= e.reply.result.choiceDescriptor.cost;

 55         humanResources.value += +e.reply.result.choiceDescriptor.cost;
 56     }

 59 function nextWeek() {
 60     period.value += 1;
 61 
 62     var qdf = lookupBean("QuestionDescriptorFacade"),
 63     items = evidences.descriptor.items;
 64 
 65     humanResources.value = humanResources.descriptor.defaultInstance.value;
 66 
 67     for (var i = 0; i < items.size(); i += 1) {
 68         var questionInstance = items.get(i).getInstance(self);
 69         for (var j = 0; j< questionInstance.replies.size(); j += 1) {
 70             var reply = questionInstance.replies.get(j);
 71             if (+reply.startTime + +reply.result.choiceDescriptor.duration + 1 === period.value) {
 72                 qdf.validateReply(self, reply);
 73             }
 74 
 75             if (+reply.startTime + 1  === +period.value ) {
 76                 humanResources.value -= reply.result.choiceDescriptor.cost;
 77             }
 78         }
 79     }
 80 }
