  5 YUI.add('wegas-leaderway-dialogue', function (Y) {
  6     "use strict";
  7     var CONTENTBOX = 'contentBox', Dialogue;
  8     Dialogue = Y.Base.create("wegas-dialogue", Y.Widget, [Y.Wegas.Widget], {
  9         chart: null,
 10         seriesName: null,
 11         seriesValue: null,
 12         handlers: null,
 13         timers: null,
 14         currentDialogue: null,
 15         resourceDescriptor: null,
 16         responseIsDisplayed: null,
 17         availableActions: null,
 18         state: null,
 19         /***Lifecycle methode***/
 20         initializer: function () {
 21             this.responseIsDisplayed = false;
 22             this.seriesName = [];
 23             this.seriesValue = [];
 24             this.handlers = {};
 25             this.timers = [];
 26         },
 27         /**
 28          * Render the widget.
 29          */
 30         renderUI: function () {
 31             var cb = this.get(CONTENTBOX);
 32             cb.setContent(
 33                     '<div class="pictures">\n\
 34                     <div class="backgroundLayer" style="absolute"></div>\n\
 35                     <div class="questionLayer" style="absolute"></div>\n\
 36                     <div class="answerLayer" style="absolute"></div>\n\
 37                  </div>\n\
 38                  <div style="width: 250px; height: 200px;" class="chart"></div>\n\
 39                  <div class="speaker-name"></div>\n\
 40                  <div class="dialogue"><div class="talk"></div><div class="response"></div></div>'
 41                     );
 42         },
 43         /**
 44          * Bind some function at nodes of this widget
 45          */
 46         bindUI: function () {
 47             var cb = this.get(CONTENTBOX);
 48             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this);
 49             this.handlers.dialogueResponse = cb.one('.dialogue .response').delegate('click', function (e) {
 50                 var no = parseInt(e.currentTarget._node.attributes[0].nodeValue),
 51                         dialogue = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.currentDialogue);
 52                 this.responseIsDisplayed = false;
 53                 if (this.availableActions[no]) {
 54                     dialogue.doTransition(this.availableActions[no]);
 55                 }
 56             }, '.responseElements li', this);
 57         },
 58         /**
 59          * Synchronise the content of this widget.
 60          * Recreat the chart widget
 61          */
 62         syncUI: function () {
 63             var cb = this.get(CONTENTBOX);
 64             if (!this.currentDialogue)
 65                 return;
 66             if (!this.responseIsDisplayed) {
 67                 this.responseIsDisplayed = true;
 68                 this.clear(cb);
 69                 this.createChart(this.resourceDescriptor);
 70                 this.readStateMachine(cb);
 71             }
 72         },
 73         /*
 74          * Destroy all child widget and all function
 75          */
 76         destructor: function () {
 77             var k;
 78             if (!this.chart) {
 79                 this.chart.destroy();
 80             }
 81             for (k in this.handlers) {
 82                 this.handlers[k].detach();
 83             }
 84             for (k = 0; k < this.timers.length; k += 1) {
 85                 this.timers[k].cancel();
 86             }
 87         },
 88         /***Particular Methode***/
 89         /**
 90          *Create tooltips for the chart's values
 91          */
 92         chartTooltip: {
 93             markerLabelFunction: function (categoryItem, valueItem, itemIndex, series, seriesIndex) {
 94                 var msg = document.createElement("div"),
 95                         boldTextBlock = document.createElement("div");
 96                 boldTextBlock.appendChild(document.createTextNode(valueItem.displayName + ': ' + valueItem.axis.get("labelFunction").apply(this, [valueItem.value])));
 97                 msg.appendChild(boldTextBlock);
 98                 return msg;
 99             }
100         },
101         /**
102          * Clear each node created in the renderUI function
103          * Delete the chart too.
104          * @param String cb, the widget's contentbox.
105          */
106         clear: function (cb) {
107             this.chart = null;
108             this.seriesName.length = 0;
109             this.seriesValue.length = 0;
110             cb.one('.pictures .backgroundLayer').setHTML();
111             cb.one('.pictures .questionLayer').setHTML();
112             cb.one('.pictures .answerLayer').setHTML();
113             cb.one('.chart').setHTML();
114             cb.one('.speaker-name').setHTML();
115             cb.one('.dialogue .talk').setHTML();
116             cb.one('.dialogue .response').setHTML();
117         },
118         /**
119          * Creat a YUI3 Charts combospline' with values of a resource's moral and confidence historic values.
120          * If any resource is given, the chart will be not created.
121          * @ Param ResourceDescriptor resourceDescriptor, the source of chart's values
122          */
123         createChart: function (resourceDescriptor) {
124             if (this.chart)
125                 this.chart.destroy();
126             if (!resourceDescriptor)
127                 return;
128             var i, resourceInstance = resourceDescriptor.getInstance(),
129                     seriesCollection = [
130                 {
131                     yDisplayName: 'moral'
132                 },
133                 {
134                     yDisplayName: 'confiance'
135                 }
136             ],
137                     rawSeries = [
138                     resourceInstance.get('moralHistory'),
139                     resourceInstance.get('confidenceHistory')
140             ];
141             this.chart = new Y.Chart({
142                 type: 'combospline',
143                 seriesCollection: seriesCollection,
144                 axes: {
145                     values: {
146                         minimum: 0,
147                         maximum: 100
148                     }
149                 },
150                 legend: {
151                     styles: {
152                         gap: 0
153                     },
154                     chart: this.chart,
155                     position: "bottom",
156                     width: 250,
157                     height: 50
158                 },
159                 tooltip: this.chartTooltip,
160                 dataProvider: this.getChartValues(5, rawSeries),
161                 horizontalGridlines: true,
162                 verticalGridlines: true
163             });
164             this.chart.render(".chart");
165         },
166         /**
167          * Create series for the chart.
168          * i = numberOfValues
169          * For each series, If number of values is smaller than i, copy the last value to create a serie with i values.
170          * If number of values is greater than i, keep only the i last values.
171          * @param Integer numberOfValues, the number of value wanted in the series.
172          * @param Array rawSeries, an array of array of Integer.
173          */
174         getChartValues: function (numberOfValues, rawSeries) {
175             var i, j, fitSeries = new Array(), serieRawData = new Array(), serieFitData = new Array();
176             for (i = 0; i < numberOfValues; i++) {
177                 serieFitData.push(i);
178             }
179             fitSeries.push(serieFitData.slice());
180             for (i = 0; i < rawSeries.length; i++) {
181                 serieRawData = rawSeries[i];
182                 serieFitData.length = 0;
183                 for (j = numberOfValues - 1; j >= 0; j--) {
184                     if (serieRawData.length - 1 < j) {
185                         serieFitData.push(serieRawData[0]);
186                     } else {
187                         serieFitData.push(serieRawData[serieRawData.length - (j + 1)]);
188                     }
189                 }
190                 fitSeries.push(serieFitData.slice());
191             }
192             return fitSeries;
193         },
194         /**
195          * Get the dialogue corresponding the currentDialogue's name.
196          * Get the current state.
197          * Get availableAction for this state.
198          * Hide div for images and div for responses.
199          *@param String cb, the widget's contentbox.
200          */
201         readStateMachine: function (cb) {
202             //Read state
203             var dialogue;
204             if (!this.currentDialogue) {
205                 this.responseIsDisplayed = false;
206                 return;
207             }
208             dialogue = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.currentDialogue);
209             if (!dialogue) {
210                 cb.one('.dialogue .talk').insert("Aucun dialogue n'est disponible.");
211                 this.responseIsDisplayed = false;
212                 return;
213             }
214             cb.one('.pictures .backgroundLayer').hide();
215             cb.one('.pictures .questionLayer').hide();
216             cb.one('.pictures .answerLayer').hide();
217             cb.one('.response').hide();
218             this.state = dialogue.getCurrentState();
219             if (!this.state.getAvailableActions) {
220                 this.responseIsDisplayed = false;
221                 Y.log("State isn't a dialogue state.", 'error', 'wegas-leaderway-dialogue.js');
222                 return;
223             }
224             this.state.getAvailableActions(Y.bind(this.readStateContent, this));
225         },
226         /**
227          * Read a dialogue corresponding with the current dialogue value in this widget.
228          * if current state is empty, doTransition with the first available actions.
229          * else if state contain JSON with param "subPageId" and "targetPageLoaderId", call the function "displayWidget()" in this widget.
230          * else, the JSON can contain a array of text (one will be randomly choosed), one string correspnding of a param "speakerName",
231          * a object "backgroundImages", an object "questionImages" and a object "answerImages".
232          * backgroundImages is always displayed.
233          * questionImages is displayed during the display of the text then hidden if answerImages exist
234          * answerImages is displayed when the text is completely displayed.
235          * For more information about object image, read comments on the function "renderJSONImages()" in this widget.
236          * hide the layer "answerImage".
237          */
238         readStateContent: function (availableActions) {
239             var dialogue = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.currentDialogue),
240                     content, rawContent, texts, splittedText = [], cb = this.get(CONTENTBOX);
241             //get availableActions
242             this.availableActions = availableActions;
243             rawContent = this.state.get('text');
244             //Do semi-auto transition
245             if (!rawContent || rawContent.length === 0) {
246                 this.responseIsDisplayed = false;
247                 dialogue.doTransition(this.availableActions[0]);
248                 return;
249             }
250             //Change widget (exit dialogue)
251             content = JSON.parse(rawContent);
252             if (content.subPageId && content.targetPageLoaderId) {
253                 this.displayWidget(dialogue, content);
254                 return;
255             }
256             //Preparing texte
257             texts = content.texts[Math.floor(Math.random() * content.texts.length)];
258             //while(texts.indexOf('&code') != -1){ don't do this !!! out of memory exception.
259             if (texts.indexOf('&code') > -1) {
260                 splittedText.push(texts.substring(0, texts.indexOf('&code')));
261                 texts = texts.substring(texts.indexOf('&code') + 5);
262                 try {
263                     splittedText.push(eval(texts.substring(0, texts.indexOf('&code'))));
264                 } catch (e) {
265                     Y.log('unable to eval script', 'warn', 'wegas.leaderway.dialogue');
266                 }
267                 texts = texts.substring(texts.indexOf('&code') + 5);
268                 splittedText.push(texts);
269                 texts = splittedText.join("");
270             }
271             splittedText = texts.split(new RegExp(" ", "g"));
272             cb.one('.speaker-name').setHTML(content.speakerName);
273             cb.one('.dialogue .talk').insert('<p></p>');
274             //Display image
275             if (content.backgroundImages) {
276                 this.renderImages(cb.one('.pictures .backgroundLayer'), content.backgroundImages);
277             }
278             if (content.questionImages) {
279                 this.renderImages(cb.one('.pictures .questionLayer'), content.questionImages);
280                 this.renderImages(cb.one('.pictures .answerLayer'), (content.answerImages) ? content.answerImages : content.questiocnImages);
281             }
282             cb.one('.pictures .backgroundLayer').show();
283             cb.one('.pictures .questionLayer').show();
284             //Display text
285             //this.timers.push(Y.later(400, this, Y.bind(this.displayText, this, cb, splittedText))); // add time to load picture before displying text.
286             this.displayText(cb, splittedText);
287         },
288         /**
289          * Destroy this widget and open a newer corresponding to the subPageId at place defined by the targetPageLoaderId.
290          * A function can be evaluated after the transition if the JSON contain a param "functionAfterTransition";
291          * Show the node corresponding to the "toHide" ATTRS of this widget.
292          * @param Dialogue dialogue, the readed dialogue.
293          * @param Object content, a object with obligatory param "targetPageLoaderId" and "subPageId" and a non-obligatory param "functionAfterTransition"
294          */
295         displayWidget: function (dialogue, content) {
296             if (content.subPageId && content.targetPageLoaderId) {
297                 var targetPageLoader = Y.Wegas.PageLoader.find(content.targetPageLoaderId);
298                 targetPageLoader.once("widgetChange", function (e) {
299                     if (content.functionAfterTransition) {
300                         try {
301                             eval(content.functionAfterTransition);
302                         } catch (e) {
303                             Y.log('unable to execute function : ' + content.functionAfterTransition, 'warn', 'wegas.leaderway.dialogue');
304                         }
305                     }
306                 });
307                 dialogue.doTransition(this.availableActions[0]);
308                 targetPageLoader.set("pageId", content.subPageId);
309             }
310         },
311         /**
312          * Insert a image to the DOM with given parametre
313          * @param Node node, node where the image must be render.
314          * @param Array[Object] imageObject, a array of objects. Each object can contain a link, an possible height, a possible width, a possible array named "css"  with css properties.
315          */
316         renderImages: function (node, imageObjects) {
317             var i, key, imageHTML = new Array();
318             for (i = 0; i < imageObjects.length; i++) {
319                 imageHTML.push('<img data-file="');
320                 imageHTML.push(imageObjects[i].link);
321                 imageHTML.push('" ');
322                 if (imageObjects[i].height) {
323                     imageHTML.push('height="');
324                     imageHTML.push(imageObjects[i].height);
325                     imageHTML.push('" ');
326                 }
327                 if (imageObjects[i].width) {
328                     imageHTML.push('width="');
329                     imageHTML.push(imageObjects[i].width);
330                     imageHTML.push('" ');
331                 }
332                 imageHTML.push('style="');
333                 imageHTML.push('position:absolute;');
334                 for (key in imageObjects[i].css) {
335                     imageHTML.push(key);
336                     imageHTML.push(':');
337                     imageHTML.push(imageObjects[i].css[key]);
338                     imageHTML.push('; ');
339                 }
340                 imageHTML.push('" />');
341                 node.insert(imageHTML.join(""));
342             }
343         },
344         /**
345          * Recursive function that display text each 70 milisseconde in the ".dialogue .talk p" node of this widget.
346          * call the function "displayResponse" when all the text is displayed.
347          * @param String cb, the widget's contentbox.
348          * @param Array[String] textParts, an array with a splitted string.
349          */
350         displayText: function (cb, textParts) {
351             cb.one('.dialogue .talk p').insert(textParts[0] + '  ');
352             textParts.shift();
353             if (textParts.length > 0) {
354                 this.timers.push(Y.later(50, this, Y.bind(this.displayText, this, cb, textParts)));
355             } else {
356                 cb.one('.pictures .questionLayer').hide();
357                 cb.one('.pictures .answerLayer').show();
358                 this.displayResponse();
359             }
360         },
361         /**
362          * Show all available transitions for the current state.
363          * hide the .questionLayer and sho the .answerLayer to display images.
364          * @param Event object containing actionsAvailable (available transitions)
365          */
366         displayResponse: function (e) {
367             var i, cb = this.get(CONTENTBOX);
368             if (!this.availableActions) {
369                 return;
370             }
371             if (cb.one('.dialogue .response .responseElements')) {
372                 cb.one('.dialogue .response .responseElements').empty(true);
373             } else {
374                 cb.one('.dialogue .response').insert('<ul class="responseElements"></ul>');
375             }
376 
377             for (i = 0; i < this.availableActions.length; i++) {
378                 cb.one('.dialogue .response .responseElements').insert('<li response_no="' + i + '">' + this.availableActions[i].get('actionText') + '</li>');
379             }
380             cb.one('.response').show();
381         },
382         /**
383          * Set the current dialogue and search the resource owner of this new dialogue.
384          * If a resource is finded, set the resourceDescriptor of this widget.
385          * call the function syncUI of this widget.
386          * @param String newDialogueRef, the new DialogueDescriptor's name
387          */
388         setCurrentDialogue: function (newDialogueRef) {
389             var i, listResource = Y.Wegas.VariableDescriptorFacade.rest.find("name", 'resources'),
390                     resourceDescriptor;
391             if (typeof newDialogueRef !== "string" || this.currentDialogue === newDialogueRef) {
392                 return;
393             }
394             this.currentDialogue = newDialogueRef;
395             this.resourceDescriptor = null;
396             for (i = 0; i < listResource.get('items').length; i++) {
397                 resourceDescriptor = listResource.get('items')[i];
398                 if (resourceDescriptor.getInstance().get('properties').dialogue === this.currentDialogue) {
399                     this.resourceDescriptor = resourceDescriptor;
400                     break;
401                 }
402             }
403             this.syncUI();
404         },
405         /**
406          * Set the resourceDescriptor used by this widget.
407          * if the new resource have a dialogue set the current dialogue used by this widget.
408          * @param ResourceDescriptor resourceDescriptor, the new resourceDescriptor
409          */
410         setResourceDescriptor: function (resourceDescriptor) {
411             if (!resourceDescriptor)
412                 return;
413             this.resourceDescriptor = resourceDescriptor;
414             this.setCurrentDialogue(resourceDescriptor.getInstance().get('properties').dialogue);
415         },
416         // *** hack Methods *** //
417         /**
418          * if current week > max value of week value, then
419          * change the current widget to go on the "dialogue" widget.
420          */
421         goToFinalPage: function () {
422             var currentWeek = Y.Wegas.VariableDescriptorFacade.rest.find("name", "week"),
423                     targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
424             if (parseInt(currentWeek.getInstance().get('value')) > currentWeek.get('maxValue')) {
425                 targetPageLoader.once("widgetChange", function (e) {
426                     e.newVal.setCurrentDialogue("dialogueFinal");
427                 });
428                 targetPageLoader.set("pageId", this.get('dialoguePageId'));
429             }
430         }
431 
432     }, {
433         ATTRS: {
434             dialoguePageId: {
435                 value: null,
436                 validator: function (s) {
437                     return s === null || Y.Lang.isString(s);
438                 }
439             },
440             targetPageLoaderId: {
441                 value: null,
442                 validator: function (s) {
443                     return s === null || Y.Lang.isString(s);
444                 }
445             }
446         }
447     });
448     Y.namespace('Wegas').Dialogue = Dialogue;
