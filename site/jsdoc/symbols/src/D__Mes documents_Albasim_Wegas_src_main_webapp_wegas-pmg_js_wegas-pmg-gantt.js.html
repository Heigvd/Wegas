 14 YUI.add( "wegas-pmg-gantt", function ( Y ) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Gantt;
 18 
 19     Gantt = Y.Base.create( "wegas-pmg-gantt", Y.Wegas.PmgDatatable, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable], {
 20         
 21         handlers:null,
 22         schedule:null,
 23         
 24         //*** Private Methods ***/
 25         checkRealization: function(){
 26             var i, cb = this.get(CONTENTBOX), tasks, taskDesc, taskInst, realized, allRow;
 27             if(this.data == null
 28                 || this.data.length == 0
 29                 || this.get("columnValues").indexOf('realized')<=-1){
 30                 return; 
 31             }
 32             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
 33             allRow = cb.all(".yui3-datatable-data tr");
 34             allRow.removeClass("notstarted").removeClass("started").removeClass("completed");
 35             allRow.each(function (node){
 36                 for(i=0; i<tasks.get('items').length; i++){
 37                     taskDesc = tasks.get('items')[i];
 38                     taskInst = taskDesc.getInstance();
 39                     realized = (taskInst.get('properties').realized)?taskInst.get('properties').realized:null;
 40                     if(realized){
 41                         if(node.one("*").getContent() == taskDesc.get('name')){
 42                             if(realized >= 100){
 43                                 node.addClass("completed");
 44                             } else if(realized > 0){
 45                                 node.addClass("started");
 46                             } else{
 47                                 node.addClass("notstarted");
 48                             }
 49                             break;
 50                         }
 51                     }
 52                 }
 53             });
 54         },
 55         
 56         displayDescription: function(e){
 57             var i, name, label, tasks, node, divDesc, taskDesc, description;
 58             node = e.currentTarget;
 59             if(this.get("viewDescription") == "false"
 60                 || node.one(".description")
 61                 || node.get("className").indexOf("noDescription")>-1
 62                 || node.get("className").indexOf("cell-gantt")>-1) return;
 63             name = node.ancestor().one("*").getContent();
 64             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
 65             if(!name || !tasks) return;
 66             for (i = 0; i < tasks.get('items').length; i++) {
 67                 taskDesc = tasks.get('items')[i];
 68                 if(taskDesc.get('name') === name){
 69                     label = (taskDesc.get("label") || name);
 70                     description = taskDesc.get("description");
 71                     break;
 72                 } 
 73             }
 74             divDesc = Y.Node.create("<div class='description'></div>");
 75             divDesc.append("<p class='task_name'>"+label+"</p>").append("<p class='content'>"+description+"</p>");
 76             node.append(divDesc);
 77         },
 78         
 79         removeDescription: function(e){
 80             var disappearAnim, node;
 81             node = this.get(CONTENTBOX).one('.description');
 82             if(!node) return;
 83             disappearAnim = new Y.Anim({
 84                 node: node,
 85                 to: {
 86                     opacity: 0
 87                 },
 88                 duration: 0.2
 89             });
 90             disappearAnim.run();
 91             disappearAnim.on('end', function(){
 92                 node.remove();
 93             });
 94         },
 95         
 96         syncGantt:function(){
 97             var i, cb = this.get(CONTENTBOX), week, type, row;
 98             cb.all('.yui3-datatable-data .yui3-datatable-cell').each(function(node){
 99                 if(node.get("className").indexOf(".yui3-datatable-col-week")>-1){
100                     node.setHTML();   
101                 }
102             });
103             for(var key in this.schedule) {
104                 cb.all('.yui3-datatable-data tr').each(function(node){
105                     if(node.one("*").getContent() == key){
106                         row = node;
107                     }  
108                 })
109                 for(i=0; i<this.schedule[key].length; i++){
110                     week = this.schedule[key][i].week;
111                     type = this.schedule[key][i].type;
112                     if(type == "suppressible"){
113                         row.one('.yui3-datatable-col-week'+week).append("<span class='scheduled'></span>");  
114                     }
115                 }
116                 
117             }
118         },
119         
120         toggleBooking: function(e){
121             var node, week, taskName;
122             node = e.currentTarget;
123             if(node.get("className").indexOf("previous-week")>-1)return;
124             taskName = node.ancestor().one("*").getContent();
125             week = this.getCellWeek(node);
126             if(!node.one(".scheduled")){
127                 node.append("<span class='scheduled'></span>");
128                 this.scheduleTask(taskName, week);
129             } else{
130                 node.one(".scheduled").remove();
131                 this.unScheduleTask(taskName, week);
132             }
133 
134         },
135         
136         scheduleTask: function(taskName, week){
137             var i, exist = false;
138             if(!this.schedule[taskName]){
139                 this.schedule[taskName] = new Array(); 
140             }
141             for(i=0; i<this.schedule[taskName].length; i++){
142                 if(this.schedule[taskName][i].week == week){
143                     exist = true;
144                     break;
145                 }
146             }   
147             if(!exist){
148                 this.schedule[taskName].push({
149                     week:week,
150                     type:"suppressible"
151                 });     
152             }
153         },
154         
155         unScheduleTask: function(taskName, week){
156             var i;
157             if(this.schedule[taskName]){
158                 for(i=0; i<this.schedule[taskName].length; i++){
159                     if(this.schedule[taskName][i].week == week){
160                         if(this.schedule[taskName][i].type == "suppressible"){
161                             this.schedule[taskName].splice(i,1);   
162                         }
163                         break;
164                     }
165                 }
166             }
167         },
168         
169         getCellWeek:function(cell){
170             var week;
171             if(cell.get("className").indexOf("yui3-datatable-col-week")<=-1) return null;
172             week = cell.get("className").substring(cell.get("className").indexOf("yui3-datatable-col-week")+23)
173             if(week.indexOf(" ")>-1){
174                 week = week.substring(0, week.indexOf(" "));   
175             }
176             return week;
177         },
178         
179         displayCurrentWeek:function(currentWeek){
180             var week, cb = this.get(CONTENTBOX);
181             if(!currentWeek) return; //add class "previous-week", "current-week" and "next-week" to the column title in Gantt
182             cb.all(".yui3-datatable-columns th, .yui3-datatable-data td").each(function (node){
183                 if(node.get('className').indexOf("yui3-datatable-col-week")>-1){
184                     if(node.get("nodeName")=="td" || node.get("nodeName")=="TD"){
185                         node.addClass("cell-gantt");   
186                     }
187                     week = this.getCellWeek(node);
188                     if(week < currentWeek){
189                         node.addClass('previous-week');
190                     } else if(week == currentWeek){
191                         node.addClass('current-week');
192                     } else if(week > currentWeek){
193                         node.addClass('next-week');
194                     }
195                 }
196             }, this);
197             
198         },
199         
200         initializer: function(){
201             this.handlers = new Array();
202             this.schedule = {};
203         },   
204         
205         renderUI: function(){
206             var i, periods;
207             Gantt.superclass.renderUI.apply(this);
208             periods = this.get("periodsDesc");
209             if(!periods)return;
210             for(i=periods.get("minValue"); i<=periods.get("maxValue"); i++){
211                 this.datatable.addColumn({
212                     key:'week'+i,
213                     label:''+i
214                 });
215             }
216         },
217         
218         bindUI: function(){
219             Gantt.superclass.bindUI.apply(this);
220             this.handlers.push(Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this));
221             this.handlers.push(Y.Wegas.app.after('currentPlayerChange', this.syncUI, this));
222             this.handlers.push(this.datatable.after('sort', this.syncUI, this));
223             this.handlers.push(this.datatable.delegate('click', function (e) {
224                 this.displayDescription(e);
225             }, '.yui3-datatable-data td', this));
226             this.handlers.push(this.datatable.delegate('mouseout', function (e) {
227                 this.removeDescription(e);
228             }, '.yui3-datatable-data tr', this));
229             this.handlers.push(this.datatable.delegate('click', function (e) {
230                 this.toggleBooking(e);
231             }, '.yui3-datatable-data .cell-gantt', this));
232         },
233         
234         syncUI: function(){
235             var cb = this.get(CONTENTBOX), currentWeek;
236             Gantt.superclass.syncUI.apply(this);
237             currentWeek = (this.get("periodsDesc"))?this.get("periodsDesc").getInstance().get("value"):null;
238             this.checkRealization();
239             this.displayCurrentWeek(currentWeek);
240             this.syncGantt();
241         },
242         
243         destructor: function(){
244             var i;
245             for (i=0; i<this.handlers.length;i++) {
246                 this.handlers[i].detach();
247             } 
248         }  
249 
250     }, {
251         ATTRS : {
252             viewDescription:{
253                 value: true,
254                 validator: function (b){
255                     return b == "false" || b == "true";
256                 }
257             },
258             periods: {}, // to change to accept global expresssion or simple variable.
259             periodsDesc: {
260                 getter: function () {
261                     return Y.Wegas.VariableDescriptorFacade.rest.findById(
262                         Y.Wegas.VariableDescriptorFacade.script.scopedEval( this.get( "periods" ) ) );
263                 }
264             }
265         }
266     });
267 
268     Y.namespace( "Wegas" ).PmgGantt = Gantt;
269 });
