 14 YUI.add("wegas-pmg-datatable", function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Datatable;
 18 
 19     Datatable = Y.Base.create("wegas-pmg-datatable", Y.Widget, [Y.Wegas.Widget, Y.Wegas.Editable], {
 20         handlers: null,
 21         datatable: null,
 22         data: null,
 23         // *** Lifecycle Methods *** //
 24         initializer: function () {
 25             var i, ct = this.get("columnTitles"), columnTitles = new Array();
 26             this.handlers = {};
 27             this.data = [];
 28             if (ct == null || ct.length == 0 || ct.length != this.get("columnValues").length) {
 29                 return;
 30             }
 31             columnTitles.push({
 32                 key: "_id",
 33                 label: "_id"
 34             });                                                                 //First column is always the name (but not displayed)
 35             for (i = 0; i < ct.length; i++) {                                         //construct Datatable's columns
 36                 columnTitles.push(
 37                         {
 38                             key: ct[i],
 39                             label: ct[i],
 40                             sortable: true
 41                         }
 42                 );
 43             }
 44 //            this.datatable = new Y.Treeble({                                  //Using Treeble (comment also the sorting methode belove)
 45             this.datatable = new Y.DataTable({//Using simple database
 46                 columns: columnTitles
 47             });
 48             if (this.get("defaultSort") && this.get("defaultSort").indexOf(this.get("columnValues") > -1)) { //Add columns sorting possibilities
 49                 this.datatable.sort(this.get("defaultSort"));
 50             } else {
 51                 this.datatable.sort(this.get("columnTitles")[0]);
 52             }
 53         },
 54         renderUI: function () {
 55             var cb = this.get(CONTENTBOX);
 56             if (!this.datatable)
 57                 return;
 58             this.datatable.render(cb);
 59         },
 60         bindUI: function () {
 61             this.handlers.update = Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this);
 62         },
 63         syncUI: function () {
 64             if (this.datatable == null || this.get("variables") == null)
 65                 return;
 66             this.datatable.set("data", []);
 67             this.data.length = 0;
 68             this.getData();
 69             this.datatable.addRows(this.data);
 70         },
 71         destructor: function () {
 72             var i;
 73             for (i = 0; i < this.handlers.length; i++) {
 74                 this.handlers[i].detach();
 75             }
 76             this.datatable.destroy();
 77         },
 78         //*** Private Methods ***/
 79         getData: function () {
 80             var i, j, variables, variableDesc, variableInst, oneRowDatas, data,
 81                     ct = this.get("columnTitles"), cv = this.get("columnValues");
 82             if (cv == null)
 83                 return;
 84             variables = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("variables"));
 85             for (i = 0; i < variables.get('items').length; i++) {
 86                 variableDesc = variables.get('items')[i];
 87                 variableInst = variableDesc.getInstance();
 88                 oneRowDatas = {};
 89                 oneRowDatas["_id"] = variableDesc.get("id");
 90                 for (j = 0; j < ct.length; j++) {
 91                     if (variableDesc.get(cv[j])) {
 92                         data = variableDesc.get(cv[j]);
 93                     } else if (variableInst.get(cv[j])) {
 94                         data = variableInst.get(cv[j]);
 95                     } else {
 96                         data = variableInst.get('properties')[cv[j]];
 97                     }
 98                     //change texte-number in number to prepare the value to be sorted.
 99                     if (parseFloat(data)) {
100                         data = parseFloat(data);
101                     }
102                     oneRowDatas[ct[j]] = data;
103                 }
104                 this.data.push(oneRowDatas);
105             }
106         }
107     }, {
108         ATTRS: {
109             variables: {
110                 value: null,
111                 validator: function (s) {
112                     return s === null || Y.Lang.isString(s);
113                 }
114             },
115             columnTitles: {
116                 validator: Y.Lang.isArray
117             },
118             columnValues: {
119                 validator: Y.Lang.isArray
120             },
121             defaultSort: {
122                 value: null,
123                 validator: function (s) {
124                     return s === null || Y.Lang.isString(s);
125                 }
126             }
127         }
128     });
129 
130     Y.namespace("Wegas").PmgDatatable = Datatable;
131 });
