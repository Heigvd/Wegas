 15 YUI.add("wegas-inputex-roleselect", function (Y) {
 16     "use strict";
 17 
 18     var RoleSelect;
 19 
 20     /**
 21      * Adds an url regexp, and display the favicon at this url
 22      * @class inputEx.UrlField
 23      * @extends inputEx.StringField
 24      * @constructor
 25      * @param {Object} options inputEx.Field options object
 26      * <ul>
 27      *   <li>favicon: boolean whether the domain favicon.ico should be displayed or not (default is true, except for https)</li>
 28      * </ul>
 29      */
 30     RoleSelect = function (options) {
 31         RoleSelect.superclass.constructor.call(this, options);
 32     };
 33 
 34     Y.extend(RoleSelect, Y.inputEx.SelectField, {
 35         setValue: function (val, sendUpdatedEvent) {
 36             this.value = val;
 37             RoleSelect.superclass.setValue.call(this, val.id, sendUpdatedEvent);
 38         },
 39         getValue: function () {
 40             return {
 41                 "@class": "Role",
 42                 name: "",
 43                 description: "",
 44                 id: RoleSelect.superclass.getValue.call(this)
 45             };
 46         },
 47         setOptions: function (options) {
 48             this.options.choices = [];
 49             RoleSelect.superclass.setOptions.call(this, options);
 50         },
 51         render: function () {
 52             RoleSelect.superclass.render.call(this, options);
 53             Y.Wegas.RoleFacade.rest.sendRequest({
 54                 request: "/",
 55                 on: {
 56                     success: Y.bind(function () {
 57                         this.renderOptions(Y.Wegas.RoleFacade.data)
 58                     }, this)
 59                 }
 60             });
 61 
 62         },
 63         renderOptions: function (options) {
 64             var i;
 65             for (i = 0; i < options.length; i = i + 1) {
 66                 this.createChoiceNode()
 67                 this.options.choices.push({
 68                     value: Y.Wegas.RoleFacade.data[i].get("id"),
 69                     label: Y.Wegas.RoleFacade.data[i].get("name")
 70                 });
 71             }
 72 
 73         }
 74     });
 75 
 76     Y.namespace("inputEx.Wegas").RoleSelect = RoleSelect;
 77     Y.inputEx.registerType("roleselect", RoleSelect);               // Register this class as "wegasurl" type
 78 });
