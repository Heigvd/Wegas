1 YUI.add('wegas-langselector', function (Y) {
  2     "use strict";
  3 
  4     var CONTENTBOX = 'contentBox', LangSelector;
  5 
  6     LangSelector = Y.Base.create("wegas-langselector", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
  7         
  8         items: null,
  9         handlers:new Array(),
 10         
 11         renderUI: function() {
 12             this.items = this.get("items");
 13             this.get(CONTENTBOX).insert("<select class='wegas-langselector-select'></select>");
 14         },
 15         
 16         bindUI: function(){
 17             var cb = this.get(CONTENTBOX);
 18             this.handlers.push(cb.one('.wegas-langselector-select').on('change', function (e) {
 19                var lang = e.currentTarget.get('value'),
 20                url = window.location.href,
 21                urlStart, urlEnd;
 22                if(url.indexOf("&lang=") > -1){
 23                   urlStart = url.substring(0, url.indexOf("&lang="));
 24                   urlEnd = url.substring(url.indexOf("&lang=")+1);
 25                     if(urlEnd.indexOf("&")>-1){
 26                         urlEnd = urlEnd.substring(urlEnd.indexOf("&"));   
 27                     }else{
 28                         urlEnd = "";
 29                     }
 30                     location.replace(urlStart+urlEnd+"&lang="+lang);
 31                }else{
 32                     location.replace(url+"&lang="+lang);
 33                }
 34             },'option', this));
 35         },
 36         
 37         syncUI: function () {
 38             if(this.items == null) return;
 39             var i, cb=this.get(CONTENTBOX), browserLang, pageLang, url, selected = false;
 40             url = window.location.href;
 41             browserLang = (navigator.language) ? navigator.language : navigator.userLanguage; 
 42             if(url.indexOf("&lang=")>-1){
 43                 pageLang = url.substring(url.indexOf("&lang=")+6);
 44                 if(pageLang.indexOf("&")>-1){
 45                     pageLang = pageLang.substring(0, pageLang.indexOf("&"));
 46                 }
 47             }
 48             for(i=0;i<this.items.length;i++){
 49                 if(!selected){
 50                     if(pageLang && (this.items[i].indexOf(pageLang)>-1 || pageLang.indexOf(this.items[i])>-1)){
 51                         cb.one('.wegas-langselector-select').insert("<option selected='selected'>"+this.items[i]+"</option>");
 52                         selected = true;
 53                     }else if((this.items[i].indexOf(browserLang)>-1 || browserLang.indexOf(this.items[i])>-1)&&url.indexOf("&lang=")<=-1){
 54                         cb.one('.wegas-langselector-select').insert("<option selected='selected'>"+this.items[i]+"</option>");
 55                         selected = true;
 56                     }
 57                     else{
 58                         cb.one('.wegas-langselector-select').insert("<option>"+this.items[i]+"</option>");
 59                     }
 60                 }else{
 61                     cb.one('.wegas-langselector-select').insert("<option>"+this.items[i]+"</option>");
 62                 }
 63             }
 64         },
 65 
 66         destroy: function(){
 67             for (var i=0; i<this.handlers.length;i++) {
 68                 this.handlers[i].detach();
 69             }
 70         }
 71         
 72     }, {
 73         ATTRS : {
 74             items:{
 75                 value:null,
 76                 validator: Y.Lang.isArray
 77             }
 78         }
 79     });
 80 
 81     Y.namespace('Wegas').LangSelector = LangSelector;
 82 });
