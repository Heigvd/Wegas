 15 YUI.add('wegas-form', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', FormWidget;
 19 
 20     FormWidget = Y.Base.create("wegas-form", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 21         // ** Private Fields ** //
 22 
 23         // ** Lifecycle Methods ** //
 24         initializer: function () {
 25             this.plug(Y.Plugin.WidgetToolbar);
 26             this.publish("submit", {
 27                 emitFacade: true
 28             });
 29         },
 30         renderUI: function () {
 31             this.renderToolbar();
 32         },
 33         // ** Private Methods ** //
 34         renderToolbar: function () {
 35             var toolbarNode = this.toolbar.get('header');
 36 
 37             this.saveButton = new Y.Button({
 38                 label: "<span class=\"wegas-icon wegas-icon-save\" ></span>Save",
 39                 on: {
 40                     click: Y.bind(function () {
 41                         var form = this.get("form"),
 42                                 val = form.getValue();
 43 
 44                         if (!form.validate()) {
 45                             return;
 46                         }
 47                         form.fire("afterValidation");
 48                         if (val.valueselector) {
 49                             val = val.valueselector;
 50                         }
 51                         this.fire("submit", {
 52                             value: val
 53                         });
 54                     }, this)
 55                 }
 56             }).render(toolbarNode);
 57 
 58             this.cancelButton = new Y.Button({
 59                 label: "<span class=\"wegas-icon wegas-icon-cancel\" ></span>Cancel",
 60                 on: {
 61                     click: Y.bind(function () {
 62                         this.fire("cancel");
 63                     }, this)
 64                 }
 65             }).render(toolbarNode);
 66         },
 67     
 68         destroyForm: function () {
 69             this.set("form", null);
 70         }
 71 
 72     }, {
 73         ATTRS: {
 74             values: {
 75                 value: {
 76                     setter: function (val) {
 77                         if (this.get("form")) {
 78                             this.form.setValue(val);
 79                         }
 80                     }
 81                 }
 82             },
 83             form: {
 84                 setter: function (val) {
 85                     if (this.get("form")) {                                     // If there is alread a form instantiated, destroy it
 86                         this.get("form").destroy();
 87                     }
 88                     return val;
 89                 }
 90             },
 91             cfg: {
 92                 setter: function (val) {
 93                     val.parentEl = this.get(CONTENTBOX);                        // Set up the form parentEl attribute, so it knows where to render
 94                     val.className = "wegas-form-ix";
 95                     Y.inputEx.use(val, Y.bind(function (cfg) {                  // Load form dependencies
 96                         var form = Y.inputEx(cfg);                              // Initialize and render form
 97                         form.setValue(this.get("values"));                      // Sync form with "values" ATTR
 98                         this.set("form", form);
 99                     }, this, val));
100                     return val;
101                 }
102             }
103         }
104     });
105 
106     Y.namespace("Wegas").FormWidget = FormWidget;
107 
108     var inputEx = Y.inputEx,
109             lang = Y.Lang;
110 
111     /**
112      *  @hack So we can easily change classs on inputex fields
113      */
114     Y.inputEx.Field.prototype.addClassName = function (className) {
115         Y.one(this.divEl).addClass(className);
116     };
117     Y.inputEx.Field.prototype.removeClassName = function (className) {
118         Y.one(this.divEl).removeClass(className);
119     };
120     /*
121      * @hack Automatically add the "optional" message when necessary
122      */
123     Y.inputEx.StringField.prototype.setOptions = function (options) {
124         Y.inputEx.StringField.superclass.setOptions.call(this, options);
125 
126         this.options.regexp = options.regexp;
127         this.options.size = options.size;
128         this.options.maxLength = options.maxLength;
129         this.options.minLength = options.minLength;
130         this.options.typeInvite = options.typeInvite;
131         if (!this.options.required && this.options.typeInvite === undefined) {  // !!!MODIFIED!!!
132             this.options.typeInvite = "optional";
133         }
134         this.options.readonly = options.readonly;
135         this.options.autocomplete = lang.isUndefined(options.autocomplete) ?
136                 inputEx.browserAutocomplete :
137                 (options.autocomplete === false || options.autocomplete === "off") ? false : true;
138         this.options.trim = (options.trim === true) ? true : false;
139     };
140     /**
141      * @hack Let inputex also get requirement from selectfields, lists
142      */
143     Y.inputEx.getRawModulesFromDefinition = function (inputexDef) {
144 
145         var type = inputexDef.type || 'string',
146                 module = YUI_config.groups.inputex.modulesByType[type],
147                 modules = [module || type],
148                 //set fields if they exist
149                 fields = inputexDef.fields
150                 //else see if we have elementType for lists - if neither then we end up with null
151                 || inputexDef.availableFields || [];
152 
153         if (inputexDef.elementType) {
154             fields.push(inputexDef.elementType);
155         }
156 
157         // recursive for group,forms,list,combine, etc...
158         Y.Array.each(fields, function (field) {
159             modules = modules.concat(this.getModulesFromDefinition(field));
160         }, this);
161 
162         // TODO: inplaceedit  editorField
163 
164         return modules;
165     };
166 });
