 27     destroySelf = function(){
 28 
 29         if ( !this._node ) {
 30             return;                                                             // The node has already been destroyed
 31         }
 32 
 33         if(this.timeout){
 34             this.timeout.cancel();
 35         }
 36         this.closeHandler.detach();
 37         var anim = new Y.Anim({
 38             node: this,
 39             to: {
 40                 opacity: 0
 41             },
 42             duration: 0.2
 43         });
 44         anim.on( "end", this.remove, this, true );
 45         anim.run();
 46     };

 48     function Widget () {
 49         this.after( "render", function () {
 50             var bb = this.get( BOUNDING_BOX );
 51             bb.addClass( "wegas-widget" );
 52             if ( this.get( "cssClass" ) ) {
 53                 bb.addClass( this.get( "cssClass" ) );
 54             }
 55         });
 56         this.constructor.CSS_PREFIX = this.constructor.CSS_PREFIX               // If no prefix is set, use the name (without
 57         || this.constructor.NAME.toLowerCase();                                 // the usual "yui3-" prefix)
 58         this._cssPrefix = this.constructor.CSS_PREFIX;
 59     }

 61     Y.mix( Widget.prototype, {
 62 
 63         showOverlay: function(){
 64             this.get( BOUNDING_BOX ).prepend( "<div class='wegas-widget-loading'></div>" );
 65         },
 66 
 67         hideOverlay: function(){
 68             var overlayNode = this.get( BOUNDING_BOX ).all( ".wegas-widget-loading" );
 69             if ( overlayNode ) {
 70                 overlayNode.remove( true );
 71             }
 72         },
 73 
 74         emptyMessage: function () {						// Form msgs logic
 75             var msgNode = this.get(BOUNDING_BOX).one(".wegas-systemmessage");
 76             if ( !msgNode ) {
 77                 return;
 78             }
 79             msgNode.empty();
 80         },
 81 
 82         showMessage: function ( level, txt, timeout ) {
 83             var msgNode = this.getMessageNode(),
 84             message = Y.Node.create( "<div class='" + ( LEVEL[level] ? LEVEL[level] : "" ) + "'><span class='icon'></span><span class='content'>" + txt + "</span><span class='close'></span></div>");
 85             if ( level === "success" && !timeout ) {                            // @hack successful messages disapear automatically
 86                 if(this.toolbar instanceof Y.Plugin.WidgetToolbar){
 87                     this.setStatusMessage(txt);
 88                     return;
 89                 }else{
 90                     timeout = 800;
 91                 }
 92             }
 93             msgNode.append(message);
 94 
 95             message.closeHandler = message.one(".close").once( "click", destroySelf, message );
 96 
 97             if ( timeout ) {
 98                 message.timeout = Y.later( timeout, message, destroySelf);
 99             }
100         },
101 
102         getMessageNode: function () {
103             var msgNode = this.get( BOUNDING_BOX ).one( ".wegas-systemmessage" );
104             if ( !msgNode ) {
105                 this.get( BOUNDING_BOX ).append( "<div class='wegas-systemmessage'></div>" );
106                 return this.get( BOUNDING_BOX ).one( ".wegas-systemmessage" );
107             }
108             return msgNode;
109         },
110         setStatusMessage: function(txt){
111             var statusNode = this._getStatusNode();
112             if(statusNode === null){
113                 return false;
114             }
115             statusNode.setContent(txt);
116             return true;
117         },
118         _getStatusNode: function(){
119             var statusNode;
120             if(!(this.toolbar instanceof Y.Plugin.WidgetToolbar)){
121                 return null;
122             }
123             statusNode = this.toolbar.get("header").one(".wegas-status-message");
124             if(!statusNode){
125                 statusNode = new Y.Node.create("<span class='wegas-status-message'></span>");
126                 this.toolbar.get("header").append(statusNode);
127             }
128             return statusNode;
129         }

247         /**
248         * Return recursively the inputex modules from their 'type' property using (modulesByType from loader.js)
249         */
250         getRawModulesFromDefinition: function(cfg) {
251 
252             var i, props, type = cfg.type || "text",
253             module = YUI_config.groups.wegas.modulesByType[type],
254             modules = [];
255 
256             if (module) {
257                 modules.push(module);
258             }
259 
260             props = [ "children" ];
261             for ( i = 0; i < props.length; i = i + 1) {
262                 if (cfg[props[i]]) {                                            // Get definitions from children (for Y.WidgetParent widgets)
263                     Y.Array.each(cfg[props[i]], function(field) {
264                         modules = modules.concat( this.getModulesFromDefinition(field) );
265                     }, this);
266                 }
267             }
268 
269             props = ["left", "right", "center", "top", "bottom"];               // Get definitions from children (for Y.Wegas.Layout widgets)
270             for ( i = 0; i < props.length; i = i + 1) {
271                 if (cfg[props[i]]) {
272                     modules = modules.concat(this.getModulesFromDefinition(cfg[props[i]]));
273                 }
274             }
275 
276             return modules;
277         },
278         /**
279          * Return unique modules definitions
280          */
281         getModulesFromDefinition: function(cfg) {
282             var modules = Y.Wegas.Widget.getRawModulesFromDefinition(cfg);
283             return Y.Object.keys(Y.Array.hash(modules));
284         },
285         /**
286          * Load the modules from an Wegas widget definition
287          */
288         use: function(cfg, cb) {
289             var modules = Y.Wegas.Widget.getModulesFromDefinition(cfg);
290             modules.push(cb);
291             Y.use.apply( Y, modules);
292         }

300     Y.WidgetParent.prototype._createChild = function (config) {
301         var defaultType = this.get("defaultChildType"),
302         altType = config.childType || config.type,
303         child,
304         Fn,
305         FnConstructor;
306 
307         if (altType) {
308             Fn = Lang.isString(altType) ? Y.Wegas[altType] || Y[altType] : altType;           // @hacked
309         }
310 
311         if (Lang.isFunction(Fn)) {
312             FnConstructor = Fn;
313         } else if (defaultType) {
314             // defaultType is normalized to a function in it's setter
315             FnConstructor = defaultType;
316         }
317 
318         if (FnConstructor) {
319             child = new FnConstructor(config);
320         } else {
321             Y.error("Could not create a child instance because its constructor is either undefined or invalid (" + altType + ")");
322         }
323         return child;
324     };

330     var newPlug = Y.DataSource.IO.prototype.plug = function(Plugin, config) {
331         var i, ln, ns;
332 
333         if (Lang.isArray(Plugin)) {
334             for (i = 0, ln = Plugin.length; i < ln; i++) {
335                 this.plug(Plugin[i]);
336             }
337         } else {
338             if (Plugin && !Lang.isFunction(Plugin)) {
339                 config = Plugin.cfg;
340                 Plugin = Plugin.fn;
341             }
342             if (Plugin && !Lang.isFunction(Plugin)) {			// @hacked
343                 Plugin = Y.Plugin[Plugin];
344             }
345 
346             // Plugin should be fn by now
347             if (Plugin && Plugin.NS) {
348                 ns = Plugin.NS;
349 
350                 config = config || {};
351                 config.host = this;
352 
353                 if (this.hasPlugin(ns)) {
354                     // Update config
355                     this[ns].setAttrs(config);
356                 } else {
357                     // Create new instance
358                     this[ns] = new Plugin(config);
359                     this._plugins[ns] = Plugin;
360                 }
361             }
362         }
363         return this;
364     };
