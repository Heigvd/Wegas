 15 YUI.add('wegas-crimesim-scheduledisplay', function(Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19             ScheduleDisplay;
 20 
 21     /**
 22      *  The schedule display class.
 23      */
 24     ScheduleDisplay = Y.Base.create("wegas-crimesim-scheduledisplay", Y.Widget, [Y.Wegas.Widget, Y.WidgetChild, Y.Wegas.Editable], {
 25         CONTENT_TEMPLATE: '<div><div class="schedule-questions"></div>'
 26                 + '<div class="schedule-detail"><div class="schedule-icon-close"></div><h1></h1><div class="content"></div>'
 27                 + '<div class="schedule-gallery"></div><h2>Anaylses</h2><div class="schedule-analysis"></div>'
 28                 + '</div></div>',
 29         // *** Fields *** /
 30         menu: null,
 31         handlers: null,
 32         gallery: null,
 33         datatable: null,
 34         data: null,
 35         currentQuestionId: null,
 36         // *** Lifecycle Methods *** //
 37         initializer: function() {
 38             this.data = [];
 39             this.menu = new Y.Wegas.Menu();
 40             this.menuDetails = new Y.Wegas.Menu({
 41                 width: "250px"
 42             });
 43         },
 44         renderUI: function() {
 45             // cb.on("clickoutside", this.hideMenu, this);
 46             this.menu.on("button:mouseenter", function(e) {
 47                 var choice = e.target.get("data").choice;
 48                 this.menuDetails.set("align", {
 49                     node: this.menu.get("boundingBox"),
 50                     points: (e.details[0].domEvent.clientX > Y.DOM.winWidth() / 2) ? ["tr", "tl"] : ["tl", "tr"]
 51                 });
 52                 this.menuDetails.get("contentBox").setHTML('<div style="padding:5px 10px">'
 53                         + (choice.get("description") || "<i>No description</i>")
 54                         + "<br /><br />Human resources needed: " + choice.get("cost")
 55                         + "<br />Duration: " + choice.get("duration")
 56                         + '</div>');
 57                 this.menuDetails.show();
 58             }, this);
 59 
 60             this.menu.on("visibleChange", function(e) {                 // When the menu is hidden, hide the details panel
 61                 if (!e.newVal) {
 62                     this.menuDetails.hide();
 63                 }
 64             }, this);
 65 
 66 
 67             this.renderDetailsPanel(this.get(CONTENTBOX).one(".schedule-analysis"));
 68 
 69             this.gallery = new Y.Wegas.FileExplorerGallery({
 70                 render: this.get(CONTENTBOX).one(".schedule-gallery"),
 71                 selectedHeight: 150,
 72                 selectedWidth: 235
 73             });
 74         },
 75         bindUI: function() {
 76             var cb = this.get(CONTENTBOX);
 77             this.handlers = {};
 78 
 79             cb.delegate("click", function(e) {                            // Show the available menu options on cell click
 80                 var questionId = e.target.ancestor("tr").getAttribute("data-questionid"),
 81                         startTime = +e.target.ancestor("td").getAttribute("data-startTime"),
 82                         question = Y.Wegas.VariableDescriptorFacade.rest.findById(questionId);
 83 
 84                 this.menu.removeAll();                                          // Populate the menu
 85                 this.menu.add(this.genMenuItems(question, startTime));
 86                 this.menu.attachTo(e.target);                                   // Display the menu button next to the arrow
 87             }, ".schedule-available .icon", this);
 88 
 89             this.menu.on("button:click", this.onMenuClick, this);               // Listen for the choice menu click event
 90 
 91             cb.delegate("click", function(e) {                                 // Show the question detail on left label click
 92                 var questionId = e.target.ancestor("tr").getAttribute("data-questionid");
 93                 this.currentQuestionId = questionId;
 94                 this.get("contentBox").all(".schedule-leftcolum-selected").removeClass("schedule-leftcolum-selected");
 95                 e.target.addClass("schedule-leftcolum-selected");
 96                 this.syncDetailsPanel();
 97             }, "td.schedule-leftcolum", this);
 98 
 99             cb.delegate("click", this.hideDetails, // Hide the question detail on close icon click
100                     ".schedule-icon-close", this);
101 
102             cb.delegate("click", this.onCancelReplyClick, ".icon .close-icon", this);// Hide the question detail on close icon click
103 
104 
105             this.handlers.response = // If data changes, refresh
106                     Y.Wegas.app.dataSources.VariableDescriptor.after("response", this.syncUI, this);
107 
108             this.handlers.playerChange = // If current user changes, refresh (editor only)
109                     Y.Wegas.app.after('currentPlayerChange', this.syncUI, this);
110         },
111         /**
112          *
113          */
114         syncUI: function() {
115             this.syncSchedule();
116             if (this.currentQuestionId) {
117                 this.syncDetailsPanel();
118             }
119             this.hideOverlay();
120             this.datatable.syncUI();
121         },
122         /**
123          *
124          */
125         destructor: function() {
126             var i;
127             for (i in this.handlers) {
128                 this.handlers[i].detach();
129             }
130         },
131         // *** Rendering methods *** //
132         syncSchedule: function() {
133             var perPeriodBudget = 15, perPeriodLoad = [], cIndex, choiceDescriptor, choiceInstance,
134                     questionInstance, reply, i, j, k, question, cols, replies, names,
135                     questionsVarDesc = Y.Wegas.VariableDescriptorFacade.rest.find('name', "evidences").get("items"),
136                     questionInstances = [],
137                     period = Y.Wegas.VariableDescriptorFacade.rest.find('name', "period"),
138                     periodInstance = period.getInstance(),
139                     maxValue = period.get("maxValue"),
140                     totalPeriods = period.get("maxValue") - period.get("minValue"),
141                     acc = ['<table class="schedule-table"><tr><th class="schedule-leftcolum">Evidences</th>'],
142                     cb = this.get(CONTENTBOX).one(".schedule-questions"),
143                     currentTime = periodInstance.get("value") - period.get("minValue");
144 
145             this.currentTime = currentTime;
146             this.perPeriodLoad = perPeriodLoad;
147 
148             if (!period) {
149                 cb.setContent("No time variable is set.");
150                 return;
151             }
152 
153             for (i = period.get("minValue"); i <= maxValue; i += 1) {
154                 acc.push('<th class="schedule-maincolum"><div>' + i + '</div></th>'); // Generate table header
155                 perPeriodLoad.push(0);                                          // Default value for perPeriodLoad calculation
156             }
157             acc.push("</tr>");
158 
159             for (i = 0; i < questionsVarDesc.length; i += 1) {                  // First pass to compute remaining time budget per period
160                 question = questionsVarDesc[i];
161                 questionInstance = question.getInstance();
162                 for (j = 0; j < questionInstance.get("replies").length; j += 1) {
163                     reply = questionInstance.get("replies")[j];
164                     choiceDescriptor = reply.getChoiceDescriptor();
165                     perPeriodLoad[reply.get("startTime")] += choiceDescriptor.get("cost");
166                 }
167                 questionInstances.push(questionInstance);
168             }
169 
170             for (i = 0; i < questionsVarDesc.length; i += 1) {                  // Generate table body
171                 question = questionsVarDesc[i];
172                 questionInstance = questionInstances[i];
173 
174                 if (!questionInstance.get("active")) {                          // Do not render inactive questions
175                     continue;
176                 }
177 
178                 acc.push('<tr data-questionId="' + question.get("id") + '"><td class="schedule-leftcolum" >' +
179                         (question.getPublicLabel() || "undefined") + "</td>");
180                 cols = [];
181                 names = [];
182                 replies = [];
183 
184                 for (j = 0; j <= totalPeriods; j += 1) {                        // Initially, all time slots are available
185                     if (j >= currentTime) {
186                         cols.push(["schedule-item", "schedule-available"]);
187                     } else {
188                         cols.push(["schedule-item"]);
189                     }
190                     replies.push(null);
191                     names.push("");
192                 }
193 
194                 for (j = 0; j < questionInstance.get("replies").length; j += 1) {
195                     reply = questionInstance.get("replies")[j];
196                     cIndex = reply.get("startTime");
197                     choiceDescriptor = reply.getChoiceDescriptor();
198                     choiceInstance = choiceDescriptor.getInstance();
199 
200                     cols[cIndex] = ["schedule-unavailable", "schedule-task",
201                         "schedule-unavailable-" + choiceDescriptor.get("duration")];
202 
203                     if (currentTime >= reply.get("startTime") && currentTime < reply.get("startTime") + choiceDescriptor.get("duration")) {
204                         cols[cIndex].push("schedule-ongoingtask");
205                     }
206 
207                     names[cIndex] = choiceDescriptor.get("name");
208                     replies[cIndex] = reply;
209 
210                     for (k = 1; k < choiceDescriptor.get("duration"); k += 1) {
211                         cols[cIndex + k] = ["schedule-unavailable"];
212                     }
213 
214                     cols[cIndex].push((choiceInstance.get("active")) ? "schedule-active" : "schedule-inactive");
215                 }
216 
217                 for (j = 0; j <= totalPeriods; j += 1) {
218                     if (j > currentTime) {
219                         cols[j].push("schedule-future");                        // Mark cells in the future
220                     } else if (j < currentTime) {
221                         cols[j].push("schedule-past");                          // Mark cells in the past
222                     } else {
223                         cols[j].push("schedule-present");                       // Mark cells in the past
224                     }
225                 }
226 
227                 for (j = 0; j < cols.length; j += 1) {                          // Render each cell
228                     acc.push('<td data-startTime="', j,
229                             '" class="', cols[j].join(" "), '"><div>');
230                     if (replies[j]) {
231                         acc.push('<div ' +
232                                 ' class="icon wegas-tooltip-trigger" title="', // Tooltip
233                                 escape(this.renderDetails(replies[j])), '"',
234                                 ' data-replyid="', replies[j].get("id"), '">', names[j],
235                                 '<div class="close-icon"></div></div>');
236                     } else {
237                         acc.push('<div class="icon"> <div class="close-icon"></div></div>');
238                     }
239                     acc.push('</div></td>');
240 
241                 }
242 
243                 acc.push("</tr>");
244             }
245 
246             acc.push('<tfoot><tr>', // Generate table footer
247                     //'<td class="schedule-leftcolum">Available human resources</td>');
248                     '<td class="schedule-leftcolum">Total human resources</td>');
249 
250             for (i = 0; i < perPeriodLoad.length; i += 1) {
251                 //acc.push('<td>' + (perPeriodBudget - perPeriodLoad[i]) + '/' + perPeriodBudget + '</td>');
252                 acc.push('<td>' + perPeriodLoad[i] + '</td>');
253             }
254             acc.push("</tr></tfoot></table>");
255 
256             cb.set("innerHTML", acc.join(""));                          // Update ContentBox
257         },
258         renderDetailsPanel: function(node) {
259             var columns = [{
260                     key: "choiceDescriptorId",
261                     className: "hidden"
262                 }, {
263                     sortable: true,
264                     key: "startTime",
265                     //className: 'hidden',
266                     label: "Period",
267                     className: "period"
268                 }, {
269                     sortable: true,
270                     key: "analyis",
271                     label: "Analyse"
272                 }, {
273                     key: "answer",
274                     label: "Result",
275                     allowHTML: true
276                 }, {
277                     sortable: true,
278                     key: "fileLinks",
279                     label: "Files",
280                     allowHTML: true,
281                     emptyCellValue: "no files"
282                 }];
283             this.datatable = new Y.Wegas.CrimeSimTreeble({
284                 columns: columns,
285                 isTreeble: true,
286                 node: node
287             });
288             this.datatable.render(this.get(CONTENTBOX).one(".schedule-analysis"));
289         },
290         syncDetailsPanel: function() {
291             var i, k, reply, status, replyData, cb = this.get(CONTENTBOX),
292                     question = Y.Wegas.VariableDescriptorFacade.rest.findById(this.currentQuestionId),
293                     questionInstance = question.getInstance();
294 
295             this.data.length = 0;
296 
297             cb.one("h1").setContent(question.getPublicLabel() || "undefined");
298             cb.one(".content").setContent(question.get("description") || "<em>No description</em>");
299 
300             while (this.datatable.datatable.getRow(0)) {
301                 this.datatable.datatable.removeRow(0);
302             }
303 
304             for (i = 0; i < questionInstance.get("replies").length; i += 1) {
305                 reply = questionInstance.get("replies")[i];
306                 replyData = Y.mix(reply.getAttrs(), reply.get("result").getAttrs());
307                 replyData.choiceDescriptorId = reply.get('result').get('choiceDescriptorId');
308                 status = reply.getStatus(this.currentTime);
309 
310                 if (status === 1) {
311                     replyData.answer = "analysis in progress";
312                 } else if (status === 2) {
313                     replyData.answer = "analysis planified";
314                 } else {
315                     replyData.fileLinks = "";
316                     for (k = 0; k < replyData.files.length; k = k + 1) {
317                         replyData.fileLinks += '<a target="_blank" href="' +
318                                 Y.Plugin.CRDataSource.getFullpath(replyData.files[k]) + '">' +
319                                 Y.Plugin.CRDataSource.getFilename(replyData.files[k]) + '</a><br />';
320                     }
321                     if (!replyData.fileLinks) {
322                         delete replyData.fileLinks;
323                     }
324                 }
325                 replyData.analyis = reply.getChoiceDescriptor().get("name");
326                 replyData.startTime = replyData.startTime + 1;
327                 this.data.push(replyData);
328             }
329             this.datatable.syncUI(this.data);
330 
331             cb.one(".schedule-detail").setStyles({
332                 position: 'display',
333                 display: "block",
334                 overflowX: "auto"
335             });
336             /* gallery : [{srcUrl:'url', description:'text'},{}, ...]*/
337             this.gallery.set("gallery", question.get("pictures"));
338         },
339         renderDetails: function(reply) {
340             var choiceDescriptor = reply.getChoiceDescriptor(),
341                     status = reply.getStatus(this.currentTime),
342                     ret = ['<div class="schedule-detail-reply"><h3>Period ',
343                 reply.get("startTime") + 1, ': ', choiceDescriptor.get("name") || "undefined",
344                 '</h3><div class="content">'];
345 
346             if (status === 0) {
347                 ret.push(reply.get("result").get("answer"));
348             } else if (status === 1) {
349                 ret.push("analysis in progress");
350             } else {
351                 ret.push("analysis planified");
352             }
353             ret.push("</div>");
354             return ret.join("");
355         },
356         hideDetails: function() {
357             var cb = this.get(CONTENTBOX);
358             cb.one(".schedule-detail").setStyles({
359                 position: 'display',
360                 display: "none"
361             });
362             cb.all(".schedule-leftcolum-selected").removeClass("schedule-leftcolum-selected");
363             this.currentQuestionId = null;
364         },
365         // *** Events Methods *** /
366 
367         onCancelReplyClick: function(e, args) {
368             var replyId = e.target.ancestor(".icon").getAttribute("data-replyid");
369 
370             this.showOverlay();
371             Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
372                 request: "/QuestionDescriptor/CancelReply/" + replyId + "/Player/" + Y.Wegas.app.get('currentPlayer')
373             });
374         },
375         onMenuClick: function(e) {
376             var data = e.target.get("data");
377 
378             this.showOverlay();
379             Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
380                 request: "/QuestionDescriptor/SelectChoice/" + data.choice.get("id")
381                         + "/Player/" + Y.Wegas.app.get('currentPlayer') + "/StartTime/" + data.startTime + "/"
382             });
383         },
384         /**
385          * Return a list of possible actions on a given action at a given time.
386          */
387         genMenuItems: function(question, startTime) {
388             var perPeriodBudget = 15, ret = [], i, j, disabled, choice, choiceInstance;
389             for (i = 0; i < question.get("items").length; i += 1) {
390                 choice = question.get("items")[i];
391                 choiceInstance = choice.getInstance();
392 
393                 if (!choiceInstance.get("active")) {                            // Check if the choice is active
394                     continue;
395                 }
396 
397                 disabled = this.perPeriodLoad[startTime] + choice.get("cost")   // and if we have enough resources
398                         > perPeriodBudget;
399 
400                 for (j = 0; j < choice.get("duration"); j = j + 1) {
401                     disabled = disabled                                         // finally we check if there is no other task assigned for this timeslot
402                             || question.getRepliesByStartTime(startTime + j).length > 0;
403                 }
404                 ret.push({
405                     type: "Button",
406                     label: '<span ' +
407                             //'class="wegas-tooltip-trigger" title="' +
408                             //escape( choice.get("description")) + '"' +
409                             '>' + (choice.getPublicLabel() || "undefined") + "</span>",
410                     data: {
411                         choice: choice,
412                         startTime: startTime
413                     },
414                     disabled: disabled
415                 });
416             }
417             return ret;
418         }
419     });
420 
421     Y.namespace('Wegas').ScheduleDisplay = ScheduleDisplay;
422 });
