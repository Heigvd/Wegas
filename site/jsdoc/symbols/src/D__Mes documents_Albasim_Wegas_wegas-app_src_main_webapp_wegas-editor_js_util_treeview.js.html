  6 YUI.add('treeview', function(Y) {
  7     var getClassName = Y.ClassNameManager.getClassName,
  8         TREEVIEW = 'treeview',
  9         TREENODE = 'treenode',
 10         TREELEAF = 'treeleaf',
 11         CONTENT_BOX = "contentBox",
 12         BOUNDING_BOX = "boundingBox",
 13         classNames = {
 14             loading: getClassName(TREENODE, "loading"),
 15             collapsed: getClassName(TREENODE, "collapsed"),
 16             visibleRightWidget: getClassName(TREEVIEW, "visible-right")
 17         },
 18         RIGHTWIDGETSETTERFN = function(v) {
 19             if (this.get("rightWidget") !== v && this.get("rightWidget")) {// Remove existing child
 20                 if (this.get("rightWidget") instanceof Y.Node) {        // Case 1: Y.Node
 21                     this.get("rightWidget").remove();
 22                 } else {
 23                     this.get("rightWidget").get(BOUNDING_BOX).remove(); // Case 2: Y.Widget
 24                     this.get("rightWidget").removeTarget(this);
 25                     this.set("parent", null);
 26                 }
 27             }
 28             if (v) {                                                    // Append the new widget
 29                 if (v instanceof Y.Node) {                              // Case 1: Y.Node
 30                     v.appendTo("#" + this.get("id") + "_right");
 31                 } else {                                                // Case 2: Y.Widget
 32                     if (v.get("rendered")) {
 33                         v.get(BOUNDING_BOX).appendTo("#" + this.get("id") + "_right");
 34                     } else {
 35                         v.render("#" + this.get("id") + "_right");
 36                     }
 37                     v.set("parent", this);
 38                     v.addTarget(this);
 39                 }
 40             }
 41             return v;
 42         };

 44     Y.TreeView = Y.Base.create("treeview", Y.Widget, [Y.WidgetParent], {
 45         BOUNDING_TEMPLATE: "<div></div>",
 46         CONTENT_TEMPLATE: "<ul></ul>",
 47         initializer: function() {
 48         },
 49         bindUI: function() {
 50             this.on("*:click", function(e) {
 51                 if (e.node && e.node !== this) {
 52                     this.deselectAll();
 53                     e.node.set("selected", 2);
 54                 }
 55             });
 56         },
 57         renderUI: function() {
 58             if (this.get("visibleRightWidget")) {
 59                 this.get(CONTENT_BOX).addClass(classNames.visibleRightWidget);
 60             }
 61         },
 62         expandAll: function() {
 63             this.each(function(item) {
 64                 if (item.expandAll) {
 65                     item.expandAll();
 66                 }
 67             });
 68         },
 69         collapseAll: function() {
 70             this.each(function(item) {
 71                 if (item.collapseAll) {
 72                     item.collapseAll();
 73                 }
 74             });
 75         }
 76     }, {
 77         NAME: 'treeview',
 78         ATTRS: {
 79             visibleRightWidget: {
 80                 value: false,
 81                 validator: Y.Lang.isBoolean
 82             },
 83             defaultChildType: {
 84                 value: "TreeLeaf"
 85             },
 86             multiple: {
 87                 value: false
 88             }
 89         }

 92     Y.TreeNode = Y.Base.create("treenode", Y.Widget, [Y.WidgetParent, Y.WidgetChild], {
 93         BOUNDING_TEMPLATE: "<li></li>",
 94         CONTENT_TEMPLATE: "<ul></ul>",
 95         toggleNode: null,
 96         labelNode: null,
 97         iconNode: null,
 98         currentIconCSS: null,
 99         menuNode: null,
100         eventInstances: {},
101         initializer: function() {
102             this.publish("toggleClick", {
103                 bubbles: false,
104                 broadcast: false,
105                 defaultFn: this.toggleTree
106             });
107             this.publish("nodeExpanded", {
108                 bubbles: true
109             });
110             this.publish("nodeCollapsed", {
111                 bubbles: true
112             });
113             this.publish("iconClick", {
114                 bubbles: true
115             });
116             this.publish("labelClick", {
117                 bubbles: true
118             });
119             this.publish("click", {
120                 bubbles: true
121             });
122         },
123         renderUI: function() {
124             var bb = this.get(BOUNDING_BOX), header;
125             header = Y.Node.create("<div class='content-header " + this.getClassName("content", "header") + "'></div>");
126             bb.insertBefore(header, this.get(CONTENT_BOX));
127             this.toggleNode = Y.Node.create("<span class='" + this.getClassName("content", "toggle") + "'></span>");
128             this.iconNode = Y.Node.create("<span class='" + this.getClassName("content", "icon") + "'></span>");
129             this.labelNode = Y.Node.create("<span class='" + this.getClassName("content", "label") + "'></span>");
130             header.append(this.toggleNode);
131             header.append(this.iconNode);
132             header.append(this.labelNode);
133             header.append("<div id=\"" + this.get("id") + "_right\" class=\"" + this.getClassName("content", "rightwidget") + "\">");
134             if (this.get('collapsed') && !bb.hasClass(classNames.collapsed)) {
135                 bb.addClass(classNames.collapsed);
136             }
137         },
138         bindUI: function() {
139             /*
140              * Force event firing for an exisiting and none 0 attribute
141              * "selected" on initialization
142              */
143             this.onceAfter("renderedChange", function(e) {
144                 var val = this.get("selected");
145                 if (val && !this.get("selection")) {                            /* check for last selected node */
146                     this.set("selected", 0);
147                     this.set("selected", val);
148                 }
149             });
150             this.after("selectedChange", function(e) {
151                 if (e.newVal && !e.target.get("selection")) {
152                     e.target.get(BOUNDING_BOX).addClass("selected");
153                 } else {
154                     e.target.get(BOUNDING_BOX).removeClass("selected");
155                 }
156             });
157             this.eventInstances.click = this.toggleNode.on("click", function(e) {
158                 e.stopPropagation();
159                 this.fire("toggleClick", {
160                     node: this
161                 });
162             },
163             this);
164             this.eventInstances.fullClick = this.get(BOUNDING_BOX).one("." + this.getClassName("content", "header")).on("click", function(e) {
165                 var node = e.target;
166                 e.stopPropagation();
167                 if (node.hasClass(this.getClassName("content", "icon"))) {
168                     this.fire("iconClick", {
169                         node: this
170                     });
171                 }
172                 if (node.hasClass(this.getClassName("content", "label"))) {
173                     this.fire("labelClick", {
174                         node: this
175                     });
176                 }
177                 this.fire("click", {
178                     node: this,
179                     domEvent: e
180                 });
181 
182             }, this);
183             this.get(BOUNDING_BOX).on("click", function(e) {
184                 e.stopPropagation();
185             });
186         },
187         syncUI: function() {
188             this.set("loading", this.get("loading"));
189             this.set("iconCSS", this.get("iconCSS"));
190             this.set("label", this.get("label"));
191             this.set("rightWidget", this.get("rightWidget"));
192             this.set("collapsed", this.get("collapsed"));
193         },
194         destructor: function() {
195             var event;
196             this.blur();                                                        //remove a focused node generates some errors
197             for (event in this.eventInstances) {
198                 this.eventInstances[event].detach();
199             }
200             if (this.get("rightWidget")) {
201                 this.get("rightWidget").destroy();
202             }
203         },
204         toggleTree: function() {
205             this.get(BOUNDING_BOX).toggleClass(classNames.collapsed);
206             if (this.get(BOUNDING_BOX).hasClass(classNames.collapsed)) {
207                 this.fire('nodeCollapsed', {
208                     node: this
209                 });
210             } else {
211                 this.fire('nodeExpanded', {
212                     node: this
213                 });
214             }
215         },
216         expand: function(fireEvent) {
217             this.set("collapsed", false);
218             fireEvent = (fireEvent === undefined) ? true : fireEvent;
219             if (fireEvent) {
220                 this.fire('nodeExpanded', {
221                     node: this
222                 });
223             }
224         },
225         collapse: function(fireEvent) {
226             this.set("collapsed", true);
227             fireEvent = (fireEvent === undefined) ? true : fireEvent;
228             if (fireEvent) {
229                 this.fire('nodeCollapsed', {
230                     node: this
231                 });
232             }
233         },
234         expandAll: function() {
235             this.expand(false);
236             this.each(function(item) {
237                 if (item.expandAll) {
238                     item.expandAll();
239                 }
240             });
241         },
242         collapseAll: function() {
243             this.collapse(false);
244             this.each(function(item) {
245                 if (item.collapseAll) {
246                     item.collapseAll();
247                 }
248             });
249         },
250         destroyChildren: function() {
251             var i, widgets = this.removeAll();
252             for (i in widgets) {
253                 widgets.each(this.destroyChild, this);
254             }
255         },
256         destroyChild: function(item, index) {
257             item.destroy();
258         }
259     }, {
260         NAME: "TreeNode",
261         ATTRS: {
262             label: {
263                 value: "",
264                 validator: Y.Lang.isString,
265                 setter: function(v) {
266                     if (this.labelNode) {
267                         this.labelNode.setContent(v);
268                     }
269                     return v;
270                 }
271             },
272             initialSelected: {
273                 writeOnce: "initOnly"
274             },
275             collapsed: {
276                 value: true,
277                 validator: Y.Lang.isBoolean,
278                 setter: function(v) {
279                     if (v) {
280                         this.get(BOUNDING_BOX).addClass(classNames.collapsed);
281                     } else {
282                         this.get(BOUNDING_BOX).removeClass(classNames.collapsed);
283                     }
284                     return v;
285                 }
286             },
287             tabIndex: {
288                 value: 1
289             },
290             rightWidget: {
291                 value: null,
292                 validator: function(o) {
293                     return o instanceof Y.Widget || o instanceof Y.Node || o === null;
294                 },
295                 setter: RIGHTWIDGETSETTERFN
296             },
297             loading: {
298                 value: false,
299                 validator: Y.Lang.isBoolean,
300                 setter: function(v) {
301                     if (v) {
302                         this.get(BOUNDING_BOX).addClass(classNames.loading);
303                     } else {
304                         this.get(BOUNDING_BOX).removeClass(classNames.loading);
305                     }
306                     return v;
307                 }
308             },
309             iconCSS: {
310                 value: getClassName("treenode", "default-icon"),
311                 validator: Y.Lang.isString,
312                 setter: function(v) {
313                     if (this.currentIconCSS) {
314                         this.iconNode.removeClass(this.currentIconCSS);
315                     }
316                     this.iconNode.addClass(v);
317                     this.currentIconCSS = v;
318                     return v;
319                 }
320             },
321             defaultChildType: {
322                 value: "TreeLeaf"
323             },
324             multiple: {
325                 value: false
326             },
327             data: {}
328         }
329     });

331     /**
332      * TreeLeaf widget. Default child type for TreeView.
333      * It extends  WidgetChild, please refer to it's documentation for more info.
334      * @class TreeLeaf
335      * @constructor
336      * @uses WidgetChild
337      * @extends Widget
338      * @param {Object} config User configuration object.
339      */
340     Y.TreeLeaf = Y.Base.create("treeleaf", Y.Widget, [Y.WidgetChild], {
341         CONTENT_TEMPLATE: "<div></div>",
342         BOUNDING_TEMPLATE: "<li></li>",
343         menuNode: null,
344         iconNode: null,
345         currentIconCSS: null,
346         labelNode: null,
347         events: {},
348         initializer: function() {
349             this.publish("iconClick", {
350                 bubbles: true
351             });
352             this.publish("labelClick", {
353                 bubbles: true
354             });
355             this.publish("click", {
356                 bubbles: true
357             });
358         },
359         renderUI: function() {
360             var cb = this.get(CONTENT_BOX), header;
361             header = Y.Node.create("<div class='content-header " + this.getClassName("content", "header") + "'></div>");
362             cb.append(header);
363             this.iconNode = Y.Node.create("<span class='" + this.getClassName("content", "icon") + "'></span>");
364             this.labelNode = Y.Node.create("<span class='" + this.getClassName("content", "label") + "'></span>");
365             header.append(this.iconNode);
366             header.append(this.labelNode);
367             header.append("<div id=\"" + this.get("id") + "_right\" class=\"" + this.getClassName("content", "rightwidget") + "\">");
368         },
369         bindUI: function() {
370             this.onceAfter("renderedChange", function(e) {
371                 var val = this.get("selected");
372                 if (val) {
373                     this.set("selected", 0);
374                     this.set("selected", val);
375                 }
376             });
377             this.after("selectedChange", function(e) {
378                 if (e.newVal && !e.target.get("selection")) {
379                     e.target.get(BOUNDING_BOX).addClass("selected");
380                 } else {
381                     e.target.get(BOUNDING_BOX).removeClass("selected");
382                 }
383             });
384             this.events.fullClick = this.get(CONTENT_BOX).one("." + this.getClassName("content", "header")).on("click", function(e) {
385                 var node = e.target;
386                 e.stopImmediatePropagation();
387                 if (node.hasClass(this.getClassName("content", "icon"))) {
388                     this.fire("iconClick", {
389                         node: this
390                     });
391                 }
392                 if (node.hasClass(this.getClassName("content", "label"))) {
393                     this.fire("labelClick", {
394                         node: this
395                     });
396                 }
397                 this.fire("click", {
398                     node: this,
399                     domEvent: e
400                 });
401             }, this);
402 
403             //one line, prevent special chars
404             this.labelNode.on("blur", function(e) {
405                 e.target.setContent(e.target.getContent().replace(/&[^;]*;/gm, "").replace(/(\r\n|\n|\r|<br>|<br\/>)/gm, "").replace(/(<|>|\|\\|:|;)/gm, "").replace(/^\s+/g, '').replace(/\s+$/g, ''));
406             }, this);
407         },
408         syncUI: function() {
409             this.set("label", this.get("label"));
410             this.set("iconCSS", this.get("iconCSS"));
411             this.set("editable", this.get("editable"));
412             this.set("loading", this.get("loading"));
413             this.set("rightWidget", this.get("rightWidget"));
414         },
415         destructor: function() {
416             this.blur();                                                        //remove a focused node generates some errors
417             if (this.get("rightWidget") && this.get("rightWidget").destroy) {
418                 try {
419                     this.get("rightWidget").destroy();
420                 } catch (e) {
421                 }
422             }
423             this.iconNode.remove(true);
424             this.labelNode.remove(true);
425         }
426     }, {
427         NAME: "TreeLeaf",
428         ATTRS: {
429             label: {
430                 value: "",
431                 validator: Y.Lang.isString,
432                 setter: function(v) {
433                     if (this.labelNode) {
434                         this.labelNode.setContent(v);
435                     }
436                     return v;
437                 },
438                 getter: function(v) {
439                     return this.labelNode.getContent();
440                 }
441             },
442             tabIndex: {
443                 value: -1
444             },
445             initialSelected: {
446                 writeOnce: "initOnly"
447             },
448             rightWidget: {
449                 value: null,
450                 validator: function(o) {
451                     return o instanceof Y.Widget || o instanceof Y.Node || o === null;
452                 },
453                 setter: RIGHTWIDGETSETTERFN
454             },
455             editable: {
456                 value: false,
457                 validator: Y.Lang.isBoolean,
458                 setter: function(v) {
459                     if (v) {
460                         this.labelNode.setAttribute("contenteditable", "true");
461                     } else {
462                         this.labelNode.setAttribute("contenteditable", "false");
463                     }
464                     return v;
465                 }
466             },
467             loading: {
468                 value: false,
469                 validator: Y.Lang.isBoolean,
470                 setter: function(v) {
471                     if (v) {
472                         this.get(CONTENT_BOX).addClass(classNames.loading);
473                     } else {
474                         this.get(CONTENT_BOX).removeClass(classNames.loading);
475                     }
476                     return v;
477                 }
478             },
479             iconCSS: {
480                 value: getClassName("treeleaf", "default-icon"),
481                 validator: Y.Lang.isString,
482                 setter: function(v) {
483                     if (this.currentIconCSS) {
484                         this.iconNode.removeClass(this.currentIconCSS);
485                     }
486                     this.iconNode.addClass(v);
487                     this.currentIconCSS = v;
488                     return v;
489                 }
490             },
491             data: {}
492         }
493     });
