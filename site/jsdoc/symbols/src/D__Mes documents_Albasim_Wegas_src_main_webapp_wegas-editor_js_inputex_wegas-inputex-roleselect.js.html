1 /**
  2  * @module inputex-url
  3  */
  4 YUI.add("wegas-inputex-roleselect",function(Y){
  5 
  6     var inputEx = Y.inputEx;
  7 
  8     /**
  9      * Adds an url regexp, and display the favicon at this url
 10      * @class inputEx.UrlField
 11      * @extends inputEx.StringField
 12      * @constructor
 13      * @param {Object} options inputEx.Field options object
 14      * <ul>
 15      *   <li>favicon: boolean whether the domain favicon.ico should be displayed or not (default is true, except for https)</li>
 16      * </ul>
 17      */
 18     Y.namespace("inputEx.Wegas").RoleSelect = function(options) {
 19         inputEx.Wegas.RoleSelect.superclass.constructor.call(this,options);
 20     };
 21 
 22     Y.extend(inputEx.Wegas.RoleSelect, inputEx.SelectField, {
 23 
 24         setValue: function ( val, sendUpdatedEvent ) {
 25             inputEx.Wegas.RoleSelect.superclass.setValue.call( this, val.id, sendUpdatedEvent );
 26         },
 27         getValue: function ( ) {
 28             return {
 29                 "@class": "Role",
 30                 name: "",
 31                 description: "",
 32                 id: inputEx.Wegas.RoleSelect.superclass.getValue.call( this )
 33             }
 34         },
 35         setOptions: function (options) {
 36             inputEx.Wegas.RoleSelect.superclass.setOptions.call( this, options );
 37             this.options.choices = [];
 38             for ( var i = 0; i < Y.Wegas.RoleFacade.data.length; i = i + 1 ) {
 39                 this.options.choices.push({
 40                     value: Y.Wegas.RoleFacade.data[i].get( "id" ),
 41                     label: Y.Wegas.RoleFacade.data[i].get( "name" )
 42                 });
 43             }
 44         }
 45     });
 46 
 47     inputEx.registerType( "roleselect", inputEx.Wegas.RoleSelect );             // Register this class as "wegasurl" type
 48 });
