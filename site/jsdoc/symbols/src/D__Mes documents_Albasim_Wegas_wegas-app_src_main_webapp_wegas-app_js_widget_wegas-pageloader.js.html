 15 YUI.add('wegas-pageloader', function(Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', PageLoader;
 19 
 20     PageLoader = Y.Base.create("wegas-pageloader", Y.Widget, [Y.WidgetChild, Y.WidgetParent, Y.Wegas.Widget, Y.Wegas.Editable], {
 21 
 22         // *** Lifecycle Methods ***/
 23         initializer: function() {
 24             PageLoader.pageLoaderInstances[this.get("pageLoaderId")] = this;    // We keep a references of all loaded PageLoaders
 25             this.currentPageId = null;
 26             if (this.get("defaultPageId")) {
 27                 this.set("pageId", this.get("defaultPageId"));
 28             }
 29             this.handlers = [];
 30 
 31         },
 32         bindUI: function() {
 33             //Y.Wegas.app.dataSources.Page.after("response", this.syncUI, this);
 34             this.handlers.push(Y.Wegas.PageFacade.rest.after("pageUpdated", function(e) {
 35                 if (e.page && (+e.page["@pageId"] === +this.get("pageId"))) {
 36                     this.currentPageId = null; // @hack force update
 37                     this.syncUI();
 38                 }
 39             }, this));
 40             var onUpdate = function(e) {
 41                 if (+this.get("variable.evaluated") !== +this.get('pageId')) {
 42                     this.syncUI();
 43                 }
 44             };
 45             this.handlers.push(
 46                 Y.Wegas.app.dataSources.VariableDescriptor.after("response", onUpdate, this));
 47             this.handlers.push(
 48                 Y.Wegas.app.after('currentPlayerChange', onUpdate, this));
 49 
 50             this.on("*:exception", function(e) {
 51                 var test;
 52 
 53                 e.halt(true);
 54 
 55                 if (test = e.message.match(/ConstraintViolationException: (.*) is out of bound/)) {
 56                     this.showMessage("error", "Insufficient " + test[1] + ".");
 57                 } else {
 58                     this.showMessage("error", e.message);
 59                 }
 60             });
 61         },
 62 
 63         syncUI: function() {
 64             var val = this.get("variable.evaluated");
 65             if (val && val.getInstance().get('value')) {                        // If there is a variable to refresh
 66                 this.set("pageId", val.getInstance().get('value'));
 67             } else {                                                            // Otherwise use pageId (in case the setter has not been called yet)
 68                 this.set("pageId", this.get("pageId"));
 69             }
 70         },
 71 
 72         destructor: function () {
 73             var i;
 74             if (this.get("widget")) {
 75                 this.get("widget").destroy();
 76             }
 77             for (i = 0; i < this.handlers.length; i += 1) {
 78                 this.handlers[i].detach();
 79             }
 80             delete PageLoader.pageLoaderInstances[this.get("pageLoaderId")];
 81         },
 82 
 83         // *** Private Methods ***/
 84         isLoadingALoop: function(pageId) {                                     //Page loader mustn't load the page who contain himself.
 85             var k, isALoop = false;
 86             for (k in PageLoader.pageLoaderInstances) {
 87                 if (PageLoader.pageLoaderInstances[k].get('id') === this.get('id')) { //don't check pageId of current instance and contained childrens
 88                     break;
 89                 }
 90                 if (pageId === PageLoader.pageLoaderInstances[k].get('pageId')
 91                     || pageId === PageLoader.pageLoaderInstances[k].get('variable.evaluated')) {
 92                     isALoop = true;
 93                     Y.log("Attempt to load the PageLoader of this PageLoader's instance is aborted.", 'warn', 'Wegas.PageLoader');
 94                 }
 95             }
 96             return isALoop;
 97         }
 98     }, {
 99         ATTRS: {
100             pageLoaderId: {
101                 type: "string",
102                 _inputex: {
103                     label: "Pageloader id"
104                 }
105             },
106             defaultPageId: {
107                 type: "string",
108                 _inputex: {
109                     label: "Default page",
110                     _type: "pageselect",
111                     required: false
112                 }
113             },
114             pageId: {
115                 type: "string",
116                 "transient": true,
117                 setter: function(val) {
118                     if (!val || val === this.currentPageId || this.isLoadingALoop(val)) {// If the widget is currently being loaded, escape
119                         return val;
120                     }
121                     var widgetCfg = Y.Wegas.PageFacade.rest.getPage(val);
122 
123                     if (!widgetCfg) {
124                         return val;
125                     }
126                     this.currentPageId = val;
127                     if (this.get("widget")) {
128                         this.get("widget").destroy();                           // @fixme we should remove the widget instead of destroying it
129                     }
130                     this.get(CONTENTBOX).empty();
131                     this.showOverlay();
132                     try {
133                         Y.Wegas.Widget.use(widgetCfg, Y.bind(function(cfg) {    // Load the subwidget dependencies
134                             var widget = Y.Wegas.Widget.create(cfg);            // Render the subwidget
135                             widget.render(this.get(CONTENTBOX));
136                             widget['@pageId'] = cfg['@pageId'];
137                             this.set("widget", widget);
138                             widget.addTarget(this);                             // Event on the loaded widget will be forwarded
139                             this.hideOverlay();
140                         }, this, widgetCfg));
141                     } catch (e) {
142                         Y.log('renderUI(): Error rendering widget: ' + (e.stack || e), 'error', 'Wegas.PageLoader');
143                     }
144                     return val;
145                 }
146             },
147             /**
148              * The target variable, returned either based on the name attribute,
149              * and if absent by evaluating the expr attribute.
150              */
151             variable: {
152                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER
153             },
154             widget: {
155                 "transient": true
156             }
157         },
158 
159         pageLoaderInstances: [],
160 
161         find: function (id) {
162             return PageLoader.pageLoaderInstances[id];
163         }
164     });
165 
166     Y.namespace('Wegas').PageLoader = PageLoader;
167 });
