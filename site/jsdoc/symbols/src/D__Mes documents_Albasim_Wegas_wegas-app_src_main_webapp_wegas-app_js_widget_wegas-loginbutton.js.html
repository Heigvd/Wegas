 15 YUI.add("wegas-loginbutton", function (Y) {
 16     "use strict";
 17 
 18     var LoginButton;
 19 
 20     /**
 21      * Login button
 22      */
 23     LoginButton = Y.Base.create("wegas-login", Y.Wegas.Button, [], {
 24         bindUI: function () {
 25             Y.Wegas.LoginButton.superclass.bindUI.apply(this, arguments);
 26             Y.Wegas.UserFacade.after("update", this.syncUI, this);
 27 
 28             if (this.menu) {                                                    // Don't add the plugin if it already exist.
 29                 return;
 30             }
 31 
 32             this.plug(Y.Plugin.WidgetMenu, {
 33                 children: [{
 34                     type: "Button",
 35                     label: "Preferences",
 36                     plugins: [{
 37                         "fn": "OpenPageAction",
 38                         "cfg": {
 39                             "subpageId": this.get("preferencePageId"), // @fixme
 40                             "targetPageLoaderId": this.get("targetPageLoader")
 41                         }
 42                     }]
 43                 }, {
 44                     type: "Button",
 45                     label: "Logout",
 46                     "plugins": [{
 47                         fn: "OpenUrlAction",
 48                         cfg: {
 49                             url: "wegas-app/logout",
 50                             target: "self"
 51                         }
 52                     }
 53                     ]
 54                 }]
 55             });
 56         },
 57         syncUI: function () {
 58             Y.Wegas.LoginButton.superclass.syncUI.apply(this, arguments);
 59 
 60             var cUser = Y.Wegas.app.get("currentUser"),
 61             cPlayer = Y.Wegas.GameFacade.rest.getCurrentPlayer(),
 62             cTeam = Y.Wegas.GameFacade.rest.getCurrentTeam(),
 63             name = cUser.name || "undefined";
 64             if (!this.get('labelIsUser')) {
 65                 if (cPlayer) {
 66                     name = cPlayer.get("name");
 67                 }
 68                 if (cTeam) {
 69                     name = cTeam.get("name") + " : " + name;
 70                 }
 71             }
 72             this.set("label", name);
 73         }
 74     }, {
 75         ATTRS: {
 76             labelIsUser: {
 77                 value: false,
 78                 validator: function(b) {
 79                     return (b === 'true' || b === true);
 80                 }
 81             },
 82             type: {
 83                 value: "LoginButton"
 84             },
 85             plugins: {
 86                 "transient": true,
 87                 getter: function () {
 88                     return [];
 89                 }
 90             },
 91             preferencePageId: {
 92                 value: 1000                                                     //@fixme
 93             },
 94             targetPageLoader: {
 95                 value: "maindisplayarea"
 96             }
 97         }
 98     });
 99     Y.namespace('Wegas').LoginButton = LoginButton;
100 
101 });
