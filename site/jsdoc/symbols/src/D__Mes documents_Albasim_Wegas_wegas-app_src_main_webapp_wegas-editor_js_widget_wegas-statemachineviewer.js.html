182             if (!this.panel.loader) {
183                 this.cacheDialogue = Y.Wegas.VariableDescriptorFacade.rest.filter("@class", "DialogueDescriptor");
184                 tmp = "<select><option>Select</option>";
185                 for (i in this.cacheDialogue) {

190                 this.panel.append(this.panel.loader);

193                         this.set("entity", Y.Wegas.VariableDescriptorFacade.rest.find("id", parseInt(e.target.getDOMNode().value)));

195                     this.panel.loader.remove(true);
196                     this.panel.loader = null;

200         rebuild: function() {
201             if (!this.jpLoaded) {
202                 this.hideOverlay();
203                 return false;
204             }
205             var state, states, sm = this.get("entity");
206             this.stateId = 1;
207             this.nodes = {};
208             states = this.removeAll();
209             states.each(function(item) {
210                 item.destroy();
211             });
212             this.renderPanel();
213             if (this.get("entity")) {
214                 for (state in sm.get("states")) {
215                     this.addState(sm.get("states")[state].get("editorPosition") ? sm.get("states")[state].get("editorPosition").get("x") || 30 : 30, sm.get("states")[state].get("editorPosition") ? sm.get("states")[state].get("editorPosition").get("y") || 30 : 30, parseInt(state), sm.get("states")[state]);
216                 }
217 
218                 this.each(function() {
219                     try {
220                         this.makeAllOutgoingTransitions();
221                     } catch (e) {
222                         Y.error("Failed creating transition", e, "Y.Wegas.StateMachineViewer");
223                     }
224                 });
225             }
226             this.hideOverlay();
227             this.syncUI();
228             return true;

231         addState: function(x, y, id, entity) {
232             if (!this.jpLoaded) {
233                 return null;
234             }
235             var state, config;
236             config = {
237                 sid: id,
238                 initial: (+id === +this.get("entity").getInitialStateId()),
239                 entity: entity || null,
240                 x: x,
241                 y: y
242             };
243             if (!config.entity) {
244                 delete config.entity;
245             }
246             state = new Y.Wegas.State(config);
247             this.add(state);
248             this.nodes[id.toString()] = state;
249             this.get("entity").get("states")[id.toString()] = state.get("entity");
250             this.stateId = Math.max(this.stateId, parseInt(id) + 1);
251             return state;
252         },

254             if (!this.get("entity") || !this.get("entity").get("scope")) {
255                 return false;
256             }
257             this.panel.append(this.panel.name);
258             this.panel.append(this.panel.scope);
259 
260             this.panel.name.getDOMNode().value = this.get("entity").get("name");
261             this.panel.scope.getDOMNode().value = this.get("entity").get("scope").get("@class");
262             return true;

264         processMenu: function(type) {
265             var entity,
266                 DEFAULTCB = {
267                     success: Y.bind(function(e) {
268                         this.showMessage("success", "States successfully saved", 1500);
269                         this.hideOverlay();
270                     }, this),
271                     failure: Y.bind(function(e) {
272                         this.showMessage("error", e.response.data.message);
273                         this.hideOverlay();
274                     }, this)
275                 };
276             switch (type) {
277                 case "load":
278                     this.loader();
279                     break;
280                 case "new":
281                     entity = new Y.Wegas.persistence.DialogueDescriptor();
282                     entity.setInitialStateId(1);
283                     this.set("entity", entity);
284                     break;
285                 case "save":
286                     entity = this.get("entity");
287                     if (entity) {
288                         this.showOverlay();
289                         entity = JSON.parse(JSON.stringify(entity));
290                         if (entity.id) {
291                             Y.Wegas.VariableDescriptorFacade.rest.put(entity, DEFAULTCB);
292                         } else {
293                             Y.Wegas.VariableDescriptorFacade.rest.post(entity, DEFAULTCB);
294                         }
295                     }
296                     break;
297                 default:
298                     Y.log("Not Implemented yet: " + type, "warn", "Y.Wegas.StateMachineViewer");
299             }
300         },
301         zoom: function(event) {
302             return; //zoom disabled
303             if (event.wheelDelta < 0) {
304                 this.currentZoom = (this.currentZoom < 0.35) ? 0.3 : this.currentZoom - 0.05;
305             } else {
306                 this.currentZoom = (this.currentZoom > 1.95) ? 2 : this.currentZoom + 0.05;
307             }
308             this.get(CONTENT_BOX).setStyle("zoom", this.currentZoom);
309         },

313                 return;
314             }

316                 this.removeClass("currentState");
317             });
318             this.nodes[sm.getInstance().get("currentStateId")].get("boundingBox").addClass("currentState");
319         }

321         ATTRS: {
322             entity: {
323                 value: null,
324                 validator: function(o) {
325                     return o instanceof Y.Wegas.persistence.FSMDescriptor || o === null;
326                 }
327             }
328         }

655             this.get(BOUNDING_BOX).appendTo(this.get("parent").get("parent").get(CONTENT_BOX));
656             var nextStateId = this.get("entity").get("nextStateId");
657             this.source = this.get('parent');
658             this.target = this.get("parent").get("parent").nodes[nextStateId.toString()];
659             try {
660                 this.connector = jp.connect({
661                     source: this.source.get(BOUNDING_BOX),
662                     target: this.target.get(BOUNDING_BOX),
663                     deleteEndpointsOnDetach: true,
664                     uniqueEndpoint: false,
665                     parameters: {
666                         transition: this
667                     }
668                 });
669                 this.addTarget(this.target);
670                 this.target.transitionsTarget.push(this);
671                 //this could be if we listen to click events on complete connector(ie arrow + label)
672                 // this.connector.canvas.setAttribute("cursor", "pointer");
673             } catch (e) {
674                 Y.error("Connection failed", e, "Y.Wegas.StateMachineViewer");
675             }
676             this.createLabel();
677         },
678         //
679         disconnect: function(e) {
680             jp.detach(this.connector, {
681                 fireEvent: true
682             });
683         },
684         setTriggerCondition: function(entity) {
685             entity = entity instanceof Y.Wegas.persistence.Script ? entity : Y.Wegas.Editable.reviver(entity);
686             this.get("entity").set("triggerCondition", entity);
687         },
688         createLabel: function() {
689             if (this.get("entity") instanceof Y.Wegas.persistence.DialogueTransition) {
690                 this.connector.setLabel({
691                     label: this.get("entity").get("actionText"),
692                     cssClass: "transition-label"
693                 });
694             } else {
695                 this.connector.setLabel({
696                     label: (this.get("entity").get("triggerCondition") ? this.get("entity").get("triggerCondition").get("content") : ""),
697                     cssClass: "transition-label"
698                 });
699             }
700             this.labelNode = this.connector.getLabelOverlay();
701             if (this.labelNode) {
702                 this.events.labelClick = this.labelNode.bind("click", function(e) {
703                     var that = e.component.getParameter("transition");
704                     that.editor();
705                 });
706             }
707             //Listen to complete connector
708             //            this.events.conClick = this.connector.bind("click", function (e){
709             //                var that = e.getParameter("transition");
710             //                that.editor();
711             //            });
712 
713         },
714         editor: function() {
715             var x, y;
716             x = parseInt(this.labelNode.getElement().style.left);
717             y = parseInt(this.labelNode.getElement().style.top);
718             this.get(CONTENT_BOX).setStyle("left", (x - 5) + "px");
719             this.get(CONTENT_BOX).setStyle("top", (y - 5) + "px");
720             this.show();
721         },
722         destructor: function() {
723             jp.detach(this.connector, {
724                 forceDetach: true
725             });
726             for (var i in this.events) {
727                 try {
728                     this.events[i].detach();
729                 } catch (e) {
730                     this.events[i].unbind();
731                 }
732             }
733         }
734     }, {
735         ATTRS: {
736             tid: {
737                 value: null,
738                 validator: Y.Lang.isNumber
739             },
740             entity: {
741                 valueFn: function() {
742                     var e = new Y.Wegas.persistence.Transition();
743                     return e;
744                 }
745             }
746         }
