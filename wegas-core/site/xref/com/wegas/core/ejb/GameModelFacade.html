63      @Override
64      public GameModel duplicate(final Long entityId) throws IOException {
65          ObjectMapper mapper = JacksonMapperProvider.getMapper();                // Retrieve a jackson mapper instance
66  
67          GameModel oldEntity = this.find(entityId);                              // Retrieve the entity to duplicate
68  
69          String serialized = mapper.writerWithView(Views.Export.class).
70                  writeValueAsString(oldEntity);
71          GameModel newEntity = mapper.readValue(serialized, GameModel.class);    // and deserialize it
72  
73          boolean added = false;
74          int suffix = 1;
75          String newName = null;
76          while (!added) {
77              newName = oldEntity.getName() + "(" + suffix + ")";
78              try {
79                  this.findByName(newName);
80                  suffix++;
81              }
82              catch (NoResultException ex) {
83                  added = true;
84              }
85          }
86  
87          newEntity.setName(newName);
88          this.create(newEntity);                                                 // store it db
89          em.flush();
90  
91          try {                                                                   //Clone jcr FILES
92              ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(newEntity.getId());
93              connector.cloneWorkspace(oldEntity.getId());
94          }
95          catch (RepositoryException ex) {
96              System.err.println(ex);
97          }
98          return newEntity;
99      }

114     /**
115      *
116      * @param name
117      * @return
118      */
119     public GameModel findByName(String name) throws NoResultException {
120         CriteriaBuilder cb = em.getCriteriaBuilder();
121         CriteriaQuery cq = cb.createQuery();
122         Root<GameModel> gameModel = cq.from(GameModel.class);
123         cq.where(cb.equal(gameModel.get(GameModel_.name), name));
124         Query q = em.createQuery(cq);
125         return (GameModel) q.getSingleResult();
126     }
