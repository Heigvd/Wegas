28  public class PerformFunctionAndGoBackServlet extends HttpServlet implements Servlet {
29  
30      private static transient final Logger log = LoggerFactory.getLogger(PerformFunctionAndGoBackServlet.class);
31      private static final long serialVersionUID = -7896114563632467947L;
32  
33      /**
34       *
35       * @param req
36       * @param resp
37       * @throws ServletException
38       * @throws IOException
39       */
40      @Override
41      protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
42          doPost(req, resp);
43      }
44  
45      /**
46       *
47       * @param request
48       * @param response
49       * @throws ServletException
50       * @throws IOException
51       */
52      @Override
53      protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
54          String action = request.getParameter("action");
55          String actionResult = performAction(action);
56          request.setAttribute("actionResultMessage", actionResult);
57  
58          // forward the request and response back to original page
59          String originalPage = request.getParameter("originalPage");
60          RequestDispatcher dispatcher = getServletContext().getRequestDispatcher(originalPage);
61          dispatcher.forward(request, response);
62      }
63  
64      private String performAction(String actionName) {
65          try {
66              Actions action = findAction(actionName);
67              String result = action == null ? null : action.doIt();
68              log.debug("Performed function with result: " + result);
69              return result;
70          }
71          catch (ShiroException ex) {
72              log.debug("Function failed with " + ex.getMessage() + " message.");
73              return "Error: " + ex.getMessage();
74          }
75      }
76  
77      private Actions findAction(String actionName) {
78          if (actionName == null) {
79              return null;
80          }
81  
82          Actions[] values = Actions.values();
83  
84          for (Actions action : values) {
85              if (actionName.equals(action.getName())) {
86                  return action;
87              }
88          }
89          return null;
90      }
91  
92      /**
93       *
94       * @param request
95       * @return
96       */
97      public String getUrl(HttpServletRequest request) {
98          String reqUrl = request.getRequestURL().toString();
99          String queryString = request.getQueryString(); // d=789
100         if (queryString != null) {
101             reqUrl += "?" + queryString;
102         }
103         return reqUrl;
104     }
