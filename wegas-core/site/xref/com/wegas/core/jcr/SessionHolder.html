11  package com.wegas.core.jcr;
12  
13  import java.util.HashMap;
14  import java.util.Map;
15  import java.util.ResourceBundle;
16  import javax.annotation.PreDestroy;
17  import javax.ejb.DependsOn;
18  import javax.ejb.Singleton;
19  import javax.jcr.Repository;
20  import javax.jcr.RepositoryException;
21  import javax.jcr.Session;
22  import javax.jcr.SimpleCredentials;
23  
24  /**
25   *
26   * @author Cyril Junod <cyril.junod at gmail.com>
27   */
28  
29  @Singleton
30  @DependsOn("JackrabbitConnector")
31  public class SessionHolder {
32  
33      final static private ResourceBundle resourceBundle = ResourceBundle.getBundle("wegas");
34      final static private SimpleCredentials admin = new SimpleCredentials(resourceBundle.getString("jcr.admin.username"), resourceBundle.getString("jcr.admin.password").toCharArray());
35      private static Map<String, Session> sessionMap = new HashMap<>();
36  
37      public static Session getSession(String repository) throws RepositoryException {
38          Session session;
39          if (!SessionHolder.sessionMap.containsKey(repository) || !SessionHolder.sessionMap.get(repository).isLive()) {
40              Repository repo = (Repository) new JackrabbitConnector().getRepo();
41              sessionMap.remove(repository);
42              try {
43                  session = repo.login(admin, repository);
44              } catch (javax.jcr.NoSuchWorkspaceException ex) {
45                  createWorkspace(repository);
46                  session = repo.login(admin, repository);
47              }
48              sessionMap.put(repository, session);
49          }
50          return sessionMap.get(repository);
51      }
52  
53      public static void closeSession(String repository) throws RepositoryException {
54          if (SessionHolder.sessionMap.containsKey(repository) && SessionHolder.sessionMap.get(repository).isLive()) {
55              SessionHolder.sessionMap.get(repository).save();
56              SessionHolder.sessionMap.get(repository).logout();
57          }
58          SessionHolder.sessionMap.remove(repository);
59      }
60  
61      protected static void createWorkspace(String repository) throws RepositoryException {
62          Session s;
63  
64          s = getSession(null);
65          s.getWorkspace().createWorkspace(repository);
66      }
67  
68      @PreDestroy
69      private void onDestroy() {
70          for (Session s : sessionMap.values()) {
71              if (s.isLive()) {
72                  s.logout();
73              }
74          }
75          sessionMap.clear();
76      }
77  }
