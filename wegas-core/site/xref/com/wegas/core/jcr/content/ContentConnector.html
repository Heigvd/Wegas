160     /**
161      * Compress directory and children to ZipOutputStream. Warning: metadatas are not included due to zip limitation
162      *
163      * @param out a ZipOutputStream to write files to
164      * @param path root path to compress
165      * @throws RepositoryException
166      * @throws IOException
167      */
168     public void zipDirectory(ZipOutputStream out, String path) throws RepositoryException, IOException {
169         AbstractContentDescriptor node = DescriptorFactory.getDescriptor(path, this);
170         DirectoryDescriptor root;
171         if (node.isDirectory()) {
172             root = (DirectoryDescriptor) node;
173         } else {
174             return;
175         }
176         List<AbstractContentDescriptor> list = root.list();
177 
178         ZipEntry entry;
179         for (Iterator<AbstractContentDescriptor> it = list.iterator(); it.hasNext();) {
180             AbstractContentDescriptor item = it.next();
181             entry = item.getZipEntry();
182             try {
183                 out.putNextEntry(entry);
184             } catch (ZipException ex) {
185                 logger.warn("error");
186             }
187             if (item.isDirectory()) {
188                 zipDirectory(out, item.getFullPath());
189             } else {
190                 byte[] write = ((FileDescriptor) item).getBytesData();
191                 out.write(write, 0, write.length);
192             }
193         }
194         out.closeEntry();
195     }

203     public void deleteWorkspace() throws RepositoryException {
204         //throw new UnsupportedOperationException("Jackrabbit: There is currently no programmatic way to delete workspaces. You can delete a workspace by manually removing the workspace directory when the repository instance is not running.");
205         String name = session.getWorkspace().getName();
206         SessionHolder.closeSession(workspace);
207         Session s = SessionHolder.getSession(null);
208         try {
209             s.getWorkspace().deleteWorkspace(name);
210         } catch (UnsupportedRepositoryOperationException ex) {
211             logger.warn("UnsupportedRepositoryOperationException : fallback to clear workspace. Further : improve to remove workspace");
212             session = SessionHolder.getSession(workspace);
213             this.clearWorkspace();
214             SessionHolder.closeSession(workspace);
215         }
216     }

261     public void importXML(InputStream input)
262             throws IOException, SAXException, ParserConfigurationException,
263             TransformerException, RepositoryException {
264         try {
265             DocumentBuilderFactory bf = DocumentBuilderFactory.newInstance();
266             bf.setIgnoringElementContentWhitespace(true);
267             DocumentBuilder rootBuilder = bf.newDocumentBuilder();
268             Document root = rootBuilder.parse(input);
269             NodeList list = root.getFirstChild().getChildNodes();
270             this.clearWorkspace();                                              // Remove nodes first
271             for (Integer i = 0; i < list.getLength(); i++) {
272                 if (list.item(i).getNodeName().equals("sv:node")) {
273                     Document node = bf.newDocumentBuilder().newDocument();
274                     node.appendChild(node.importNode(list.item(i), true));
275                     DOMSource source = new DOMSource(node);
276                     StringWriter writer = new StringWriter();
277                     StreamResult result = new StreamResult(writer);
278                     TransformerFactory.newInstance().newTransformer().transform(source, result);
279                     try (InputStream nodeAsStream = new ByteArrayInputStream(writer.getBuffer().toString().getBytes())) {
280                         session.getWorkspace().importXML("/", nodeAsStream, ImportUUIDBehavior.IMPORT_UUID_COLLISION_REMOVE_EXISTING);
281                     }
282                 }
283             }
284         } finally {
285             input.close();
286         }

292      * @throws RepositoryException

294     private void initializeNamespaces() throws RepositoryException {
295         for (String prefix : WFSConfig.namespaces.keySet()) {
296             try {
297                 session.getWorkspace().getNamespaceRegistry().getURI(prefix);
298             } catch (NamespaceException e) {
299                 session.getWorkspace().getNamespaceRegistry().registerNamespace(prefix, WFSConfig.namespaces.get(prefix));
300             }
301         }
