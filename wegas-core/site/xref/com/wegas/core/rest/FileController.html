67       * @param gameModelId

76      @POST
77      @Consumes(MediaType.MULTIPART_FORM_DATA)
78      @Produces(MediaType.APPLICATION_JSON)
79      @Path("upload{directory : .*?}")
80      public AbstractContentDescriptor upload(@PathParam("gameModelId") String gameModelId,
81              @FormDataParam("name") String name,
82              @FormDataParam("note") String note,
83              @FormDataParam("note") String description,
84              @PathParam("directory") String path,
85              @FormDataParam("file") InputStream file,
86              @FormDataParam("file") FormDataBodyPart details) throws RepositoryException, WegasException {
87          logger.debug("File name: {}", details.getContentDisposition().getFileName());
88          if (name == null) {
89              name = details.getContentDisposition().getFileName();
90          }
91          Pattern pattern = Pattern.compile(FILENAME_REGEXP);
92          Matcher matcher = pattern.matcher(name);
93          if (name.equals("") || !matcher.matches()) {
94              throw new WegasException(name + " is not a valid filename.");
95          }
96          ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
97  
98          AbstractContentDescriptor detachedFile = null;
99          AbstractContentDescriptor dir = DescriptorFactory.getDescriptor(path, connector);
100         if (dir.exist()) {                                                  //directory has to exist
101             if (details == null || details.getContentDisposition().getFileName() == null || details.getContentDisposition().getFileName().equals("")) {       //Assuming an empty filename means a directory
102                 detachedFile = new DirectoryDescriptor(name, path, connector);
103             } else {
104                 logger.debug("File name: {}", details.getContentDisposition().getFileName());
105                 detachedFile = new FileDescriptor(name, path, connector);
106             }
107             if (!detachedFile.exist()) {                                        //Node should not exist
108                 note = note == null ? "" : note;
109                 detachedFile.setNote(note);
110                 detachedFile.setDescription(description);
111                 if (detachedFile instanceof FileDescriptor) {
112                     //TODO : check allowed mime-types
113                     try {
114                         ((FileDescriptor) detachedFile).setBase64Data(file, details.getMediaType().toString());
115                         logger.info(details.getFormDataContentDisposition().getFileName() + "(" + details.getMediaType() + ") uploaded as \"" + name + "\"");
116                     } catch (IOException ex) {
117                         logger.error("Error reading uploaded file :", ex);
118                         connector.save();
119                     }
120                 } else {
121                     detachedFile.sync();
122                     logger.info("Directory {} created at {}", detachedFile.getName(), detachedFile.getPath());
123                 }
124             } else {
125                 throw new WegasException(detachedFile.getPath() + "/" + name + " already exists");
126             }
127         } else {
128             logger.debug("Parent Directory does not exist");
129         }
130         connector.save();
131         return detachedFile;
132     }

140     @GET
141     @Path("read{absolutePath : .*?}")
142     public Response read(@PathParam("gameModelId") String gameModelId, @PathParam("absolutePath") String name) {
143         logger.debug("Asking file (/{})", name);
144         AbstractContentDescriptor fileDescriptor;
145         ContentConnector connector = null;
146         Response.ResponseBuilder response = Response.status(404);
147         try {
148             connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
149             fileDescriptor = DescriptorFactory.getDescriptor(name, connector);
150         } catch (PathNotFoundException e) {
151             logger.debug("Asked path does not exist: {}", e.getMessage());
152             connector.save();
153             return response.build();
154         } catch (RepositoryException e) {
155             logger.error("Need to check those errors", e);
156             return response.build();
157         }
158         if (fileDescriptor instanceof FileDescriptor) {
159             response = Response.ok(((FileDescriptor) fileDescriptor).getBase64Data());
160             response.header("Content-Type", fileDescriptor.getMimeType());
161             response.header("Description", fileDescriptor.getDescription());
162             CacheControl cc = new CacheControl();
163             cc.setMaxAge(3600);
164             cc.setPrivate(true);
165             response.cacheControl(cc);
166             response.lastModified(((FileDescriptor) fileDescriptor).getDataLastModified().getTime());
167 
168             try {
169                 ((FileDescriptor) fileDescriptor).getBase64Data().close();
170             } catch (IOException ex) {
171                 Logger.getLogger(FileController.class.getName()).log(Level.SEVERE, null, ex);
172             } finally {
173                 connector.save();
174             }
175         }
176         return response.build();
177     }

192             AbstractContentDescriptor dir = DescriptorFactory.getDescriptor(directory, connector);
193             if (!dir.exist() || dir instanceof FileDescriptor) {

195                 return null;

275     @POST
276     @Path("importXML")
277     @Consumes(MediaType.MULTIPART_FORM_DATA)
278     @Produces(MediaType.APPLICATION_JSON)
279     public List<AbstractContentDescriptor> importXML(@PathParam("gameModelId") String gameModelId,
280             @FormDataParam("file") InputStream file,
281             @FormDataParam("file") FormDataBodyPart details)
282             throws RepositoryException, IOException, SAXException,
283             ParserConfigurationException, TransformerException, WegasException {
284         try {
285             final ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
286             switch (details.getMediaType().getSubtype()) {
287                 case "x-gzip":
288                     try (GZIPInputStream in = new GZIPInputStream(file)) {
289                         connector.importXML(in);
290                     }
291                     break;
292                 case "xml":
293                     connector.importXML(file);
294                     break;
295                 default:
296                     throw new WegasException("Uploaded file mimetype does not match requirements [XML or Gunzip], found:" + details.getMediaType().toString());
297             }
298             connector.save();
299         } finally {
300             file.close();
301         }
302         return this.listDirectory(gameModelId, "/");
303     }

312     @DELETE

314     @Produces(MediaType.APPLICATION_JSON)
315     public Object delete(@PathParam("gameModelId") String gameModelId,
316             @PathParam("absolutePath") String absolutePath,
317             @PathParam("force") String force) {
318 
319         boolean recursive = force.equals("") ? false : true;
320         logger.debug("Asking delete for node ({}), force {}", absolutePath, recursive);
321         try {
322             ContentConnector connector = ContentConnectorFactory.getContentConnectorFromGameModel(extractGameModelId(gameModelId));
323             AbstractContentDescriptor descriptor = DescriptorFactory.getDescriptor(absolutePath, connector);
324             if (descriptor.exist()) {
325                 descriptor.sync();
326                 if (descriptor instanceof DirectoryDescriptor && ((DirectoryDescriptor) descriptor).isRootDirectory()) {
327                     return Response.notModified("Unable to erase Root Directory").build();
328                 }
329                 try {
330                     descriptor.delete(recursive);
331                 } catch (ItemExistsException e) {
332                     return Response.notModified(e.getMessage()).build();
333                 }
334                 connector.save();
335                 return descriptor;
336             } else {
337                 connector.save();
338                 return Response.notModified("Path" + absolutePath + " does not exist").build();
339             }
340         } catch (RepositoryException ex) {
341             logger.error("Really what append here ??", ex);
342         }
343         return null;
344     }
