 14 YUI.add( "wegas-book-dice", function ( Y ) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", Dice;
 18 
 19     Dice = Y.Base.create( "wegas-book-dice", Y.Widget, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable ], {
 20         
 21         result:0,
 22         handlers: new Array(),
 23         rollButton:null,
 24         isRolling:false,
 25         
 26         rollDice: function(when, iteration){
 27             var cb = this.get(CONTENTBOX), result,
 28             min = parseInt(this.get("min")), max = parseInt(this.get("max"));
 29             result = Math.ceil(Math.random()*(max-min+1))+min-1;
 30             cb.one(".result").setHTML();
 31             cb.one(".result").append("<p class='result-value-"+result+"'>"+result+"</p>");
 32             if(this.get("animated").indexOf("true")>-1 && iteration>0){
 33                 Y.later(when, this, Y.bind(this.rollDice, this, when+Math.ceil(when/4), iteration-1));
 34             }
 35             else{
 36                 this.result = result;
 37                 this.fire("diceRolled");
 38                 this.isRolling = false;
 39             }
 40         },
 41         
 42         initializer: function(){
 43             this.publish("diceRolling", {});
 44             this.publish("diceRolled", {});
 45             this.rollButton = new Y.Wegas.Button({
 46                 label:this.get("label"),
 47                 tooltip: this.get("tooltip")
 48             });
 49         },
 50         
 51         renderUI: function(){
 52             var cb = this.get(CONTENTBOX);
 53             cb.append("<div class='wegas-dice'></div>");
 54             cb.one(".wegas-dice").append("<div class='button'></div>");
 55             cb.one(".wegas-dice").append("<div class='result'></div>");
 56             this.rollButton.render(cb.one('.wegas-dice .button'));
 57         },
 58         
 59         bindUI: function(){
 60             var cb = this.get(CONTENTBOX);
 61             this.handlers.push(cb.one(".wegas-dice .button").delegate('click', function(){
 62                 if(this.isRolling || this.rollButton.get("disabled")) return;
 63                 this.isRolling = true;
 64                 this.fire("diceRolling");
 65                 this.rollDice(30, 12);
 66             }, "button", this));
 67         },
 68         
 69         destructor: function(){
 70             var i;
 71             for (i=0; i<this.handlers.length;i++) {
 72                 this.handlers[i].detach();
 73             }
 74             this.rollButton.destroy();
 75         }  
 76 
 77     }, {
 78         ATTRS : {
 79             min: {
 80                 type: "Integer",
 81                 value: 1
 82             },
 83             max: {
 84                 type: "Integer",
 85                 value: 6                
 86             },
 87             animated: {
 88                 type : "Boolean",
 89                 value : "false"
 90             },
 91             label:{
 92                 type : "String",
 93                 value : "Lancer le d�"
 94             },
 95             tooltip:{
 96                 type : "String",
 97                 value : null
 98             }
 99         }
100     });
101 
102     Y.namespace( "Wegas" ).Dice = Dice;
103 });
