 15 YUI.add('wegas-gaugedisplay', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', GaugeDisplay;
 19 
 20     GaugeDisplay = Y.Base.create("wegas-gauge", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable], {
 21 
 22         CONTENT_TEMPLATE: '<div style="text-align: center;line-height:3px"><canvas height="50px" width="100px"></canvas><center class="label"></center><center class="percent"></center></div>',
 23         MAXVAL: 200,
 24 
 25         // ** Lifecycle Methods ** //
 26 
 27         renderUI: function () {
 28             var opts = {
 29                 lines: 12,                                                      // The number of lines to draw
 30                 angle: 0.15,                                                    // The length of each line
 31                 lineWidth: 0.44,                                                // The line thickness
 32                 pointer: {
 33                     length: 0.5,                                                // The radius of the inner circle
 34                     strokeWidth: 0.035,                                         // The rotation offset
 35                     color: '#000000'                                            // Fill color
 36                 },
 37                 colorStart: '#0981A9',                                          // Colors
 38                 colorStop: '#000000',
 39                 //strokeColor: '#E0E0E0',
 40                 strokeColor: '#FFFFFF',
 41                 generateGradient: true
 42             };
 43             this.gauge = new Gauge(this.get("contentBox").one("canvas").getDOMNode());// create the  gauge!conso
 44             this.gauge.setOptions(opts);
 45             this.gauge.maxValue = this.MAXVAL;                                  // set max gauge value
 46             this.gauge.animationSpeed = 32;                                     // set animation speed (32 is default value)
 47         },
 48 
 49         bindUI: function () {
 50             this.handlers = [];
 51             this.handlers.push(Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this));
 52             this.handlers.push(Y.Wegas.app.after('currentPlayerChange', this.syncUI, this));
 53         },
 54 
 55         syncUI: function () {
 56             var maxVal, minVal, value, label,
 57             variableDescriptor = this.get("variable.evaluated");
 58             if (!variableDescriptor) {
 59                 return;
 60             }
 61 
 62             label = this.get("label") || variableDescriptor.getPublicLabel();
 63             minVal = variableDescriptor.get("minValue");
 64             maxVal = variableDescriptor.get("maxValue") - minVal;
 65             value = (variableDescriptor.getInstance().get("value") - minVal) / maxVal * this.MAXVAL;
 66             if (!value) {
 67                 value = 0.1;                                                    // @hack @fixme unkown bug, value seams to be treated by gauge as false...
 68             }
 69 
 70             this.gauge.set(value);                                              // set actual value
 71             this.get(CONTENTBOX).one(".label").setContent(label);
 72             this.get(CONTENTBOX).one(".percent").setContent(Math.round(value / this.MAXVAL * 100) + "%");
 73         },
 74 
 75         destructor: function () {
 76             var i;
 77             for (i = 0; i < this.handler.length; i = i + 1) {
 78                 this.handlers[i].detach();
 79             }
 80         }
 81     }, {
 82         ATTRS : {
 83             variable: {
 84                 getter: Y.Wegas.persistence.Editable.VARIABLEDESCRIPTORGETTER
 85             },
 86             label : {
 87                 type: "string",
 88                 optional: true,
 89                 validator: Y.Lang.isString
 90             }
 91         }
 92     });
 93 
 94     Y.namespace('Wegas').GaugeDisplay = GaugeDisplay;
 95 });
