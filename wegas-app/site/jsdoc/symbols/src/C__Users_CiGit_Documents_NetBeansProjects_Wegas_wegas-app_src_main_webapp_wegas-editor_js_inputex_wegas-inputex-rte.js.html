 15 YUI.add("wegas-inputex-rte", function (Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx;
 19 
 20     /**
 21      * Wrapper for the Rich Text Editor from YUI
 22      * @class inputEx.RTEField
 23      * @extends inputEx.Field
 24      * @constructor
 25      * @param {Object} options
 26      */
 27     inputEx.RTEField = function (options) {
 28         inputEx.RTEField.superclass.constructor.call(this, options);
 29     };
 30 
 31     Y.extend(inputEx.RTEField, inputEx.Textarea, {
 32 
 33         destroy: function () {
 34             inputEx.RTEField.superclass.destroy.call(this);
 35         },
 36 
 37         /**
 38          * Set the default values of the options
 39          * @param {Object} options Options object as passed to the constructor
 40          */
 41         setOptions: function (options) {
 42             inputEx.RTEField.superclass.setOptions.call(this, options);
 43 
 44             this.options.opts = options.opts || {};
 45             this.options.typeInvite = null;
 46         },
 47 
 48         /**
 49 	 * Render the field using the YUI Editor widget
 50 	 */
 51         renderComponent: function () {
 52             inputEx.RTEField.superclass.renderComponent.call(this);
 53             if (!inputEx.RTEField.init) {
 54                 inputEx.RTEField.init = true;
 55                 tinyMCE.init({
 56                     // General options
 57                     mode : "none",                                              // "none", "textares"
 58                     theme : "advanced",                                         // "simple", "advanced"
 59 
 60                     // autolink, lists,spellchecker,pagebreak,style,
 61                     // layer,table,save,advhr,advimage,advlink,emotions,iespell,
 62                     // inlinepopups,insertdatetime,preview,media,searchreplace,
 63                     // print,contextmenu,paste,directionality,fullscreen,noneditable,
 64                     // visualchars,nonbreaking,xhtmlxtras,template,wordcount,advlist,
 65                     // autosave,visualblocks
 66                     plugins : "autolink,autoresize,lists,spellchecker,style,layer,table," +
 67                     "advimage,advlink,emotions,iespell,inlinepopups,media," +
 68                     "searchreplace,contextmenu,fullscreen,visualchars",
 69 
 70                     // Theme options
 71                     theme_advanced_buttons1 : "bold,italic,underline,|,justifyleft,justifycenter,justifyright,justifyfull,styleselect,|,bullist,numlist,|,outdent,indent,|,undo,redo,|,link,unlink,image,media,charmap,emotions,iespell,|,forecolor,backcolor",
 72                     theme_advanced_buttons2 : "tablecontrols,|,hr,removeformat,cleanup,styleprops,iespell,spellchecker,visualaid,|,insertlayer,moveforward,movebackward,absolute,|,search,replace,|,fullscreen,code",
 73                     theme_advanced_toolbar_location : "top",
 74                     theme_advanced_toolbar_align : "left",
 75                     theme_advanced_statusbar_location : "none",                 // top, bottom, none
 76                     theme_advanced_resizing : false,
 77                     relative_urls : false,
 78 
 79                     file_browser_callback: function (field_name, url, type, win) {
 80 
 81                         if (!inputEx.RTEField.filePanel) {
 82                             inputEx.RTEField.filePanel = new Y.Panel({
 83                                 headerContent: 'Choose a file from library',
 84                                 bodyContent: '',
 85                                 width: 600,
 86                                 height: Y.DOM.winHeight() - 150,
 87                                 zIndex: 303000,
 88                                 modal: true,
 89                                 render: true,
 90                                 centered: true
 91                             });
 92 
 93                             inputEx.RTEField.filePanel.explorer = new Y.Wegas.FileExplorer().render(inputEx.RTEField.filePanel.getStdModNode(Y.WidgetStdMod.BODY));
 94 
 95                             inputEx.RTEField.filePanel.explorer.on("*:fileSelected", function (e, path) {
 96                                 e.stopImmediatePropagation();
 97                                 e.preventDefault();
 98                                 inputEx.RTEField.filePanel.hide();
 99 
100                                 var win = inputEx.RTEField.filePanel.win,
101                                 field_name = inputEx.RTEField.filePanel.field_name,
102                                 targetInput = win.document.getElementById(field_name);
103                                 targetInput.value = Y.Plugin.CRDataSource.getFullpath(path);  // update the input field
104 
105                                 if (typeof (win.ImageDialog) !== "undefined") { // are we an image browser
106                                     if (win.ImageDialog.getImageData) {         // we are, so update image dimensions...
107                                         win.ImageDialog.getImageData();
108                                     }
109 
110                                     if (win.ImageDialog.showPreviewImage) {     // ... and preview if necessary
111                                         win.ImageDialog.showPreviewImage(Y.Plugin.CRDataSource.getFullpath(path));
112                                     }
113                                 }
114                                 if (win.Media) {                                // If in an editor window
115                                     win.Media.formToData("src");                // update the data
116                                 }
117                             });
118                         }
119 
120                         inputEx.RTEField.filePanel.show();
121                         inputEx.RTEField.filePanel.win = win;
122                         inputEx.RTEField.filePanel.field_name = field_name;
123                         return false;
124                     }
125 
126                 //content_css : "css/content.css",                              // Example content CSS (should be your site CSS)
127 
128                 // Drop lists for link/image/media/template dialogs
129                 //                    template_external_list_url : "lists/template_list.js",
130                 //                    external_link_list_url : "lists/link_list.js",
131                 //                    external_image_list_url : "lists/image_list.js",
132                 //                    media_external_list_url : "lists/media_list.js",
133 
134                 // Style formats
135                 //style_formats : [
136                 //{
137                 //    title : 'Bold text',
138                 //    inline : 'b'
139                 //},{
140                 //    title : 'Red text',
141                 //    inline : 'span',
142                 //    styles : {
143                 //        color : '#ff0000'
144                 //    }
145                 //},{
146                 //    title : 'Red header',
147                 //    block : 'h1',
148                 //    styles : {
149                 //        color : '#ff0000'
150                 //    }
151                 //},{
152                 //    title : 'Example 1',
153                 //    inline : 'span',
154                 //    classes : 'example1'
155                 //},{
156                 //    title : 'Example 2',
157                 //    inline : 'span',
158                 //    classes : 'example2'
159                 //},{
160                 //    title : 'Table styles'
161                 //},{
162                 //    title : 'Table row 1',
163                 //    selector : 'tr',
164                 //    classes : 'tablerow1'
165                 //}],
166 
167                 // Replace values for the template plugin
168                 //template_replace_values : {
169                 //    username : "Some User",
170                 //    staffid : "991234"
171                 //}
172                 });
173             }
174             Y.once("domready", function () {
175                 tinyMCE.execCommand('mceAddControl', false, this.el.id);
176             }, this);
177         },
178 
179         /**
180          * Set the html content
181          * @param {String} value The html string
182          * @param {boolean} [sendUpdatedEvt] (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
183          */
184         setValue: function (value) {
185             inputEx.RTEField.superclass.setValue.apply(this, arguments);
186 
187             var tmceI = tinyMCE.get(this.el.id);
188             if (tmceI) {
189                 tmceI.setContent(value);
190             }
191         },
192 
193         /**
194          * Get the html string
195          * @return {String} the html string
196          */
197         getValue: function () {
198             tinyMCE.triggerSave();
199             return inputEx.RTEField.superclass.getValue.call(this);
200         },
201 
202         /**
203          * @static
204          */
205         filePanel: null
206     });
207 
208 
209     inputEx.registerType("html", inputEx.RTEField, []);                        // Register this class as "html" type
210 
211 });
