 11 YUI.add('wegas-scripteval', function(Y) {
 12     "use strict";
 13 
 14     var ScriptEval;
 15 
 16     ScriptEval = Y.Base.create("ScriptEval", Y.Plugin.Base, [], {
 17         context: null,
 18         upToDate: false,
 19         initializer: function() {
 20             this.context = {};
 21             this.upToDate = false;
 22             this.afterHostEvent("response", function(e) {
 23                 this.upToDate = false;
 24             }, this);
 25             this.publish("evaluated");
 26             this.publish("failure");
 27         },
 28         /**
 29          *  the id should be generated and returned  by the function, not passed as a
 30          *  a parameter
 31          *
 32          *  @param script The script to evaluate
 33          *  @param id The transaction id
 34          *  @param cb A callback object, containing success, failure and scope objects
 35          */
 36         scopedEval: function(script, id, cb) {
 37             var result, response, url;
 38 
 39             try {
 40                 result = this.localEval(script);
 41                 this.fire("evaluated", result, id);
 42                 if (cb && cb.success)
 43                     cb.success.call(cb.scope || this, result);
 44                 return result;
 45             } catch (error) {
 46                 url = Y.Wegas.VariableDescriptorFacade.get("source") + "/Script/Run/Player/" + Y.Wegas.app.get('currentPlayer');
 47                 return Y.io(url, {
 48                     headers: {
 49                         'Content-Type': 'application/json; charset=iso-8859-1',
 50                         'Managed-Mode': 'false'
 51                     },
 52                     sync: false,
 53                     method: "POST",
 54                     data: Y.JSON.stringify({
 55                         "@class": "Script",
 56                         "language": "JavaScript",
 57                         "content": script
 58                     }),
 59                     on: {
 60                         success: Y.bind(function(id, req_id, response) {
 61                             this.fire("evaluated", Y.JSON.parse(response.responseText), id);
 62                         }, this, id),
 63                         failure: Y.bind(function(id, req_id, response) {
 64                             try {
 65                                 this.fire("failure", Y.JSON.parse(response.responseText), id);
 66                             } catch (e) {
 67                                 this.fire("failure", null, id);
 68                             }
 69                         }, this, id)
 70                     }
 71                 });
 72             }
 73         },
 74         /**
 75          * Tries to evaluate the script locally, using variables cache
 76          * @param {String} script The script to eval localy
 77          */
 78         localEval: function(script) {
 79             if (!this.upToDate) {                                                 //Only compute if new value
 80                 this.buildContext();
 81             }
 82             return (new Function("with(this) { return " + script + ";}")).call(this.context);
 83         },
 84         buildContext: function() {
 85             var data = this.get("host").data;
 86             this.upToDate = true;
 87             this.context = {};
 88             for (var i in data) {
 89                 this.context[data[i].get('name')] = JSON.parse(JSON.stringify(data[i].getInstance()));
 90                 if (data[i] instanceof Y.Wegas.persistence.ListDescriptor) {
 91                     this.context[data[i].get('name')].items = [];
 92                     for (var j in data[i].get("items")) {
 93                         this.context[data[i].get('name')].items.push(JSON.parse(JSON.stringify(data[i].get("items")[j].getInstance())));
 94                     }
 95                 }
 96             }
 97             /*SANDBOX*/
 98             Y.mix(this.context, {
 99                 window: undefined,
100                 Y: undefined,
101                 YUI: undefined
102             });
103         }
104 
105     }, {
106         NS: "script",
107         NAME: "scriptEval"
108     });
109 
110     Y.namespace('Plugin').ScriptEval = ScriptEval;
111 
112 });
