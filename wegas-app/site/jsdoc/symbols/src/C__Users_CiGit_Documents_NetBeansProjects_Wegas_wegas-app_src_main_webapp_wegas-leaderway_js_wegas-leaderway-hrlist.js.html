  5 YUI.add('wegas-leaderway', function (Y) {
  6     "use strict";
  7 
  8     var CONTENTBOX = 'contentBox', HRList;
  9 
 10     HRList = Y.Base.create("wegas-hrlist", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 11         
 12         // *** Fields *** /
 13         table: null,
 14         data: new Array(),
 15         handlers: new Array(),
 16 
 17         //*** Particular Methods ***/
 18         /**
 19          * Add rows to the datatable. Get informations on the resources  
 20          * @param ListDescriptor listResourcesDescriptor, A list of all resources.
 21          */
 22         getResourcesData: function(listResourcesDescriptor){
 23             var i, picture, resourceDescriptor, resourceInstance, needPicture = false;
 24             if(listResourcesDescriptor.get('items')[0].getInstance().get('properties').picture != null){
 25                 needPicture = true;
 26                 this.table.addColumn({ key: 'picture', label: 'Portrait', allowHTML: true }, 0);
 27             }
 28             for (i = 0; i < listResourcesDescriptor.get('items').length; i++) {
 29                 resourceDescriptor = listResourcesDescriptor.get('items')[i];
 30                 resourceInstance = resourceDescriptor.getInstance();
 31                 if(needPicture){
 32                      picture = '<img src="'+resourceInstance.get('properties').picture+'" alt="picture" width="70" height="70" />';
 33                      this.data.push({
 34                         picture:picture,
 35                         name:resourceDescriptor.get('name'),
 36                         surname:resourceInstance.get('properties').surname,
 37                         occupation:this.getOccupation(resourceInstance),
 38                         folder:i,
 39                         speak:i
 40                     })                   
 41                 }
 42                 else{
 43                     this.data.push({
 44                         name:resourceDescriptor.get('name'),
 45                         surname:resourceInstance.get('properties').surname,
 46                         occupation:this.getOccupation(resourceInstance),
 47                         folder:i,
 48                         speak:i
 49                     })
 50                 }
 51             }
 52         },
 53         
 54         /**
 55          * Return the text of the occupation.
 56          * @deprecated check only if a assignement exist in the given resource.
 57          * @param ResourceInstance resourceInstance, the resource to get the occupation text.
 58          * @return String 'libre' or the name of the occupation 
 59          */
 60         getOccupation: function(resourceInstance){
 61             var occupation, listDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "tasks"), taskDescriptor;
 62             if(resourceInstance.get('assignments').length == 0){
 63                 occupation = 'Libre';
 64             }
 65             else{
 66                 for (var i = 0; i < listDescriptor.get('items').length; i = i + 1) {
 67                         taskDescriptor = listDescriptor.get('items')[i];
 68                         if(taskDescriptor.get('id') == resourceInstance.get('assignments')[0].get('taskDescriptorId')){
 69                             occupation = taskDescriptor.get('name');
 70                             break;
 71                         }
 72                 }
 73             }
 74             return occupation;
 75         },
 76 
 77         // *** Lifecycle Methods *** //
 78         /**
 79          * Render the widget.
 80          * Create the child widget "table"  
 81          */
 82         renderUI: function (){
 83             this.table = new Y.DataTable({
 84                 columns: [
 85                 {
 86                     key:"name", 
 87                     label:"Nom",
 88                     sortable:true
 89                 },
 90 
 91                 {
 92                     key:"surname", 
 93                     label:"Prenom",
 94                     sortable:true
 95                 },
 96 
 97                 {
 98                     key:"occupation", 
 99                     label:"Occupation",
100                     sortable:true
101                 },
102                 {
103                     key: "folder",
104                     formatter: '<input class="folder" type="button" name="folder" value="dossier">',
105                     label: ' ',
106                     allowHTML: true
107                 },
108                 ]
109             });
110             this.table.render(this.get(CONTENTBOX));
111         },
112           
113         /**
114          * Bind some function at nodes of this widget
115          */
116         bindUI: function() {
117             this.handlers.push(Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this));
118             this.handlers.push(Y.Wegas.app.after('currentPlayerChange', this.syncUI, this));
119             this.table.delegate('click', function (e) {
120                 alert('todo');
121             }, '.yui3-datatable-data .folder', this.table);
122         },
123         
124         /**
125          * Synchronise the content of this widget.
126          */
127         syncUI: function (){
128             var listResourcesDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "resources");
129             if(listResourcesDescriptor == null) return;
130             this.data.length = 0;
131             this.getResourcesData(listResourcesDescriptor);
132             this.table.addRows(this.data);
133             if(this.data[0] == null){
134                 this.table.showMessage("Personne n'est disponible.");
135             }
136             this.goToFinalPage();// ! hack function            
137         },
138         
139         /*
140          * Destroy all child widget and all remanent function
141          */
142         destructor: function(){
143             var i;
144             this.table.destroy();
145             for (i=0; i<this.handlers.length;i++) {
146                 this.handlers[i].detach();
147             }
148         },
149         
150         // *** hack Methods *** //
151         /**
152          * if current week > max value of week value, then
153          * change the current widget to go on the "dialogue" widget.
154          */
155         goToFinalPage: function(){
156             var currentWeek = Y.Wegas.VariableDescriptorFacade.rest.find("name", "week"),
157             targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
158             if(parseInt(currentWeek.getInstance().get('value')) > currentWeek.get('maxValue')){
159                 targetPageLoader.once("widgetChange", function(e) {
160                     e.newVal.setCurrentDialogue("dialogueFinal");
161                 });
162                 targetPageLoader.set("pageId", this.get('dialoguePageId'));    
163             }
164         }
165     },
166     {
167         ATTRS : {
168             dialoguePageId: {
169                 value:null,
170                 validator: function (s){
171                     return s === null || Y.Lang.isString(s);
172                 }
173             },
174             targetPageLoaderId: {
175                 value:null,
176                 validator: function (s){
177                     return s === null || Y.Lang.isString(s);
178                 }
179             }
180         }
181     });
182 
183     Y.namespace('Wegas').HRList = HRList;
184 });
