 15 YUI.add('wegas-widgettoolbar', function (Y) {
 16     "use strict";
 17 
 18     /**
 19      *  @class Wegas.WidgetToolbar
 20      *  @module Wegas
 21      *  @constructor
 22      */
 23     var  WidgetToolbar = function () {
 24         WidgetToolbar.superclass.constructor.apply(this, arguments);
 25     };
 26 
 27     WidgetToolbar.NS = "toolbar";
 28     WidgetToolbar.NAME = "toolbar";
 29 
 30 
 31     Y.extend(WidgetToolbar, Y.Plugin.Base, {
 32 
 33         // *** Lifecycle methods *** //
 34 
 35         initializer: function () {
 36             //            var host = this.get("host");
 37             //            this.afterHostEvent("render", function () {
 38             //
 39             //                }, this);
 40             this.render();
 41         },
 42 
 43         destructor: function (){
 44             var i, children = this.get("children");
 45             for ( i = 0; i < children.length; i = i + 1 ){
 46                 children[i].destroy();
 47             }
 48         },
 49 
 50 
 51         // *** Private methods *** //
 52         render: function () {
 53             var host = this.get("host"),
 54             bb = host.get('boundingBox'),
 55             children = this.get("children");
 56 
 57             bb.append('<div class="wegas-toolbar"><div class="wegas-toolbar-header"></div><div class="wegas-toolbar-panel"></div></div>');
 58             host.get('contentBox').setStyles({
 59                 position: "absolute",
 60                 bottom: "26px",
 61                 overflow: "auto",
 62                 padding:"0",
 63                 left:"0px",
 64                 right:"0px",
 65                 top: "31px"
 66             });
 67 
 68             for (var i = 0; i < children.length; i = i + 1) {
 69                 children[i] = this.add( children[ i ] );
 70             }
 71         },
 72 
 73         add: function( widget ) {
 74             if ( !( widget instanceof Y.Widget )) {
 75                 widget = Y.Wegas.Widget.create( widget );
 76             }
 77             widget.render(this.get("header"));
 78             widget.addTarget(this.get("host"));
 79             return widget;
 80         }
 81 
 82     }, {
 83         ATTRS: {
 84             children: {
 85                 value: []
 86             },
 87             header: {
 88                 lazyAdd: false,
 89                 value: false,
 90                 getter : function () {
 91                     return this.get("host").get('boundingBox').one(".wegas-toolbar-header");
 92                 }
 93             },
 94             panel: {
 95                 lazyAdd: false,
 96                 value: false,
 97                 getter : function () {
 98                     return this.get("host").get('boundingBox').one(".wegas-toolbar-panel");
 99                 }
100             }
101         }
102     });
103 
104     Y.namespace('Plugin').WidgetToolbar = WidgetToolbar;
105 });
