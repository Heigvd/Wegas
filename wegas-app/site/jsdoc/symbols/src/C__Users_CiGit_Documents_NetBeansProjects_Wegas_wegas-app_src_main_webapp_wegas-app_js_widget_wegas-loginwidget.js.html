 15 YUI.add('wegas-loginwidget', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19     BOUNDINGBOX = 'boundingBox',
 20     LoginWidget;
 21 
 22     /**
 23      *
 24      *  @class Y.Wegas.JoinGameWidget
 25      *
 26      */
 27     LoginWidget = Y.Base.create("wegas-loginwidget", Y.Widget, [ Y.WidgetChild, Y.Wegas.Widget ], {
 28 
 29         // *** Private fields *** //
 30         defaultRedirect: "wegas-app/view/lobby.html",
 31 
 32         // *** Lifecycle Methods *** //
 33 
 34         renderUI: function () {
 35             var cb = this.get(CONTENTBOX),
 36             cUser = Y.Wegas.app.get("currentUser");
 37 
 38             if (cUser.accounts[0]["@class"] !== "GuestAccount") {
 39                 this.showMessage("success", "You are already logged in.", 4000);
 40                 this.redirect();
 41             }
 42 
 43             this.loginForm = new Y.inputEx.Group({
 44                 fields: [{
 45                     name: "email",
 46                     label: "Email",
 47                     required: true,
 48                     type: "email"
 49                 }, {
 50                     name: "password",
 51                     label: "Password",
 52                     required: true,
 53                     type: "password",
 54                     capsLockWarning: true
 55                 }, {
 56                     label: "",
 57                     type: "boolean",
 58                     name: "remember",
 59                     rightLabel: " Remember me"
 60                 }],
 61                 parentEl: cb
 62             });
 63 
 64             var newEntity = new Y.Wegas.persistence.JpaAccount(),
 65             formCfg = newEntity.getFormCfg();
 66             this.createAccountForm = Y.inputEx(formCfg);
 67 
 68             formCfg.parentEl = cb;
 69             this.createAccountForm = new Y.inputEx.Group({
 70                 parentEl: cb,
 71                 fields: [{
 72                     name: "id",
 73                     type: "hidden"
 74                 }, {
 75                     name: "@class",
 76                     required: true,
 77                     type: "hidden",
 78                     value: "JpaAccount"
 79                 }, {
 80                     label: "First name",
 81                     name: "firstname",
 82                     required: true,
 83                     type: "string",
 84                     showMsg: true
 85                 }, {
 86                     label: "Last name",
 87                     name: "lastname",
 88                     required: true,
 89                     type: "string",
 90                     showMsg: true
 91                 }, {
 92                     label: "Email",
 93                     name: "email",
 94                     required: true,
 95                     type: "email",
 96                     showMsg: true
 97                 }, {
 98                     label: "Password",
 99                     name: "password",
100                     strengthIndicator: true,
101                     capsLockWarning: true,
102                     id: "password",
103                     required: true,
104                     type: "password",
105                     showMsg: true
106                 }, {
107                     label: "Password (confirm)",
108                     name: "passwordConfirm",
109                     showMsg: true,
110                     required: true,
111                     confirm: "password",
112                     type: "password"
113                 }]
114             });
115 
116             this.loginButton = new Y.Button();
117             this.loginButton.render(cb);
118 
119             cb.append(' or <a class="alt-link" href="#"></a>');
120         },
121 
122         bindUI: function () {
123             var cb = this.get(CONTENTBOX);
124 
125             cb.one(".alt-link").on("click", this.toggleMode, this);
126 
127             this.loginButton.on("click", function (e) {                      // join a game based on a token
128 
129                 if (this.get("mode") === "login") {
130                     if (!this.loginForm.validate()) {
131                         return;
132                     }
133 
134                     var value = this.loginForm.getValue();
135                     this.doLogin(value.email, value.password, value.remember);
136 
137                 } else {
138                     if (!this.createAccountForm.validate()) {
139                         return;
140                     }
141 
142                     Y.Wegas.UserFacade.rest.sendRequest({
143                         request: "/Signup/",
144                         cfg: {
145                             method: "POST",
146                             data: this.createAccountForm.getValue()
147                         },
148                         on: {
149                             success: Y.bind(function (e) {
150                                 this.showMessage("success", "User created, you can now use it to login", 4000);
151                                 this.set("mode", "login");
152                             }, this),
153                             failure: Y.bind(function (e) {
154                                 this.showMessage("error", e.response.results.message || "Error creating user", 4000);
155                             }, this)
156                         }
157                     });
158                 }
159             }, this);
160         },
161 
162         syncUI: function () {
163             this.set("mode", this.get("mode"));
164         },
165 
166         toggleMode: function () {
167             if (this.get("mode") === "login") {
168                 this.set("mode", "createaccount");
169             } else {
170                 this.set("mode", "login");
171             }
172         },
173 
174         doLogin: function (email, password, remember) {
175             Y.Wegas.UserFacade.rest.sendRequest({
176                 request: "/Authenticate/?email=" + email
177                 + "&password=" + password
178                 + "&remember=" + remember,
179                 cfg: {
180                     method: "POST"
181                 },
182                 on: {
183                     success: Y.bind(function (e) {
184                         this.showMessage("success", "Login successful", 4000);
185                         this.redirect();
186                         return;
187                     }, this),
188                     failure: Y.bind(function (e) {
189                         this.showMessage("error", e.response.results.message || "Email/password combination not found", 4000);
190                     }, this)
191                 }
192             });
193         },
194 
195         redirect: function () {
196             window.location = this.getRedirect();
197         },
198 
199         getRedirect: function () {
200             return this.getQueryParameter("redirect") ||
201             (Y.Wegas.app.get("base") + this.defaultRedirect);
202         },
203 
204         /**
205          * Returns a parameter from the GET parameters.
206          */
207         getQueryParameter: function (name) {
208             var query = window.location.search.substring(1),
209             vars = query.split("&"),
210             i, pair;
211 
212             for (i = 0; i < vars.length; i = i + 1) {
213                 pair = vars[i].split("=");
214                 if (pair[0] === name) {
215                     return decodeURIComponent(pair[1]);
216                 }
217             }
218             return null;
219         }
220 
221     }, {
222         ATTRS: {
223             mode: {
224                 value: "login",
225                 setter: function (val) {
226                     var cb = this.get(CONTENTBOX);
227 
228                     if (val === "login") {
229                         //this.loginForm.clear();
230                         this.loginForm.show();
231                         this.createAccountForm.hide();
232                         this.loginButton.set("label", "Login");
233                         cb.one(".alt-link").setContent("create a new user");
234                     } else {
235                         this.loginForm.hide();
236                         this.createAccountForm.show();
237                         //this.createAccountForm.clear();
238                         this.loginButton.set("label", "Submit");
239                         cb.one(".alt-link").setContent("login with existing account");
240                     }
241                 }
242             }
243         }
244     });
245 
246     Y.namespace('Wegas').LoginWidget = LoginWidget;
247 });
