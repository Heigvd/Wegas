  5 YUI.add('wegas-datatable', function(Y) {
  6 
  7     var CONTENTBOX = 'contentBox',
  8 
  9     DataTable = Y.Base.create("wegas-datatable", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 10 
 11 	_dataSource: null,
 12 	_pushButton: null,
 13 	_table: null,
 14 
 15 	initializer: function(cfg) {
 16 	    this._dataSource = Y.Wegas.app.dataSources[this.get('dataSource')];
 17 	},
 18 
 19 	renderUI: function () {
 20 	    var cb = this.get(CONTENTBOX);
 21 
 22 	    this._table = new Y.DataTable({				// Instantiate the data table
 23 		columns: this.get('columnset')
 24 	    });
 25 	    this._table.plug(Y.Plugin.DataTableDataSource, {
 26 		datasource: this._dataSource
 27 		//initialRequest: "/"
 28 	    });
 29 	    this._table.render(cb);
 30 	},
 31 	bindUI: function() {
 32             return;
 33             
 34 	    var that = this;
 35 	    this._dataSource.after("response", function (e) {			// Listen for datasource updates
 36 		this._table.set('recordset', e.data);
 37 	    }, this);
 38 
 39 	    Y.delegate('click', function(e) {					// Listen for click events on the table
 40 		var target = e.currentTarget,
 41 		record = this._table.getRecord(target.ancestor('tr').get('id'));
 42 
 43 		//cellIndex = Y.Node.getDOMNode(target).cellIndex,
 44 		//columnSet = this.get('columnset');
 45 		//col = columnSet._conf.data.value.definitions[cellIndex],
 46 		//name= recordSet.get(target.ancestor('tr').get('id')).getValue('name');
 47 
 48 		this._dataSource.rest.getById(record.get('id'));
 49 
 50 		this._dataSource.once('response', function(e) {
 51 		    Y.Wegas.editor.edit(e.response.results[0], function(cfg) {
 52 			that._dataSource.rest.put(cfg);
 53 		    });
 54 		})
 55 
 56 	    }, this.get(CONTENTBOX), 'td', this);
 57 	},
 58 	syncUI: function() {
 59 	}
 60     }, {
 61 	ATTRS : {
 62 	    classTxt: {
 63 		value: 'DataTable'
 64 	    },
 65 	    type: {
 66 		value: "DataTable"
 67 	    },
 68 	    columnset: {},
 69 	    dataSource: {}
 70 	}
 71     });
 72 
 73     Y.namespace('Wegas').DataTable = DataTable;
 74 });
