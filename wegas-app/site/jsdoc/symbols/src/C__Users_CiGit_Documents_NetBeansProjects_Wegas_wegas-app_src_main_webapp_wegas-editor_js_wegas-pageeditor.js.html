 15 YUI.add('wegas-pageeditor', function(Y) {
 16     "use strict";
 17 
 18     /**
 19      *  @class WidgetMenu
 20      *  @module Wegas
 21      *  @constructor
 22      */
 23     var BOUNDINGBOX = "boundingBox",
 24             CONTENTBOX = "contentBox",
 25             Alignable = Y.Base.create("wegas-pageeditor-overlay", Y.Widget, [Y.WidgetPosition, Y.WidgetPositionAlign, Y.WidgetStack], {
 26         CONTENT_TEMPLATE: '<div><span class="wegas-icon wegas-icon-edit"></span><div>'
 27     }, {
 28         CSS_PREFIX: "wegas-pageeditor-overlay"
 29     }),
 30     PageEditor = function() {
 31         PageEditor.superclass.constructor.apply(this, arguments);
 32     };
 33 
 34 
 35     Y.extend(PageEditor, Y.Plugin.Base, {
 36         // *** Lifecycle methods *** //
 37         initializer: function() {
 38             this.afterHostEvent("render", this.render);
 39             this.handlers = [];
 40         },
 41         render: function() {
 42             var el, host = this.get('host');
 43 
 44             if (host.toolbar) {
 45                 el = host.toolbar.get('header');
 46                 this.designButton = new Y.ToggleButton({
 47                     label: "<span class=\"wegas-icon wegas-icon-designmode\"></span>Edit page",
 48                     on: {
 49                         click: Y.bind(function(e) {
 50                             this.get("host").get(BOUNDINGBOX)
 51                                     .toggleClass("wegas-pageeditor-designmode", e.target.get("pressed"));
 52                             if (e.target.get("pressed")) {
 53                                 this.bind();
 54                             } else {
 55                                 this.detach();
 56                             }
 57                         }, this)
 58                     }
 59                 }).render(el);
 60             }
 61 
 62             this.highlightOverlay = new Alignable({// Init the highlighting overlay
 63                 zIndex: 1,
 64                 render: true,
 65                 visible: false
 66             });
 67 
 68             this.highlightOverlay.plug(Y.Plugin.WidgetMenu, {
 69                 children: [{
 70                         type: "Button",
 71                         label: "Edit",
 72                         on: {
 73                             click: Y.bind(function() {                            // Display the edit form
 74                                 Y.Plugin.EditEntityAction.showEditForm(this.targetWidget, Y.bind(function(targetWidget, val, e, f) {
 75                                     Y.Plugin.EditEntityAction.hideEditFormOverlay();
 76                                     targetWidget.setAttrs(val);
 77                                     targetWidget.syncUI();
 78                                     Y.Wegas.PageFacade.rest.patch(targetWidget.get("root").toObject());
 79                                 }, this, this.targetWidget));
 80                             }, this)
 81                         }
 82                     }, {
 83                         type: "Button",
 84                         label: "Delete",
 85                         on: {
 86                             click: Y.bind(function() {
 87                                 this.targetWidget.destroy();
 88                             }, this)
 89                         }
 90                     }],
 91                 selector: ".wegas-icon"
 92             });
 93 
 94             this.highlightOverlay.menu.on("menuOpen", function() {
 95                 this.targetWidget = this.overlayWidget;
 96             }, this);
 97         },
 98         bind: function() {
 99             var cb = this.get('host').get(CONTENTBOX);
100 
101             this.handlers.push(cb.delegate("mouseenter", function(e) {
102                 var widget = Y.Widget.getByNode(e.currentTarget);
103                 if (widget) {
104                     this.showOverlay(widget);
105                 }
106                 e.halt();
107             }, '.wegas-widget', this));
108 
109             this.handlers.push(cb.delegate("mouseleave", function(e) {
110                 //console.log("out", e.currentTarget.get('id'));
111                 this.hideOverlay();
112                 e.halt();
113 
114                 var parentWidget = Y.Widget.getByNode(e.currentTarget.get('parentNode'));
115                 if (parentWidget && parentWidget.get('root') !== parentWidget) {
116                     this.showOverlay(parentWidget);
117                 }
118             }, '.yui3-widget', this));
119         },
120         detach: function() {
121             for (var i = 0; i < this.handlers.length; i = i + 1) {
122                 this.handlers[ i ].detach();
123             }
124         },
125         showOverlay: function(widget) {
126             var targetNode = widget.get(BOUNDINGBOX);
127 
128             if (!widget.toObject || this.overlayWidget === widget) {
129                 return;
130             }
131 
132             Y.log("Highlight " + widget.get('id'), "debug", "Y.Wegas.PageEditor");
133             this.overlayWidget = widget;
134 
135             targetNode.prepend(this.highlightOverlay.get(BOUNDINGBOX));
136             this.highlightOverlay.get(CONTENTBOX).setStyle("height", targetNode.getHeight());
137             this.highlightOverlay.get(CONTENTBOX).setStyle("width", targetNode.getWidth());
138             this.highlightOverlay.align(targetNode, ["tl", "tl"]);
139             this.highlightOverlay.show();
140         },
141         hideOverlay: function() {
142             this.highlightOverlay.hide();
143         }
144 
145     }, {
146         NS: "pageeditor",
147         NAME: "pageeditor",
148         ATTRS: {}
149     });
150     Y.namespace('Plugin').PageEditor = PageEditor;
151 
152     Y.Node.prototype.getWidth = function() {
153         return parseInt(this.getComputedStyle('width'))
154                 + parseInt(this.getComputedStyle('margin-left')) + parseInt(this.getComputedStyle('margin-right'))
155                 + parseInt(this.getComputedStyle('padding-left')) + parseInt(this.getComputedStyle('padding-right'));
156     };
157     Y.Node.prototype.getHeight = function() {
158         return parseInt(this.getComputedStyle('height'))
159                 + parseInt(this.getComputedStyle('margin-top')) + parseInt(this.getComputedStyle('margin-bottom'))
160                 + parseInt(this.getComputedStyle('padding-top')) + parseInt(this.getComputedStyle('padding-bottom'));
161     };
162 });
