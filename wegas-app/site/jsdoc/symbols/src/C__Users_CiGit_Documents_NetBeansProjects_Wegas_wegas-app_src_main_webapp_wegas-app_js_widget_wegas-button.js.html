 24     Y.Button.prototype._uiSetLabel = function (value) {
 25         var node = this._host,
 26         attr = (node.get('tagName').toLowerCase() === 'input') ? 'value' : 'text';

 28         // node.set(attr, value);
 29         node.setContent(value);
 30         return value;
 31     };

 33     /**
 34      *  Custom Button implementation. Adds Y.WidgetChild and Y.Wegas.Widget extensions
 35      *  to the original Y.Button
 36      *
 37      *  @class Y.Wegas.Button
 38      *
 39      */
 40     Button = Y.Base.create("button", Y.Button, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable ], {
 41         // *** Private fields *** //
 42 
 43         // *** Lifecycle Methods *** //
 44         initializer: function () {
 45             Button.superclass.initializer.apply(this, arguments);
 46 
 47             //this.constructor.CSS_PREFIX = "yui3-button";                      // Revert changes done by Y.Wegas.Widget so styling will work
 48             this._cssPrefix = "yui3-button";
 49 
 50             if (this.get("cssClass")) {
 51                 this.get(CONTENTBOX).addClass(this.get("cssClass"));
 52             }
 53 
 54             if (this.get("tooltip")) {
 55                 this.plug(Y.Plugin.Tooltip, {
 56                     content: this.get("tooltip")
 57                 });
 58             }
 59         },
 60 
 61         renderUI: function () {
 62             Button.superclass.renderUI.apply(this, arguments);
 63             this.get(BOUNDINGBOX).addClass("wegas-button");
 64         }
 65     }, {
 66         ATTRS: {
 67             label: {
 68                 type: "string"
 69             },
 70             tooltip: {
 71                 type: "string",
 72                 optional: true,
 73                 "transient": true
 74             },
 75             data: {
 76                 "transient": true
 77             },
 78             cssClass: {
 79                 value : null
 80             }
 81         }
 82     });

 85     /**
 86      * Plugin which adds an unread message counter to a widget.
 87      *
 88      * @class Y.Wegas.UnreadCount
 89      */
 90     var UnreadCount = function () {
 91         UnreadCount.superclass.constructor.apply(this, arguments);
 92     };
 93 
 94     Y.mix(UnreadCount, {
 95         NS: "button",
 96         NAME: "UnreadCount",
 97         ATTRS: {
 98             variable: {},
 99             expr: {},
100             /**
101              * The target variable, returned either based on the variableName attribute,
102              * and if absent by evaluating the expr attribute.
103              */
104             variableDesc: {
105                 getter: function () {
106                     if (this.get("variable")) {
107                         return Y.Wegas.VariableDescriptorFacade.rest.find('name', this.get("variable"));
108                     } else {
109                         return Y.Wegas.VariableDescriptorFacade.rest.findById(
110                             Y.Wegas.VariableDescriptorFacade.script.scopedEval(this.get("expr")));
111                     }
112                 }
113             }
114         }
115     });
116 
117     Y.extend(UnreadCount, Y.Plugin.Base, {
118 
119         initializer: function () {
120             Y.Wegas.app.dataSources.VariableDescriptor.after("response",        // If data changes, refresh
121                 this.syncUI, this);
122             this.afterHostEvent("render", this.syncUI, this);
123         },
124         syncUI: function () {
125             var cb = this.get('host').get(CONTENTBOX),
126             target = cb.one(".unread-count"),
127             unreadCount = this.getUnreadCount();
128 
129             if (!target) {                                                      // If the counter span has not been rendered, do it
130                 cb.append('<span class="unread-count"></span>');
131                 target = cb.one(".unread-count");
132             }
133 
134             if (unreadCount > 0) {                                              // Update the content
135                 target.setContent(" <span class='bracket'>(</span><span class='value'>" + unreadCount + "</span><span class='bracket'>)</span>");
136             } else {
137                 target.setContent("");
138             }
139         },
140 
141         getUnreadCount:  function () {
142             var i, instance, messages, count = 0,
143             descriptor = this.get('variableDesc');
144 
145             if (!descriptor) {
146                 return 0;
147             }
148 
149             if (descriptor.get("items")) {                                      // For ListDescriptors, we count the children instance's
150                 for (i = 0; i < descriptor.get("items").length; i = i + 1) {
151                     instance = descriptor.get("items")[i].getInstance();
152                     //count += instance.get("unread") ? 1 : 0;
153                     count += instance.get("replies").length === 0 && instance.get("active") ? 1 : 0; // only count if it is active
154                 }
155             }
156 
157             messages = descriptor.getInstance().get("messages");                // For InboxVariableDescriptors, we count the replies
158             if (messages) {
159                 for (i = 0; i < messages.length; i = i + 1) {
160                     count += messages[i].get("unread") ? 1 : 0;
161                 }
162             }
163 
164             return count;
165         }
166     });
167     Y.namespace('Plugin').UnreadCount = UnreadCount;

169     /**
170      * Login button
171      */
172     LoginButton = Y.Base.create("wegas-login", Y.Wegas.Button, [], {
173         bindUI: function () {
174             Y.Wegas.LoginButton.superclass.bindUI.apply(this, arguments);
175 
176             Y.Wegas.GameFacade.after("response", this.syncUI, this);
177             Y.Wegas.app.after("currentPlayerChange", this.syncUI, this);
178             this.plug(Y.Plugin.WidgetMenu, {
179                 children: [{
180                     type: "Button",
181                     label: "Preferences",
182                     disabled: true
183                 }, {
184                     type: "Button",
185                     label: "Logout",
186                     plugins: [{
187                         fn: "OpenUrlAction",
188                         cfg: {
189                             url: "wegas-app/logout",
190                             target: "self"
191                         }
192                     }]
193                 }]
194             });
195         },
196         syncUI: function () {
197             Y.Wegas.LoginButton.superclass.syncUI.apply(this, arguments);
198 
199             var cUser = Y.Wegas.app.get("currentUser"),
200             cPlayer = Y.Wegas.GameFacade.rest.getCurrentPlayer(),
201             cTeam = Y.Wegas.GameFacade.rest.getCurrentTeam(),
202             name = cUser.name || "undefined";
203 
204             if (cPlayer) {
205                 name = cPlayer.get("name");
206             }
207             if (cTeam) {
208                 name = cTeam.get("name") + " : " + name;
209             }
210             this.set("label", name);
211         }
212     }, {
213         ATTRS : {
214             classTxt: {
215                 value: 'LoginButton'
216             },
217             type: {
218                 value: "LoginButton"
219             }
220         }
221     });
222     Y.namespace('Wegas').LoginButton = LoginButton;
