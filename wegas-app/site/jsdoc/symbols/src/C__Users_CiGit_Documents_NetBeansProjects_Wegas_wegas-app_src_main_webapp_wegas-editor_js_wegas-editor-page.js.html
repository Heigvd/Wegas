 12 YUI.add('wegas-editor-page', function(Y){
 13     var PageEditor, CONTENT_BOX = "contentBox", BOUNDING_BOX = "boundingBox";
 14 
 15     PageEditor = Y.Base.create("wegas-editor-page", Y.Widget, [Y.WidgetChild],{
 16         initializer: function(){
 17             this.dataSource = Y.Wegas.PageFacade.rest;
 18             this.plug(Y.Plugin.WidgetToolbar);
 19         },
 20         renderUI: function(){
 21             this.tw = new Y.TreeView({
 22                 render:this.get(CONTENT_BOX)
 23             });
 24         },
 25         syncUI: function(){
 26             this.dataSource.getIndex(Y.bind(this.buildIndex, this));
 27         },
 28         bindUI: function(){
 29             this.tw.on("treeleaf:click", function(e){
 30                 this.get("pageLoader").set("pageId", e.node.get("data"));
 31             }, this);
 32             this.dataSource.after("pageUpdated", this.syncUI);
 33         },
 34         buildIndex: function(index){
 35             var i;
 36             this.tw.removeAll();
 37             for(i in index){
 38                 this.tw.add(new Y.TreeLeaf({
 39                     label: "Page: "+ index[i] + " (" + i+ ")",
 40                     data: i
 41                 }));
 42             }
 43         },
 44         destructor: function(){
 45             this.tw.destroy();
 46         }
 47     },{
 48         ATTRS:{
 49             pageLoader:{
 50                 value:"previewPageLoader",
 51                 getter: function(v){
 52                     return Y.Wegas.PageLoader.find(v);
 53                 }
 54             }
 55         }
 56     });
 57 
 58     Y.namespace("Wegas").PageEditor = PageEditor;
 59 });
