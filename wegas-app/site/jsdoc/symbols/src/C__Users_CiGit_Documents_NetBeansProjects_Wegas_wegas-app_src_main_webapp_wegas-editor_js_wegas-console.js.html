  5 YUI.add('wegas-console', function (Y) {
  6     var CONTENTBOX = 'contentBox',
  7     Console;
  8 
  9     Console = Y.Base.create("wegas-console", Y.Widget, [Y.WidgetChild,  Y.Wegas.Widget], {
 10 
 11         form: null,
 12 
 13         destructor: function () {
 14         },
 15 
 16         renderUI: function () {
 17             this.plug( Y.Plugin.WidgetToolbar );
 18 
 19             var cb = this.get(CONTENTBOX),
 20             el = this.toolbar.get('header');
 21 
 22             this.aceField = new Y.inputEx.AceField({
 23                 parentEl: cb,
 24                 typeInvite: 'Enter script here',
 25                 rows: 7
 26             });
 27             cb.append('<div class="results"></div>');
 28 
 29             this.runButton = new Y.Button({
 30                 label: "<span class=\"wegas-icon wegas-icon-play\"></span>Run script",
 31                 on: {
 32                     click: Y.bind(function () {
 33                         Y.Wegas.app.dataSources.VariableDescriptor.rest.sendRequest({
 34                             request: "/Script/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
 35                             cfg: {
 36                                 method: "POST",
 37                                 data: Y.JSON.stringify({
 38                                     "@class": "Script",
 39                                     language: "JavaScript",
 40                                     content: this.aceField.getValue()
 41                                 })
 42                             },
 43                             on: {
 44                                 scope: this,
 45                                 success: function(e) {
 46                                     cb.one(".results").prepend('<div class="result">Script exectuted. Returned value: '
 47                                         + e.response.results.entities[0] + "</div>");
 48                                 },
 49                                 failure: function(e) {
 50                                     cb.one(".results").prepend('<div class="result">Error executing script: '
 51                                         + e.response.results.message + "</div>");
 52                                 }
 53                             }
 54                         });
 55                     }, this)
 56                 }
 57             }).render(el);
 58         }
 59     });
 60 
 61 
 62     Y.namespace('Wegas').Console = Console;
 63 });
