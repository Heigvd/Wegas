103 Object.prototype.clone = function() {
104   var newObj = (this instanceof Array) ? [] : {};
105   for (var i in this) {
106     if (i == 'clone') continue;
107     if (this[i] && typeof this[i] == "object") {
108       newObj[i] = this[i].clone();
109     } else newObj[i] = this[i]
110   } return newObj;
111 };
