181         /**
182          * Get the occupation of the given resource. this resource can be vacant, sick or on work.
183          * @param ResourceInstance resourceInstance, the resource to get the occupation.
184          * @return Object with two argument : a code (Integer) and a task if the resource is sick or on work. The code must be 0 (vacant), 1 (on work), 2 (sick)
185          */
186         getOccupationObject: function(resourceInstance){
187             var i, j, occupationObject = null, sick=false,
188             taskListDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "tasks"),
189             listAbsenceDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "absences"),
190             taskDescriptor;
191             for (i = 0; i < listAbsenceDescriptor.get('items').length; i++) {
192                 taskDescriptor = listAbsenceDescriptor.get('items')[i];
193                 if(taskDescriptor.getInstance().get('active')){
194                     for(j = 0; j < resourceInstance.get('assignments').length; j++){
195                         if(taskDescriptor.get('id') == resourceInstance.get('assignments')[j].get('taskDescriptorId')){
196                             sick=true;
197                             occupationObject = {
198                                 code:2, 
199                                 taskDescriptor: taskDescriptor
200                             };
201                             break;
202                         }
203                     }
204                 }
205             }
206             if(!sick){
207                 for (i = 0; i < taskListDescriptor.get('items').length; i++) {
208                     for(j = 0; j < resourceInstance.get('assignments').length; j++){
209                         taskDescriptor = taskListDescriptor.get('items')[i];
210                         if(taskDescriptor.get('id') == resourceInstance.get('assignments')[j].get('taskDescriptorId')){
211                             occupationObject = {
212                                 code:1, 
213                                 taskDescriptor: taskDescriptor
214                             };
215                         }
216                     }
217                 }
218             }
219             if(occupationObject == null){
220                 occupationObject = {
221                     code:0, 
222                     taskDescriptor: null
223                 };
224             }
225             return occupationObject;
226         },

268         /**
269          * Create a DOM element usable as a gauge.
270          * @param String label, the label of the gauge (must be between 0 and 100)
271          * @param Integer nombreOfUnits, the nombre of div in the gauge container (the value of the gauge).
272          * @return String div container of the gauge
273          */
274         createGauge: function(label, nomberOfUnits){
275             var gauge = new Array("");
276             if(typeof nomberOfUnits === 'number'){
277                 if(nomberOfUnits>=0 && nomberOfUnits<=100){
278                     gauge.push('<span class="gauge-label">');
279                     gauge.push(label);
280                     gauge.push(' <span class="gauge-value">(');
281                     gauge.push(nomberOfUnits);
282                     gauge.push('/100)</span><span class="gauge-units">');
283                     for(var i=0; i<nomberOfUnits; i++){
284                         gauge.push('<div class="gauge-unit"></div>');
285                     }
286                     gauge.push('</span></div>');
287                 }
288                 else{
289                     gauge.push('<span class="error">The number for the gauge "'+label+'" must to be between 0 to 100.</span>');
290                 }
291             }
292             else{
293                 gauge.push('<span class="error">Unvalid number to create gauge : '+ label+"</span>");
294             }
295             return gauge.join("");
296         },
