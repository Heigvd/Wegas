 29         destroy: function () {

 34         /**
 35          *

 37         setOptions: function (options) {
 38             inputEx.WysiwygScript.superclass.setOptions.call(this, options);
 39             this.options.className = options.className || 'inputEx-Field inputEx-WysiwigScript';
 40             this.options.mode = options.mode || "wysiwyg";
 41             this.options.expects = options.expects || "expression";             // condition or expression

 44         /**
 45          *
 46          */
 47         getValue: function () {
 48             if (this.options.mode === "wysiwyg") {
 49                 var ct = "";
 50                 if (this.exprList.getArray().length > 0) {
 51                     if (this.options.expects === "condition") {
 52                         ct = this.exprList.getArray().join(" && ");
 53                     } else {
 54                         ct = this.exprList.getArray().join(";\n") + ";";
 55                     }
 56                 }
 57                 return {
 58                     '@class': "Script",
 59                     language: "JavaScript",
 60                     content: ct
 61                 };
 62             } else {
 63                 return inputEx.WysiwygScript.superclass.getValue.apply(this, arguments);
 64             }
 65         },

 76                 label: "<span class=\"wegas-icon wegas-icon-viewsrc\"></span>"
 77             });

129         updateExpressionList: function () {
130             var i, tree,
131             container = new Y.Node(this.fieldContainer),
132             fields = [];
133 
134             container.one(".msg").setContent("");                               // Reset layout
135 
136             try {
137                 tree = window.esprima.parse(this.el.value, {                    // Generate the syntaxic tree using esprima
138                     raw: true
139                 });
140 
141                 for (i = 0; i < tree.body.length; i = i + 1) {
142                     fields = fields.concat(this.generateExpression(tree.body[i].expression));
143                 }
144             } catch (e) {
145                 //Y.error("Error evaluating line: " + window.escodegen.generate(tree.body[i].expression, {indent: true}));
146                 this.setMode("text");
147                 this.viewSrc.set("disabled", true);
148                 container.one(".msg").setContent("Unable to read this impact, displaying source only.");
149                 return;
150             }
151 
152             this.viewSrc.set("disabled", false);
153             if (this.exprList) {
154                 this.exprList.destroy();
155             }
156             if (this.options.expects === "condition") {
157                 for (i = 0; i < fields.length; i += 1) {
158                     fields[i].type = "variabledescriptorcondition";
159                 }
160             }
161             this.exprList = Y.inputEx({                                         // Render the expression as a Y.inputEx.Wegas.ListField
162                 type: "inputlist",
163                 fields: fields,
164                 useButtons: true,
165                 parentEl: this.fieldContainer,
166                 addType: (this.options.expects === "condition") ? "variabledescriptorcondition" : "variabledescriptorsetter"
167             });
168             this.exprList.addButton.get("boundingBox").setStyle("display", "inline-block");
169 
170         // this.setMode(this.options.mode);
171         },

173         /**
174          *
175          */
176         generateExpression: function (expression) {
177             Y.log("generateExpression(" + expression.type + ")");
178             switch (expression.type) {
179 
180                 case "Literal":
181                     // @fixme GOTCHA catch for the true hack in condition
182                     return [];
183 
184                 case "BinaryExpression":
185                     var vdSelect = this.generateExpression(expression.left)[0], args = [];
186                     vdSelect.type = "variabledescriptorcondition";
187                     vdSelect.operator = expression.operator;
188                     vdSelect.rightValue = expression.right.raw;
189                     return [vdSelect];
190 
191                 case "LogicalExpression":
192                     //return [{
193                     //    type: "inputlist",
194                     //    fields: this.generateExpression(expression.left).concat(this.generateExpression(expression.right),
195                     //    useButtons: true,
196                     //    addType: "variabledescriptorcondition"
197                     //}]
198                     return this.generateExpression(expression.left).concat(this.generateExpression(expression.right));
199 
200                 case "CallExpression":
201                     switch (expression.callee.object.type) {
202                         case "Identifier":
203                             switch (expression.callee.object.name) {
204                                 case "VariableDescriptorFacade":
205                                     return {
206                                         type: "variabledescriptorsetter",
207                                         value: expression.arguments[0].value
208                                     };
209                             }
210                             break;
211                         default:
212                             //return new MethodSelect({
213                             //    object: this.generateExpression(expression.callee.object),
214                             //    name: expression.callee.property.value,
215                             //    arguments: expression.callee.arguments
216                             var vdSelect = this.generateExpression(expression.callee.object), args = [];
217 
218                             Y.Array.each(expression.arguments, function (i) {
219                                 args.push(i.value || i.name);
220                             });
221                             Y.mix(vdSelect, {
222                                 //type: "variabledescriptormethodselect",
223                                 //object: this.generateExpression(expression.callee.object),
224                                 //fields: [],
225                                 method: expression.callee.property.name,
226                                 arguments: args
227 
228                             });
229                             return [vdSelect];
230                     }
231                     break;
232             }
233             throw new Error("Unable to parse expression.");
234         }
235     });

254      * @constructor
255      * @extends

258     var VariableDescriptorSelect = function (options) {
259         VariableDescriptorSelect.superclass.constructor.call(this, options);
260     };
261 
262     Y.extend(VariableDescriptorSelect, inputEx.Group, {
263 
264         entityId: null,
265         /**
266 	 * Setup the options.fields from the availableFields option
267 	 */
268         setOptions: function (options) {
269             options.fields = options.fields || [];
270             VariableDescriptorSelect.superclass.setOptions.call(this, options);
271         },
272 
273         render: function () {
274             VariableDescriptorSelect.superclass.render.call(this);
275             this.syncUI();
276         },
277 
278         setValue: function (val) {
279         // Set value should not ba called directly
280         //Y.log("VariableDescriptorSelect.setValue", val);
281         },
282 
283         getValue: function () {
284             return "VariableDescriptorFacade.find(" + this.inputs[this.inputs.length - 1].getValue() + ")";
285         },
286         getEntityId: function () {
287             return this.inputs[this.inputs.length - 3].getValue();
288         },
289         getEntity: function () {
290             return Y.Wegas.VariableDescriptorFacade.rest.findById(this.getEntityId());
291         },
292         getMethods: function (entity) {
293             var i, methods = entity.getMethodCfgs(), ret = [];
294             for (i in methods) {
295                 if (!this.options.returnsFilter                                 // Apply filter on the method return type
296                     || this.options.returnsFilter.indexOf(methods[i].returns || "void") >= 0) {
297 
298                     methods[i].value = i;
299                     methods[i].label =  methods[i].label || i;
300                     ret.push(methods[i]);
301                 /* ret.push({
302                         label: methods[i].label || i,
303                         value: i
304                     });*/
305                 }
306             }
307             return ret;
308         },
309 
310         syncUI: function () {
311             this.empty();
312 
313             var i, ret = [],
314             currentId = this.entityId || this.options.value,
315             rootEntities = Y.Wegas.VariableDescriptorFacade.rest.getCache(),
316             currentEntity = Y.Wegas.VariableDescriptorFacade.rest.findById(currentId) || rootEntities[0];
317 
318             while (this.getMethods(currentEntity).length === 0                  // If the current entity has no methods,
319                 && currentEntity.get("items") && currentEntity.get("items").length > 0) { // but it has a child
320                 currentEntity = currentEntity.get("items")[0];                  // select its first child
321             }
322 
323             this.currentEntity = currentEntity;                                 // Keeps a reference to the current entity
324 
325             while (currentEntity.parentDescriptor) {                            // Add the current entity hierarchy
326                 ret.push(this.generateSelectConfig(currentEntity.parentDescriptor,
327                     currentEntity, currentEntity.parentDescriptor.get("items")));
328                 currentEntity = currentEntity.parentDescriptor;
329             }
330             ret.push(this.generateSelectConfig(null,                            // Add the root context (entities that are at the root of the gameModel
331                 currentEntity, rootEntities));
332 
333             ret = ret.reverse();
334             for (i = 0; i < ret.length; i += 1) {
335                 this.addField(ret[i]);
336             }
337         },
338         /**
339          *
340          * @overrride Y.inputEx.Group.onChange()
341          */
342         onChange: function (fieldValue, fieldInstance) {
343 
344             if (Y.Lang.isNumber(fieldValue)) {                                  // value is an number, it is the new current entity's id'
345                 this.entityId = fieldValue;
346                 this.options.method = null;
347                 this.options.arguments = null;
348                 this.syncUI();
349             } else if (Y.Lang.isString(fieldValue)) {                           // The id is a string, it's the new current mehtod
350                 this.entityId = fieldInstance.options.parentEntity.get("id");
351                 this.options.method = fieldValue;
352                 this.options.arguments = null;
353                 this.syncUI();
354             } else {                                                            // Otherwise current args value
355                 this.options.arguments = this.inputs[this.inputs.length - 1].getValue();
356             }
357             this.fireUpdatedEvt();
358         },
359 
360         empty: function () {
361             while (this.inputs.length > 0) {
362                 this.inputs.pop().destroy();
363             }
364         },
365 
366         /**
367          *  Overriden to add reference to parententity
368          */
369         addField: function (fieldOptions) {
370             VariableDescriptorSelect.superclass.addField.call(this, fieldOptions);
371             this.inputs[this.inputs.length - 1].options.parentEntity = fieldOptions.parentEntity;
372         },
373         /**
374          * Generate
375          */
376         generateSelectConfig: function (entity, selectedEntity, items) {
377             var value;
378 
379             if (entity === this.currentEntity) {
380                 value = this.options.method;
381             }
382             if (selectedEntity) {
383                 value = selectedEntity.get("id");
384             }
385             return {
386                 type: 'select',
387                 choices: this.genChoices(entity, items),
388                 value: value,
389                 parentEntity: entity
390             };
391         },
392         genChoices: function (entity, items) {
393             var i, choices = [];
394 
395             if (items) {
396                 for (i = 0; i < items.length; i++) {
397                     choices.push({
398                         value: items[i].get("id"),
399                         label: items[i].get("editorLabel")
400                     });
401                 }
402             }
403             return choices;
404         }
405     });

415     var VariableDescriptorMethod = function (options) {
416         VariableDescriptorMethod.superclass.constructor.call(this, options);
417     };
418 
419     Y.extend(VariableDescriptorMethod, VariableDescriptorSelect, {
420 
421         syncUI: function () {
422             VariableDescriptorMethod.superclass.syncUI.call(this);
423 
424             this.addField(this.generateSelectConfig(this.currentEntity,         // Pushes the current entity methods and children to the stack
425                 null, this.currentEntity.get("items")));
426 
427             var i, args, methods  = this.getMethods(this.currentEntity),
428             cMethod = methods[this.options.method];
429 
430             if (!cMethod && Y.Object.values(methods).length > 0) {
431                 cMethod = Y.Object.values(methods)[0];                          // By default select the first method available
432             }
433 
434             args = (cMethod && cMethod.arguments) ? cMethod.arguments : [];
435 
436             for (i = 0; i < args.length; i = i + 1) {
437                 args[i].entity = this.currentEntity;                            // Adds a reference to the target entity to the argument Fields;
438             }
439 
440             this.currentMethod = cMethod;
441 
442             this.addField(Y.mix({
443                 type: "combine",
444                 fields: args,
445                 value: this.options.arguments,
446                 label: null
447             }, cMethod));
448 
449         // Same as above, but using json object format for method definitions
450         //var schemaMap = {
451         //    Entity: {
452         //        type: "array",
453         //        items: cMethod.arguments
454         //    }
455         //}, builder = new Y.inputEx.JsonSchema.Builder({
456         //    'schemaIdentifierMap': schemaMap,
457         //    'defaultOptions':{
458         //        'showMsg':true
459         //    }
460         //}), field  = builder.schemaToInputEx(schemaMap.Entity);
461         },
462         setOptions: function (options) {
463             VariableDescriptorMethod.superclass.setOptions.call(this, options);
464             this.options.method = options.method;
465             this.options.arguments = options.arguments;
466             this.argsOffset = 1;
467         },
468 
469         getValue: function () {
470             var i, l = this.inputs.length,
471             args = this.inputs[l - this.argsOffset].getValue(),
472             method = this.inputs[l - this.argsOffset - 1].getValue();
473 
474             if (!method) {
475                 return "true";
476             }
477             for (i = 0; i < args.length; i = i + 1) {
478                 if (this.currentMethod.arguments[i].scriptType === "string") {
479                     args[i] = '"' + args[i].replace('"', '\\"') + '"';
480                 }
481             }
482             return "VariableDescriptorFacade.find(" + this.inputs[l - this.argsOffset - 2].getValue() + ")" +
483             "." + method +
484             "(" + args.join(", ") + ")";
485         },
486         /**
487          * Generate choices for a given entity: add it's methods and then pass up to parent class
488          * to add children variable descriptors.
489          */
490         genChoices: function (entity, items) {
491             var choices = [];
492             if (entity) {
493                 choices = choices.concat(this.getMethods(entity));              // Push the methods to the select choices
494             }
495 
496             if (items && choices.length > 0) {                                  // If required, push separator
497                 choices.push({
498                     value: "----------"
499                 });
500             }
501             return choices.concat(VariableDescriptorMethod.superclass.genChoices.apply(this, arguments));
502         }
503     });
