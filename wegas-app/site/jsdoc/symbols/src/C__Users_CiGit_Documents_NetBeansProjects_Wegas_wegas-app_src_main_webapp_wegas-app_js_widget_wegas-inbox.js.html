 15 YUI.add('wegas-inbox', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19     InboxDisplay;
 20 
 21     InboxDisplay = Y.Base.create("wegas-inbox", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 22 
 23         // *** Private Fields *** //
 24         tabView: null,
 25         dataSource: null,
 26 
 27         // *** Lifecycle Methods *** //
 28         initializer: function () {
 29             this.dataSource = Y.Wegas.app.dataSources.VariableDescriptor;
 30         },
 31 
 32         renderUI: function () {
 33             this.tabView = new Y.TabView();
 34             this.tabView.render(this.get(CONTENTBOX));
 35         },
 36 
 37         bindUI: function () {
 38             this.handlers = {};
 39 
 40             this.tabView.after("selectionChange", this.onTabSelected, this);
 41 
 42             this.handlers.respone = this.dataSource.after("response", this.syncUI, this);
 43             this.handlers.playerChange = Y.Wegas.app.after('currentPlayerChange', this.syncUI, this);
 44         },
 45 
 46         syncUI: function () {
 47             var i, msg, tab, from,
 48             inboxVariable = this.get('variable.evaluated').getInstance(),
 49             messages = inboxVariable.get("messages"),
 50             selectedIndex = 0,
 51             tabs = [];
 52 
 53             this.isSyncing = true;
 54             this.tabView.removeAll();
 55             for (i = messages.length - 1; i >= 0; i -= 1) {
 56                 msg = messages[i];
 57                 from  = msg.get("from") || "admin@wegas.com";
 58                 tab = new Y.Tab({
 59                     label: '<div class="' + (msg.get("unread") ? "unread" : "read") + '"><div class="left">' + from + '</div>'
 60                     + '<div class="right">' + msg.get("subject") + '</div></div>',
 61                     content: '<div class="msg-subject">Subject: ' + msg.get("subject") + '</div>'
 62                     + '<div class="msg-from">From: ' + from + '</div>'
 63                     + '<div class="msg-body">' + msg.get("body") + '</div>'
 64                 });
 65                 tab.msg = msg;
 66                 tabs.push(tab);
 67 
 68                 if (this.msg && msg.id === this.msg.id) {
 69                     selectedIndex = i;
 70                 }
 71             }
 72             this.tabView.add(tabs);
 73 
 74             if (messages.length === 0) {
 75                 this.tabView.add({
 76                     label: '',
 77                     content: '<center><em>You have no messages</em></center>'
 78                 });
 79             }
 80 
 81             this.isSyncing = false;
 82             this.tabView.selectChild(selectedIndex);
 83         },
 84 
 85         destructor: function () {
 86             this.tabView.destroy();
 87             var i;
 88             for (i in this.handlers) {
 89                 this.handlers[i].detach();
 90             }
 91         },
 92 
 93         // *** Private Methods *** //
 94         onTabSelected: function (e) {
 95             if (this.isSyncing) {
 96                 return;
 97             }
 98 
 99             if (this.timer) {                                                   // If there is an active unread
100                 this.timer.cancel();                                            // message timer, we cancel it.
101             }
102 
103             if (e.newVal && e.newVal.msg) {
104                 this.msg = e.newVal.msg;
105 
106                 if (e.newVal.msg.get("unread")) {                                      // If the message is currently unread,
107                     this.timer = Y.later(2000, this, function () {              // we send a request to mark it as read
108                         Y.log("Sending message read update", "info",  "InboxDisplay");
109                         this.msg.set("unread", false);
110                         this.dataSource.rest.sendRequest({
111                             request: "/InboxDescriptor/Message/" + this.msg.get("id"),
112                             cfg: {
113                                 method: "PUT",
114                                 data: Y.JSON.stringify(this.msg)
115                             }
116                         });
117                     });
118                 }
119             }
120         }
121 
122     }, {
123         ATTRS : {
124             /**
125              * The target variable, returned either based on the name attribute,
126              * and if absent by evaluating the expr attribute.
127              */
128             variable: {
129                 getter: Y.Wegas.persistence.Editable.VARIABLEDESCRIPTORGETTER
130             }
131         }
132     });
133 
134     Y.namespace('Wegas').InboxDisplay = InboxDisplay;
135 });
