 15 YUI.add('wegas-joingamewidget', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox',
 19     BOUNDINGBOX = 'boundingBox',
 20     JoinGameWidget;
 21 
 22     /**
 23      *
 24      *  @class Y.Wegas.JoinGameWidget
 25      */
 26     JoinGameWidget = Y.Base.create("wegas-joingamewidget", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 27         // *** Private fields *** //
 28 
 29         // *** Lifecycle Methods *** //
 30         renderUI: function () {
 31             var cb = this.get(CONTENTBOX);
 32 
 33             this.tokenField = new Y.inputEx.StringField({                       // Render
 34                 required: "true",
 35                 parentEl: cb,
 36                 label: "Enter a key phrase to join a game"
 37             });
 38             this.joinGameButton = new Y.Button({
 39                 label: "Join game"
 40             });
 41             this.joinGameButton.render(cb);
 42 
 43             this.teamsField = new Y.inputEx.SelectField({                       // Render team selection
 44                 required: "true",
 45                 parentEl: cb,
 46                 label: "Select a new you want to join"
 47             });
 48             this.teamsField.addClassName("inputEx-hidden");
 49             this.joinTeamButton = new Y.Button({
 50                 label: "Join team",
 51                 visible: false
 52             });
 53             this.joinTeamButton.render(cb);
 54 
 55             this.createTeamField = new Y.inputEx.StringField({                  // Render team creation
 56                 required: "true",
 57                 parentEl: cb,
 58                 label: "OR Create a new team:"
 59             });
 60             this.createTeamField.addClassName("inputEx-hidden");
 61             this.createButton = new Y.Button({
 62                 label: "Create team",
 63                 visible: false
 64             });
 65             this.createButton.render(cb);
 66         },
 67 
 68         bindUI: function () {
 69             this.joinGameButton.on("click", function (e) {                      // join a game based on a token
 70                 if (this.tokenField.validate()) {
 71                     Y.Wegas.GameFacade.rest.sendRequest({
 72                         request: "/JoinGame/" + this.tokenField.getValue(),
 73                         on: {
 74                             success: Y.bind(function (e) {
 75                                 if (e.response.entity                           // If the returned value is a Team enity
 76                                     instanceof Y.Wegas.persistence.Team) {
 77                                     this.sendJoinTeamRequest(                   // it means we can join this team directly
 78                                         e.response.entity.get("id"));
 79                                 } else {
 80                                     this.showTeams();                           // otherwise the player can choose or create its team
 81                                 }
 82                             }, this),
 83                             failure: Y.bind(function (e) {
 84                                 this.showMessage("error", e.response.results.message || "Invalid token", 4000);
 85                             }, this)
 86                         }
 87                     });
 88                 }
 89             }, this);
 90 
 91             this.joinTeamButton.on("click", function (e) {                      // Join an existing team
 92                 if (this.teamsField.validate()) {
 93                     this.sendJoinTeamRequest(this.teamsField.getValue());
 94                 }
 95             }, this);
 96             this.createButton.on("click", function (e) {                      // Create a new team
 97                 if (this.createTeamField.validate()) {
 98                     var team = new Y.Wegas.persistence.Team({
 99                         name: this.createTeamField.getValue()
100                     });
101                     Y.Wegas.GameFacade.rest.post(team.toObject(), this.currentGame.toObject(), {
102                         success: Y.bind(function (e) {
103                             this.sendJoinTeamRequest(e.response.entity.get("id"));
104                         }, this),
105                         failure: Y.bind(function (e) {
106                             this.showMessage("error", e.response.results.message || "Error creating team", 4000);
107                         }, this)
108                     });
109                 }
110             }, this);
111         },
112 
113         showTeams: function () {
114             //this.msg.empty();
115 
116             this.joinGameButton.hide();
117             this.tokenField.addClassName("inputEx-hidden");
118             this.joinTeamButton.show();
119             this.teamsField.removeClassName("inputEx-hidden");
120             this.createButton.show();
121             this.createTeamField.removeClassName("inputEx-hidden");
122 
123             this.currentGame = Y.Wegas.GameFacade.rest.getCache()[0];
124 
125             var i, teams = this.currentGame.get("teams");
126 
127             for (i = 0; i < teams.length; i = i + 1) {
128                 this.teamsField.addChoice({
129                     label: teams[i].get("name"),
130                     value: teams[i].get("id")
131                 });
132             }
133         },
134 
135         sendJoinTeamRequest: function (teamId) {
136             Y.Wegas.GameFacade.rest.sendRequest({
137                 request: "/JoinTeam/" + teamId,
138                 on: {
139                     success: Y.bind(function (e) {
140 
141                         this.showMessage("success", "Game joined, it has been added to your games", 10000);
142 
143                         Y.Wegas.RegisteredGamesFacade.rest.clearCache();
144                         Y.Wegas.RegisteredGamesFacade.sendInitialRequest();
145 
146                         this.createButton.hide();
147                         this.createTeamField.addClassName("inputEx-hidden");
148                         this.joinTeamButton.hide();
149                         this.teamsField.addClassName("inputEx-hidden");
150                         this.joinGameButton.hide();
151                         this.tokenField.addClassName("inputEx-hidden");
152                     }, this),
153                     failure: Y.bind(function (e) {
154                         this.showMessage("error", "Error joinging team");
155                     }, this)
156                 }
157             });
158         }
159     });
160 
161     Y.namespace('Wegas').JoinGameWidget = JoinGameWidget;
162 });
