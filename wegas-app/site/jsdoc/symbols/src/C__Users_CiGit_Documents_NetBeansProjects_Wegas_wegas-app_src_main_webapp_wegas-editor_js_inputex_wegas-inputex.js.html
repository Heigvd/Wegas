 15 YUI.add('wegas-inputex', function (Y) {
 16     "use strict";
 17 
 18     var inputEx = Y.inputEx,
 19     lang = Y.Lang;
 20 
 21     /**
 22      *  @hack Set up inputex providers for Wegas,
 23      *  so inputex will use our own widgets when using f = Y.inpuex(cfg)
 24      */
 25     YUI_config.groups.inputex.modulesByType.html = "wegas-inputex-rte";
 26     YUI_config.groups.inputex.modulesByType.hashlist = "wegas-inputex-hashlist";
 27     YUI_config.groups.inputex.modulesByType.ace = "wegas-inputex-ace";
 28     YUI_config.groups.inputex.modulesByType.wegasurl = "wegas-inputex-url";
 29     YUI_config.groups.inputex.modulesByType.script = "wegas-inputex-wysiwygscript";
 30     YUI_config.groups.inputex.modulesByType.entityarrayfieldselect = "wegas-inputex-wysiwygscript";
 31     YUI_config.groups.inputex.modulesByType.roleselect = "wegas-inputex-roleselect";
 32 
 33     /**
 34      *  @hack So we can easily change classs on inputex fields
 35      */
 36     Y.inputEx.Field.prototype.addClassName = function (className) {
 37         Y.one(this.divEl).addClass(className);
 38     };
 39     Y.inputEx.Field.prototype.removeClassName = function (className) {
 40         Y.one(this.divEl).removeClass(className);
 41     };
 42 
 43     /**
 44      * @hack Let inputex also get requirement from selectfields, lists
 45      */
 46     Y.inputEx.getRawModulesFromDefinition = function (inputexDef) {
 47         function walk(cfg) {                                                    // recursive for group,forms,list,combine, etc...
 48             var type = cfg.type || 'string',
 49             modules = [YUI_config.groups.inputex.modulesByType[type]];
 50 
 51             if (cfg.fields) {
 52                 Y.Array.each(cfg.fields, function (field) {
 53                     modules = modules.concat(walk(field));
 54                 }, this);
 55             }
 56             if (cfg.availableFields) {
 57                 Y.Array.each(cfg.availableFields, function (field) {
 58                     modules = modules.concat(walk(field));
 59                 }, this);
 60             }
 61 
 62             if (cfg.elementType) {
 63                 modules = modules.concat(walk(cfg.elementType));
 64             }
 65             return modules;
 66         // TODO: list elementType
 67         // TODO: inplaceedit  editorField
 68         }
 69         var ret =  walk(inputexDef);
 70         return ret;
 71     };
 72 
 73     /*
 74      * @hack prevents KeyValueField to return the selector field
 75      */
 76     Y.inputEx.KeyValueField.prototype.getValue = function () {
 77         return this.inputs[1].getValue();
 78     };
 79     /*
 80      * @hack prevents KeyValueField to return the selector field
 81      */
 82     //    Y.inputEx.KeyValueField.prototype.setValue = function (val, e) {
 83     //        Y.inputEx.KeyValueField.superclass.setValue.apply(this, arguments)
 84     //        this.inputs[1].setValue(val);
 85     //    };
 86     /*
 87      * @hack Automatically add the "optional" message when necessary
 88      */
 89     Y.inputEx.StringField.prototype.setOptions = function (options) {
 90         Y.inputEx.StringField.superclass.setOptions.call(this, options);
 91 
 92         this.options.regexp = options.regexp;
 93         this.options.size = options.size;
 94         this.options.maxLength = options.maxLength;
 95         this.options.minLength = options.minLength;
 96         this.options.typeInvite = options.typeInvite;
 97         if (!this.options.required && this.options.typeInvite === undefined) {  // !!!MODIFIED!!!
 98             this.options.typeInvite = "optional";
 99         }
100         this.options.readonly = options.readonly;
101         this.options.autocomplete = lang.isUndefined(options.autocomplete) ?
102         inputEx.browserAutocomplete :
103         (options.autocomplete === false || options.autocomplete === "off") ? false : true;
104         this.options.trim = (options.trim === true) ? true : false;
105     };
106 });
