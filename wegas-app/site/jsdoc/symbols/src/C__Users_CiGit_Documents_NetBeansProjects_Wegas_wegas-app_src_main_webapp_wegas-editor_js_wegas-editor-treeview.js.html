 15 YUI.add('wegas-editor-treeview', function (Y) {
 16     "use strict";
 17 
 18     var CONTENTBOX = 'contentBox', EditorTreeView,
 19     EDITBUTTONTPL = "<span class=\"wegas-treeview-editmenubutton\"></span>";
 20 
 21     EditorTreeView = Y.Base.create("wegas-editor-treeview", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 22 
 23         // *** Private fields ** //
 24         treeView: null,
 25         expandedIds: {},
 26 
 27         // ** Lifecycle methods ** //
 28 
 29         renderUI: function () {
 30             this.treeView = new Y.TreeView();
 31             this.treeView.render(this.get( CONTENTBOX ));
 32             this.menu = new Y.Wegas.Menu();
 33         },
 34 
 35         bindUI: function () {
 36             if (this.get("dataSource")) {
 37                 this.get("dataSource").after("response", this.syncUI, this);    /* Listen updates on the target datasource */
 38                 this.get("dataSource").after("error", function(e) {             /* GLOBAL error message */
 39                     this.showMessage("error", e.response.results.message);
 40                 }, this);
 41             }
 42             this.treeView.on("*:click", this.onTreeViewClick, this);
 43 
 44             this.treeView.before("*:nodeExpanded", function (e) {
 45                 this.expandedIds[e.node.get( "data" ).entity.get( "id" )] = true;
 46             }, this);
 47 
 48             this.treeView.before("*:nodeCollapsed", function(e) {
 49                 delete this.expandedIds[e.node.get("data").entity.get("id")];
 50             }, this);
 51         },
 52 
 53         syncUI: function() {
 54             this.set("dataSource", this.get("dataSource"));
 55 
 56             if (!this.get("dataSource")) {
 57                 this.get(CONTENTBOX).append("Unable to find datasource");
 58                 return;
 59             }
 60 
 61             var ds = this.get("dataSource"),
 62             selector = this.get("dataSelector"),
 63             entities =  (selector) ? ds.rest.find( selector.key, selector.val ) : ds.rest.getCache(),
 64             msg = this.get(CONTENTBOX).one(".wegas-smallmessage");
 65 
 66             if (msg) {
 67                 msg.remove(true);
 68             }
 69             this.treeView.removeAll();
 70             if (entities.length == 0) {
 71                 this.get(CONTENTBOX).append('<div class="wegas-smallmessage">' + this.get( "emptyMessage" ) + '</div>');
 72                 return;
 73             }
 74             this.treeView.add(this.genTreeViewElements(entities));
 75             this.treeView.syncUI();
 76         },
 77 
 78         // *** Private Methods *** //
 79         onTreeViewClick: function(e) {
 80             Y.log(e.target.get("label") + " label was clicked", "info", "Wegas.EditorTreeView");
 81 
 82             var menuItems, data = e.node.get("data"),
 83             domTarget = e.domEvent.target;
 84             this.currentSelection = data.entity.get("id");
 85             data.dataSource = this.get( "dataSource" );
 86 
 87             menuItems = data.entity.getMenuCfg(data);
 88 
 89             if (menuItems.length == 0) {
 90                 return;
 91             }
 92 
 93             this.menu.removeAll();                                              // Populate the menu with the elements associated to the
 94             this.menu.add(menuItems);
 95 
 96             if (domTarget.hasClass( "wegas-treeview-editmenubutton")) {       // If user clicked on the edit button
 97                 this.menu.attachTo( domTarget );                                // Display the edit button next to it
 98             } else {                                                            // Otherwise the user clicked on the node
 99                 this.menu.item( 0 ).fire("click");                            // Excute the actions associated to the first item of the menu
100             }
101         },
102 
103         isNodeExpanded: function(id) {
104             return this.expandedIds[id] || false;
105         },
106 
107         genTreeViewElements: function (elements) {
108             var ret = [], i, el, elClass, text, collapsed, selected;
109 
110             for (i in elements) {
111                 if (elements.hasOwnProperty(i)) {
112                     el = elements[i];
113                     elClass = (el.get) ? el.get('@class') : el['type'];
114                     collapsed = (el.get) ? !this.isNodeExpanded(el.get("id")) : true;
115                     selected = (el.get) ? ((this.currentSelection == el.get("id")) ? 2 : 0) : 0;
116 
117                     if ((this.get("excludeClasses") === null
118                         || !this.get('excludeClasses').hasOwnProperty(elClass))
119                     && (this.get('includeClasses') === null
120                         || this.get('includeClasses').hasOwnProperty(elClass))) {
121 
122                         switch (elClass) {
123                             case 'StringDescriptor':
124                             case 'NumberDescriptor':
125                             case 'InboxDescriptor':
126                             case 'TriggerDescriptor':
127                             case 'TaskDescriptor':
128                             case 'ResourceDescriptor':
129                             case 'DialogueDescriptor':
130                                 text = el.get('@class').replace("Descriptor", "") + ': ' + el.getPrivateLabel();
131                                 ret.push({
132                                     type: 'TreeNode',
133                                     label: text,
134                                     children: this.genScopeTreeViewElements(el),
135                                     data: {
136                                         entity: el
137                                     },
138                                     collapsed: collapsed,
139                                     selected: selected,
140                                     rightWidget: Y.Node.create(EDITBUTTONTPL),
141                                     iconCSS: "wegas-icon-variabledescriptor"
142                                 //iconCSS: "wegas-icon-" + el.get('@class')
143                                 });
144                                 break;
145 
146                             case 'ListDescriptor':
147                                 text = el.get('@class').replace("Descriptor", "") + ': ' + el.getPrivateLabel();
148                                 ret.push({
149                                     type: 'TreeNode',
150                                     label: text,
151                                     collapsed: collapsed,
152                                     selected: selected,
153                                     children: this.genTreeViewElements(el.get("items")),
154                                     data: {
155                                         entity: el
156                                     },
157                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
158                                 });
159                                 break;
160 
161                             case 'QuestionDescriptor':
162                                 text = el.get('@class').replace("Descriptor", "") + ': ' + el.getPrivateLabel();
163                                 ret.push({
164                                     type: 'TreeNode',
165                                     label: text,
166                                     collapsed: collapsed,
167                                     selected: selected,
168                                     children: this.genTreeViewElements(el.get("items")),
169                                     data: {
170                                         entity: el
171                                     },
172                                     iconCSS: "wegas-icon-variabledescriptor",
173                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
174                                 });
175                                 break;
176 
177                             case 'ChoiceDescriptor':
178                                 text = el.get('@class').replace("Descriptor", "") + ': ' + el.getPrivateLabel();
179 
180                                 var l, result, children = [];
181                                 for (l = 0; l < el.get("results").length ; l += 1) {
182                                     result = el.get( "results" )[l];
183                                     //TODO : result should be an entity
184                                     if (!(result instanceof Y.Wegas.persistence.Entity)) {
185                                         result = Y.Wegas.persistence.Editable.readObject(result);
186                                     }
187                                     children.push({
188                                         label: "Result: " + result.get( "name" ),
189                                         selected: (result.get("id") == this.currentSelection) ? 2 : 0,
190                                         data: {
191                                             entity: result,
192                                             parentEntity: el
193                                         },
194                                         rightWidget: Y.Node.create(EDITBUTTONTPL),
195                                         iconCSS: "wegas-icon-variabledescriptor"
196                                     });
197                                 }
198 
199                                 ret.push({
200                                     type: 'TreeNode',
201                                     label: text,
202                                     children: children,
203                                     data: {
204                                         entity: el
205                                     },
206                                     collapsed: collapsed,
207                                     selected: selected,
208                                     rightWidget: Y.Node.create(EDITBUTTONTPL),
209                                     iconCSS: "wegas-icon-variabledescriptor"
210                                 });
211                                 break;
212 
213                             case 'SingleResultChoiceDescriptor':
214                                 text = 'Choice: ' + el.getPrivateLabel();
215                                 ret.push({
216                                     type: 'TreeLeaf',
217                                     label: text,
218                                     selected: selected,
219                                     data: {
220                                         entity: el
221                                     },
222                                     rightWidget: Y.Node.create(EDITBUTTONTPL),
223                                     iconCSS: "wegas-icon-variabledescriptor"
224                                 });
225                                 break;
226 
227                             case 'Game':
228                                 text = 'Game: ' + el.get("name") + ' (token:' + el.get("token") + ')';
229                                 ret.push({
230                                     type: 'TreeNode',
231                                     label: text,
232                                     collapsed:  false,
233                                     selected: selected,
234                                     children: this.genTreeViewElements(el.get("teams")),
235                                     data: {
236                                         entity: el
237                                     },
238                                     iconCSS: 'wegas-icon-game',
239                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
240                                 });
241                                 break;
242 
243                             case 'Team':
244                                 text = 'Team: ' + el.get("name") + ' (token: ' + el.get("token") + ")";
245                                 ret.push({
246                                     type: 'TreeNode',
247                                     collapsed: collapsed,
248                                     selected: selected,
249                                     label: text,
250                                     children: this.genTreeViewElements(el.get("players")),
251                                     data: {
252                                         entity: el
253                                     },
254                                     iconCSS: 'wegas-icon-team',
255                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
256                                 });
257                                 break;
258 
259                             case 'Player':
260                                 ret.push({
261                                     label: 'Player: ' + el.get("name"),
262                                     selected: selected,
263                                     data: {
264                                         entity: el
265                                     },
266                                     iconCSS: 'wegas-icon-player',
267                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
268                                 });
269                                 break;
270 
271                             case 'GameModel':
272                                 text = 'Game model: ' + el.get("name");
273                                 ret.push({
274                                     label: text,
275                                     selected: selected,
276                                     data: {
277                                         entity: el
278                                     },
279                                     iconCSS: 'wegas-icon-gamemodel',
280                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
281                                 });
282                                 break;
283 
284                             case 'User':
285                                 ret.push({
286                                     label: 'User:' + el.get( "name" ),
287                                     selected: selected,
288                                     data: {
289                                         entity: el.getMainAccount()
290                                     },
291                                     iconCSS: 'wegas-icon-player',
292                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
293                                 });
294                                 break;
295 
296                             case 'Role':
297                                 ret.push({
298                                     label: 'Group:' + el.get( "name" ),
299                                     selected: selected,
300                                     data: {
301                                         entity: el
302                                     },
303                                     iconCSS: 'wegas-icon-team',
304                                     rightWidget: Y.Node.create(EDITBUTTONTPL)
305                                 });
306                                 break;
307 
308                             case 'List':
309                             case 'Folder':
310                             case "TaskList":
311                             case "InboxDisplay":
312                             case "Score":
313                             case "Layout":
314                             case "Dialogue":
315                                 ret = ret.concat(this.genPageTreeViewElements(el));
316                                 break;

318                             default:
319                                 text = el.get('@class') + ': ' + el.get("name");
320                                 ret.push({

332         genScopeTreeViewElements: function (el) {
333             var children = [], i, label, team, player, instance;
334 
335             for (i in el.get("scope").get("variableInstances")) {
336                 if (el.get("scope").get("variableInstances").hasOwnProperty(i)) {
337                     instance = el.get("scope").get("variableInstances")[i];
338                     label = '';
339                     switch (el.get("scope").get('@class')) {
340                         case 'PlayerScope':
341                             player = Y.Wegas.GameFacade.rest.getPlayerById(i);
342                             label = (player) ? player.get("name") : "undefined";
343                             break;
344                         case 'TeamScope':
345                             team = Y.Wegas.GameFacade.rest.getTeamById(i);
346                             label = (team) ? team.get("name") : "undefined";
347                             break;
348                         case 'GameScope':
349                         case 'GameModelScope':
350                             label = 'Global';
351                             break;
352                     }
353                     children.push(this.genVariableInstanceElements(label, instance));
354                 }
355             }
356             return children;
357         },
358 
359         genVariableInstanceElements: function (label, el) {
360             var l,
361             selected = (el.get) ? ((this.currentSelection == el.get("id")) ? 2 : 0) : 0;
362             switch (el.get('@class')) {
363                 case 'StringInstance':
364                 case 'NumberInstance':
365                 case 'ListInstance':
366                     return {
367                         label: label + ': ' + el.get("value"),
368                         selected: selected,
369                         data: {
370                             entity: el
371                         }
372                     };
373 
374                 case 'QuestionInstance':
375                     l = label + ((el.get("replies").length > 0) ? ': ' + el.get("replies").get("name") : ': unanswered');
376                     return {
377                         label: l,
378                         selected: selected,
379                         data: {
380                             entity: el
381                         }
382                     };
383 
384                 case 'InboxInstance':
385                     var k, children = [];
386 
387                     label += "(" + el.get("messages").length + ")";
388 
389                     for (k = 0; k < el.get("messages").length; k += 1) {
390                         children.push({
391                             label: el.get("messages")[k].get("subject")
392                         });
393                     }
394                     return {
395                         type: 'TreeNode',
396                         label: label,
397                         selected: selected,
398                         data: {
399                             entity: el
400                         },
401                         children: children
402                     };
403 
404                 default:
405                     return {
406                         label: label,
407                         selected: selected,
408                         data: {
409                             entity: el
410                         }
411                     };
412             }
413         },
414 
415         genPageTreeViewElements: function (elts) {
416             var ret = [], j, text, el;
417             if (!Y.Lang.isArray(elts)) {
418                 elts = [ elts ]
419             }
420 
421             for (j = 0; j < elts.length; j += 1) {
422                 el = elts[j];
423                 text =  el.type + ': ' + (el.label || el.name || el.id || 'unnamed');
424                 switch (el.type) {
425                     case 'List':
426                         ret.push({
427                             type: 'TreeNode',
428                             label: 'List: ' + (el.label || 'unnamed'),
429                             data: {
430                                 entity: el
431                             },
432                             children: this.genPageTreeViewElements(el.children)
433                         });
434                         break;
435                     case 'VariableDisplay':
436                         text = 'Variable displayer: ' + (el.variable);
437                         ret.push({
438                             label: text,
439                             data: {
440                                 entity: el
441                             }
442                         });
443                         break;
444                     case 'Text':
445                         ret.push({
446                             label: 'Text: ' + el.content.substring(0, 15) + "...",
447                             data: {
448                                 entity: el
449                             }
450                         });
451                         break;
452                     case 'Button':
453                         ret.push({
454                             label: text,
455                             data: {
456                                 entity: el
457                             }
458                         });
459                         break;
460                     default:
461                         ret.push({
462                             label: text,
463                             data: {
464                                 entity: el
465                             }
466                         });
467                         break;
468 
469                 }
470             }
471             return ret;
472         }
473     }, {

477             },
478             excludeClasses: {
479                 value: null
480             },
481             emptyMessage: {
482                 value: "No data to display"
483             },
484             dataSelector: {},
485             dataSource: {
486                 setter: function ( val ) {
487                     if ( Y.Lang.isString( val )) {
488                         val = Y.Wegas.app.dataSources[val];
489                     }
490                     return val;
491                 }
492             }
493         }
494     });
