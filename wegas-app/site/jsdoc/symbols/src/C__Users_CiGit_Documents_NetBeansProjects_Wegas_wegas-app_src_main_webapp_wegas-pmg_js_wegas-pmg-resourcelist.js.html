 14 YUI.add( "wegas-pmg-resourcelist", function ( Y ) {
 15     "use strict";
 16 
 17     var CONTENTBOX = "contentBox", ResourceList;
 18 
 19     ResourceList = Y.Base.create( "wegas-pmg-resourcelist", Y.Wegas.PmgGantt, [ Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.persistence.Editable], {
 20         
 21         handlers:null,
 22         menu:null,
 23         menuDetails:null,
 24         drag:null,
 25         drop:null,
 26         
 27         addButtonsAssignement:function(){
 28             var cb = this.get(CONTENTBOX);
 29             cb.all(".yui3-datatable-data tr .yui3-datatable-col-assignements").each(function(node){
 30                 node.append("<div class='assignement'></div>");
 31                 node.one('.assignement').append("<span class='assign'></span>");
 32                 node.one('.assignement').append("<span class='remove'></span>");
 33                 node.addClass('noDescription');
 34             });
 35         },
 36         
 37         createMenu: function(e, add){
 38             var i, tasks, resources, resourceDesc, resourceName;
 39             resourceName = e.target.ancestor().ancestor().ancestor().one('*').getContent();
 40             resources = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get('variables'));
 41             for (i = 0; i < resources.get('items').length; i++) {
 42                 if(resources.get('items')[i].get('name') == resourceName){
 43                     resourceDesc = resources.get('items')[i];
 44                     break;
 45                 }
 46             }
 47             this.menu.removeAll();
 48             tasks = this.getTasks(resourceDesc, add);
 49             if(!tasks || tasks.lenght <= 0) return
 50             this.menu.add(tasks);
 51             this.menu.attachTo(e.target);
 52         },
 53         
 54         compareTask: function(a,b) {
 55             var sort = (this.get("taskSortBy") || "name");
 56             if(a.get(sort)){
 57                 if(a.get(sort) < b.get(sort)) return -1;
 58                 if(a.get(sort) > b.get(sort)) return 1;
 59             }else if(a.getInstance().get(sort)){
 60                 if(a.getInstance().get(sort) < b.getInstance().get(sort)) return -1;
 61                 if(a.getInstance().get(sort) > b.getInstance().get(sort)) return 1;
 62             }else {
 63                 if(a.getInstance().get('properties')[sort] < b.getInstance().get('properties')[sort]) return -1;
 64                 if(a.getInstance().get('properties')[sort] > b.getInstance().get('properties')[sort]) return 1;
 65             }
 66             return 0;
 67         },
 68         
 69         getTasks:function(resourceDesc, add){
 70             //add is a boolean to determine if target is remove or add a task
 71             //you can only add a task which isn't already added. 
 72             //you can only remove a task which is added. 
 73             var i, tasks, items, taskDesc, array = new Array(), no;
 74             if(!this.get("taskList")) return;
 75             tasks = Y.Wegas.VariableDescriptorFacade.rest.find("name", this.get("taskList"));
 76             items = tasks.get('items');
 77             items.sort(Y.bind(this.compareTask, this));
 78             for (i = 0; i < items.length; i++) {
 79                 taskDesc = items[i];
 80                 no = (taskDesc.getInstance().get('properties').no || "");
 81                 array.push({
 82                     type: "Button",
 83                     label: no+". "+(taskDesc.get("label") || taskDesc.get("name") || "undefined"),
 84                     data: {
 85                         resource : resourceDesc,
 86                         task: taskDesc
 87                     }
 88                 });
 89             }
 90             return array;
 91         },
 92         
 93         onMenuClick:function(e){
 94             var data = e.target.get("data");
 95             //console.log('to do...')
 96             //console.log(e.target.get("data"), data.resource.get('name'), data.task.get('name'));
 97         },
 98         
 99         taskToCell: function(){     
100             var goingUp = false, lastX = 0;                                     //temporary to test "drag" feature
101             var node, cb = this.get(CONTENTBOX);
102             node = cb.one('.yui3-datatable-data .yui3-datatable-col-assignements');
103             node.append("<div class='tasks'></div>");
104             node.one('.tasks').append('<span class="task">t</span>');
105             node.one('.tasks').append('<span class="task">e</span>');
106             node.one('.tasks').append('<span class="task">s</span>');
107             node.one('.tasks').append('<span class="task">t</span>');
108             node.one('.tasks').append('<span class="task">1</span>');
109             node.one('.tasks').append('<span class="task">2</span>');
110             node.all('.task').each(function(v, k){
111                 this.drag.push(new Y.DD.Drag({
112                     node: v,
113                     //Make it Drop target and pass this config to the Drop constructor
114                     target: {
115                         padding: '0 0 0 20'
116                     }
117                 }).plug(Y.Plugin.DDProxy, {
118                     //Don't move the node at the end of the drag
119                     moveOnEnd: false
120                 }).plug(Y.Plugin.DDConstrained, {
121                     //Keep it inside the #play node
122                     constrain2node: v.ancestor()
123                 }));
124             }, this);
125             node.all('.tasks').each(function(v, k) {
126                 this.drop.push(new Y.DD.Drop({
127                     node: v
128                 }));
129             }, this);
130             
131             
132             //Listen for all drop:over events
133             this.handlers.push(Y.DD.DDM.on('drop:over', function(e) {
134                 //Get a reference to our drag and drop nodes
135                 var drag = e.drag.get('node'),
136                 drop = e.drop.get('node');
137             
138                 //Are we dropping on a span node?
139                 if (drop.get('tagName').toLowerCase() === 'span') {
140                     //Are we not going up?
141                     if (!goingUp) {
142                         drop = drop.get('nextSibling');
143                     }
144                     //Add the node to this list
145                     e.drop.get('node').get('parentNode').insertBefore(drag, drop);
146                     //Resize this nodes shim, so we can drop on it later.
147                     e.drop.sizeShim();
148                 }
149             }));
150             
151             
152             //Listen for all drag:drag events
153             this.handlers.push(Y.DD.DDM.on('drag:drag', function(e) {
154                 //Get the last y point
155                 var x = e.target.lastXY[0];
156                 //is it greater than the lastY var ? (-1, debbug leftmost element)
157                 if (x-1 < lastX) {
158                     //We are going up
159                     goingUp = true;
160                 } else {
161                     //We are going down.
162                     goingUp = false;
163                 }
164                 //Cache for next check
165                 lastX = x;
166             }));
167             
168             
169             //Listen for all drag:start events
170             this.handlers.push(Y.DD.DDM.on('drag:start', function(e) {
171                 //Get our drag object
172                 var drag = e.target;
173                 //Set some styles here
174                 drag.get('node').setStyle('opacity', '.25');
175                 drag.get('dragNode').set('innerHTML', drag.get('node').get('innerHTML'));
176                 drag.get('dragNode').setStyles({
177                     opacity: '.5',
178                     borderColor: drag.get('node').getStyle('borderColor'),
179                     backgroundColor: drag.get('node').getStyle('backgroundColor')
180                 });
181             }));
182             
183             
184             //Listen for a drag:end events
185             this.handlers.push(Y.DD.DDM.on('drag:end', function(e) {
186                 var drag = e.target;
187                 //Put our styles back
188                 drag.get('node').setStyles({
189                     visibility: '',
190                     opacity: '1'
191                 });
192             }));
193             
194             
195             this.handlers.push(Y.DD.DDM.on('drag:drophit', function(e) {
196                 var drop = e.drop.get('node'),
197                 drag = e.drag.get('node');
198 
199                 //if we are not on an span, we must have been dropped on a div
200                 if(drop.get('tagName').toLowerCase() !== 'span'){
201                     if (!drop.contains(drag)) {
202                         drop.appendChild(drag);
203                     }
204                 }
205 
206             }));
207         },
208     
209         initializer: function(){
210             this.handlers = new Array();
211             this.menu = new Y.Wegas.Menu();
212             this.menuDetails = new Y.Wegas.Menu({
213                 width: "250px"
214             });
215             this.drag = new Array();
216             this.drop = new Array();
217         },
218         
219         renderUI: function(){
220             var i, columns;
221             ResourceList.superclass.renderUI.apply(this);
222             columns = this.datatable.head.columns[0];
223             for(i=0; i<columns.length; i++){
224                 if(columns[i].key == 'week1'){
225                     break;
226                 }
227             }
228             this.datatable.addColumn({
229                 key:'assignements',
230                 label:"Assignements"
231             }, i);
232         },
233         
234         bindUI: function(){
235             ResourceList.superclass.bindUI.apply(this);
236             this.handlers.push(Y.Wegas.VariableDescriptorFacade.after("response", this.syncUI, this));
237             this.handlers.push(Y.Wegas.app.after('currentPlayerChange', this.syncUI, this));
238             
239             this.handlers.push(this.datatable.delegate('click', function(e) {            // fill the "add" menu on click
240                 this.createMenu(e, true);
241             }, '.yui3-datatable-data .assignement .assign', this));
242             
243             this.handlers.push(this.datatable.delegate('click', function(e) {            // fill the "remove" menu on click
244                 this.createMenu(e, false);
245             }, '.yui3-datatable-data .assignement .remove', this));
246             
247             this.handlers.push(this.menu.on("button:mouseenter", function(e) {           // align the menu
248                 this.menuDetails.set("align", {
249                     node: this.menu.get("boundingBox"),
250                     points: (e.details[0].domEvent.clientX > Y.DOM.winWidth()/2) ?
251                     ["tr", "tl"] : ["tl", "tr"]
252                 });
253                 this.menuDetails.show();
254                 this.menuDetails.get("contentBox").setHTML('<div style="padding:5px 10px"><i>No description</i></div>');
255             }, this));
256             
257             this.handlers.push(this.menu.on("visibleChange", function(e){                 // When the menu is hidden, hide the details panel
258                 if (!e.newVal) {
259                     this.menuDetails.hide();
260                 }
261             }, this));
262             
263             this.handlers.push(this.menu.on("button:click", this.onMenuClick, this));     // assign a task to a resource
264         },
265         
266         syncUI: function(){
267             ResourceList.superclass.syncUI.apply(this);
268             this.addButtonsAssignement();
269             this.taskToCell();                                                  //temporary
270         },
271         
272         destructor: function(){
273             var i;
274             for (i=0; i<this.handlers.length;i++) {
275                 this.handlers[i].detach();
276             } 
277             for (i=0; i<this.drag.length;i++) {
278                 this.drag[i].destroy();
279             } 
280             for (i=0; i<this.drop.length;i++) {
281                 this.drop[i].destroy();
282             } 
283             this.menu.destroy();
284             this.menuDetails.destroy();
285         } 
286         
287     }, {
288         ATTRS : {
289             taskList:{
290                 value: null,
291                 validator: function (s){
292                     return s === null || Y.Lang.isString(s);
293                 }
294             },
295             taskSortBy:{
296                 value: null,
297                 validator: function (s){
298                     return s === null || Y.Lang.isString(s);
299                 }
300             }
301         }
302     });
303 
304     Y.namespace( "Wegas" ).PmgResourcelist = ResourceList;
305 });
