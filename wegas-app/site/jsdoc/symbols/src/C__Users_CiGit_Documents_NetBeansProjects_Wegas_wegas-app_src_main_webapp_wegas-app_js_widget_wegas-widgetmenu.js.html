 15 YUI.add('wegas-widgetmenu', function (Y) {
 16     "use strict";
 17 
 18     /**
 19      *  @class WidgetMenu
 20      *  @module Wegas
 21      *  @constructor
 22      */
 23     var  WidgetMenu = function () {
 24         WidgetMenu.superclass.constructor.apply(this, arguments);
 25     };
 26 
 27     WidgetMenu.NS = "menu";
 28     WidgetMenu.NAME = "widgetmenu";
 29 
 30     Y.extend(WidgetMenu, Y.Plugin.Base, {
 31 
 32         // *** Lifecycle methods *** //
 33         initializer: function () {
 34             this.afterHostEvent("render", function () {
 35                 var cb = this.get("host").get("contentBox");
 36                 cb.delegate(this.get("event"), function (e) {
 37                     var menu = this.getMenu(e.target);                                  // Get a menu instance
 38 
 39                     menu.attachTo(e.target);                                  // Attach it to the target node
 40                     e.halt();                                                   // Prevent event from bubbling
 41                     this.fire("menuOpen");                                    // Notify the parent the menu has been opened
 42                 }, this.get("selector"), this);
 43             });
 44         },
 45 
 46         // *** Private methods *** //
 47 
 48         getMenu: function () {
 49             if (!this.menu) {
 50                 var cfg = this.get("menuCfg"),
 51                 host = this.get("host"),
 52                 parentWidget = host.get("parent");
 53                 cfg.children = this.get("children");
 54 
 55                 this.menu = new Y.Wegas.Menu(cfg);
 56                 this.menu.addTarget(this);                                    // Catch any event generated by the menu
 57 
 58                 if (parentWidget && parentWidget instanceof Y.Wegas.Menu) {   // Handle nested menus, events are forwarded to the parent widget
 59                     this.menu.on("timerCanceled", function () {
 60                         parentWidget.cancelMenuTimer();
 61                     }, this);
 62                     this.menu.on("timerStarted", function () {
 63                         parentWidget.startMenuHideTimer();
 64                     }, this);
 65 
 66                 }
 67                 //else {
 68                 host.get("contentBox").delegate("mouseleave", function () {
 69                     this.menu.startMenuHideTimer(false);
 70                 }, this.get("selector"), this);
 71                 // }
 72             }
 73             return this.menu;
 74         }
 75     }, {
 76         ATTRS: {
 77             children: {
 78                 value: []
 79             },
 80             selector: {
 81                 value: "*"
 82             },
 83             menuCfg: {
 84                 value: {}
 85             },
 86             event: {
 87                 value: "click"
 88             }
 89         }
 90         //menu: null
 91     });
 92 
 93     Y.namespace('Plugin').WidgetMenu = WidgetMenu;
 94 
 95     Y.namespace('Wegas').Menu = Y.Base.create("menu", Y.Widget, [ Y.WidgetPosition,  Y.WidgetPositionAlign, Y.WidgetStack, Y.WidgetParent, Y.WidgetPositionConstrain ], {
 96 
 97         // *** private fields *** //
 98 
 99         timer: null,
100 
101         // *** Lifecycle methods *** //
102         initializer: function () {
103             this.publish("click", {
104                 emitFacade: true,
105                 bubbles: true
106             });
107             this.publish("cancelMenuTimer", {
108                 bubbles: true
109             });
110         },
111 
112         renderUI: function () {
113             var bb = this.get("boundingBox");
114 
115             bb.on("clickoutside", this.clickOutside, this);
116             bb.on("click", this.menuClick, this);
117             bb.on("mouseenter", this.cancelMenuTimer, this);
118             bb.on("mouseleave", this.startMenuHideTimer, this);
119         },
120 
121         bindUI: function () {
122             this.on("*:click", function (e) {                            // @hack in order for event to be bubbled up
123                 //Y.log("fix");
124             }, this);
125         },
126 
127         hide: function () {
128             Y.Wegas.Menu.superclass.hide.call(this);
129         },
130         // *** Public methods *** //
131         /**
132          *
133          *  Displays the menu next to the provided node and add mouseenter and
134          *  mouseleave callbacks to the node
135          *
136          * @method attachTo
137          */
138 
139         attachTo: function (node) {
140 
141             //node.on("mouseenter", this.show, this);
142             //node.on("mouseleave", this.hide, this);
143             this.currentNode = node;
144 
145             this.set("align", {
146                 node: node,
147                 points: this.get("points")
148             });
149 
150             this.cancelMenuTimer();
151             this.show();
152             // console.log("attachTo", this.get("contentBox").one("button").getHTML());
153         },
154 
155         // *** Private methods *** //
156         menuClick: function () {
157             this.hide();
158             this.fire("menuClick");
159         },
160 
161         clickOutside: function (e) {
162             if (this.currentNode !== e.target) {
163                 this.hide();
164             }
165         },
166 
167         startMenuHideTimer: function (fireEvent) {
168             //console.log("startMenuHideTimer",this.get("contentBox").one("button").getHTML());
169             this.cancelMenuTimer();
170             this.timer = Y.later(500, this, this.hide);
171 
172             if (!!fireEvent) {
173                 this.fire("timerStarted");
174             }
175         },
176         cancelMenuTimer: function () {
177             //console.log("cancelMenuTimer", this.get("contentBox").one("button").getHTML());
178             if (this.timer) {
179                 this.timer.cancel();
180             }
181             this.fire("timerCanceled");
182         }
183     }, {
184         ATTRS: {
185             points: {
186                 value: [ "tl", "bl" ]
187             },
188             constrain: {
189                 value: true
190             },
191             zIndex:{
192                 value: 25
193             },
194             width: {
195                 value: "12em"
196             },
197             render: {
198                 value: true
199             },
200             visible: {
201                 value: false
202             },
203             defaultChildType: {
204                 value: "Button"
205             }
206         }
207     });
208 });
