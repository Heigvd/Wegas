  5 YUI.add('wegas-leaderway-tasklist', function (Y) {
  6     "use strict";
  7 
  8     var CONTENTBOX = 'contentBox', TaskList;
  9 
 10     TaskList = Y.Base.create("wegas-tasklist", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 11 
 12         // *** Fields *** /
 13         table: null,
 14         data: new Array(),
 15         selectedTaskDescriptor:null,
 16         resourceDescriptor: null,
 17         nextPageId: null,
 18         handlers: new Array(),
 19         pickingMode:false,
 20 
 21         //*** Particular Methods ***/
 22         /**
 23          * Change this widget to its picking mode
 24          * The picking mode is used to add a task to a resource.
 25          * Add class "modePicking".
 26          * Display the surname of the selected resoure
 27          * Display the button used to assign the task and display the informations of the selected task.
 28          * Set the identifiant of the next page reached by quitting this widget by a hit on the picking-mode's buttons
 29          * @param ResourceDescriptor resourceDescriptor, the new resourceDescriptor
 30          * @param String nextPageId the identifiant of the nexte page id.
 31          */
 32         switchToPickingMode: function(resourceDescriptor, nextPageId){
 33             var cb = this.get(CONTENTBOX);
 34             this.resourceDescriptor = resourceDescriptor;
 35             if(this.resourceDescriptor != null){
 36                 this.pickingMode=true;
 37                 this.nextPageId = nextPageId;
 38                 cb.addClass('modePicking');
 39                 cb.one('.resourceName p').setHTML('Assigner un mandat ï¿½ '+this.resourceDescriptor.getInstance().get('properties').surname);
 40                 cb.one('.resourceName').show();
 41                 cb.one('.footer').show();
 42             }
 43         },
 44 
 45         /**
 46          * Add rows to the datatable. Get informations on the valide tasks
 47          * @param ListDescriptor listTasksDescriptor, A list of all tasks.
 48          */
 49         getTasksData: function(listTasksDescriptor){
 50             var i, j, k, termData, workers = new Array(), taskDescriptor, taskInstance, resourceDescriptor, resourceInstance, comment,
 51             listResourcesDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "resources"),
 52             currentWeekInstance = Y.Wegas.VariableDescriptorFacade.rest.find("name", "week").getInstance();
 53             for (i = 0; i < listTasksDescriptor.get('items').length; i++) {
 54                 workers.length = 0;
 55                 taskDescriptor = listTasksDescriptor.get('items')[i];
 56                 taskInstance = taskDescriptor.getInstance();
 57                 for(j=0; j<listResourcesDescriptor.get('items').length; j++){
 58                     resourceDescriptor = listResourcesDescriptor.get('items')[j];
 59                     resourceInstance = resourceDescriptor.getInstance();
 60                     for(k=0; k<resourceInstance.get('assignments').length; k++){
 61                         if(taskDescriptor.get('id') == resourceInstance.get('assignments')[k].get('taskDescriptorId')){
 62                             workers.push(resourceInstance.get('properties').surname);
 63                             break;
 64                         }
 65                     }
 66                 }
 67                 if(taskInstance.get('active')){
 68                     termData = (workers.length <= 0)?(taskInstance.get('properties').disappearAtWeek - currentWeekInstance.get('value')) : "-";
 69                     comment = new Array();
 70                     if(taskInstance.get('properties').comment) comment.push(taskInstance.get('properties').comment);
 71                     if(taskInstance.get('properties').workWithLeader == 'true') comment.push("S'effectue en coopï¿½ration avec le leader.");
 72                     if(comment.length <= 0) comment.push("-");
 73                     if(workers.length <= 0) workers.push("-");
 74                     this.data.push({
 75                         id:taskDescriptor.get('id'),
 76                         task:taskDescriptor.get('name'),
 77                         skill:this.getSkillsets(taskInstance),
 78                         duration:taskInstance.get('duration'),
 79                         term:termData,
 80                         salary:taskInstance.get('properties').salary,
 81                         comment:comment.join(" "),
 82                         worker: workers.join(",")
 83                     })
 84                 }
 85             }
 86         },
 87 
 88         /**
 89          * @param TaskInstance taskInstance, the task to get skillsets
 90          * @return String, a texte including all the skillset of the given task.
 91          */
 92         getSkillsets: function(taskInstance){
 93             var temp = new Array();
 94             for (var key in taskInstance.get('skillset')){
 95                 temp.push(key, ' (', taskInstance.get('skillset')[key], ')\n');
 96             }
 97             return temp.join("");
 98         },
 99 
100         /**
101          * @return the current selected task.
102          */
103         getSelectedTaskDescriptor: function(){
104             return this.selectedTaskDescriptor;
105         },
106 
107         /**
108          * This function must be called by a click event.
109          * set the current selected task
110          * @param e, a click event on a row from the datatable.
111          */
112         selectRow: function(e){
113             var i, cb = this.get(CONTENTBOX),
114             listTasksDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "tasks"),taskDescriptorId;
115             //deselect old row
116             if(this.pickingMode){
117                 cb.all('.yui3-datatable-content .selected').removeClass('selected');
118             }
119             //select new row
120             taskDescriptorId = e.currentTarget.one("*").getContent();
121             if(this.pickingMode){
122                 e.currentTarget.addClass('selected');
123             }
124             //get new task descriptor
125             for (i = 0; i < listTasksDescriptor.get('items').length; i++) {
126                 if(listTasksDescriptor.get('items')[i].get('id') == taskDescriptorId){
127                     this.selectedTaskDescriptor = listTasksDescriptor.get('items')[i];
128                     break;
129                 }
130             }
131             if(this.selectedTaskDescriptor!=null){
132                 cb.one('.footer .assignInfo').setHTML('Assigner le mandat : '+this.selectedTaskDescriptor.get('name'));
133                 cb.one('.footer .buttonOK').show();
134             }
135         },
136 
137         /**
138          * Assign the given task to the given ressource.
139          * @param ResourceDescriptor resourceDescriptor, the resource to assign a task.
140          * @param TaskDescriptor taskDescriptor, the task to assign at a ressource.
141          */
142         assignTask: function(resourceDescriptor, taskDescriptor){
143             if(taskDescriptor != null && resourceDescriptor != null){
144                 Y.Wegas.VariableDescriptorFacade.rest.sendRequest({
145                     request: "/Script/Run/Player/" + Y.Wegas.app.get('currentPlayer'),
146                     headers:{
147                         'Content-Type': 'application/json; charset=ISO-8859-1',
148                         'Managed-Mode': 'true'
149                     },
150                     cfg: {
151                         method: "POST",
152                         data: Y.JSON.stringify({
153                             "@class": "Script",
154                             "language": "JavaScript",
155                             "content": "importPackage(com.wegas.core.script);\nassignTask("+resourceDescriptor.get('id')+","+taskDescriptor.get('id')+");"
156                         })
157                     },
158                     on: {
159                         //success: Y.bind(this.assignTaskResult, this, true), // useless, players expect a success
160                         failure: Y.bind(this.assignTaskResult, this, false)
161                     }
162                 });
163             }
164             this.syncUI();
165         },
166 
167         /**
168          * Display a feedback on the operation "add task to a resource"
169          * This feedback will disappear after 5 seconde.
170          * the feedback message appear in node ".leaderway-feedback"
171          * @param Boolean success, true if the opreation was a success, false otherwise.
172          */
173         assignTaskResult: function(success){
174             Y.one('.leaderway-feedback').setStyle('display', 'block');
175             if(success){
176                 Y.one('.leaderway-feedback').one('p').addClass('green');
177                 Y.one('.leaderway-feedback').one('p').insert("Le mandat ï¿½ ï¿½tï¿½ dï¿½lï¿½guï¿½ !");
178             }
179             else{
180                 Y.one('.leaderway-feedback').one('p').addClass('red');
181                 Y.one('.leaderway-feedback').one('p').insert("Le mandat n'a pas pu ï¿½tre dï¿½lï¿½guï¿½.");
182             }
183             setTimeout(function(){
184                 Y.one('.leaderway-feedback').setHTML('<p></p>');
185                 Y.one('.leaderway-feedback').setStyle('display', 'none');
186                 Y.one('.leaderway-feedback').one('p').removeClass('green');
187                 Y.one('.leaderway-feedback').one('p').removeClass('red');
188             }, 5000);
189         },

193         /**
194          * Render the widget.
195          * Hide node corresponding used only with the "picking mode" of this widget
196          */
197         renderUI: function (){
198             var cb = this.get(CONTENTBOX);
199             this.table = new Y.DataTable({
200                 columns: [
201                 {
202                     key:"id",
203                     className: 'hidden'
204                 },
205                 {
206                     key:"task",
207                     label:"Mandat",
208                     sortable:true
209                 },
210                 {
211                     key:"skill",
212                     label:"Compï¿½tence",
213                     sortable:true
214                 },
215                 {
216                     key:"duration",
217                     label:"Durï¿½e",
218                     sortable:true
219                 },
220                 {
221                     key:"term",
222                     label:"Echï¿½ance",
223                     sortable:true
224                 },
225                 {
226                     key:"salary",
227                     label:"Remunï¿½ration",
228                     sortable:true
229                 },
230                 {
231                     key:"comment",
232                     label:"Remarque",
233                     sortable: true
234                 },
235                 {
236                     key:"worker",
237                     label:"Employï¿½",
238                     sortable: true
239                 }
240                 ]
241             });
242             cb.insert('<div class="resourceName"><p></p></div>');
243             this.table.render(cb);
244             cb.insert('<div class="footer"><div class="assignInfo"></div><div class="buttons"><div class="buttonOK">Assigner</div><div class="buttonCancel">Annuler</div></div></div>');
245             cb.one('.resourceName').hide();
246             cb.one('.footer').hide();
247             cb.one('.footer .buttonOK').hide();
248         },
249 
250         /**
251          * Bind some function at nodes of this widget
252          */
253         bindUI: function() {
254             var cb = this.get(CONTENTBOX);
255             this.handlers.push(Y.Wegas.app.dataSources.VariableDescriptor.after("response", this.syncUI, this));
256             this.handlers.push(Y.Wegas.app.after('currentPlayerChange', this.syncUI, this));
257             this.handlers.push(this.table.delegate('click', function (e) {
258                 this.selectRow(e);
259             }, '.yui3-datatable-data tr', this));
260 
261             this.handlers.push(cb.one('.buttons').delegate('click', function (e) {
262                 var targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
263                 targetPageLoader.once("widgetChange", function(e) {
264                     if(e.newVal.setResourceDescriptor){
265                         e.newVal.setResourceDescriptor(this.resourceDescriptor);
266                     }
267                     if(e.newVal.decreaseResourceState){
268                         e.newVal.decreaseResourceState();
269                     }
270                 },{
271                     resourceDescriptor:this.resourceDescriptor
272                     });
273                 this.assignTask(this.resourceDescriptor, this.selectedTaskDescriptor);
274                 targetPageLoader.set("pageId", this.nextPageId);
275             }, '.buttonOK', this));
276 
277             this.handlers.push(cb.one('.buttons').delegate('click', function (e) {
278                 var targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
279                 targetPageLoader.once("widgetChange", function(e) {
280                     if(e.newVal.setResourceDescriptor){
281                         e.newVal.setResourceDescriptor(this.resourceDescriptor);
282                     }
283                 },{
284                     resourceDescriptor:this.resourceDescriptor
285                     });
286                 targetPageLoader.set("pageId", this.nextPageId);
287             }, '.buttonCancel', this));
288 
289         },
290 
291         /**
292          * Synchronise the content of this widget.
293          */
294         syncUI: function (){
295             var listTasksDescriptor = Y.Wegas.VariableDescriptorFacade.rest.find("name", "tasks");
296             if(!listTasksDescriptor) return;
297             this.data.length = 0;
298             this.getTasksData(listTasksDescriptor);
299             this.table.addRows(this.data);
300             if(this.data[0] == null){
301                 this.table.showMessage("Aucun mandat n'est disponible.");
302             }
303             else{
304                 this.table.hideMessage();
305             }
306             this.goToFinalPage();// ! hack function
307         },
308 
309         /*
310          * Destroy all child widget and all function
311          */
312         destructor: function(){
313             var i;
314             this.table.destroy();
315             for (i=0; i<this.handlers.length;i++) {
316                 this.handlers[i].detach();
317             }
318         },
319 
320         // *** hack Methods *** //
321         /**
322          * if current week > max value of week value, then
323          * change the current widget to go on the "dialogue" widget.
324          */
325         goToFinalPage: function(){
326             var currentWeek = Y.Wegas.VariableDescriptorFacade.rest.find("name", "week"),
327             targetPageLoader = Y.Wegas.PageLoader.find(this.get('targetPageLoaderId'));
328             if(parseInt(currentWeek.getInstance().get('value')) > currentWeek.get('maxValue')){
329                 targetPageLoader.once("widgetChange", function(e) {
330                     e.newVal.setCurrentDialogue("dialogueFinal");
331                 });
332                 targetPageLoader.set("pageId", this.get('dialoguePageId'));
333             }
334         }
335 
336     },
337     {
338         ATTRS : {
339             dialoguePageId: {
340                 value:null,
341                 validator: function (s){
342                     return s === null || Y.Lang.isString(s);
343                 }
344             },
345             targetPageLoaderId: {
346                 value:null,
347                 validator: function (s){
348                     return s === null || Y.Lang.isString(s);
349                 }
350             }
351         }
352     });
353 
354     Y.namespace('Wegas').TaskList = TaskList;
355 });
