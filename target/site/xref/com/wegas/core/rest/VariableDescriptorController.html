30  @Stateless
31  @Path("GameModel/{gameModelId : [1-9][0-9]*}/VariableDescriptor")
32  public class VariableDescriptorController extends AbstractRestController<VariableDescriptorEntityFacade> {
33  
34      private static final Logger logger = Logger.getLogger("Authoring_GM_VariableDescriptor");
35      /**
36       *
37       */
38      @EJB
39      private VariableDescriptorEntityFacade variableDescriptorFacade;
40      /**
41       *
42       */
43      @EJB
44      private GameModelEntityFacade gameModelFacade;
45      /**
46       *
47       */
48      @EJB
49      private ScriptManager scriptManager;
50  
51      /**
52       *
53       * @return
54       */
55      @Override
56      @GET
57      @Produces(MediaType.APPLICATION_JSON)
58      public Collection<AbstractEntity> index() {
59          Long gameModelId = this.getGameModelId();
60          GameModelEntity gameModel = gameModelFacade.find(gameModelId);
61          return (Collection) gameModel.getVariableDescriptors();
62      }
63  
64      @Override
65      public AbstractEntity create(AbstractEntity entity) {
66          this.variableDescriptorFacade.create(new Long(this.getPathParam("gameModelId")), (VariableDescriptorEntity) entity);
67          return entity;
68      }
69  
70      /**
71       * Resets all the variables of a given game model
72       *
73       * @param gameModelId game model id
74       * @return OK
75       */
76      @GET
77      @Path("reset")
78      @Produces(MediaType.APPLICATION_JSON)
79      public Collection<VariableDescriptorEntity> reset(@PathParam("gameModelId") Long gameModelId) {
80          return gameModelFacade.reset(gameModelId).getVariableDescriptors();
81      }
82  
83      /**
84       *
85       * @param gameModelId
86       * @param playerId
87       * @param script
88       * @return
89       */
90      @PUT
91      @Path("Player/{playerId : [1-9][0-9]*}/RunScript")
92      @Produces(MediaType.APPLICATION_JSON)
93      public Collection<VariableDescriptorEntity> runScript(@PathParam("gameModelId") Long gameModelId,
94              @PathParam("playerId") Long playerId, ScriptEntity script) {
95          return (Collection) this.scriptManager.runScript(gameModelId, playerId, script);
96      }
97  
98      private Long getGameModelId() {
99          return new Long(this.getPathParam("gameModelId"));
100     }
101 
102     /**
103      *
104      * @return
105      */
106     @Override
107     protected VariableDescriptorEntityFacade getFacade() {
108         return this.variableDescriptorFacade;
109     }
