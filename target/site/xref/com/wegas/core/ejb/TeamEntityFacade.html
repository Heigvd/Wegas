25  @Stateless
26  public class TeamEntityFacade extends AbstractFacade<TeamEntity> {
27  
28      /**
29       *
30       */
31      @EJB
32      private UserEntityFacade userFacade;
33      /**
34       *
35       */
36      @EJB
37      private GameEntityFacade gameFacade;
38      /**
39       *
40       */
41      @PersistenceContext(unitName = "wegasPU")
42      private EntityManager em;
43  
44      /**
45       *
46       * @param gameId
47       * @param t
48       */
49      public void create(Long gameId, TeamEntity t) {
50          GameEntity g = gameFacade.find(gameId);
51          g.addTeam(t);
52          em.flush();
53          em.refresh(t);
54          g.getGameModel().propagateDefaultVariableInstance(false);
55      }
56  
57      /**
58       *
59       * @param teamId
60       * @param userId
61       * @return
62       */
63      public PlayerEntity createPlayer(Long teamId, Long userId) {
64          // logger.log(Level.INFO, "Adding user " + userId + " to team: " + teamId + ".");
65          UserEntity u = userFacade.find(userId);
66          TeamEntity t = this.find(teamId);
67          PlayerEntity p = new PlayerEntity();
68          p.setUser(u);
69          t.addPlayer(p);
70          em.flush();
71          em.refresh(p);
72          t.getGame().getGameModel().propagateDefaultVariableInstance(false);
73          return p;
74      }
75  
76      /**
77       *
78       * @return
79       */
80      @Override
81      protected EntityManager getEntityManager() {
82          return em;
83      }
