32  public class GameController implements Serializable {

34      /**
35       *
36       */
37      @ManagedProperty(value = "#{param.id}")
38      private Long id;
39      /**
40       *
41       */
42      @ManagedProperty(value = "#{param.gameId}")
43      private Long gameId;
44      /**
45       *
46       */
47      @EJB
48      private PlayerEntityFacade playerEntityFacade;
49      /**
50       *
51       */
52      @EJB
53      private GameModelEntityFacade gameModelEntityFacade;
54      /**
55       *
56       */
57      private PlayerEntity currentPlayer;
58  
59      /**
60       *
61       * @throws IOException
62       */
63      @PostConstruct
64      public void init() throws IOException {
65          if (this.id != null) {
66              currentPlayer = playerEntityFacade.find(getId());
67              if (currentPlayer == null) {
68                  FacesContext.getCurrentInstance().getExternalContext().dispatch("/wegas-app/view/lobby.xhtml");
69              }
70          } else if (this.gameId != null) {
71              currentPlayer = gameModelEntityFacade.find(this.gameId).getGames().get(0).getTeams().get(0).getPlayers().get(0);
72              if (currentPlayer == null) {
73                  FacesContext.getCurrentInstance().getExternalContext().dispatch("/wegas-editor/view/lobby.xhtml");
74              }
75          } else {
76              FacesContext.getCurrentInstance().getExternalContext().dispatch("/wegas-editor/view/lobby.xhtml");
77          }
78      }
79  
80      /**
81       *

86      }
