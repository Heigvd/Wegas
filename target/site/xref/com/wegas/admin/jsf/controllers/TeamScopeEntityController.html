188     private void updateCurrentItem() {
189         int count = getFacade().count();
190         if (selectedItemIndex >= count) {
191             // selected index cannot be bigger than number of items:
192             selectedItemIndex = count - 1;
193             // go to previous page if last page disappeared:
194             if (pagination.getPageFirstItem() >= count) {
195                 pagination.previousPage();
196             }
197         }
198         if (selectedItemIndex >= 0) {
199             current = getFacade().findRange(new int[]{selectedItemIndex, selectedItemIndex + 1}).get(0);
200         }
201     }
