63          MCQVariableDescriptorEntity vd = (MCQVariableDescriptorEntity) a;
64          this.setDescription(vd.getDescription());
65          this.setLabel(vd.getLabel());
66          this.setAllowMultipleReplies(vd.getAllowMultipleReplies());
67  
68          List<MCQReplyVariableDescriptorEntity> newReplies = new ArrayList<MCQReplyVariableDescriptorEntity>();
69          //newReplies.addAll(vd.getReplies());
70          for (MCQReplyVariableDescriptorEntity nReply : vd.getReplies()) {
71              int pos = this.replies.indexOf(nReply);
72              if (pos == -1) {
73                  newReplies.add(nReply);
74              } else {
75                  MCQReplyVariableDescriptorEntity oReply = this.replies.get(pos);
76                  oReply.merge(nReply);
77                  newReplies.add(oReply);
78              }
79          }
80          this.setReplies(newReplies);
81      }
